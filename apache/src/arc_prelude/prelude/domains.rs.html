1//! Domain Separation Constants for HKDF
2//!
3//! This module provides domain separation strings used in HKDF key derivation
4//! to ensure cryptographic isolation between different cryptographic operations.
5//!
6//! Domain separation prevents key reuse across different protocols and ensures
7//! that keys derived for one purpose cannot be used for another.
8
9#![deny(unsafe_code)]
10#![deny(missing_docs)]
11#![deny(clippy::unwrap_used)]
12#![deny(clippy::panic)]
13
14/// Domain for hybrid KEM key derivation.
15///
16/// Used when deriving keys from hybrid key encapsulation mechanisms
17/// combining X25519 classical key exchange with ML-KEM-1024 post-quantum KEM.
18pub const HYBRID_KEM: &[u8] = b"LatticeArc-v1-HybridKEM-X25519-MLKEM1024";
19
20/// Domain for cascaded encryption outer layer.
21///
22/// Used for the outer encryption layer when applying cascaded encryption
23/// with ChaCha20-Poly1305 for defense in depth.
24pub const CASCADE_OUTER: &[u8] = b"LatticeArc-v1-Cascade-ChaCha20Poly1305";
25
26/// Domain for cascaded encryption inner layer.
27///
28/// Used for the inner encryption layer when applying cascaded encryption
29/// with AES-256-GCM for defense in depth.
30pub const CASCADE_INNER: &[u8] = b"LatticeArc-v1-Cascade-AES256GCM";
31
32/// Domain for signature binding.
33///
34/// Used when binding dual signatures combining Ed25519 classical signatures
35/// with ML-DSA-87 post-quantum signatures for hybrid authentication.
36pub const SIGNATURE_BIND: &[u8] = b"LatticeArc-v1-DualSignature-Ed25519-MLDSA87";
