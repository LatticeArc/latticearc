20/// NTT processor for fast polynomial operations
21pub struct NttProcessor {
22    /// Precomputed twiddle factors for forward NTT
23    forward_twiddles: Vec<i32>,
24    /// Precomputed twiddle factors for inverse NTT
25    inverse_twiddles: Vec<i32>,
26    /// NTT size (must be power of 2)
27    pub(crate) n: usize,
28    /// Modulus for arithmetic
29    pub(crate) modulus: i64,
30    /// Primitive NTT root of unity
31    primitive_root: i32,
32}
33
34impl NttProcessor {
35    /// Get precomputed forward NTT twiddle factors
36    #[must_use]
37    pub fn forward_twiddles(&self) -> &[i32] {
38        &self.forward_twiddles
39    }
40
41    /// Get precomputed inverse NTT twiddle factors
42    #[must_use]
43    pub fn inverse_twiddles(&self) -> &[i32] {
44        &self.inverse_twiddles
45    }
46    /// Find primitive Nth root of unity modulo modulus
47    ///
48    /// Uses precomputed roots for known parameter sets (Kyber, Dilithium).
49    /// For other parameters, returns an error as finding primitive roots
50    /// modulo arbitrary moduli is computationally expensive.
51    fn find_primitive_root(n: usize, modulus: i64) -> Result<i32> {
52        match (n, modulus) {
53            (256, 3329) => Ok(17),   // Kyber parameters
54            (512, 12289) => Ok(49),  // Dilithium parameters
55            (1024, 12289) => Ok(49), // Dilithium parameters
56            _ => Err(LatticeArcError::InvalidInput(format!(
57                "No known primitive root for N={}, modulus={}",
58                n, modulus
59            ))),
60        }
61    }
62
63    /// Compute forward NTT twiddle factors
64    fn compute_twiddles(n: usize, primitive_root: i32, modulus: i64) -> Result<Vec<i32>> {
65        let mut twiddles = vec![0i32; n];
66        let root_pow = mod_pow(i64::from(primitive_root), (modulus - 1) / n as i64, modulus);
67
68        twiddles[0] = 1;
69        for (i, twiddle) in twiddles.iter_mut().enumerate().skip(1).take(n - 1) {
70            *twiddle = mod_pow(root_pow, i as i64, modulus) as i32;
71        }
72
73        Ok(twiddles)
74    }
75
76    /// Compute inverse NTT twiddle factors
77    fn compute_inverse_twiddles(n: usize, primitive_root: i32, modulus: i64) -> Result<Vec<i32>> {
78        let mut twiddles = vec![0i32; n];
79        let root_pow = mod_pow(i64::from(primitive_root), (modulus - 1) / n as i64, modulus);
80        let inv_root_pow = mod_inverse(root_pow, modulus)?;
81
82        twiddles[0] = 1;
83        for (i, twiddle) in twiddles.iter_mut().enumerate().skip(1).take(n - 1) {
84            *twiddle = mod_pow(inv_root_pow, i as i64, modulus) as i32;
85        }
86
87        Ok(twiddles)
88    }

90    /// Create new NTT processor for given size and modulus
91    ///
92    /// # Errors
93    /// Returns an error if n is not a power of 2, modulus is invalid, or no primitive root is known.
94    pub fn new(n: usize, modulus: i64) -> Result<Self> {
95        if !n.is_power_of_two() {
96            return Err(LatticeArcError::InvalidInput("NTT size must be a power of 2".to_string()));
97        }
98
99        if modulus <= 1 {
100            return Err(LatticeArcError::InvalidInput(
101                "Modulus must be greater than 1".to_string(),
102            ));
103        }
104
105        // Find primitive NTT root of unity
106        let primitive_root = Self::find_primitive_root(n, modulus)?;
107
108        // Precompute twiddle factors
109        let forward_twiddles = Self::compute_twiddles(n, primitive_root, modulus)?;
110        let inverse_twiddles = Self::compute_inverse_twiddles(n, primitive_root, modulus)?;
111
112        Ok(Self { forward_twiddles, inverse_twiddles, n, modulus, primitive_root })
113    }

115    /// Forward NTT transform (coefficient to evaluation domain)
116    ///
117    /// # Errors
118    /// Returns an error if input length does not match NTT size.
119    pub fn forward(&self, coeffs: &[i32]) -> Result<Vec<i32>> {
120        if coeffs.len() != self.n {
121            return Err(LatticeArcError::InvalidInput(format!(
122                "Input length {} doesn't match NTT size {}",
123                coeffs.len(),
124                self.n
125            )));
126        }
127
128        let mut result = coeffs.to_vec();
129        self.ntt(&mut result, false)?;
130        Ok(result)
131    }
132
133    /// Inverse NTT transform (evaluation to coefficient domain)
134    ///
135    /// # Errors
136    /// Returns an error if input length does not match NTT size.
137    pub fn inverse(&self, evaluations: &[i32]) -> Result<Vec<i32>> {
138        if evaluations.len() != self.n {
139            return Err(LatticeArcError::InvalidInput(format!(
140                "Input length {} doesn't match NTT size {}",
141                evaluations.len(),
142                self.n
143            )));
144        }
145
146        let mut result = evaluations.to_vec();
147        self.ntt(&mut result, true)?;
148
149        // Scale by inverse of n
150        // n is bounded to small values (256, 512, 1024) by find_primitive_root
151        let n_i64 = i64::try_from(self.n).map_err(|_e| {
152            LatticeArcError::InvalidInput("NTT size exceeds i64 range".to_string())
153        })?;
154        let n_inv_i64 = mod_inverse(n_i64, self.modulus)?;
155        let n_inv = i32::try_from(n_inv_i64).map_err(|_e| {
156            LatticeArcError::InvalidInput("NTT inverse exceeds i32 range".to_string())
157        })?;
158        for coeff in &mut result {
159            *coeff = self.mod_mul(*coeff, n_inv);
160        }
161
162        Ok(result)
163    }

165    /// Fast polynomial multiplication using NTT
166    ///
167    /// # Errors
168    /// Returns an error if polynomial lengths do not match NTT size.
169    pub fn multiply(&self, a: &[i32], b: &[i32]) -> Result<Vec<i32>> {
170        if a.len() != self.n || b.len() != self.n {
171            return Err(LatticeArcError::InvalidInput(
172                "Polynomial lengths must match NTT size".to_string(),
173            ));
174        }
175
176        // Transform to evaluation domain
177        let a_eval = self.forward(a)?;
178        let b_eval = self.forward(b)?;
179
180        // Pointwise multiplication
181        let mut c_eval = vec![0i32; self.n];
182        for i in 0..self.n {
183            if let (Some(&a_val), Some(&b_val)) = (a_eval.get(i), b_eval.get(i))
184                && let Some(c_val) = c_eval.get_mut(i)
185            {
186                *c_val = self.mod_mul(a_val, b_val);
187            }
188        }
189
190        // Transform back to coefficient domain
191        self.inverse(&c_eval)
192    }

194    /// Modular multiplication
195    fn mod_mul(&self, a: i32, b: i32) -> i32 {
196        ((i64::from(a) * i64::from(b)) % self.modulus) as i32
197    }
198
199    /// Modular addition
200    fn mod_add(&self, a: i32, b: i32) -> i32 {
201        let sum = i64::from(a) + i64::from(b);
202        (sum % self.modulus) as i32
203    }
204
205    /// Modular subtraction
206    fn mod_sub(&self, a: i32, b: i32) -> i32 {
207        let diff = i64::from(a) - i64::from(b);
208        let result = diff % self.modulus;
209        if result < 0 { (result + self.modulus) as i32 } else { result as i32 }
210    }

212    /// Core NTT implementation (Cooley-Tukey FFT-like algorithm)
213    fn ntt(&self, data: &mut [i32], inverse: bool) -> Result<()> {
214        let n = data.len();
215
216        // Bit-reversal permutation
217        let mut j = 0;
218        for i in 1..n {
219            let mut bit = n >> 1;
220            while j & bit != 0 {
221                j ^= bit;
222                bit >>= 1;
223            }
224            j ^= bit;
225
226            if i < j {
227                data.swap(i, j);
228            }
229        }
230
231        // Iterative Cooley-Tukey NTT
232        //
233        // The algorithm performs butterfly operations at each level:
234        //     u -----> + -----> u'
235        //              |
236        //              | * w
237        //              |
238        //     v -----> - -----> v'
239        //
240        // Where u' = u + w*v, v' = u - w*v (mod modulus)
241        let root = if inverse {
242            let inv = mod_inverse(i64::from(self.primitive_root), self.modulus)?;
243            i32::try_from(inv).map_err(|_e| {
244                LatticeArcError::InvalidInput("Root inverse exceeds i32 range".to_string())
245            })?
246        } else {
247            self.primitive_root
248        };
249
250        let mut length = 2;
251        while length <= n {
252            let length_i64 = i64::try_from(length).map_err(|_e| {
253                LatticeArcError::InvalidInput("NTT length exceeds i64 range".to_string())
254            })?;
255            let wlen_i64 = mod_pow(i64::from(root), (self.modulus - 1) / length_i64, self.modulus);
256            let wlen = i32::try_from(wlen_i64).map_err(|_e| {
257                LatticeArcError::InvalidInput("Twiddle factor exceeds i32 range".to_string())
258            })?;
259            let mut i = 0;
260            while i < n {
261                let mut w = 1;
262                let half_len = length / 2;
263
264                // Split the current block [i..i+length] into two halves:
265                // u_slice: [i..i+half_len]
266                // v_slice: [i+half_len..i+length]
267                let (left, right) = data.split_at_mut(i + half_len);
268                let u_slice = &mut left[i..];
269                let v_slice = &mut right[..half_len];
270
271                for j in 0..half_len {
272                    if let (Some(&u), Some(&v_data)) = (u_slice.get(j), v_slice.get(j)) {
273                        let v = self.mod_mul(v_data, w);
274                        u_slice[j] = self.mod_add(u, v);
275                        v_slice[j] = self.mod_sub(u, v);
276                    }
277                    w = self.mod_mul(w, wlen);
278                }
279                i += length;
280            }
281            length *= 2;
282        }
283        Ok(())
284    }

286    /// Scalar NTT implementation using precomputed twiddle factors
287    ///
288    /// This is a low-level method for custom NTT implementations.
289    /// For standard use, prefer `forward()` and `inverse()`.
290    #[inline]
291    pub fn ntt_scalar(
292        &self,
293        data: &mut [i32],
294        twiddles: &[i32],
295        size: usize,
296        half_size: usize,
297        step: usize,
298    ) {
299        for i in (0..data.len()).step_by(size) {
300            let mut k = 0;
301            let (first_half, second_half) = data.split_at_mut(i + size);
302            let (first_part, _) = first_half.split_at_mut(i);
303            let u_slice = &mut first_part[i..i + half_size];
304            let v_slice = &mut second_half[..half_size];
305
306            for j in 0..half_size {
307                if let (Some(&u), Some(&v_data)) = (u_slice.get(j), v_slice.get(j))
308                    && let Some(twiddle) = twiddles.get(k)
309                {
310                    let v = self.mod_mul(v_data, *twiddle);
311                    u_slice[j] = self.mod_add(u, v);
312                    v_slice[j] = self.mod_sub(u, v);
313                }
314                k += step;
315            }
316        }
317    }
