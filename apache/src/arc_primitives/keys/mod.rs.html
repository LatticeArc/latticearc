9//! # Cryptographic Key Management
10//!
11//! High-level key types and management operations for LatticeArc.
12//!
13//! ## Key Types
14//!
15//! - `KeyPair`: Complete hybrid keypair (ML-KEM + X25519)
16//! - `PublicKey`: Hybrid public key
17//! - `SecretKey`: Hybrid secret key
18//!
19//! ## Security Features
20//!
21//! - Automatic zeroization on drop
22//! - Secure memory handling
23//! - Post-quantum + classical hybrid cryptography
24//!
25//! ## Example
26//!
27//! ```ignore
28//! use arc_primitives::keys::KeyPair;
29//!
30//! let keypair = KeyPair::generate()?;
31//!
32//! // Access keys
33//! let public_key = keypair.public_key();
34//! let secret_key = keypair.secret_key();
35//! # Ok::<(), arc_primitives::keys::KeyError>(())
36//! ```

45/// Error types for key operations
46#[derive(Debug, Clone, PartialEq, Eq, Error)]
47pub enum KeyError {
48    /// Key generation operation failed.
49    #[error("Key generation failed: {0}")]
50    GenerationFailed(String),
51
52    /// Key serialization to bytes failed.
53    #[error("Key serialization failed: {0}")]
54    SerializationFailed(String),
55
56    /// Key deserialization from bytes failed.
57    #[error("Key deserialization failed: {0}")]
58    DeserializationFailed(String),
59
60    /// The provided key is invalid or malformed.
61    #[error("Invalid key: {0}")]
62    InvalidKey(String),
63}
64
65impl From<rand::Error> for KeyError {
66    fn from(e: rand::Error) -> Self {
67        KeyError::GenerationFailed(e.to_string())
68    }
69}

75/// A complete hybrid keypair for encryption/decryption operations.
76///
77/// This hybrid keypair combines post-quantum (ML-KEM) and classical (X25519)
78/// cryptography for defense-in-depth security.
79///
80/// **Note:** This type requires the `non-fips` feature. For FIPS-only builds,
81/// use the ML-KEM types directly from `arc_primitives::kem::ml_kem`.
82#[derive(Debug)]
83pub struct KeyPair {
84    /// The public key part
85    pub(crate) public_key: PublicKey,
86    /// The secret key (must be kept private)
87    pub(crate) secret_key: SecretKey,
88}
89
90/// Hybrid public key for encryption and signature verification.
91///
92/// Contains both ML-KEM (post-quantum) and X25519 (classical) public keys.
93///
94/// **Note:** This type requires the `non-fips` feature.
95///
96/// ## Security Notes
97///
98/// - Can be freely shared
99/// - Used for encrypting messages
100/// - Used for verifying signatures
101///
102/// ## Serialization
103///
104/// Public keys implement secure serialization with format validation.
105#[derive(Clone)]
106pub struct PublicKey {
107    /// ML-KEM public key bytes (post-quantum)
108    pub(crate) ml_pk: Vec<u8>,
109    /// ECDH public key (classical)
110    pub(crate) ecc_pk: EccPublicKey,
111}
112
113impl std::fmt::Debug for PublicKey {
114    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
115        f.debug_struct("PublicKey").finish_non_exhaustive()
116    }
117}
118
119impl PublicKey {
120    /// Get the ML-KEM public key bytes
121    #[must_use]
122    pub fn ml_kem(&self) -> &[u8] {
123        &self.ml_pk
124    }
125
126    /// Get the ECDH public key
127    #[must_use]
128    pub fn ecc(&self) -> &EccPublicKey {
129        &self.ecc_pk
130    }
131
132    /// Serialize to bytes
133    #[must_use]
134    pub fn to_bytes(&self) -> Vec<u8> {
135        let mut bytes = Vec::new();
136        bytes.extend_from_slice(&self.ml_pk);
137        bytes.extend_from_slice(self.ecc_pk.as_bytes());
138        bytes
139    }
140
141    /// Deserialize from bytes
142    ///
143    /// # Errors
144    /// Returns an error if there are insufficient bytes for deserialization.
145    pub fn from_bytes(bytes: &[u8]) -> Result<Self, KeyError> {
146        if bytes.len() < 32 {
147            return Err(KeyError::DeserializationFailed(
148                "Insufficient bytes for public key".to_string(),
149            ));
150        }
151
152        // Safe: already validated bytes.len() >= 32 above
153        let ecc_start = bytes.len().saturating_sub(32);
154        let ecc_pk_bytes: [u8; 32] = bytes
155            .get(ecc_start..)
156            .ok_or_else(|| KeyError::DeserializationFailed("ECC key out of bounds".to_string()))?
157            .try_into()
158            .map_err(|_e| KeyError::DeserializationFailed("Invalid ECC key bytes".to_string()))?;
159
160        let ml_pk = bytes
161            .get(..ecc_start)
162            .ok_or_else(|| KeyError::DeserializationFailed("ML-KEM key out of bounds".to_string()))?
163            .to_vec();
164        let ecc_pk = EccPublicKey::from_bytes(&ecc_pk_bytes).map_err(|_e| {
165            KeyError::DeserializationFailed("Invalid ECC public key format".to_string())
166        })?;
167
168        Ok(Self { ml_pk, ecc_pk })
169    }
170
171    /// Get the length in bytes
172    #[must_use]
173    pub fn len(&self) -> usize {
174        self.ml_pk.len().saturating_add(32)
175    }
176
177    /// Always false for valid keys
178    #[must_use]
179    pub fn is_empty(&self) -> bool {
180        false
181    }
182}

184/// Hybrid secret key for decryption and digital signing.
185///
186/// Contains both ML-KEM (post-quantum) and X25519 (classical) secret keys.
187///
188/// **Note:** This type requires the `non-fips` feature.
189///
190/// This key must be kept absolutely private and secure. It is used to:
191/// - Decrypt messages encrypted with corresponding public key
192/// - Create digital signatures that can be verified with the public key
193///
194/// ## Security Notes
195///
196/// - **NEVER share or expose this key**
197/// - Automatically zeroized when dropped to prevent memory leaks
198/// - Contains sensitive cryptographic material
199/// - Use hardware security modules (HSM) for production storage
200///
201/// ## Memory Safety
202///
203/// This struct implements `ZeroizeOnDrop` to ensure sensitive key material
204/// is securely erased from memory when key goes out of scope.
205#[derive(Zeroize, ZeroizeOnDrop)]
206pub struct SecretKey {
207    /// ML-KEM secret key bytes (post-quantum)
208    pub(crate) ml_sk: Zeroizing<Vec<u8>>,
209    /// ECDH secret key (classical)
210    pub(crate) ecc_sk: Zeroizing<EccSecretKey>,
211}
212
213impl std::fmt::Debug for SecretKey {
214    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
215        f.debug_struct("SecretKey").finish_non_exhaustive()
216    }
217}
218
219impl SecretKey {
220    /// Get the ML-KEM secret key bytes
221    #[must_use]
222    pub fn ml_kem(&self) -> &[u8] {
223        &self.ml_sk
224    }
225
226    /// Get the ECDH secret key
227    #[must_use]
228    pub fn ecc(&self) -> &EccSecretKey {
229        &self.ecc_sk
230    }
231
232    /// Serialize to bytes
233    #[must_use]
234    pub fn to_bytes(&self) -> Vec<u8> {
235        let mut bytes = Vec::new();
236        bytes.extend_from_slice(&self.ml_sk);
237        bytes.extend_from_slice((*self.ecc_sk).as_bytes());
238        bytes
239    }
240
241    /// Deserialize from bytes
242    ///
243    /// # Errors
244    /// Returns an error if there are insufficient bytes for deserialization.
245    pub fn from_bytes(bytes: &[u8]) -> Result<Self, KeyError> {
246        if bytes.len() < 32 {
247            return Err(KeyError::DeserializationFailed(
248                "Insufficient bytes for secret key".to_string(),
249            ));
250        }
251
252        // Safe: already validated bytes.len() >= 32 above
253        let ecc_start = bytes.len().saturating_sub(32);
254        let ecc_sk_bytes: [u8; 32] = bytes
255            .get(ecc_start..)
256            .ok_or_else(|| KeyError::DeserializationFailed("ECC key out of bounds".to_string()))?
257            .try_into()
258            .map_err(|_e| KeyError::DeserializationFailed("Invalid ECC key bytes".to_string()))?;
259
260        let ml_sk = Zeroizing::new(
261            bytes
262                .get(..ecc_start)
263                .ok_or_else(|| {
264                    KeyError::DeserializationFailed("ML-KEM key out of bounds".to_string())
265                })?
266                .to_vec(),
267        );
268        let ecc_sk = EccSecretKey::from_bytes(&ecc_sk_bytes).map_err(|_e| {
269            KeyError::DeserializationFailed("Invalid ECC secret key format".to_string())
270        })?;
271
272        Ok(Self { ml_sk, ecc_sk: Zeroizing::new(ecc_sk) })
273    }
274}

538impl KeyPair {
539    /// Generate a new hybrid keypair (ML-KEM + X25519)
540    ///
541    /// # Errors
542    /// Returns an error if key generation fails.
543    pub fn generate() -> Result<Self, KeyError> {
544        let mut rng = rand::rngs::OsRng;
545        let (ml_pk, ml_sk) = crate::kem::ml_kem::MlKem::generate_keypair(
546            &mut rng,
547            crate::kem::ml_kem::MlKemSecurityLevel::MlKem768,
548        )
549        .map_err(|e| KeyError::GenerationFailed(e.to_string()))?;
550
551        let x25519_kp = X25519StaticKeyPair::generate()
552            .map_err(|e| KeyError::GenerationFailed(e.to_string()))?;
553        let ecc_pk = EccPublicKey::from_bytes(x25519_kp.public_key_bytes())
554            .map_err(|e| KeyError::GenerationFailed(e.to_string()))?;
555        let seed = x25519_kp.seed_bytes().map_err(|e| KeyError::GenerationFailed(e.to_string()))?;
556        let ecc_sk = EccSecretKey::from_bytes(seed.as_ref())
557            .map_err(|e| KeyError::GenerationFailed(e.to_string()))?;
558
559        Ok(Self {
560            public_key: PublicKey { ml_pk: ml_pk.into_bytes(), ecc_pk },
561            secret_key: SecretKey { ml_sk: ml_sk.into_bytes(), ecc_sk: Zeroizing::new(ecc_sk) },
562        })
563    }
564
565    /// Get the public key
566    #[must_use]
567    pub fn public_key(&self) -> &PublicKey {
568        &self.public_key
569    }
570
571    /// Get the secret key
572    #[must_use]
573    pub fn secret_key(&self) -> &SecretKey {
574        &self.secret_key
575    }
