1//! Convenience API for cryptographic operations
2//!
3//! This module provides a unified, high-level API for encryption, decryption,
4//! signing, and verification with automatic algorithm selection.
5//!
6//! ## Unified API
7//!
8//! All operations use `CryptoConfig` for configuration:
9//!
10//! ```rust,ignore
11//! use latticearc::{
12//!     encrypt, decrypt, generate_signing_keypair, sign_with_key, verify,
13//!     CryptoConfig, UseCase,
14//! };
15//!
16//! // Encrypt with use case (recommended)
17//! let encrypted = encrypt(data, &key, CryptoConfig::new()
18//!     .use_case(UseCase::FileStorage))?;
19//!
20//! // Decrypt
21//! let plaintext = decrypt(&encrypted, &key, CryptoConfig::new())?;
22//!
23//! // Sign (generate keypair once, sign with persistent key)
24//! let (pk, sk, scheme) = generate_signing_keypair(CryptoConfig::new())?;
25//! let signed = sign_with_key(message, &sk, &pk, CryptoConfig::new())?;
26//!
27//! // Verify
28//! let is_valid = verify(&signed, CryptoConfig::new())?;
29//! ```
30//!
31//! ## With Zero Trust Session
32//!
33//! ```rust,ignore
34//! use latticearc::{encrypt, CryptoConfig, UseCase, VerifiedSession};
35//!
36//! let session = VerifiedSession::establish(&pk, &sk)?;
37//! let encrypted = encrypt(data, &key, CryptoConfig::new()
38//!     .session(&session)
39//!     .use_case(UseCase::FileStorage))?;
40//! ```
