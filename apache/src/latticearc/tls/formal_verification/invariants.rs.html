4//! TLS security invariants for formal verification.
5//!
6//! This module defines and checks security properties that must hold throughout
7//! the TLS connection lifecycle. These invariants are checked at construction
8//! time and during state transitions.
9
10use crate::tls::{TlsConfig, TlsMode};
11
12/// TLS security invariant checker.
13///
14/// Validates that TLS configurations and state transitions uphold
15/// security invariants required for correct operation.
16pub struct TlsInvariants {
17    /// Require PQ-safe key exchange
18    require_pq: bool,
19    /// Minimum allowed TLS version (as u16)
20    min_tls_version: u16,
21    /// Require forward secrecy
22    require_forward_secrecy: bool,
23}
24
25/// Result of invariant checking
26#[derive(Debug, Clone)]
27pub struct InvariantResult {
28    /// Whether all invariants passed
29    pub passed: bool,
30    /// List of invariant violations (empty if passed)
31    pub violations: Vec<String>,
32}
33
34impl TlsInvariants {
35    /// Creates a new invariant checker with default security requirements.
36    ///
37    /// Defaults:
38    /// - PQ not required (hybrid recommended but not enforced)
39    /// - Minimum TLS 1.3 (0x0304)
40    /// - Forward secrecy required
41    #[must_use]
42    pub fn new() -> Self {
43        Self {
44            require_pq: false,
45            min_tls_version: 0x0304, // TLS 1.3
46            require_forward_secrecy: true,
47        }
48    }
49
50    /// Create strict invariants requiring PQ-safe key exchange.
51    #[must_use]
52    pub fn strict() -> Self {
53        Self { require_pq: true, min_tls_version: 0x0304, require_forward_secrecy: true }
54    }
55
56    /// Set whether PQ key exchange is required.
57    #[must_use]
58    pub fn with_pq_required(mut self, required: bool) -> Self {
59        self.require_pq = required;
60        self
61    }
62
63    /// Set minimum TLS version.
64    #[must_use]
65    pub fn with_min_tls_version(mut self, version: u16) -> Self {
66        self.min_tls_version = version;
67        self
68    }
69
70    /// Check all invariants against a TLS configuration.
71    #[must_use]
72    pub fn check(&self, config: &TlsConfig) -> InvariantResult {
73        let mut violations = Vec::new();
74
75        // INV-1: Mode must be PQ-safe if required
76        if self.require_pq && config.mode == TlsMode::Classic {
77            violations
78                .push("INV-1: PQ key exchange required but Classic mode selected".to_string());
79        }
80
81        // INV-2: TLS version must meet minimum
82        if let Some(min_ver) = config.min_protocol_version {
83            let ver_num: u16 = min_ver.into();
84            if ver_num < self.min_tls_version {
85                violations.push(format!(
86                    "INV-2: Minimum TLS version 0x{:04x} is below required 0x{:04x}",
87                    ver_num, self.min_tls_version
88                ));
89            }
90        }
91
92        // INV-3: Forward secrecy - all TLS 1.3 modes use ephemeral keys
93        // TLS 1.3 inherently provides forward secrecy, so this only checks
94        // if TLS 1.2 is allowed (which may use static key exchange)
95        if self.require_forward_secrecy
96            && let Some(min_ver) = config.min_protocol_version
97        {
98            let ver_num: u16 = min_ver.into();
99            if ver_num < 0x0303 {
100                violations.push("INV-3: Forward secrecy requires TLS 1.2+ minimum".to_string());
101            }
102        }
103
104        // INV-4: Key logging must not be enabled in strict mode
105        if self.require_pq && config.enable_key_logging {
106            violations
107                .push("INV-4: Key logging enabled with strict security requirements".to_string());
108        }
109
110        InvariantResult { passed: violations.is_empty(), violations }
111    }
112}
113
114impl Default for TlsInvariants {
115    fn default() -> Self {
116        Self::new()
117    }
118}

120#[cfg(test)]
121#[allow(clippy::indexing_slicing)]
122#[allow(clippy::float_cmp)]
123mod tests {
124    use super::*;
125
126    #[test]
127    fn test_default_invariants_pass_default_config() {
128        let inv = TlsInvariants::new();
129        let config = TlsConfig::new();
130        let result = inv.check(&config);
131        assert!(result.passed);
132        assert!(result.violations.is_empty());
133    }
134
135    #[test]
136    fn test_strict_invariants_fail_classic_mode() {
137        let inv = TlsInvariants::strict();
138        let config = TlsConfig { mode: TlsMode::Classic, ..TlsConfig::default() };
139        let result = inv.check(&config);
140        assert!(!result.passed);
141        assert!(result.violations.iter().any(|v| v.contains("INV-1")));
142    }
143
144    #[test]
145    fn test_strict_invariants_pass_hybrid_mode() {
146        let inv = TlsInvariants::strict();
147        let config = TlsConfig::new(); // default is Hybrid
148        let result = inv.check(&config);
149        assert!(result.passed);
150    }
151
152    #[test]
153    fn test_strict_invariants_fail_key_logging() {
154        let inv = TlsInvariants::strict();
155        let config = TlsConfig::new().with_key_logging();
156        let result = inv.check(&config);
157        assert!(!result.passed);
158        assert!(result.violations.iter().any(|v| v.contains("INV-4")));
159    }
160
161    #[test]
162    fn test_custom_min_version() {
163        let inv = TlsInvariants::new().with_min_tls_version(0x0304);
164        let mut config = TlsConfig::new();
165        config.min_protocol_version = Some(rustls::ProtocolVersion::TLSv1_2);
166        let result = inv.check(&config);
167        assert!(!result.passed);
168        assert!(result.violations.iter().any(|v| v.contains("INV-2")));
169    }
170
171    #[test]
172    fn test_pq_required_builder() {
173        let inv = TlsInvariants::new().with_pq_required(true);
174        let config = TlsConfig { mode: TlsMode::Classic, ..TlsConfig::default() };
175        let result = inv.check(&config);
176        assert!(!result.passed);
177    }
178
179    #[test]
180    fn test_invariant_result_debug() {
181        let result = InvariantResult { passed: true, violations: vec![] };
182        let debug = format!("{:?}", result);
183        assert!(debug.contains("passed: true"));
184    }
185}
