1//! SP 800-57 Key Lifecycle Management
2//!
3//! This module implements formal key lifecycle management per NIST SP 800-57
4//! requirements, including state transitions, custodianship, and audit trails.
5//!
6//! # Key States (SP 800-57 Section 3)
7//!
8//! - **Generation**: Key material is being generated
9//! - **Active**: Key is ready for use
10//! - **Rotating**: Key rotation in progress (overlap period)
11//! - **Retired**: Key scheduled for retirement
12//! - **Destroyed**: Key material zeroized
13//!
14//! # Example
15//!
16//! ```
17//! use latticearc::types::key_lifecycle::{KeyLifecycleRecord, KeyLifecycleState};
18//!
19//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
20//! let mut record = KeyLifecycleRecord::new(
21//!     "key-123".to_string(),
22//!     "ML-KEM-768".to_string(),
23//!     3,   // security level (1-5, NIST PQC categories)
24//!     365, // rotation interval (days, > 0)
25//!     30,  // overlap period (days)
26//! )?;
27//!
28//! // Activate the key
29//! record.transition(
30//!     KeyLifecycleState::Active,
31//!     "alice".to_string(),
32//!     "Key generation complete".to_string(),
33//!     Some("approval-123".to_string()),
34//! )?;
35//!
36//! assert!(record.is_valid_for_use());
37//! # Ok(())
38//! # }
39//! ```
40
41use crate::types::error::{Result, TypeError};
42use serde::{Deserialize, Serialize};
43
44/// Maximum length (in bytes) for a single audit-trail string field.
45/// Keeps state_history bounded and JSONL audit consumers happy.
46const MAX_AUDIT_FIELD_LEN: usize = 512;
47
48/// Cap on the number of `StateTransition` entries in
49/// `KeyLifecycleRecord::state_history`. The state machine has 5
50/// terminal states; legitimate records have under a dozen entries.
51/// 1024 is generous slack for Active⇄Rotating cycling, while
52/// preventing a tampered or pathologically-cycled record from driving
53/// unbounded growth (and making the linear scans in
54/// `KeyLifecycleRecordRaw::try_from` O(n²) — see L8).
55const MAX_STATE_HISTORY: usize = 1024;
56
57/// Cap on the number of approver IDs in
58/// `KeyLifecycleRecord::approvers`. Real-world quorum schemes top out
59/// at a few dozen approvers; 256 is generous slack with the same
60/// motivation as `MAX_STATE_HISTORY`.
61const MAX_APPROVERS: usize = 256;
62
63/// Validate an audit-trail string. Rejects:
64/// - empty
65/// - control characters (including newlines, which break JSONL)
66/// - more than `MAX_AUDIT_FIELD_LEN` bytes
67///
68/// `Option<String>` callers should validate inside the `Some(_)` arm:
69/// presence is decided at the caller; content is validated here.
70fn validate_audit_field(name: &str, value: &str) -> Result<()> {
71    if value.is_empty() {
72        return Err(TypeError::InvalidAuditInput(format!("{} must not be empty", name)));
73    }
74    if value.len() > MAX_AUDIT_FIELD_LEN {
75        return Err(TypeError::InvalidAuditInput(format!(
76            "{} exceeds {} bytes",
77            name, MAX_AUDIT_FIELD_LEN
78        )));
79    }
80    if value.chars().any(char::is_control) {
81        return Err(TypeError::InvalidAuditInput(format!("{} contains control characters", name)));
82    }
83    Ok(())
84}

86/// SP 800-57 Section 3: Key Lifecycle States
87#[non_exhaustive]
88#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
89#[cfg_attr(kani, derive(kani::Arbitrary))]
90pub enum KeyLifecycleState {
91    /// Key material generation
92    Generation,
93    /// Key activation - ready for use
94    Active,
95    /// Key rotation in progress (overlap period)
96    Rotating,
97    /// Key scheduled for retirement
98    Retired,
99    /// Key destruction - material zeroized
100    Destroyed,
101}
102
103/// SP 800-57 Key State Transitions (formally verifiable)
104pub struct KeyStateMachine;
105
106impl KeyStateMachine {
107    /// Verify state transition is valid per SP 800-57
108    ///
109    /// Valid transitions:
110    /// - None -> Generation (initial state)
111    /// - Generation -> Active (initialization complete)
112    /// - Active -> Rotating (rotation initiated)
113    /// - Active -> Retired (direct retirement)
114    /// - Rotating -> Retired (rotation complete)
115    /// - Retired -> Destroyed (cleanup)
116    #[must_use]
117    pub fn is_valid_transition(from: Option<KeyLifecycleState>, to: KeyLifecycleState) -> bool {
118        match (from, to) {
119            // Generation is always valid initial state
120            (None, KeyLifecycleState::Generation) => true,
121
122            // Generation -> Active (initialization complete)
123            (Some(KeyLifecycleState::Generation), KeyLifecycleState::Active) => true,
124
125            // Active -> Rotating (rotation initiated)
126            (Some(KeyLifecycleState::Active), KeyLifecycleState::Rotating) => true,
127
128            // Rotating -> Retired (rotation complete)
129            (Some(KeyLifecycleState::Rotating), KeyLifecycleState::Retired) => true,
130
131            // Active -> Retired (direct retirement)
132            (Some(KeyLifecycleState::Active), KeyLifecycleState::Retired) => true,
133
134            // Retired -> Destroyed (cleanup)
135            (Some(KeyLifecycleState::Retired), KeyLifecycleState::Destroyed) => true,
136
137            // All other transitions are invalid
138            _ => false,
139        }
140    }
141
142    /// Get allowed next states from current state
143    #[must_use]
144    pub fn allowed_next_states(current: KeyLifecycleState) -> Vec<KeyLifecycleState> {
145        match current {
146            KeyLifecycleState::Generation => vec![KeyLifecycleState::Active],
147            KeyLifecycleState::Active => {
148                vec![KeyLifecycleState::Rotating, KeyLifecycleState::Retired]
149            }
150            KeyLifecycleState::Rotating => vec![KeyLifecycleState::Retired],
151            KeyLifecycleState::Retired => vec![KeyLifecycleState::Destroyed],
152            KeyLifecycleState::Destroyed => vec![],
153        }
154    }
155}

186/// Key lifecycle record with audit trail.
187///
188/// State-machine fields are private so the only way to mutate
189/// `current_state`, `state_history`, the per-state timestamps, or
190/// custodianship is through [`Self::transition`] /
191/// [`Self::add_approver`]. Construction-time fields (`key_id`,
192/// `key_type`, `security_level`, `generated_at`,
193/// `rotation_interval_days`, `overlap_period_days`) stay public — they
194/// are set in [`Self::new`] and never reassigned.
195#[derive(Debug, Clone, Serialize, Deserialize)]
196#[serde(try_from = "KeyLifecycleRecordRaw")]
197pub struct KeyLifecycleRecord {
198    /// Unique key identifier
199    pub key_id: String,
200    /// Algorithm/key type (e.g., "ML-KEM-768", "ML-DSA-65")
201    pub key_type: String,
202    /// Security level (1-5)
203    pub security_level: u32,
204
205    // State management — private so transitions only happen via `transition()`
206    current_state: KeyLifecycleState,
207    state_history: Vec<StateTransition>,
208
209    // Custodianship — private so updates only happen via `transition()` /
210    // `add_approver()`
211    generator: Option<String>,
212    approvers: Vec<String>,
213    destroyer: Option<String>,
214
215    /// When the key was generated
216    pub generated_at: chrono::DateTime<chrono::Utc>,
217
218    // Per-state timestamps — private so they can only be set by
219    // `transition()`
220    activated_at: Option<chrono::DateTime<chrono::Utc>>,
221    rotated_at: Option<chrono::DateTime<chrono::Utc>>,
222    retired_at: Option<chrono::DateTime<chrono::Utc>>,
223    destroyed_at: Option<chrono::DateTime<chrono::Utc>>,
224
225    // SP 800-57 requirements
226    /// How often the key should be rotated (days)
227    pub rotation_interval_days: u32,
228    /// Overlap period during rotation (days)
229    pub overlap_period_days: u32,
230}

232/// Wire-shape mirror of `KeyLifecycleRecord` used by the serde
233/// `try_from` hook. Loading a persisted record goes
234/// JSON → `KeyLifecycleRecordRaw` → `validate()` → `KeyLifecycleRecord`,
235/// so a tampered or corrupted file with state-machine inconsistencies
236/// (e.g. `Destroyed` with empty `state_history`, or `Generation` with
237/// a populated `activated_at`) is rejected at deserialize time rather
238/// than producing a silently-broken record.
239#[derive(Debug, Clone, Deserialize)]
240#[doc(hidden)]
241struct KeyLifecycleRecordRaw {
242    key_id: String,
243    key_type: String,
244    security_level: u32,
245    current_state: KeyLifecycleState,
246    state_history: Vec<StateTransition>,
247    generator: Option<String>,
248    approvers: Vec<String>,
249    destroyer: Option<String>,
250    generated_at: chrono::DateTime<chrono::Utc>,
251    activated_at: Option<chrono::DateTime<chrono::Utc>>,
252    rotated_at: Option<chrono::DateTime<chrono::Utc>>,
253    retired_at: Option<chrono::DateTime<chrono::Utc>>,
254    destroyed_at: Option<chrono::DateTime<chrono::Utc>>,
255    rotation_interval_days: u32,
256    overlap_period_days: u32,
257}

259impl TryFrom<KeyLifecycleRecordRaw> for KeyLifecycleRecord {
260    type Error = TypeError;
261
262    fn try_from(raw: KeyLifecycleRecordRaw) -> Result<Self> {
263        // Re-validate state-machine invariants on load so a corrupted
264        // or tampered persisted record cannot bypass the rules
265        // `transition()` enforces in memory.
266
267        // (0) Vector caps: same bounds the in-memory mutators
268        //     enforce. Without these, the linear scans below
269        //     (`entered`, last-history check) become O(n²) on a
270        //     pathologically-large persisted record, and the
271        //     in-memory record can grow past `transition()`'s cap
272        //     after a roundtrip.
273        if raw.state_history.len() > MAX_STATE_HISTORY {
274            return Err(TypeError::InvalidAuditInput(format!(
275                "state_history length {} exceeds cap of {}",
276                raw.state_history.len(),
277                MAX_STATE_HISTORY
278            )));
279        }
280        if raw.approvers.len() > MAX_APPROVERS {
281            return Err(TypeError::InvalidAuditInput(format!(
282                "approvers length {} exceeds cap of {}",
283                raw.approvers.len(),
284                MAX_APPROVERS
285            )));
286        }
287
288        // (1) `current_state` must be reachable from the persisted
289        //     `activated_at` value. Rotating and Retired both require
290        //     having passed through Active, so `activated_at` MUST be
291        //     `Some` for those — and Generation MUST NOT have one.
292        match (raw.current_state, raw.activated_at) {
293            (KeyLifecycleState::Generation, Some(_)) => {
294                return Err(TypeError::InvalidAuditInput(
295                    "current_state=Generation with activated_at set".to_string(),
296                ));
297            }
298            (
299                KeyLifecycleState::Active
300                | KeyLifecycleState::Rotating
301                | KeyLifecycleState::Retired,
302                None,
303            ) => {
304                return Err(TypeError::InvalidAuditInput(format!(
305                    "current_state={:?} requires activated_at",
306                    raw.current_state
307                )));
308            }
309            _ => {}
310        }
311
312        // (2) For non-Generation states, the last state_history entry
313        //     must agree with `current_state`. Otherwise a tampered
314        //     file with `current_state = Active` and a `state_history`
315        //     ending in `Rotating` (or empty) would deserialize cleanly.
316        match (raw.current_state, raw.state_history.last()) {
317            (KeyLifecycleState::Generation, _) => {}
318            (_, None) => {
319                return Err(TypeError::InvalidAuditInput(format!(
320                    "current_state={:?} with empty state_history",
321                    raw.current_state
322                )));
323            }
324            (current, Some(t)) if t.to_state != current => {
325                return Err(TypeError::InvalidAuditInput(format!(
326                    "current_state={:?} but last state_history entry is {:?}",
327                    current, t.to_state
328                )));
329            }
330            _ => {}
331        }
332
333        // (3) Per-state timestamp consistency: the timestamp for a
334        //     state must be present iff that state has been entered
335        //     (i.e. appears in state_history).
336        let entered = |st: KeyLifecycleState| raw.state_history.iter().any(|t| t.to_state == st);
337        if entered(KeyLifecycleState::Rotating) != raw.rotated_at.is_some() {
338            return Err(TypeError::InvalidAuditInput(
339                "rotated_at presence disagrees with state_history".to_string(),
340            ));
341        }
342        if entered(KeyLifecycleState::Retired) != raw.retired_at.is_some() {
343            return Err(TypeError::InvalidAuditInput(
344                "retired_at presence disagrees with state_history".to_string(),
345            ));
346        }
347        if entered(KeyLifecycleState::Destroyed) != raw.destroyed_at.is_some() {
348            return Err(TypeError::InvalidAuditInput(
349                "destroyed_at presence disagrees with state_history".to_string(),
350            ));
351        }
352
353        // (4) state_history timestamps must be non-decreasing. A
354        //     persisted record with `state_history[i].timestamp >
355        //     state_history[i+1].timestamp` is a tamper signal — the
356        //     in-memory `transition()` path always appends with
357        //     `Utc::now()`, which is monotone in practice and forbids
358        //     out-of-order entries by construction.
359        for pair in raw.state_history.windows(2) {
360            // `windows(2)` always yields 2-element slices; pattern-
361            // bind so clippy's indexing-may-panic gate is satisfied
362            // without an `[allow]` waiver.
363            let [prev, curr] = pair else { continue };
364            if curr.timestamp < prev.timestamp {
365                return Err(TypeError::InvalidAuditInput(format!(
366                    "state_history timestamps decrease between {:?} ({}) and {:?} ({})",
367                    prev.to_state, prev.timestamp, curr.to_state, curr.timestamp
368                )));
369            }
370        }
371
372        // (5) Approver identifiers must be unique. The in-memory
373        //     `add_approver` path explicitly deduplicates; a persisted
374        //     record with `approvers = ["alice", "alice"]` would
375        //     otherwise inflate threshold-quorum counts.
376        let mut seen: std::collections::HashSet<&String> =
377            std::collections::HashSet::with_capacity(raw.approvers.len());
378        for approver in &raw.approvers {
379            if !seen.insert(approver) {
380                return Err(TypeError::InvalidAuditInput(format!(
381                    "approvers contains duplicate id {:?}",
382                    approver
383                )));
384            }
385        }
386
387        // (6) Each `StateTransition` entry's audit-trail strings
388        //     (`custodian_id`, `justification`, `approval_id`) must
389        //     pass the same sanitization as the in-memory
390        //     `transition()` path — empty / control-char / oversized
391        //     strings are rejected. Without this check a tampered
392        //     persisted record could load a `StateTransition` with
393        //     embedded `\n` (corrupting JSONL audit consumers),
394        //     empty attribution (defeating the audit trail), or
395        //     a 1 MiB `justification` (memory amplification on the
396        //     in-memory side after deserialization).
397        for (idx, t) in raw.state_history.iter().enumerate() {
398            validate_audit_field(&format!("state_history[{idx}].custodian_id"), &t.custodian_id)?;
399            validate_audit_field(&format!("state_history[{idx}].justification"), &t.justification)?;
400            if let Some(ref approval_id) = t.approval_id {
401                validate_audit_field(&format!("state_history[{idx}].approval_id"), approval_id)?;
402            }
403        }
404
405        // (7) Top-level per-state timestamps must respect the
406        //     state-machine's temporal ordering:
407        //         generated_at ≤ activated_at ≤ rotated_at ≤ retired_at ≤ destroyed_at
408        //     The state_history monotonicity check (step 4) only
409        //     covers entries within the history vector — the top-
410        //     level timestamps are populated from independent JSON
411        //     fields and could otherwise carry impossible orderings
412        //     (e.g. `retired_at < activated_at`). The in-memory
413        //     `transition()` path always assigns these via
414        //     `Utc::now()` after the monotonicity-enforced state
415        //     change, so the relationship holds by construction
416        //     there; deserialization needs an explicit gate.
417        type TsPair = (
418            &'static str,
419            Option<chrono::DateTime<chrono::Utc>>,
420            &'static str,
421            Option<chrono::DateTime<chrono::Utc>>,
422        );
423        let pairs: [TsPair; 5] = [
424            ("generated_at", Some(raw.generated_at), "activated_at", raw.activated_at),
425            ("activated_at", raw.activated_at, "rotated_at", raw.rotated_at),
426            ("activated_at", raw.activated_at, "retired_at", raw.retired_at),
427            ("rotated_at", raw.rotated_at, "retired_at", raw.retired_at),
428            ("retired_at", raw.retired_at, "destroyed_at", raw.destroyed_at),
429        ];
430        for (earlier_name, earlier, later_name, later) in pairs {
431            if let (Some(e), Some(l)) = (earlier, later)
432                && l < e
433            {
434                return Err(TypeError::InvalidAuditInput(format!(
435                    "{later_name} ({l}) precedes {earlier_name} ({e}) — state-machine ordering violated"
436                )));
437            }
438        }
439
440        // Re-validate the numeric-bound invariants that
441        // `KeyLifecycleRecord::new` enforces on construction. The
442        // state-machine-invariant pass above does not cover these,
443        // so without this gate a tampered persisted record could
444        // carry security_level = 999 or rotation_interval_days = 0
445        // past the round-trip.
446        validate_security_level(raw.security_level)?;
447        validate_rotation_interval(raw.rotation_interval_days)?;
448
449        Ok(KeyLifecycleRecord {
450            key_id: raw.key_id,
451            key_type: raw.key_type,
452            security_level: raw.security_level,
453            current_state: raw.current_state,
454            state_history: raw.state_history,
455            generator: raw.generator,
456            approvers: raw.approvers,
457            destroyer: raw.destroyer,
458            generated_at: raw.generated_at,
459            activated_at: raw.activated_at,
460            rotated_at: raw.rotated_at,
461            retired_at: raw.retired_at,
462            destroyed_at: raw.destroyed_at,
463            rotation_interval_days: raw.rotation_interval_days,
464            overlap_period_days: raw.overlap_period_days,
465        })
466    }
467}

469/// Reject `security_level` outside the documented 1-5 range. Used by
470/// both `KeyLifecycleRecord::new` and the deserializer so the in-memory
471/// and on-disk paths share one gate.
472fn validate_security_level(security_level: u32) -> Result<()> {
473    if !(1..=5).contains(&security_level) {
474        return Err(TypeError::InvalidAuditInput(format!(
475            "security_level {} outside documented range 1-5",
476            security_level
477        )));
478    }
479    Ok(())
480}

482/// Reject `rotation_interval_days == 0`. A zero interval makes
483/// `requires_rotation()` return `true` immediately on activation, so
484/// every key would be flagged for rotation at the moment it goes live —
485/// almost certainly a configuration bug, not a real lifecycle policy.
486fn validate_rotation_interval(rotation_interval_days: u32) -> Result<()> {
487    if rotation_interval_days == 0 {
488        return Err(TypeError::InvalidAuditInput(format!(
489            "rotation_interval_days {} is zero; would force immediate rotation on activation",
490            rotation_interval_days
491        )));
492    }
493    Ok(())
494}

496/// Record of a state transition
497#[derive(Debug, Clone, Serialize, Deserialize)]
498pub struct StateTransition {
499    /// Previous state (None if initial)
500    pub from_state: Option<KeyLifecycleState>,
501    /// New state
502    pub to_state: KeyLifecycleState,
503    /// When the transition occurred
504    pub timestamp: chrono::DateTime<chrono::Utc>,
505    /// ID of the custodian who performed the transition
506    pub custodian_id: String,
507    /// Reason for the transition
508    pub justification: String,
509    /// Approval reference (if applicable)
510    pub approval_id: Option<String>,
511}
512
513impl KeyLifecycleRecord {
514    /// Create new key lifecycle record
515    ///
516    /// # Arguments
517    ///
518    /// * `key_id` - Unique identifier for the key
519    /// * `key_type` - Algorithm/key type (e.g., "ML-KEM-768")
520    /// * `security_level` - Security level (1-5, NIST PQC categories)
521    /// * `rotation_interval_days` - How often to rotate the key (must be > 0)
522    /// * `overlap_period_days` - Overlap period during rotation
523    ///
524    /// # Errors
525    ///
526    /// Returns [`TypeError::InvalidAuditInput`] if `security_level` is
527    /// outside the documented 1-5 range, or if `rotation_interval_days`
528    /// is zero (forcing immediate rotation on activation). The
529    /// deserializer applies the same checks so in-memory and on-disk
530    /// records share one validation gate.
531    pub fn new(
532        key_id: String,
533        key_type: String,
534        security_level: u32,
535        rotation_interval_days: u32,
536        overlap_period_days: u32,
537    ) -> Result<Self> {
538        validate_security_level(security_level)?;
539        validate_rotation_interval(rotation_interval_days)?;
540        Ok(Self {
541            key_id,
542            key_type,
543            security_level,
544            current_state: KeyLifecycleState::Generation,
545            state_history: Vec::new(),
546            generator: None,
547            approvers: Vec::new(),
548            destroyer: None,
549            generated_at: chrono::Utc::now(),
550            activated_at: None,
551            rotated_at: None,
552            retired_at: None,
553            destroyed_at: None,
554            rotation_interval_days,
555            overlap_period_days,
556        })
557    }
558
559    /// Transition key to new state with custodianship tracking
560    ///
561    /// # Arguments
562    ///
563    /// * `to_state` - Target state
564    /// * `custodian_id` - ID of the custodian performing the transition
565    /// * `justification` - Reason for the transition
566    /// * `approval_id` - Optional approval reference
567    ///
568    /// # Errors
569    ///
570    /// Returns `TypeError::InvalidStateTransition` if the transition is invalid
571    pub fn transition(
572        &mut self,
573        to_state: KeyLifecycleState,
574        custodian_id: String,
575        justification: String,
576        approval_id: Option<String>,
577    ) -> Result<()> {
578        // Audit-trail input gates: empty / control-chars / oversized
579        // strings break JSONL audit consumers and produce attribution-
580        // free entries. Reject up front rather than persisting them.
581        validate_audit_field("custodian_id", &custodian_id)?;
582        validate_audit_field("justification", &justification)?;
583        // Inside `Some(_)` the caller has explicitly chosen to send a
584        // value; empty-string is a sentinel that confuses
585        // presence-vs-content matching downstream. Reject it here.
586        if let Some(ref approval_id) = approval_id {
587            validate_audit_field("approval_id", approval_id)?;
588        }
589
590        if !KeyStateMachine::is_valid_transition(Some(self.current_state), to_state) {
591            return Err(TypeError::InvalidStateTransition {
592                from: self.current_state,
593                to: to_state,
594            });
595        }
596        // Capture `now` once so the state_history timestamp and the
597        // per-state timestamp below match exactly. An auditor comparing
598        // `state_history[i].timestamp` against `activated_at` (etc.)
599        // would otherwise see sub-tick skew on every transition.
600        let now = chrono::Utc::now();
601        let transition = StateTransition {
602            from_state: Some(self.current_state),
603            to_state,
604            timestamp: now,
605            custodian_id: custodian_id.clone(),
606            justification,
607            approval_id,
608        };
609
610        if self.state_history.len() >= MAX_STATE_HISTORY {
611            return Err(TypeError::InvalidAuditInput(format!(
612                "state_history at cap of {} entries; cannot record further transitions",
613                MAX_STATE_HISTORY
614            )));
615        }
616        self.state_history.push(transition);
617        self.current_state = to_state;
618
619        // Update timestamps (use the captured `now`).
620        match to_state {
621            KeyLifecycleState::Active => self.activated_at = Some(now),
622            KeyLifecycleState::Rotating => self.rotated_at = Some(now),
623            KeyLifecycleState::Retired => self.retired_at = Some(now),
624            KeyLifecycleState::Destroyed => self.destroyed_at = Some(now),
625            _ => {}
626        }
627
628        // Update custodianship
629        // Generator is the custodian who completes key generation (moves to Active)
630        // Check the last transition we just added to see if it was from Generation
631        if let Some(last_transition) = self.state_history.last()
632            && last_transition.from_state == Some(KeyLifecycleState::Generation)
633            && to_state == KeyLifecycleState::Active
634        {
635            self.generator = Some(custodian_id.clone());
636        }
637        if to_state == KeyLifecycleState::Destroyed {
638            self.destroyer = Some(custodian_id);
639        }
640
641        Ok(())
642    }
643
644    /// Days elapsed since `activated`, signed (negative for future
645    /// activation due to clock skew). Internal helper backing
646    /// [`Self::requires_rotation`] and [`Self::age_days`] so the
647    /// `Utc::now().signed_duration_since(...).num_days()` chain lives
648    /// in one place.
649    fn days_since_activation(activated: chrono::DateTime<chrono::Utc>) -> i64 {
650        chrono::Utc::now().signed_duration_since(activated).num_days()
651    }
652
653    /// Check if key is due for rotation per SP 800-57
654    #[must_use]
655    pub fn requires_rotation(&self) -> bool {
656        let Some(activated_at) = self.activated_at else { return false };
657        let age_days_i64 = Self::days_since_activation(activated_at);
658        if age_days_i64 < 0 {
659            // Future activation date — clock skew or misconfiguration
660            tracing::warn!(
661                "Key has negative age ({age_days_i64} days); activation_at in the future"
662            );
663            return false;
664        }
665        // Ages larger than u32::MAX → always require rotation
666        let age_days = u32::try_from(age_days_i64).unwrap_or(u32::MAX);
667        age_days >= self.rotation_interval_days
668    }
669
670    /// Get key age in days since activation
671    #[must_use]
672    pub fn age_days(&self) -> Option<u32> {
673        self.activated_at.map(|activated| {
674            // Negative ages (future activation) treated as 0.
675            u32::try_from(Self::days_since_activation(activated)).unwrap_or(0)
676        })
677    }

679    /// Check if key is in valid state for use
680    #[must_use]
681    pub fn is_valid_for_use(&self) -> bool {
682        matches!(self.current_state, KeyLifecycleState::Active | KeyLifecycleState::Rotating)
683    }

685    /// Add an approver to the key. Silently ignored if the approver
686    /// is already in the list, or if the cap (`MAX_APPROVERS`) has
687    /// been reached. Returns `false` on cap rejection so callers can
688    /// surface a warning if needed — the `#[must_use]` here forces
689    /// every caller to acknowledge the bool, since
690    /// changed the return type from `()` precisely to expose this
691    /// case and a caller that just `.;` it would silently lose the
692    /// signal.
693    #[must_use = "add_approver returns false when the cap is reached; surface that to the caller"]
694    pub fn add_approver(&mut self, approver_id: impl Into<String>) -> bool {
695        let approver_id = approver_id.into();
696        if self.approvers.contains(&approver_id) {
697            return true;
698        }
699        if self.approvers.len() >= MAX_APPROVERS {
700            return false;
701        }
702        self.approvers.push(approver_id);
703        true
704    }

796    /// Proves that the only valid initial state is Generation.
797    /// Security property: keys must begin in the Generation state.
798    #[kani::proof]
799    fn key_state_machine_only_generation_from_none() {
800        let to: KeyLifecycleState = kani::any();
801        kani::assume(to != KeyLifecycleState::Generation);
802
803        let is_valid = KeyStateMachine::is_valid_transition(None, to);
804        kani::assert(!is_valid, "Only Generation is valid from initial state (None)");
805    }
806
807    /// Proves that `is_valid_transition` matches the SP 800-57 transition
808    /// specification for all state pairs. This is an independent encoding of
809    /// the allowed transitions — if `is_valid_transition` has a bug, this
810    /// proof will catch the discrepancy.
811    ///
812    /// Note: `allowed_next_states()` returns `Vec`, which Kani cannot
813    /// efficiently verify (unbounded loop unwinding on heap iteration).
814    /// Unit tests verify `allowed_next_states()` separately.
815    #[kani::proof]
816    fn key_state_machine_transitions_match_spec() {
817        let from: KeyLifecycleState = kani::any();
818        let to: KeyLifecycleState = kani::any();
819
820        let transition_valid = KeyStateMachine::is_valid_transition(Some(from), to);
821
822        // Independent specification of SP 800-57 allowed transitions
823        let spec_allows = match from {
824            KeyLifecycleState::Generation => to == KeyLifecycleState::Active,
825            KeyLifecycleState::Active => {
826                to == KeyLifecycleState::Rotating || to == KeyLifecycleState::Retired
827            }
828            KeyLifecycleState::Rotating => to == KeyLifecycleState::Retired,
829            KeyLifecycleState::Retired => to == KeyLifecycleState::Destroyed,
830            KeyLifecycleState::Destroyed => false,
831        };
832
833        kani::assert(
834            transition_valid == spec_allows,
835            "is_valid_transition must match SP 800-57 specification",
836        );
837    }
838
839    /// Proves that retired keys can only transition to Destroyed.
840    /// Security property: once retired, the only path is secure destruction.
841    #[kani::proof]
842    fn key_state_machine_retired_only_to_destroyed() {
843        let to: KeyLifecycleState = kani::any();
844        kani::assume(to != KeyLifecycleState::Destroyed);
845
846        let is_valid = KeyStateMachine::is_valid_transition(Some(KeyLifecycleState::Retired), to);
847        kani::assert(!is_valid, "Retired keys can only transition to Destroyed");
848    }
