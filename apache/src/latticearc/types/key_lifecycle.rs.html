1//! SP 800-57 Key Lifecycle Management
2//!
3//! This module implements formal key lifecycle management per NIST SP 800-57
4//! requirements, including state transitions, custodianship, and audit trails.
5//!
6//! # Key States (SP 800-57 Section 3)
7//!
8//! - **Generation**: Key material is being generated
9//! - **Active**: Key is ready for use
10//! - **Rotating**: Key rotation in progress (overlap period)
11//! - **Retired**: Key scheduled for retirement
12//! - **Destroyed**: Key material zeroized
13//!
14//! # Example
15//!
16//! ```
17//! use latticearc::types::key_lifecycle::{KeyLifecycleRecord, KeyLifecycleState};
18//!
19//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
20//! let mut record = KeyLifecycleRecord::new(
21//!     "key-123".to_string(),
22//!     "ML-KEM-768".to_string(),
23//!     3,   // security level (1-5, NIST PQC categories)
24//!     365, // rotation interval (days, > 0)
25//!     30,  // overlap period (days)
26//! )?;
27//!
28//! // Activate the key
29//! record.transition(
30//!     KeyLifecycleState::Active,
31//!     "alice".to_string(),
32//!     "Key generation complete".to_string(),
33//!     Some("approval-123".to_string()),
34//! )?;
35//!
36//! assert!(record.is_valid_for_use());
37//! # Ok(())
38//! # }
39//! ```
40
41use crate::types::error::{Result, TypeError};
42use serde::{Deserialize, Serialize};
43
44/// Maximum length (in bytes) for a single audit-trail string field.
45/// Keeps state_history bounded and JSONL audit consumers happy.
46const MAX_AUDIT_FIELD_LEN: usize = 512;
47
48/// Cap on the number of `StateTransition` entries in
49/// `KeyLifecycleRecord::state_history`. The state machine has 5
50/// terminal states; legitimate records have under a dozen entries.
51/// 1024 is generous slack for Active⇄Rotating cycling, while
52/// preventing a tampered or pathologically-cycled record from driving
53/// unbounded growth (and making the linear scans in
54/// `KeyLifecycleRecordRaw::try_from` O(n²) — see L8).
55const MAX_STATE_HISTORY: usize = 1024;
56
57/// Cap on the number of approver IDs in
58/// `KeyLifecycleRecord::approvers`. Real-world quorum schemes top out
59/// at a few dozen approvers; 256 is generous slack with the same
60/// motivation as `MAX_STATE_HISTORY`.
61const MAX_APPROVERS: usize = 256;
62
63/// Validate an audit-trail string. Rejects:
64/// - empty
65/// - control characters (including newlines, which break JSONL)
66/// - more than `MAX_AUDIT_FIELD_LEN` bytes
67///
68/// `Option<String>` callers should validate inside the `Some(_)` arm:
69/// presence is decided at the caller; content is validated here.
70fn validate_audit_field(name: &str, value: &str) -> Result<()> {
71    if value.is_empty() {
72        return Err(TypeError::InvalidAuditInput(format!("{} must not be empty", name)));
73    }
74    if value.len() > MAX_AUDIT_FIELD_LEN {
75        return Err(TypeError::InvalidAuditInput(format!(
76            "{} exceeds {} bytes",
77            name, MAX_AUDIT_FIELD_LEN
78        )));
79    }
80    if value.chars().any(char::is_control) {
81        return Err(TypeError::InvalidAuditInput(format!("{} contains control characters", name)));
82    }
83    Ok(())
84}

86/// SP 800-57 Section 3: Key Lifecycle States
87#[non_exhaustive]
88#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
89#[cfg_attr(kani, derive(kani::Arbitrary))]
90pub enum KeyLifecycleState {
91    /// Key material generation
92    Generation,
93    /// Key activation - ready for use
94    Active,
95    /// Key rotation in progress (overlap period)
96    Rotating,
97    /// Key scheduled for retirement
98    Retired,
99    /// Key destruction - material zeroized
100    Destroyed,
101}
102
103/// SP 800-57 Key State Transitions (formally verifiable)
104pub struct KeyStateMachine;
105
106impl KeyStateMachine {
107    /// Verify state transition is valid per SP 800-57
108    ///
109    /// Valid transitions:
110    /// - None -> Generation (initial state)
111    /// - Generation -> Active (initialization complete)
112    /// - Active -> Rotating (rotation initiated)
113    /// - Active -> Retired (direct retirement)
114    /// - Rotating -> Retired (rotation complete)
115    /// - Retired -> Destroyed (cleanup)
116    #[must_use]
117    pub fn is_valid_transition(from: Option<KeyLifecycleState>, to: KeyLifecycleState) -> bool {
118        match (from, to) {
119            // Generation is always valid initial state
120            (None, KeyLifecycleState::Generation) => true,
121
122            // Generation -> Active (initialization complete)
123            (Some(KeyLifecycleState::Generation), KeyLifecycleState::Active) => true,
124
125            // Active -> Rotating (rotation initiated)
126            (Some(KeyLifecycleState::Active), KeyLifecycleState::Rotating) => true,
127
128            // Rotating -> Retired (rotation complete)
129            (Some(KeyLifecycleState::Rotating), KeyLifecycleState::Retired) => true,
130
131            // Active -> Retired (direct retirement)
132            (Some(KeyLifecycleState::Active), KeyLifecycleState::Retired) => true,
133
134            // Retired -> Destroyed (cleanup)
135            (Some(KeyLifecycleState::Retired), KeyLifecycleState::Destroyed) => true,
136
137            // All other transitions are invalid
138            _ => false,
139        }
140    }
141
142    /// Get allowed next states from current state
143    #[must_use]
144    pub fn allowed_next_states(current: KeyLifecycleState) -> Vec<KeyLifecycleState> {
145        match current {
146            KeyLifecycleState::Generation => vec![KeyLifecycleState::Active],
147            KeyLifecycleState::Active => {
148                vec![KeyLifecycleState::Rotating, KeyLifecycleState::Retired]
149            }
150            KeyLifecycleState::Rotating => vec![KeyLifecycleState::Retired],
151            KeyLifecycleState::Retired => vec![KeyLifecycleState::Destroyed],
152            KeyLifecycleState::Destroyed => vec![],
153        }
154    }
155}

265    // When the key was generated.

282/// Wire-shape mirror of `KeyLifecycleRecord` used by the serde
283/// `try_from` hook. Loading a persisted record goes
284/// JSON → `KeyLifecycleRecordRaw` → `validate()` → `KeyLifecycleRecord`,
285/// so a tampered or corrupted file with state-machine inconsistencies
286/// (e.g. `Destroyed` with empty `state_history`, or `Generation` with
287/// a populated `activated_at`) is rejected at deserialize time rather
288/// than producing a silently-broken record.
289#[derive(Debug, Clone, Deserialize)]
290#[doc(hidden)]
291struct KeyLifecycleRecordRaw {
292    key_id: String,
293    key_type: String,
294    security_level: u32,
295    current_state: KeyLifecycleState,
296    state_history: Vec<StateTransition>,
297    generator: Option<String>,
298    approvers: Vec<String>,
299    destroyer: Option<String>,
300    generated_at: chrono::DateTime<chrono::Utc>,
301    activated_at: Option<chrono::DateTime<chrono::Utc>>,
302    rotated_at: Option<chrono::DateTime<chrono::Utc>>,
303    retired_at: Option<chrono::DateTime<chrono::Utc>>,
304    destroyed_at: Option<chrono::DateTime<chrono::Utc>>,
305    rotation_interval_days: u32,
306    overlap_period_days: u32,
307}

309impl TryFrom<KeyLifecycleRecordRaw> for KeyLifecycleRecord {
310    type Error = TypeError;
311
312    fn try_from(raw: KeyLifecycleRecordRaw) -> Result<Self> {
313        // Re-validate state-machine invariants on load so a corrupted
314        // or tampered persisted record cannot bypass the rules
315        // `transition()` enforces in memory.
316
317        // (0) Vector caps: same bounds the in-memory mutators
318        //     enforce. Without these, the linear scans below
319        //     (`entered`, last-history check) become O(n²) on a
320        //     pathologically-large persisted record, and the
321        //     in-memory record can grow past `transition()`'s cap
322        //     after a roundtrip.
323        if raw.state_history.len() > MAX_STATE_HISTORY {
324            return Err(TypeError::InvalidAuditInput(format!(
325                "state_history length {} exceeds cap of {}",
326                raw.state_history.len(),
327                MAX_STATE_HISTORY
328            )));
329        }
330        if raw.approvers.len() > MAX_APPROVERS {
331            return Err(TypeError::InvalidAuditInput(format!(
332                "approvers length {} exceeds cap of {}",
333                raw.approvers.len(),
334                MAX_APPROVERS
335            )));
336        }
337
338        // (1) `current_state` must be reachable from the persisted
339        //     `activated_at` value. Rotating and Retired both require
340        //     having passed through Active, so `activated_at` MUST be
341        //     `Some` for those — and Generation MUST NOT have one.
342        match (raw.current_state, raw.activated_at) {
343            (KeyLifecycleState::Generation, Some(_)) => {
344                return Err(TypeError::InvalidAuditInput(
345                    "current_state=Generation with activated_at set".to_string(),
346                ));
347            }
348            (
349                KeyLifecycleState::Active
350                | KeyLifecycleState::Rotating
351                | KeyLifecycleState::Retired,
352                None,
353            ) => {
354                return Err(TypeError::InvalidAuditInput(format!(
355                    "current_state={:?} requires activated_at",
356                    raw.current_state
357                )));
358            }
359            _ => {}
360        }
361
362        // (2) For non-Generation states, the last state_history entry
363        //     must agree with `current_state`. Otherwise a tampered
364        //     file with `current_state = Active` and a `state_history`
365        //     ending in `Rotating` (or empty) would deserialize cleanly.
366        match (raw.current_state, raw.state_history.last()) {
367            (KeyLifecycleState::Generation, _) => {}
368            (_, None) => {
369                return Err(TypeError::InvalidAuditInput(format!(
370                    "current_state={:?} with empty state_history",
371                    raw.current_state
372                )));
373            }
374            (current, Some(t)) if t.to_state != current => {
375                return Err(TypeError::InvalidAuditInput(format!(
376                    "current_state={:?} but last state_history entry is {:?}",
377                    current, t.to_state
378                )));
379            }
380            _ => {}
381        }
382
383        // (3) Per-state timestamp consistency: the timestamp for a
384        //     state must be present iff that state has been entered
385        //     (i.e. appears in state_history).
386        let entered = |st: KeyLifecycleState| raw.state_history.iter().any(|t| t.to_state == st);
387        if entered(KeyLifecycleState::Rotating) != raw.rotated_at.is_some() {
388            return Err(TypeError::InvalidAuditInput(
389                "rotated_at presence disagrees with state_history".to_string(),
390            ));
391        }
392        if entered(KeyLifecycleState::Retired) != raw.retired_at.is_some() {
393            return Err(TypeError::InvalidAuditInput(
394                "retired_at presence disagrees with state_history".to_string(),
395            ));
396        }
397        if entered(KeyLifecycleState::Destroyed) != raw.destroyed_at.is_some() {
398            return Err(TypeError::InvalidAuditInput(
399                "destroyed_at presence disagrees with state_history".to_string(),
400            ));
401        }
402
403        // (4) state_history timestamps must be non-decreasing. A
404        //     persisted record with `state_history[i].timestamp >
405        //     state_history[i+1].timestamp` is a tamper signal — the
406        //     in-memory `transition()` path always appends with
407        //     `Utc::now()`, which is monotone in practice and forbids
408        //     out-of-order entries by construction.
409        for pair in raw.state_history.windows(2) {
410            // `windows(2)` always yields 2-element slices; pattern-
411            // bind so clippy's indexing-may-panic gate is satisfied
412            // without an `[allow]` waiver.
413            let [prev, curr] = pair else { continue };
414            if curr.timestamp < prev.timestamp {
415                return Err(TypeError::InvalidAuditInput(format!(
416                    "state_history timestamps decrease between {:?} ({}) and {:?} ({})",
417                    prev.to_state, prev.timestamp, curr.to_state, curr.timestamp
418                )));
419            }
420        }
421
422        // (5) Approver identifiers must be unique. The in-memory
423        //     `add_approver` path explicitly deduplicates; a persisted
424        //     record with `approvers = ["alice", "alice"]` would
425        //     otherwise inflate threshold-quorum counts.
426        let mut seen: std::collections::HashSet<&String> =
427            std::collections::HashSet::with_capacity(raw.approvers.len());
428        for approver in &raw.approvers {
429            if !seen.insert(approver) {
430                return Err(TypeError::InvalidAuditInput(format!(
431                    "approvers contains duplicate id {:?}",
432                    approver
433                )));
434            }
435        }
436
437        // (6) Each `StateTransition` entry's audit-trail strings
438        //     (`custodian_id`, `justification`, `approval_id`) must
439        //     pass the same sanitization as the in-memory
440        //     `transition()` path — empty / control-char / oversized
441        //     strings are rejected. Without this check a tampered
442        //     persisted record could load a `StateTransition` with
443        //     embedded `\n` (corrupting JSONL audit consumers),
444        //     empty attribution (defeating the audit trail), or
445        //     a 1 MiB `justification` (memory amplification on the
446        //     in-memory side after deserialization).
447        for (idx, t) in raw.state_history.iter().enumerate() {
448            validate_audit_field(&format!("state_history[{idx}].custodian_id"), &t.custodian_id)?;
449            validate_audit_field(&format!("state_history[{idx}].justification"), &t.justification)?;
450            if let Some(ref approval_id) = t.approval_id {
451                validate_audit_field(&format!("state_history[{idx}].approval_id"), approval_id)?;
452            }
453        }
454
455        // (7) Top-level per-state timestamps must respect the
456        //     state-machine's temporal ordering:
457        //         generated_at ≤ activated_at ≤ rotated_at ≤ retired_at ≤ destroyed_at
458        //     The state_history monotonicity check (step 4) only
459        //     covers entries within the history vector — the top-
460        //     level timestamps are populated from independent JSON
461        //     fields and could otherwise carry impossible orderings
462        //     (e.g. `retired_at < activated_at`). The in-memory
463        //     `transition()` path always assigns these via
464        //     `Utc::now()` after the monotonicity-enforced state
465        //     change, so the relationship holds by construction
466        //     there; deserialization needs an explicit gate.
467        type TsPair = (
468            &'static str,
469            Option<chrono::DateTime<chrono::Utc>>,
470            &'static str,
471            Option<chrono::DateTime<chrono::Utc>>,
472        );
473        let pairs: [TsPair; 5] = [
474            ("generated_at", Some(raw.generated_at), "activated_at", raw.activated_at),
475            ("activated_at", raw.activated_at, "rotated_at", raw.rotated_at),
476            ("activated_at", raw.activated_at, "retired_at", raw.retired_at),
477            ("rotated_at", raw.rotated_at, "retired_at", raw.retired_at),
478            ("retired_at", raw.retired_at, "destroyed_at", raw.destroyed_at),
479        ];
480        for (earlier_name, earlier, later_name, later) in pairs {
481            if let (Some(e), Some(l)) = (earlier, later)
482                && l < e
483            {
484                return Err(TypeError::InvalidAuditInput(format!(
485                    "{later_name} ({l}) precedes {earlier_name} ({e}) — state-machine ordering violated"
486                )));
487            }
488        }
489
490        // Re-validate the numeric-bound invariants that
491        // `KeyLifecycleRecord::new` enforces on construction. The
492        // state-machine-invariant pass above does not cover these,
493        // so without this gate a tampered persisted record could
494        // carry security_level = 999 or rotation_interval_days = 0
495        // past the round-trip.
496        validate_security_level(raw.security_level)?;
497        validate_rotation_interval(raw.rotation_interval_days)?;
498
499        Ok(KeyLifecycleRecord {
500            key_id: raw.key_id,
501            key_type: raw.key_type,
502            security_level: raw.security_level,
503            current_state: raw.current_state,
504            state_history: raw.state_history,
505            generator: raw.generator,
506            approvers: raw.approvers,
507            destroyer: raw.destroyer,
508            generated_at: raw.generated_at,
509            activated_at: raw.activated_at,
510            rotated_at: raw.rotated_at,
511            retired_at: raw.retired_at,
512            destroyed_at: raw.destroyed_at,
513            rotation_interval_days: raw.rotation_interval_days,
514            overlap_period_days: raw.overlap_period_days,
515        })
516    }
517}

519/// Reject `security_level` outside the documented 1-5 range. Used by
520/// both `KeyLifecycleRecord::new` and the deserializer so the in-memory
521/// and on-disk paths share one gate.
522fn validate_security_level(security_level: u32) -> Result<()> {
523    if !(1..=5).contains(&security_level) {
524        return Err(TypeError::InvalidAuditInput(format!(
525            "security_level {} outside documented range 1-5",
526            security_level
527        )));
528    }
529    Ok(())
530}

532/// Reject `rotation_interval_days == 0`. A zero interval makes
533/// `requires_rotation()` return `true` immediately on activation, so
534/// every key would be flagged for rotation at the moment it goes live —
535/// almost certainly a configuration bug, not a real lifecycle policy.
536fn validate_rotation_interval(rotation_interval_days: u32) -> Result<()> {
537    if rotation_interval_days == 0 {
538        return Err(TypeError::InvalidAuditInput(format!(
539            "rotation_interval_days {} is zero; would force immediate rotation on activation",
540            rotation_interval_days
541        )));
542    }
543    Ok(())
544}

546/// Record of a state transition.
547///
548/// Fields are private — once a transition is recorded into
549/// `KeyLifecycleRecord::state_history`, mutation of any field would
550/// invalidate the audit trail. Construction goes through
551/// [`Self::new`]; external code reads via typed accessors.
552#[non_exhaustive]
553#[derive(Debug, Clone, Serialize, Deserialize)]
554pub struct StateTransition {
555    from_state: Option<KeyLifecycleState>,
556    to_state: KeyLifecycleState,
557    timestamp: chrono::DateTime<chrono::Utc>,
558    custodian_id: String,
559    justification: String,
560    approval_id: Option<String>,
561}

563impl StateTransition {
564    /// Construct a new state-transition record. Internal callers (the
565    /// `transition_to` driver) capture `now` once and pass it in so the

577    }

588        self.to_state
589    }

613    }

616impl KeyLifecycleRecord {
617    /// Create new key lifecycle record
618    ///
619    /// # Arguments
620    ///
621    /// * `key_id` - Unique identifier for the key
622    /// * `key_type` - Algorithm/key type (e.g., "ML-KEM-768")
623    /// * `security_level` - Security level (1-5, NIST PQC categories)
624    /// * `rotation_interval_days` - How often to rotate the key (must be > 0)
625    /// * `overlap_period_days` - Overlap period during rotation
626    ///
627    /// # Errors
628    ///
629    /// Returns [`TypeError::InvalidAuditInput`] if `security_level` is
630    /// outside the documented 1-5 range, or if `rotation_interval_days`
631    /// is zero (forcing immediate rotation on activation). The
632    /// deserializer applies the same checks so in-memory and on-disk
633    /// records share one validation gate.
634    pub fn new(
635        key_id: String,
636        key_type: String,
637        security_level: u32,
638        rotation_interval_days: u32,
639        overlap_period_days: u32,
640    ) -> Result<Self> {
641        validate_security_level(security_level)?;
642        validate_rotation_interval(rotation_interval_days)?;
643        Ok(Self {
644            key_id,
645            key_type,
646            security_level,
647            current_state: KeyLifecycleState::Generation,
648            state_history: Vec::new(),
649            generator: None,
650            approvers: Vec::new(),
651            destroyer: None,
652            generated_at: chrono::Utc::now(),
653            activated_at: None,
654            rotated_at: None,
655            retired_at: None,
656            destroyed_at: None,
657            rotation_interval_days,
658            overlap_period_days,
659        })
660    }

662    /// Transition key to new state with custodianship tracking
663    ///
664    /// # Arguments
665    ///
666    /// * `to_state` - Target state
667    /// * `custodian_id` - ID of the custodian performing the transition
668    /// * `justification` - Reason for the transition
669    /// * `approval_id` - Optional approval reference
670    ///
671    /// # Errors
672    ///
673    /// Returns `TypeError::InvalidStateTransition` if the transition is invalid
674    pub fn transition(
675        &mut self,
676        to_state: KeyLifecycleState,
677        custodian_id: String,
678        justification: String,
679        approval_id: Option<String>,
680    ) -> Result<()> {
681        // Audit-trail input gates: empty / control-chars / oversized
682        // strings break JSONL audit consumers and produce attribution-
683        // free entries. Reject up front rather than persisting them.
684        validate_audit_field("custodian_id", &custodian_id)?;
685        validate_audit_field("justification", &justification)?;
686        // Inside `Some(_)` the caller has explicitly chosen to send a
687        // value; empty-string is a sentinel that confuses
688        // presence-vs-content matching downstream. Reject it here.
689        if let Some(ref approval_id) = approval_id {
690            validate_audit_field("approval_id", approval_id)?;
691        }
692
693        if !KeyStateMachine::is_valid_transition(Some(self.current_state), to_state) {
694            return Err(TypeError::InvalidStateTransition {
695                from: self.current_state,
696                to: to_state,
697            });
698        }
699        // Capture `now` once so the state_history timestamp and the
700        // per-state timestamp below match exactly. An auditor comparing
701        // `state_history[i].timestamp` against `activated_at` (etc.)
702        // would otherwise see sub-tick skew on every transition.
703        let now = chrono::Utc::now();
704        let transition = StateTransition::new(
705            Some(self.current_state),
706            to_state,
707            now,
708            custodian_id.clone(),
709            justification,
710            approval_id,
711        );
712
713        if self.state_history.len() >= MAX_STATE_HISTORY {
714            return Err(TypeError::InvalidAuditInput(format!(
715                "state_history at cap of {} entries; cannot record further transitions",
716                MAX_STATE_HISTORY
717            )));
718        }
719        self.state_history.push(transition);
720        self.current_state = to_state;
721
722        // Update timestamps (use the captured `now`).
723        match to_state {
724            KeyLifecycleState::Active => self.activated_at = Some(now),
725            KeyLifecycleState::Rotating => self.rotated_at = Some(now),
726            KeyLifecycleState::Retired => self.retired_at = Some(now),
727            KeyLifecycleState::Destroyed => self.destroyed_at = Some(now),
728            _ => {}
729        }
730
731        // Update custodianship
732        // Generator is the custodian who completes key generation (moves to Active)
733        // Check the last transition we just added to see if it was from Generation
734        if let Some(last_transition) = self.state_history.last()
735            && last_transition.from_state == Some(KeyLifecycleState::Generation)
736            && to_state == KeyLifecycleState::Active
737        {
738            self.generator = Some(custodian_id.clone());
739        }
740        if to_state == KeyLifecycleState::Destroyed {
741            self.destroyer = Some(custodian_id);
742        }
743
744        Ok(())
745    }
746
747    /// Days elapsed since `activated`, signed (negative for future
748    /// activation due to clock skew). Internal helper backing
749    /// [`Self::requires_rotation`] and [`Self::age_days`] so the
750    /// `Utc::now().signed_duration_since(...).num_days()` chain lives
751    /// in one place.
752    fn days_since_activation(activated: chrono::DateTime<chrono::Utc>) -> i64 {
753        chrono::Utc::now().signed_duration_since(activated).num_days()
754    }
755
756    /// Check if key is due for rotation per SP 800-57
757    #[must_use]
758    pub fn requires_rotation(&self) -> bool {
759        let Some(activated_at) = self.activated_at else { return false };
760        let age_days_i64 = Self::days_since_activation(activated_at);
761        if age_days_i64 < 0 {
762            // Future activation date — clock skew or misconfiguration
763            tracing::warn!(
764                "Key has negative age ({age_days_i64} days); activation_at in the future"
765            );
766            return false;
767        }
768        // Ages larger than u32::MAX → always require rotation
769        let age_days = u32::try_from(age_days_i64).unwrap_or(u32::MAX);
770        age_days >= self.rotation_interval_days
771    }
772
773    /// Get key age in days since activation
774    #[must_use]
775    pub fn age_days(&self) -> Option<u32> {
776        self.activated_at.map(|activated| {
777            // Negative ages (future activation) treated as 0.
778            u32::try_from(Self::days_since_activation(activated)).unwrap_or(0)
779        })
780    }

782    /// Check if key is in valid state for use
783    #[must_use]
784    pub fn is_valid_for_use(&self) -> bool {
785        matches!(self.current_state, KeyLifecycleState::Active | KeyLifecycleState::Rotating)
786    }

788    /// Add an approver to the key. Silently ignored if the approver
789    /// is already in the list, or if the cap (`MAX_APPROVERS`) has
790    /// been reached. Returns `false` on cap rejection so callers can
791    /// surface a warning if needed — the `#[must_use]` here forces
792    /// every caller to acknowledge the bool, since
793    /// changed the return type from `()` precisely to expose this
794    /// case and a caller that just `.;` it would silently lose the
795    /// signal.
796    #[must_use = "add_approver returns false when the cap is reached; surface that to the caller"]
797    pub fn add_approver(&mut self, approver_id: impl Into<String>) -> bool {
798        let approver_id = approver_id.into();
799        if self.approvers.contains(&approver_id) {
800            return true;
801        }
802        if self.approvers.len() >= MAX_APPROVERS {
803            return false;
804        }
805        self.approvers.push(approver_id);
806        true
807    }

935    /// Proves that the only valid initial state is Generation.
936    /// Security property: keys must begin in the Generation state.
937    #[kani::proof]
938    fn key_state_machine_only_generation_from_none() {
939        let to: KeyLifecycleState = kani::any();
940        kani::assume(to != KeyLifecycleState::Generation);
941
942        let is_valid = KeyStateMachine::is_valid_transition(None, to);
943        kani::assert(!is_valid, "Only Generation is valid from initial state (None)");
944    }
945
946    /// Proves that `is_valid_transition` matches the SP 800-57 transition
947    /// specification for all state pairs. This is an independent encoding of
948    /// the allowed transitions — if `is_valid_transition` has a bug, this
949    /// proof will catch the discrepancy.
950    ///
951    /// Note: `allowed_next_states()` returns `Vec`, which Kani cannot
952    /// efficiently verify (unbounded loop unwinding on heap iteration).
953    /// Unit tests verify `allowed_next_states()` separately.
954    #[kani::proof]
955    fn key_state_machine_transitions_match_spec() {
956        let from: KeyLifecycleState = kani::any();
957        let to: KeyLifecycleState = kani::any();
958
959        let transition_valid = KeyStateMachine::is_valid_transition(Some(from), to);
960
961        // Independent specification of SP 800-57 allowed transitions
962        let spec_allows = match from {
963            KeyLifecycleState::Generation => to == KeyLifecycleState::Active,
964            KeyLifecycleState::Active => {
965                to == KeyLifecycleState::Rotating || to == KeyLifecycleState::Retired
966            }
967            KeyLifecycleState::Rotating => to == KeyLifecycleState::Retired,
968            KeyLifecycleState::Retired => to == KeyLifecycleState::Destroyed,
969            KeyLifecycleState::Destroyed => false,
970        };
971
972        kani::assert(
973            transition_valid == spec_allows,
974            "is_valid_transition must match SP 800-57 specification",
975        );
976    }
977
978    /// Proves that retired keys can only transition to Destroyed.
979    /// Security property: once retired, the only path is secure destruction.
980    #[kani::proof]
981    fn key_state_machine_retired_only_to_destroyed() {
982        let to: KeyLifecycleState = kani::any();
983        kani::assume(to != KeyLifecycleState::Destroyed);
984
985        let is_valid = KeyStateMachine::is_valid_transition(Some(KeyLifecycleState::Retired), to);
986        kani::assert(!is_valid, "Retired keys can only transition to Destroyed");
987    }
