1//! # Secret byte containers
2//!
3//! Provides the two primitive secret types used throughout LatticeArc:
4//!
5//! - [`SecretBytes<N>`]: fixed-size secret bytes, stack-allocated `[u8; N]` backing.
6//!   Preferred whenever the length is compile-time known.
7//! - [`SecretVec`]: variable-length secret bytes, heap-allocated `Vec<u8>` backing.
8//!   Used only when the length genuinely varies at runtime.
9//!
10//! Both types uphold the full Secret Type Invariant set — see
11//! [`docs/SECRET_TYPE_INVARIANTS.md`](../../../docs/SECRET_TYPE_INVARIANTS.md)
12//! for the canonical, normative specification.
13//!
14//! ## Quick reference
15//!
16//! | Invariant | Mechanism |
17//! |---|---|
18//! | Wipe on drop | `#[derive(Zeroize, ZeroizeOnDrop)]` — volatile writes |
19//! | Redacted `Debug` | Manual `fmt::Debug` impl emits `"[REDACTED]"` |
20//! | Timing-safe equality | `impl ConstantTimeEq` |
21//! | No `PartialEq` / `Eq` | Enforced at `tests/no_partial_eq_on_secret_types.rs` |
22//! | No `Clone` | Explicit `clone_for_transmission(&self) -> Self` |
23//! | Sealed accessor | `expose_secret()` only — no `AsRef`, no `Deref` |
24//!
25//! ## When to use which
26//!
27//! ```text
28//! // 32-byte HKDF-Extract output: fixed size → SecretBytes<32>
29//! fn hkdf_extract(...) -> SecretBytes<32>;
30//!
31//! // 64-byte hybrid shared secret: fixed size → SecretBytes<64>
32//! struct EncapsulatedKey { shared_secret: SecretBytes<64>, ... }
33//!
34//! // AEAD plaintext: length varies with input → SecretVec
35//! fn decrypt_aes_gcm(...) -> SecretVec;
36//!
37//! // PBKDF2 output with user-specified key_length → SecretVec
38//! fn pbkdf2(..., key_length: usize) -> SecretVec;
39//! ```

46// ============================================================================
47// SecretBytes<N> — fixed-size, stack-allocated
48// ============================================================================
49
50/// Fixed-size secret byte container, stack-allocated.
51///
52/// The `[u8; N]` backing store never reallocates, never migrates across the heap,
53/// and is wiped by volatile writes on drop. Prefer this type over [`SecretVec`]
54/// whenever `N` is compile-time known.
55///
56/// # Security invariants
57///
58/// - **Wipe on drop**: `#[derive(Zeroize, ZeroizeOnDrop)]` emits volatile writes
59///   that the compiler cannot optimize away.
60/// - **Redacted `Debug`**: the bytes are never printed; `"[REDACTED]"` is emitted.
61/// - **Timing-safe equality**: [`ConstantTimeEq`] compares all `N` bytes in time
62///   independent of their contents; `PartialEq`/`Eq` are not implemented.
63/// - **No `Clone`**: duplication must go through [`clone_for_transmission`] so
64///   every copy is a grep-able audit checkpoint.
65/// - **Sealed access**: the bytes are exposed only through [`expose_secret`]. No
66///   `AsRef<[u8]>`, no `Deref`, no public field.
67///
68/// [`clone_for_transmission`]: Self::clone_for_transmission
69/// [`expose_secret`]: Self::expose_secret
70///
71/// # Why stack-allocated matters
72///
73/// - Heap allocators retain size metadata after `free`, which is a side-channel
74///   leak about secret sizes in adversary-controlled environments.
75/// - A `Vec<u8>` with a single `.push()` past capacity silently reallocates and
76///   frees the old buffer **without zeroization**. `[u8; N]` has no such escape.
77/// - Stack memory is reclaimed in deterministic scope order, not at allocator
78///   whim.
79///
80/// # Example
81///
82/// ```
83/// use latticearc::types::SecretBytes;
84///
85/// // Construct from raw bytes (e.g., after filling a buffer with KDF output)
86/// let key: SecretBytes<32> = SecretBytes::new([0x42u8; 32]);
87/// assert_eq!(key.len(), 32);
88/// assert_eq!(key.expose_secret()[0], 0x42);
89///
90/// // Debug is redacted
91/// let debug = format!("{:?}", key);
92/// assert!(debug.contains("[REDACTED]"));
93/// assert!(!debug.contains("42"));
94/// ```
95#[derive(Zeroize, ZeroizeOnDrop)]
96pub struct SecretBytes<const N: usize> {
97    bytes: [u8; N],
98}
99
100impl<const N: usize> SecretBytes<N> {
101    /// Construct from raw bytes.
102    ///
103    /// The passed array is moved in; no copy is made beyond the move itself.
104    #[must_use]
105    #[inline]
106    pub const fn new(bytes: [u8; N]) -> Self {
107        Self { bytes }
108    }
109
110    /// Construct an all-zero buffer.
111    ///
112    /// Build the final byte pattern in a plain `[u8; N]` on the stack, then
113    /// hand the array to [`Self::new`]; use [`Self::zero`] when the caller
114    /// wants an all-zero placeholder (e.g., a struct field that is overwritten
115    /// by a downstream operation).
116    #[must_use]
117    #[inline]
118    pub const fn zero() -> Self {
119        Self { bytes: [0u8; N] }
120    }
121
122    /// Length in bytes. Always equals the const generic `N`.
123    #[must_use]
124    #[inline]
125    pub const fn len(&self) -> usize {
126        N
127    }
128
129    /// Returns `true` if `N == 0`.
130    #[must_use]
131    #[inline]
132    pub const fn is_empty(&self) -> bool {
133        N == 0
134    }
135
136    /// Expose the inner bytes.
137    ///
138    /// This is the **only** public read accessor. Every call site is a grep-able
139    /// audit checkpoint: `rg '\.expose_secret\(\)'` enumerates every point at
140    /// which secret bytes are read.
141    ///
142    /// Consumers should hold the returned slice only as long as needed and
143    /// should never copy it into a non-secret container (a subsequent `.to_vec()`
144    /// or struct field of type `Vec<u8>` would evade zeroization).
145    #[must_use]
146    #[inline]
147    pub const fn expose_secret(&self) -> &[u8; N] {
148        &self.bytes
149    }
150
151    /// Create an independent copy.
152    ///
153    /// `Clone` is intentionally not derived. Use this method when duplication is
154    /// semantically required — for example, sending a shared secret across a
155    /// channel boundary — so every duplication is grep-able.
156    ///
157    /// # Note on memory locking
158    ///
159    /// `SecretBytes<N>` is stack-allocated, so `mlock(2)`-style page
160    /// pinning does not apply: each clone is a fresh stack location
161    /// (no heap pages to lock). If a caller specifically needs OS-level
162    /// memory-lock semantics for a duplicated secret, use
163    /// [`SecretVec`] (heap-backed) instead.
164    #[must_use]
165    #[inline]
166    pub fn clone_for_transmission(&self) -> Self {
167        Self { bytes: self.bytes }
168    }
169}
170
171impl<const N: usize> fmt::Debug for SecretBytes<N> {
172    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
173        f.debug_struct("SecretBytes").field("len", &N).field("bytes", &"[REDACTED]").finish()
174    }
175}
176
177impl<const N: usize> ConstantTimeEq for SecretBytes<N> {
178    fn ct_eq(&self, other: &Self) -> Choice {
179        self.bytes.ct_eq(&other.bytes)
180    }
181}

183// ============================================================================
184// SecretVec — variable-length, heap-allocated
185// ============================================================================
186
187/// Variable-length secret byte container, heap-allocated.
188///
189/// Use only when the length genuinely varies at runtime (e.g., user-specified
190/// PBKDF2 output length, AEAD plaintext of arbitrary input). For any
191/// compile-time-known length, prefer [`SecretBytes<N>`].
192///
193/// # Security invariants
194///
195/// Same as [`SecretBytes`]. See type-level docs for the full list.
196///
197/// # Reallocation safety
198///
199/// The internal `Vec<u8>` is never reallocated by this type's public API: there
200/// is no `push`, no `extend`, no `reserve`. All construction paths allocate the
201/// final size up front. If a caller needs to build a secret incrementally, they
202/// must build the full `Vec<u8>` first (outside this wrapper), then pass it to
203/// [`Self::new`] — at which point reallocation is no longer a concern because
204/// subsequent mutation is not exposed.
205///
206/// # OS-level memory locking (`secret-mlock` feature)
207///
208/// With the `secret-mlock` feature enabled, each `SecretVec` locks its backing
209/// buffer into RAM via `mlock(2)` (Linux/macOS) or `VirtualLock` (Windows) so
210/// the bytes cannot be swapped to disk or captured in a core dump. This
211/// implements invariant I-10. If locking fails at construction time (e.g.,
212/// `RLIMIT_MEMLOCK` exceeded), the `SecretVec` is still returned with an
213/// unlocked backing — the bytes remain zeroized on drop, but OS-level leakage
214/// protection is not active. Without the feature, `mlock` is not called and
215/// the type has no dependency on the `region` crate.
216///
217/// `SecretBytes<N>` is deliberately NOT covered by this feature: its `[u8; N]`
218/// backing is stack-allocated, so mlocking it would require per-instance
219/// page-aligned locking of arbitrary stack addresses — impractical and
220/// incompatible with Rust's move semantics. Callers who need OS-level
221/// protection for small fixed-size secrets should wrap them in `SecretVec` at
222/// a boundary where the sensitive bytes leave the stack.
223///
224/// # Example
225///
226/// ```
227/// use latticearc::types::SecretVec;
228///
229/// let key = SecretVec::new(vec![0x42u8; 48]);
230/// assert_eq!(key.len(), 48);
231/// assert_eq!(key.expose_secret()[0], 0x42);
232///
233/// let debug = format!("{:?}", key);
234/// assert!(debug.contains("[REDACTED]"));
235/// ```
236#[derive(Zeroize, ZeroizeOnDrop)]
237pub struct SecretVec {
238    // Drop order when `secret-mlock` is enabled:
239    //   1. `ZeroizeOnDrop`'s generated `Drop` runs first, calling `.zeroize()`
240    //      on `bytes` while the region is still mlocked.
241    //   2. Fields drop in declaration order: `_lock` first (unlocks the
242    //      region via `MlockGuard`'s panic-tolerant Drop), then `bytes`
243    //      (returns the zeroed buffer to the allocator).
244    // `#[zeroize(skip)]` is required because the lock holder does not
245    // implement `Zeroize` and there's nothing secret inside it — it only
246    // holds a handle to an OS lock.
247    #[cfg(all(feature = "secret-mlock", not(target_os = "windows")))]
248    #[zeroize(skip)]
249    _lock: Option<MlockGuard>,
250    bytes: Vec<u8>,
251}
252
253/// Panic-tolerant wrapper around [`region::LockGuard`].
254///
255/// # Why this wrapper exists
256///
257/// `region::LockGuard::drop` calls `munlock`/`VirtualUnlock` and panics on any
258/// error. That is too strict on Windows: `VirtualUnlock` returns
259/// `ERROR_NOT_LOCKED` (158) when the OS has already released the lock — most
260/// commonly because the working set was trimmed under memory pressure. Per
261/// Microsoft's documentation `VirtualLock` is *best-effort*: pages can be
262/// implicitly unlocked when the working set is trimmed, and a subsequent
263/// `VirtualUnlock` returning `ERROR_NOT_LOCKED` is documented behaviour, not
264/// a logic error. Panicking on it is wrong, and panicking inside `Drop` is
265/// itself a soundness hazard — secret containers can drop while unwinding
266/// from another panic, and a Drop-time double-panic aborts the process.
267///
268/// # Behaviour
269///
270/// On every platform, the wrapper hands `region::LockGuard::drop` to
271/// [`std::panic::catch_unwind`] so that any panic raised by the underlying
272/// unlock syscall (notably the Windows `ERROR_NOT_LOCKED` path) is caught
273/// at the wrapper boundary instead of escaping into the surrounding Drop
274/// chain.
275///
276/// - **Unix:** the `munlock` syscall does not have the working-set-trim
277///   quirk and reliably succeeds for any region we successfully locked, so
278///   `catch_unwind` should never fire — the guard runs through normally.
279///   The wrapper is functionally identical to a bare
280///   `Option<region::LockGuard>` on this platform.
281/// - **Windows:** if `VirtualUnlock` returns `ERROR_NOT_LOCKED` (the
282///   working-set-trim case), `region` raises a panic; we catch it,
283///   discard it, and return cleanly. If unlock succeeds (the common case)
284///   the panic is never raised. Either way the OS lock accounting is
285///   resolved before the backing `Vec` is freed.
286///
287/// # Why `catch_unwind` and not direct FFI
288///
289/// A direct `VirtualUnlock` FFI call would also work but requires
290/// `unsafe_code` (the workspace-wide deny gate would need a targeted
291/// `#[allow]` with justification) and either a new `windows-sys` dep or
292/// hand-rolled FFI declarations. Catching the panic at the Rust boundary
293/// keeps the safe-code guarantee, avoids a new platform-dep, and still
294/// achieves correct unlock-on-drop semantics. The runtime cost on the
295/// success path (no panic raised) is a single `catch_unwind` setup, which
296/// is a stack-only operation — negligible compared to the syscall.
297///
298/// # Security guarantee
299///
300/// `VirtualLock` is documented as best-effort; a working-set trim could
301/// already have swapped the page to disk *before* our drop. Whether we
302/// explicitly unlock at drop or not has no effect on whether the secret
303/// was resident while it mattered. The bytes are zeroized by
304/// `ZeroizeOnDrop` regardless.
305///
306/// # Caveat
307///
308/// `catch_unwind` is a no-op when the binary is built with
309/// `panic = "abort"`; in that mode the original `LockGuard::drop` panic
310/// would still abort the process. The workspace defaults to
311/// `panic = "unwind"` and `catch_unwind` works as expected. Downstream
312/// users who switch to `panic = "abort"` for size/perf should be aware
313/// that the Windows working-set-trim path becomes an abort hazard again.
314///
315/// # Upstream
316///
317/// If `region` ever changes its `Drop` to tolerate `ERROR_NOT_LOCKED` on
318/// Windows, this wrapper can be deleted and the field reverted to
319/// `Option<region::LockGuard>`.
320#[cfg(all(feature = "secret-mlock", not(target_os = "windows")))]
321struct MlockGuard(Option<region::LockGuard>);
322
323#[cfg(all(feature = "secret-mlock", not(target_os = "windows")))]
324impl Drop for MlockGuard {
325    fn drop(&mut self) {
326        let Some(guard) = self.0.take() else { return };
327        // Run the inner `LockGuard::drop` (which calls `munlock` /
328        // `VirtualUnlock`) inside `catch_unwind` so that any panic — most
329        // notably Windows `ERROR_NOT_LOCKED` after a working-set trim —
330        // is contained at this wrapper boundary instead of cascading into
331        // a Drop-time double-panic that would abort the process.
332        // `AssertUnwindSafe` is sound here because we discard `guard` on
333        // either path; nothing observed by other code can be left in an
334        // inconsistent state.
335        let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(move || {
336            drop(guard);
337        }));
338    }
339}

341impl SecretVec {
342    /// Construct from a `Vec<u8>`.
343    ///
344    /// The vector is moved in; its capacity is taken as-is. For best security
345    /// hygiene, construct the passed `Vec` at its final length (e.g.,
346    /// `vec![0u8; len]`) rather than growing it, so no unzeroized intermediate
347    /// buffer is ever freed.
348    #[must_use]
349    #[inline]
350    pub fn new(bytes: Vec<u8>) -> Self {
351        Self::from_bytes(bytes)
352    }
353
354    /// Construct an all-zero buffer of the given length.
355    #[must_use]
356    #[inline]
357    pub fn zero(len: usize) -> Self {
358        Self::from_bytes(vec![0u8; len])
359    }
360
361    /// Single construction path.
362    ///
363    /// Allocates a fresh `Vec<u8>` of exactly `bytes.len()` capacity, copies
364    /// the secret in, then zeroizes the caller-supplied buffer in place. This
365    /// is deliberately NOT `bytes.shrink_to_fit()`: `shrink_to_fit` may
366    /// internally call the allocator's `realloc`, which copies the secret
367    /// into a smaller allocation and frees the original WITHOUT zeroizing.
368    /// The freed buffer is then immediately available for allocator reuse
369    /// — a classic "secret in freed memory" leak that affects any caller
370    /// passing in an over-allocated `Vec` (e.g. `BASE64_ENGINE.decode(...)?`,
371    /// which pre-sizes with slack).
372    ///
373    /// The exact-size copy + caller-side wipe pattern guarantees:
374    /// 1. Backing capacity == length, so `Vec::zeroize` (which only wipes
375    ///    `..len`, not `..capacity` in `zeroize 1.8`) covers every byte at
376    ///    Drop time.
377    /// 2. The original caller-supplied buffer is wiped before its `Vec`
378    ///    Drop hands the memory back to the allocator, so no copy of the
379    ///    secret persists in any allocator free-list.
380    #[inline]
381    fn from_bytes(mut bytes: Vec<u8>) -> Self {
382        let len = bytes.len();
383        let mut owned: Vec<u8> = Vec::with_capacity(len);
384        owned.extend_from_slice(&bytes);
385        // Wipe the caller-supplied buffer in place: this covers any slack
386        // capacity in the ORIGINAL allocation, which the new `owned` does
387        // not inherit. Without this, an over-sized incoming `Vec` (e.g.
388        // base64 decode output) would leave secret material in the freed
389        // tail when `bytes` is dropped.
390        use zeroize::Zeroize;
391        bytes.zeroize();
392        drop(bytes);
393
394        #[cfg(all(feature = "secret-mlock", not(target_os = "windows")))]
395        let _lock = Self::try_lock(&owned);
396        Self {
397            #[cfg(all(feature = "secret-mlock", not(target_os = "windows")))]
398            _lock,
399            bytes: owned,
400        }
401    }
402
403    /// Attempt to lock the backing buffer into RAM. Returns `None` on empty
404    /// buffers (locking a dangling pointer is UB) or on `mlock` failure
405    /// (e.g. `RLIMIT_MEMLOCK` exceeded, or Windows working-set quota
406    /// exceeded). Failures are swallowed: the bytes are still zeroized on
407    /// drop, we simply lose OS-level leakage protection.
408    ///
409    /// The successful lock is returned wrapped in [`MlockGuard`] so its Drop
410    /// is panic-tolerant — see the type-level docs there for the Windows
411    /// `VirtualUnlock` rationale.
412    #[cfg(all(feature = "secret-mlock", not(target_os = "windows")))]
413    #[inline]
414    fn try_lock(bytes: &[u8]) -> Option<MlockGuard> {
415        if bytes.is_empty() {
416            return None;
417        }
418        region::lock(bytes.as_ptr(), bytes.len()).ok().map(|g| MlockGuard(Some(g)))
419    }
420
421    /// Length in bytes.
422    #[must_use]
423    #[inline]
424    pub fn len(&self) -> usize {
425        self.bytes.len()
426    }
427
428    /// Returns `true` if the backing buffer is empty.
429    #[must_use]
430    #[inline]
431    pub fn is_empty(&self) -> bool {
432        self.bytes.is_empty()
433    }
434
435    /// Expose the inner bytes. The only public read accessor.
436    #[must_use]
437    #[inline]
438    pub fn expose_secret(&self) -> &[u8] {
439        &self.bytes
440    }
441
442    /// Create an independent copy. See [`SecretBytes::clone_for_transmission`].
443    ///
444    /// With the `secret-mlock` feature, the cloned buffer is independently
445    /// locked into RAM (the original's `LockGuard` does not transfer).
446    #[must_use]
447    pub fn clone_for_transmission(&self) -> Self {
448        Self::from_bytes(self.bytes.clone())
449    }
450}

456            .field("bytes", &"[REDACTED]")
