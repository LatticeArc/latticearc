1//! Shared timing-equalizer dummy material for hybrid signature verify.
2//!
3//! Both [`crate::hybrid::sig_hybrid::verify`] (typed-input lower-level
4//! path) and [`crate::unified_api::convenience::api::verify_hybrid_ml_dsa_ed25519`]
5//! (raw-byte unified-API path) need shape-correct stand-in inputs to
6//! substitute when a caller-supplied PK or signature fails the shape
7//! check. Running the verify pipeline against the substitute keeps
8//! wall-clock cost equal between shape-fail and verify-fail, removing
9//! the timing oracle that opaque error returns alone do not close.
10//!
11//! Both layers consume the dummy as raw bytes (and re-parse internally
12//! via the same `verify_pq_ml_dsa_unverified` entry point). Holding raw
13//! bytes only — no pre-parsed structures — means this module contains
14//! no fallible operations: the cache stays infallibly initializable
15//! without any `expect` / `panic` / `#[allow]` escape hatches.
16
17#![deny(unsafe_code)]
18#![deny(missing_docs)]
19
20use crate::primitives::sig::ml_dsa::MlDsaParameterSet;
21
22/// Per-parameter-set cached zero-byte buffers used by both the
23/// `sig_hybrid::verify` and `unified_api::verify_hybrid_ml_dsa_ed25519`
24/// timing equalizers.
25///
26/// The Ed25519 dummies are constant-size (32 / 64 bytes) and shared
27/// across parameter sets but live in this struct for locality. No
28/// secret material; safe to keep `'static`.
29pub struct HybridVerifyDummy {
30    /// Raw zero-byte ML-DSA public key, length matches the parameter set.
31    pub pq_pk: Vec<u8>,
32    /// Raw zero-byte ML-DSA signature, length matches the parameter set.
33    pub pq_sig: Vec<u8>,
34    /// Raw zero-byte Ed25519 public key (32 bytes).
35    pub ed_pk: Vec<u8>,
36    /// Raw zero-byte Ed25519 signature (64 bytes).
37    pub ed_sig: Vec<u8>,
38}
39
40/// Lazy-initialized per-parameter-set dummy material.
41///
42/// First call for each `param_set` allocates the zero buffers; the
43/// values are cached for the process lifetime. The init closure is
44/// infallible — only `Vec` allocations, no parse, no `expect`.
45#[must_use]
46pub fn hybrid_verify_dummy_material(param_set: MlDsaParameterSet) -> &'static HybridVerifyDummy {
47    use std::sync::OnceLock;
48
49    static M44: OnceLock<HybridVerifyDummy> = OnceLock::new();
50    static M65: OnceLock<HybridVerifyDummy> = OnceLock::new();
51    static M87: OnceLock<HybridVerifyDummy> = OnceLock::new();
52
53    let cell = match param_set {
54        MlDsaParameterSet::MlDsa44 => &M44,
55        MlDsaParameterSet::MlDsa65 => &M65,
56        MlDsaParameterSet::MlDsa87 => &M87,
57    };
58    cell.get_or_init(|| HybridVerifyDummy {
59        pq_pk: vec![0u8; param_set.public_key_size()],
60        pq_sig: vec![0u8; param_set.signature_size()],
61        ed_pk: vec![0u8; 32],
62        ed_sig: vec![0u8; 64],
63    })
64}
