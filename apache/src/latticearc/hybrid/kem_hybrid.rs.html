6//! Hybrid Key Encapsulation Mechanism (KEM) Module
7//!
8//! This module provides hybrid key encapsulation combining post-quantum (ML-KEM)
9//! and classical (X25519 ECDH) algorithms for quantum-resistant key exchange
10//! with classical security guarantees.
11//!
12//! # Architecture
13//!
14//! ```text
15//! ┌─────────────────────────────────────────────────────────────────────────┐
16//! │                    HYBRID KEM: Encapsulation Flow                       │
17//! ├─────────────────────────────────────────────────────────────────────────┤
18//! │                                                                         │
19//! │  ┌───────────────┐      ┌──────────────────────────────────────────┐   │
20//! │  │  Recipient's  │      │             Sender (Encapsulator)        │   │
21//! │  │  Public Key   │      │                                          │   │
22//! │  │               │      │  ┌─────────────────────────────────────┐ │   │
23//! │  │ ┌───────────┐ │      │  │         ML-KEM-768 Encaps          │ │   │
24//! │  │ │ ML-KEM PK │─┼──────┼─►│ RNG ──► Ciphertext (1088 B)        │ │   │
25//! │  │ │ (1184 B)  │ │      │  │         Shared Secret₁ (32 B)      │ │   │
26//! │  │ └───────────┘ │      │  └────────────────────┬────────────────┘ │   │
27//! │  │               │      │                       │                  │   │
28//! │  │ ┌───────────┐ │      │  ┌─────────────────────────────────────┐ │   │
29//! │  │ │ X25519 PK │─┼──────┼─►│         X25519 ECDH                 │ │   │
30//! │  │ │  (32 B)   │ │      │  │ ephemeral_sk ──► PK_ephemeral (32 B)│ │   │
31//! │  │ └───────────┘ │      │  │ ECDH(sk, PK) ──► Shared Secret₂     │ │   │
32//! │  └───────────────┘      │  └────────────────────┬────────────────┘ │   │
33//! │                         │                       │                  │   │
34//! │                         │  ┌─────────────────────────────────────┐ │   │
35//! │                         │  │         HKDF-SHA256 Combine         │ │   │
36//! │                         │  │                                     │ │   │
37//! │                         │  │  info = "LatticeArc-Hybrid-KEM-SS" ‖ len(static_pk) ‖ static_pk ‖ len(ephemeral_pk) ‖ ephemeral_pk │ │   │
38//! │                         │  │  IKM  = SS₁ ║ SS₂ (64 bytes)        │ │   │
39//! │                         │  │            ↓                        │ │   │
40//! │                         │  │  Hybrid Shared Secret (64 B)        │ │   │
41//! │                         │  └─────────────────────────────────────┘ │   │
42//! │                         └──────────────────────────────────────────┘   │
43//! │                                                                         │
44//! │  Output Ciphertext:  ML-KEM CT (1088 B) ║ X25519 PK (32 B) = 1120 B    │
45//! │  Output Secret:      64-byte hybrid shared secret                       │
46//! └─────────────────────────────────────────────────────────────────────────┘
47//!
48//! ┌─────────────────────────────────────────────────────────────────────────┐
49//! │                    HYBRID KEM: Decapsulation Flow                       │
50//! ├─────────────────────────────────────────────────────────────────────────┤
51//! │                                                                         │
52//! │  ┌───────────────┐      ┌──────────────────────────────────────────┐   │
53//! │  │   Ciphertext  │      │           Recipient (Decapsulator)       │   │
54//! │  │   (1120 B)    │      │                                          │   │
55//! │  │               │      │  ┌─────────────────────────────────────┐ │   │
56//! │  │ ┌───────────┐ │      │  │         ML-KEM-768 Decaps          │ │   │
57//! │  │ │ ML-KEM CT │─┼──────┼─►│ SK + CT ──► Shared Secret₁ (32 B)  │ │   │
58//! │  │ │ (1088 B)  │ │      │  └────────────────────┬────────────────┘ │   │
59//! │  │ └───────────┘ │      │                       │                  │   │
60//! │  │               │      │  ┌─────────────────────────────────────┐ │   │
61//! │  │ ┌───────────┐ │      │  │         X25519 ECDH                 │ │   │
62//! │  │ │ X25519 PK │─┼──────┼─►│ ECDH(my_sk, ephemeral_pk)           │ │   │
63//! │  │ │  (32 B)   │ │      │  │         ──► Shared Secret₂ (32 B)  │ │   │
64//! │  │ └───────────┘ │      │  └────────────────────┬────────────────┘ │   │
65//! │  └───────────────┘      │                       │                  │   │
66//! │                         │  ┌─────────────────────────────────────┐ │   │
67//! │                         │  │         HKDF-SHA256 Combine         │ │   │
68//! │                         │  │  Hybrid Shared Secret (64 B)        │ │   │
69//! │                         │  └─────────────────────────────────────┘ │   │
70//! │                         └──────────────────────────────────────────┘   │
71//! └─────────────────────────────────────────────────────────────────────────┘
72//! ```
73//!
74//! # Key Sizes Summary
75//!
76//! | Component     | Public Key | Secret Key | Ciphertext | Shared Secret |
77//! |---------------|------------|------------|------------|---------------|
78//! | ML-KEM-768    | 1184 B     | 2400 B     | 1088 B     | 32 B          |
79//! | X25519        | 32 B       | 32 B       | 32 B       | 32 B          |
80//! | **Hybrid**    | **1216 B** | **2432 B** | **1120 B** | **64 B**      |
81//!
82//! # Security Properties
83//!
84//! - **IND-CCA2** security from ML-KEM (post-quantum secure)
85//! - **CDH-hardness** security from X25519 ECDH (classical secure)
86//! - **HKDF composition** ensures security if *either* component remains secure (KDF-based combiner with domain separation)
87//! - Automatic memory zeroization for secret keys via [`ZeroizeOnDrop`]
88//!
89//! # Example
90//!
91//! ```rust,no_run
92//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
93//! use latticearc::hybrid::kem_hybrid::{generate_keypair, encapsulate, decapsulate};
94//!
95//! // Generate hybrid keypair
96//! let (pk, sk) = generate_keypair()?;
97//!
98//! // Encapsulate to create shared secret and ciphertext
99//! let encapsulated = encapsulate(&pk)?;
100//!
101//! // Decapsulate to recover the shared secret
102//! let shared_secret = decapsulate(&sk, &encapsulated)?;
103//!
104//! // Both parties now have the same 64-byte shared secret
105//! assert_eq!(shared_secret.as_slice(), encapsulated.shared_secret());
106//! # Ok(())
107//! # }
108//! ```
109//!
110//! [`ZeroizeOnDrop`]: zeroize::ZeroizeOnDrop

191/// Hybrid secret key combining ML-KEM and X25519 ECDH.
192///
193/// # Security Guarantees
194///
195/// The ML-KEM decapsulation key is held in-memory via the aws-lc-rs
196/// `DecapsulationKey` (BoringSSL zeroizes on free). The X25519 private key
197/// is also managed by aws-lc-rs, which handles its own zeroization internally.
198///
199/// # Cloning
200///
201/// **Important**: This type does NOT implement [`Clone`] to prevent accidental
202/// copying of secret keys. The X25519 `PrivateKey` from aws-lc-rs is not
203/// cloneable by design.
204///
205/// # Key Serialization
206///
207/// Both components are fully serializable:
208/// - ML-KEM secret key: via [`ml_kem_sk_bytes()`](Self::ml_kem_sk_bytes)
209/// - X25519 seed: via [`ecdh_seed_bytes()`](Self::ecdh_seed_bytes)
210/// - Reconstruction: via [`from_serialized()`](Self::from_serialized)
211///
212/// # Constant-Time Comparison
213///
214/// Implements [`subtle::ConstantTimeEq`] via the serialized byte-level
215/// accessors [`ml_kem_sk_bytes()`](Self::ml_kem_sk_bytes) and
216/// [`ecdh_seed_bytes()`](Self::ecdh_seed_bytes). Two secret keys compare
217/// equal iff (a) their security levels match, (b) their ML-KEM
218/// decapsulation key bytes are equal, and (c) their X25519 seed bytes are
219/// equal — each compared in constant time via `subtle`.
220///
221/// **Intentionally no [`PartialEq`] / [`Eq`] impl.** Comparison is
222/// CT-only via `ct_eq`; attempting `==` on a `HybridKemSecretKey` is a
223/// compile error, not a silent non-CT check.
224///
225/// **Secret-material surfacing.** During a `ct_eq` call both sides'
226/// serialized ML-KEM and X25519 seed bytes briefly reside on the caller's
227/// stack (wrapped in `Zeroizing`, cleared on drop). The comparison does
228/// not leave residues beyond the call.
229///
230/// Cost: two `Zeroizing<Vec<u8>>` heap allocations (ML-KEM side) plus two
231/// stack `[u8; 32]` copies (ECDH side) per compare. Acceptable outside
232/// tight loops; for hot-path deduplication, prefer comparing key IDs.
233///
234/// # Example
235///
236/// ```rust,no_run
237/// use latticearc::hybrid::kem_hybrid::generate_keypair;
238///
239/// let (pk, sk) = generate_keypair().expect("keypair generation failed");
240/// // sk.ecdh_keypair holds the real X25519 private key
241/// drop(sk);  // ML-KEM bytes zeroized; aws-lc-rs handles X25519 cleanup
242/// ```
243pub struct HybridKemSecretKey {
244    /// ML-KEM decapsulation keypair — holds the real aws-lc-rs `DecapsulationKey`.
245    ml_kem_keypair: MlKemDecapsulationKeyPair,
246    /// X25519 ECDH static key pair for reusable key agreement.
247    ecdh_keypair: X25519StaticKeyPair,
248}

298    /// ML-KEM `DecapsulationKey` and the X25519 `StaticKeyPair` from their
299    /// serialized forms.
300    ///
301    /// # Arguments
302    /// * `security_level` - ML-KEM security level (must match original key)
303    /// * `ml_kem_sk_bytes` - Secret key bytes from `ml_kem_sk_bytes()`
304    /// * `ml_kem_pk_bytes` - Public key bytes (needed for `MlKemDecapsulationKeyPair`)
305    /// * `ecdh_seed` - X25519 seed bytes from `ecdh_seed_bytes()`
306    ///
307    /// # Errors
308    /// Returns an error if ML-KEM key reconstruction or X25519 seed import fails.
309    pub fn from_serialized(
310        security_level: MlKemSecurityLevel,
311        ml_kem_sk_bytes: &[u8],
312        ml_kem_pk_bytes: &[u8],
313        ecdh_seed: &[u8; 32],
314    ) -> Result<Self, HybridKemError> {
315        let ml_kem_keypair = MlKemDecapsulationKeyPair::from_key_bytes(
316            security_level,
317            ml_kem_sk_bytes,
318            ml_kem_pk_bytes,
319        )
320        .map_err(|e| HybridKemError::MlKemError(e.to_string()))?;
321
322        let ecdh_keypair = X25519StaticKeyPair::from_seed_bytes(ecdh_seed)
323            .map_err(|e| HybridKemError::EcdhError(e.to_string()))?;
324
325        Ok(Self { ml_kem_keypair, ecdh_keypair })
326    }

334    }
335
336    /// Decapsulate an ML-KEM ciphertext using the real decapsulation key.

346            .map_err(|e| HybridKemError::MlKemError(e.to_string()))
347    }

359impl ConstantTimeEq for HybridKemSecretKey {
360    /// Constant-time comparison of two hybrid secret keys.
361    ///
362    /// Compares (security level, ML-KEM decapsulation-key bytes, X25519 seed
363    /// bytes) unconditionally via `subtle::ConstantTimeEq` so timing does
364    /// not reveal which component differs.
365    ///
366    /// If byte serialization of either key fails (not reachable for keys
367    /// generated by this crate), the corresponding leg returns
368    /// `Choice::from(0)` — never a false positive, even if both sides fail
369    /// simultaneously. Length mismatch in the ML-KEM leg (different
370    /// security levels) likewise returns 0; the preceding `level_match`
371    /// check will have already done so, and the resulting wall-clock
372    /// divergence reveals only the non-secret security level.
373    fn ct_eq(&self, other: &Self) -> subtle::Choice {
374        use subtle::Choice;
375
376        // `MlKemSecurityLevel` is not `#[repr(u8)]`, so do not rely on
377        // discriminant-cast ordering. `PartialEq` is a total equality on
378        // the enum's variants.
379        let level_match = Choice::from(u8::from(self.security_level() == other.security_level()));
380
381        // `a.len() == b.len()` is a non-CT branch. It only fires when
382        // security levels differ, and in that case `level_match` is already
383        // 0 — the wall-clock divergence leaks only the security level,
384        // which is not secret material.
385        let ml_match = match (self.ml_kem_sk_bytes(), other.ml_kem_sk_bytes()) {
386            (Ok(a), Ok(b)) if a.len() == b.len() => a.ct_eq(&b),
387            _ => Choice::from(0u8),
388        };
389        let ec_match = match (self.ecdh_seed_bytes(), other.ecdh_seed_bytes()) {
390            (Ok(a), Ok(b)) => a.ct_eq(&*b),
391            _ => Choice::from(0u8),
392        };
393
394        level_match & ml_match & ec_match
395    }
396}

483/// Generate hybrid keypair at a specific ML-KEM security level.
484///
485/// # Arguments
486/// * `level` - ML-KEM security level (512, 768, or 1024)
487///
488/// # Entropy source
489/// aws-lc-rs supplies entropy internally; callers cannot provide an RNG.
490///
491/// # Errors
492/// Returns an error if ML-KEM or X25519 keypair generation fails.
493#[must_use = "discarding a generated keypair wastes entropy and leaks key material"]
494pub fn generate_keypair_with_level(
495    level: MlKemSecurityLevel,
496) -> Result<(HybridKemPublicKey, HybridKemSecretKey), HybridKemError> {
497    let ml_kem_keypair = MlKem::generate_decapsulation_keypair(level)
498        .map_err(|e| HybridKemError::MlKemError(e.to_string()))?;
499
500    let ecdh_keypair =
501        X25519StaticKeyPair::generate().map_err(|e| HybridKemError::EcdhError(e.to_string()))?;
502
503    let ecdh_pk = ecdh_keypair.public_key_bytes().to_vec();
504
505    let pk = HybridKemPublicKey {
506        ml_kem_pk: ml_kem_keypair.public_key_bytes().to_vec(),
507        ecdh_pk,
508        security_level: level,
509    };
510
511    let sk = HybridKemSecretKey { ml_kem_keypair, ecdh_keypair };
512
513    Ok((pk, sk))
514}

520/// Returns an error if:
521/// - The ECDH public key is not exactly 32 bytes.
522/// - ML-KEM public key construction or encapsulation fails.
523/// - ML-KEM encapsulation returns an invalid shared secret length.
524/// - The ECDH public key format is invalid for conversion.
525/// - Key derivation (HKDF) fails.
526pub fn encapsulate(pk: &HybridKemPublicKey) -> Result<EncapsulatedKey, HybridKemError> {
527    // Validate ECDH public key length
528    if pk.ecdh_pk.len() != X25519_KEY_SIZE {
529        return Err(HybridKemError::InvalidKeyMaterial(format!(
530            "ECDH public key must be {} bytes, got {}",
531            X25519_KEY_SIZE,
532            pk.ecdh_pk.len()
533        )));
534    }
535
536    // ML-KEM encapsulation at the public key's security level
537    let ml_kem_pk_struct = MlKemPublicKey::new(pk.security_level, pk.ml_kem_pk.clone())
538        .map_err(|e| HybridKemError::MlKemError(e.to_string()))?;
539    let (ml_kem_ss, ml_kem_ct_struct) = MlKem::encapsulate(&ml_kem_pk_struct)
540        .map_err(|e| HybridKemError::MlKemError(e.to_string()))?;
541    let ml_kem_ct = ml_kem_ct_struct.into_bytes();
542
543    // Validate ML-KEM shared secret
544    if ml_kem_ss.as_bytes().len() != 32 {
545        return Err(HybridKemError::MlKemError(
546            "ML-KEM encapsulation returned invalid shared secret length".to_string(),
547        ));
548    }
549
550    // Generate ephemeral ECDH keypair and perform key agreement
551    let ecdh_ephemeral =
552        X25519KeyPair::generate().map_err(|e| HybridKemError::EcdhError(e.to_string()))?;
553    let ecdh_ephemeral_public = ecdh_ephemeral.public_key_bytes().to_vec();
554
555    // Perform ECDH key agreement with peer's public key
556    let ecdh_shared_secret =
557        ecdh_ephemeral.agree(&pk.ecdh_pk).map_err(|e| HybridKemError::EcdhError(e.to_string()))?;
558
559    // Derive hybrid shared secret using HPKE-style KDF
560    let shared_secret = derive_hybrid_shared_secret(HybridSharedSecretInputs {
561        ml_kem_ss: ml_kem_ss.as_bytes(),
562        ecdh_ss: &*ecdh_shared_secret,
563        static_pk: pk.ecdh_pk.as_slice(),
564        ephemeral_pk: &ecdh_ephemeral_public,
565    })
566    .map_err(|e| HybridKemError::KdfError(e.to_string()))?;
567
568    // `shared_secret` is already `Zeroizing<Vec<u8>>` — pass directly.
569    Ok(EncapsulatedKey { ml_kem_ct, ecdh_pk: ecdh_ephemeral_public, shared_secret })
570}

572/// Decapsulate using hybrid KEM.
573///
574/// Recovers the shared secret from a hybrid ciphertext using the recipient's
575/// secret key. The X25519 component now performs **real ECDH** via aws-lc-rs
576/// `PrivateKey::agree()`.
577///
578/// # Errors
579///
580/// Returns an error if:
581/// - The ephemeral ECDH public key is not exactly 32 bytes.
582/// - ML-KEM secret key or ciphertext construction fails.
583/// - ML-KEM decapsulation fails or returns an invalid shared secret length.
584/// - ECDH key agreement fails.
585/// - Key derivation (HKDF) fails.
586pub fn decapsulate(
587    sk: &HybridKemSecretKey,
588    ct: &EncapsulatedKey,
589) -> Result<Zeroizing<Vec<u8>>, HybridKemError> {
590    // Validate ephemeral ECDH public key length
591    if ct.ecdh_pk.len() != X25519_KEY_SIZE {
592        return Err(HybridKemError::InvalidKeyMaterial(format!(
593            "Ephemeral ECDH public key must be {} bytes, got {}",
594            X25519_KEY_SIZE,
595            ct.ecdh_pk.len()
596        )));
597    }
598
599    // ML-KEM decapsulation at the secret key's security level
600    let ml_kem_ct_struct = MlKemCiphertext::new(sk.security_level(), ct.ml_kem_ct.clone())
601        .map_err(|e| HybridKemError::MlKemError(e.to_string()))?;
602    let ml_kem_ss = sk.ml_kem_decapsulate(&ml_kem_ct_struct)?;
603
604    // Validate ML-KEM shared secret
605    if ml_kem_ss.as_bytes().len() != 32 {
606        return Err(HybridKemError::MlKemError(
607            "ML-KEM decapsulation returned invalid shared secret length".to_string(),
608        ));
609    }
610
611    // Real X25519 ECDH: agree(our_static_sk, sender's_ephemeral_pk)
612    let ecdh_shared_secret = sk.ecdh_agree(&ct.ecdh_pk)?;
613
614    // Use actual public key bytes for context binding (not a hash of random bytes)
615    let static_public = sk.ecdh_public_key_bytes();
616    derive_hybrid_shared_secret(HybridSharedSecretInputs {
617        ml_kem_ss: ml_kem_ss.as_bytes(),
618        ecdh_ss: &*ecdh_shared_secret,
619        static_pk: &static_public,
620        ephemeral_pk: ct.ecdh_pk.as_slice(),
621    })
622    .map_err(|e| HybridKemError::KdfError(e.to_string()))
623}

625/// Named-field inputs to [`derive_hybrid_shared_secret`].
626///
627/// All four fields are `&[u8]` today (ML-KEM and ECDH shared secrets are
628/// both 32 bytes; public keys are fixed 32 bytes for the supported
629/// levels). Calling a 4×`&[u8]` positional function is easy to get wrong:
630/// accidentally swapping the ML-KEM and ECDH secrets would still compile
631/// but silently derive the wrong secret and break interop. This struct
632/// forces callers to name each input at the callsite, so a swap becomes
633/// a compile error on the field name.
634#[derive(Debug)]
635pub struct HybridSharedSecretInputs<'a> {
636    /// ML-KEM decapsulated shared secret (32 bytes). First leg of the IKM.
637    pub ml_kem_ss: &'a [u8],
638    /// ECDH (X25519) agreed shared secret (32 bytes). Second leg of the IKM.
639    pub ecdh_ss: &'a [u8],
640    /// Static recipient public key. Binds the derivation to the intended
641    /// peer. Included in the HKDF `info` field with a length prefix.
642    pub static_pk: &'a [u8],
643    /// Ephemeral sender public key. Binds the derivation to this specific
644    /// session. Included in the HKDF `info` field with a length prefix.
645    pub ephemeral_pk: &'a [u8],
646}
647
648/// Derive hybrid shared secret using HPKE-style KDF.
649///
650/// Combines ML-KEM and ECDH secrets using HKDF following HPKE (RFC 9180)
651/// specification with proper domain separation and context binding. See
652/// [`HybridSharedSecretInputs`] for parameter meaning — callers construct
653/// that struct with named fields so the ML-KEM/ECDH legs cannot be
654/// silently swapped at the callsite.
655///
656/// # Errors
657///
658/// Returns an error if:
659/// - The ML-KEM shared secret is not exactly 32 bytes.
660/// - The ECDH shared secret is not exactly 32 bytes.
661/// - HKDF expansion fails.
662pub fn derive_hybrid_shared_secret(
663    inputs: HybridSharedSecretInputs<'_>,
664) -> Result<Zeroizing<Vec<u8>>, HybridKemError> {
665    let HybridSharedSecretInputs { ml_kem_ss, ecdh_ss, static_pk, ephemeral_pk } = inputs;
666    if ml_kem_ss.len() != 32 {
667        return Err(HybridKemError::InvalidKeyMaterial(
668            "ML-KEM shared secret must be 32 bytes".to_string(),
669        ));
670    }
671    if ecdh_ss.len() != 32 {
672        return Err(HybridKemError::InvalidKeyMaterial(
673            "ECDH shared secret must be 32 bytes".to_string(),
674        ));
675    }
676
677    // Create input keying material following HPKE KDF approach
678    // Concatenate secrets for KDF input (zeroized on drop)
679    let mut ikm = Zeroizing::new(Vec::with_capacity(64));
680    ikm.extend_from_slice(ml_kem_ss);
681    ikm.extend_from_slice(ecdh_ss);
682
683    // Context info for domain separation and public-key binding, using
684    // length-prefixed encoding to prevent canonicalization collisions
685    // between the variable-length fields (HPKE §5.1 / RFC 9180 "LabeledExtract").
686    // Layout: [domain_label][static_pk_len: u32 BE][static_pk][ephemeral_pk_len: u32 BE][ephemeral_pk].
687    // Public keys happen to be fixed 32 bytes today, so the ambiguity is not
688    // currently exploitable, but length-prefixing keeps the construction
689    // consistent with `encrypt_hybrid::derive_encryption_key` (P5.1) and
690    // protects against any future variable-length component.
691    if static_pk.len() > u32::MAX as usize || ephemeral_pk.len() > u32::MAX as usize {
692        return Err(HybridKemError::KdfError("HKDF info component exceeds 2^32 bytes".to_string()));
693    }
694    let static_pk_len = u32::try_from(static_pk.len()).map_err(|_e| {
695        HybridKemError::KdfError("static public key exceeds 2^32 bytes".to_string())
696    })?;
697    let ephemeral_pk_len = u32::try_from(ephemeral_pk.len()).map_err(|_e| {
698        HybridKemError::KdfError("ephemeral public key exceeds 2^32 bytes".to_string())
699    })?;
700    let domain = crate::types::domains::HYBRID_KEM_SS_INFO;
701    let total = domain
702        .len()
703        .saturating_add(4)
704        .saturating_add(static_pk.len())
705        .saturating_add(4)
706        .saturating_add(ephemeral_pk.len());
707    let mut info = Vec::with_capacity(total);
708    info.extend_from_slice(domain);
709    info.extend_from_slice(&static_pk_len.to_be_bytes());
710    info.extend_from_slice(static_pk);
711    info.extend_from_slice(&ephemeral_pk_len.to_be_bytes());
712    info.extend_from_slice(ephemeral_pk);
713
714    // Use HKDF-SHA256 with domain separation (via aws-lc-rs)
715    let hkdf_result = hkdf(&ikm, None, Some(&info), 64)
716        .map_err(|e| HybridKemError::KdfError(format!("HKDF failed: {}", e)))?;
717
718    // SECURITY: wrap the derived hybrid shared secret in `Zeroizing` — otherwise
719    // `.to_vec()` on `hkdf_result.key()` copies the secret OUT of its internal
720    // `Zeroizing<Vec<u8>>` into an unprotected `Vec`.
721    Ok(Zeroizing::new(hkdf_result.key().to_vec()))
722}
