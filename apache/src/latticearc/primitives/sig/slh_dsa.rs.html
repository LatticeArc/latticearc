10//! SLH-DSA (Stateless Hash-Based Digital Signature Algorithm)
11//!
12//! This module provides SLH-DSA signatures as specified in FIPS 205.
13//! All cryptographic operations use the audited `fips205` crate.
14//!
15//! # Security Levels
16//!
17//! - **SLH-DSA-SHAKE-128s**: NIST security level 1 (quantum security ~128 bits)
18//! - **SLH-DSA-SHAKE-192s**: NIST security level 3 (quantum security ~192 bits)
19//! - **SLH-DSA-SHAKE-256s**: NIST security level 5 (quantum security ~256 bits)
20//!
21//! # Example
22//!
23//! ```rust
24//! use latticearc::primitives::sig::slh_dsa::{SlhDsaSecurityLevel, SigningKey, VerifyingKey};
25//!
26//! // Generate a key pair
27//! let (signing_key, verifying_key) = SigningKey::generate(SlhDsaSecurityLevel::Shake128s)?;
28//!
29//! // Sign a message (None = no context string)
30//! let message = b"Hello, SLH-DSA!";
31//! let signature = signing_key.sign(message, &[])?;
32//!
33//! // Verify the signature (None = no context string)
34//! let is_valid = verifying_key.verify(message, &signature, &[])?;
35//! assert!(is_valid);
36//! # Ok::<(), Box<dyn std::error::Error>>(())
37//! ```
38

53/// Errors that can occur in SLH-DSA operations
54#[derive(Error, Debug, Clone, PartialEq, Eq)]
55#[non_exhaustive]
56pub enum SlhDsaError {
57    /// Random number generation failed
58    #[error("Random number generation failed")]
59    RngError,
60
61    /// Pairwise Consistency Test (FIPS 140-3 §9.2 / IG 10.3.A) failed.
62    /// Distinct from `RngError` because PCT failure indicates a
63    /// corrupted keypair, not transient entropy depletion.
64    /// audit fix #12: a FIPS error-state monitor must be able to
65    /// distinguish "retry with fresh entropy" from "discard this
66    /// keypair, generate a new one, and investigate."
67    #[error("Pairwise Consistency Test failed (FIPS 140-3 §9.2)")]
68    PctFailed,
69
70    /// Invalid public key (malformed or corrupted)
71    #[error("Invalid public key")]
72    InvalidPublicKey,
73
74    /// Invalid secret key (malformed or corrupted)
75    #[error("Invalid secret key")]
76    InvalidSecretKey,
77
78    /// Signature verification failed
79    #[error("Signature verification failed")]
80    VerificationFailed,
81
82    // `DeserializationError` removed — declared but never
83    // returned by any production path. SLH-DSA byte-shape errors surface
84    // as `InvalidPublicKey` / `InvalidSecretKey` / `VerificationFailed`.
85    // Safe under `#[non_exhaustive]`: external matches must already have
86    // a fallback arm, and no production code ever produced this variant.
87    /// Context string too long (max 255 bytes)
88    #[error("Context string too long (max 255 bytes)")]
89    ContextTooLong,
90
91    /// Message length exceeds the configured resource limit.
92    ///
93    /// SLH-DSA signing cost scales with message length; this guard prevents
94    /// unbounded-message DoS when callers bypass the unified-api resource
95    /// checks.
96    ///
97    /// kept for ABI compatibility but no longer returned from
98    /// `sign()` — the cap-rejection now collapses to `SigningFailed` so
99    /// the sign path matches Pattern 6 opacity (verify-side already
100    /// collapsed; this completes sign-side symmetry). Will be removed
101    /// in a future major bump alongside `DeserializationError`.
102    #[deprecated(note = "sign() now returns SigningFailed for cap rejection; \
103                this variant is no longer reachable from production code.")]
104    #[error("Message exceeds signature resource limit")]
105    MessageTooLong,
106
107    /// Signing failed for an internal reason (resource cap, upstream
108    /// error, etc.). Opaque per Pattern 6 — the specific cause is
109    /// preserved via `tracing::debug!` for operator diagnostics.
110    #[error("SLH-DSA signing failed")]
111    SigningFailed,
112}

118/// SLH-DSA security levels exposed by this crate.
119///
120/// FIPS 205 defines twelve parameter sets (`{SHAKE,SHA2} × {128,192,256} ×
121/// {s,f}`). LatticeArc currently exposes only the three SHAKE-`s` variants
122/// (small signatures, slower signing); the remaining nine — SHA2 hash
123/// instantiation and the `f` (fast-signing, larger-signature) variants —
124/// are not yet wired through the unified API. The enum is
125/// `#[non_exhaustive]` so adding them later is a non-breaking change for
126/// downstream `match` statements.
127#[non_exhaustive]
128#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
129#[repr(u8)]
130pub enum SlhDsaSecurityLevel {
131    /// SLH-DSA-SHAKE-128s: NIST Level 1 (quantum security ~128 bits)
132    /// Smaller keys and signatures, slower signing
133    Shake128s = 1,
134
135    /// SLH-DSA-SHAKE-192s: NIST Level 3 (quantum security ~192 bits)
136    /// Balanced security and signing speed
137    Shake192s = 2,
138
139    /// SLH-DSA-SHAKE-256s: NIST Level 5 (quantum security ~256 bits)
140    /// Highest security, larger keys and signatures, slower signing
141    Shake256s = 3,
142}
143
144impl SlhDsaSecurityLevel {
145    /// Returns the NIST security level (1-5)
146    #[must_use]
147    pub const fn nist_level(&self) -> u8 {
148        match self {
149            SlhDsaSecurityLevel::Shake128s => 1,
150            SlhDsaSecurityLevel::Shake192s => 3,
151            SlhDsaSecurityLevel::Shake256s => 5,
152        }
153    }
154
155    /// Returns the public key size in bytes
156    #[must_use]
157    pub const fn public_key_size(&self) -> usize {
158        match self {
159            SlhDsaSecurityLevel::Shake128s => shake_128s::PK_LEN,
160            SlhDsaSecurityLevel::Shake192s => shake_192s::PK_LEN,
161            SlhDsaSecurityLevel::Shake256s => shake_256s::PK_LEN,
162        }
163    }
164
165    /// Returns the secret key size in bytes
166    #[must_use]
167    pub const fn secret_key_size(&self) -> usize {
168        match self {
169            SlhDsaSecurityLevel::Shake128s => shake_128s::SK_LEN,
170            SlhDsaSecurityLevel::Shake192s => shake_192s::SK_LEN,
171            SlhDsaSecurityLevel::Shake256s => shake_256s::SK_LEN,
172        }
173    }
174
175    /// Returns the signature size in bytes
176    #[must_use]
177    pub const fn signature_size(&self) -> usize {
178        match self {
179            SlhDsaSecurityLevel::Shake128s => shake_128s::SIG_LEN,
180            SlhDsaSecurityLevel::Shake192s => shake_192s::SIG_LEN,
181            SlhDsaSecurityLevel::Shake256s => shake_256s::SIG_LEN,
182        }
183    }
184}

190/// SLH-DSA verifying key (public key)
191///
192/// Wrapper around the audited fips205 crate's public key.
193///
194/// # Memory layout — intentional over-allocation
195///
196/// `bytes` is statically sized to `shake_256s::PK_LEN` (the maximum across
197/// all three security variants), leaving up to `shake_256s::PK_LEN -
198/// shake_128s::PK_LEN = 32` bytes unused for `Shake128s` keys (16 bytes for
199/// `Shake192s`). The trailing bytes are zero-initialized and ignored — `len`
200/// records the active prefix.
201///
202/// This trade-off was chosen deliberately:
203/// - **Stack-allocated** ([`u8`; N]) per invariant I-2 (see
204///   `docs/SECRET_TYPE_INVARIANTS.md`); no heap allocation per key.
205/// - **Non-generic** — a single concrete type that works across all three
206///   security levels without propagating `<L: SlhDsaSecurityLevel>` generics
207///   through every consumer.
208/// - **Public material** — the unused tail bytes carry no information about
209///   the actual public key (they are pre-zero), so the over-allocation has
210///   no security cost.
211///
212/// Alternative designs were considered and rejected: a `Box<[u8; N]>` per
213/// variant trades the stack waste for one heap allocation per key plus
214/// indirection; a `Vec<u8>` adds heap + capacity tracking; an enum with one
215/// variant per security level still uses `max(variant sizes)` of stack
216/// (Rust's enum sizing) so it does not actually save bytes.
217#[derive(Debug, Clone, PartialEq, Eq)]
218pub struct VerifyingKey {
219    /// The security level used
220    security_level: SlhDsaSecurityLevel,
221
222    /// The underlying public key bytes. Sized to fit the largest variant
223    /// (Shake256s = 64 bytes); the trailing `PK_LEN - len` bytes are
224    /// guaranteed zero. The `[u8; PK_LEN]` is initialized via `[0u8;
225    /// PK_LEN]` in [`Self::new`] before the variant-specific prefix is
226    /// `copy_from_slice`-d into it, so trailing bytes are never
227    /// uninitialized. `Clone` (derived) preserves all bytes verbatim,
228    /// keeping the invariant under copy. See type-level "Memory layout"
229    /// docs for rationale.
230    bytes: [u8; shake_256s::PK_LEN],
231
232    /// The actual length of the public key (per `security_level`).
233    len: usize,
234}
235
236impl VerifyingKey {
237    /// Creates a new verifying key from bytes
238    ///
239    /// # Errors
240    /// Returns `SlhDsaError::InvalidPublicKey` if the key length is
241    /// incorrect or the key is malformed.
242    ///
243    /// # Why not collapsed to `VerificationFailed`?
244    /// M6 collapsed the verify-side error variants for
245    /// Pattern-6 opacity, but kept this constructor returning
246    /// `InvalidPublicKey`. Constructors run synchronously on caller-
247    /// supplied bytes (key load, deserialize, CLI parse) — the
248    /// adversary in the verify-side threat model can't reach
249    /// `VerifyingKey::new`. Returning a structured error here helps
250    /// CLI error messages and KAT-replay scripts distinguish "wrong
251    /// length" from "valid key, bad signature." Symmetric with
252    /// `MlDsaPublicKey::new`.
253    pub fn new(security_level: SlhDsaSecurityLevel, bytes: &[u8]) -> Result<Self, SlhDsaError> {
254        let expected_len = security_level.public_key_size();
255        if bytes.len() != expected_len {
256            return Err(SlhDsaError::InvalidPublicKey);
257        }
258
259        let mut key_bytes = [0u8; shake_256s::PK_LEN];
260        key_bytes[..expected_len].copy_from_slice(bytes);
261
262        // the previous implementation called
263        // `try_from_bytes` here as eager validation, then `verify()`
264        // called it again on every signature check — two parses per
265        // PK, with the constructor's parsed result immediately
266        // discarded. The struct stores raw bytes only, so caching the
267        // parsed PK would require a typestate enum; since the parse
268        // is re-done in verify anyway, drop the eager call. Length is
269        // still validated above (`expected_len` check), and content
270        // validity is detected on first verify with the same opaque
271        // `InvalidPublicKey` error. PCT-based keygen flows
272        // (`SigningKey::generate` → `pct_slh_dsa`) still exercise
273        // both sign and verify on a fresh keypair so correctness
274        // regressions are caught at construction time.
275
276        Ok(VerifyingKey { security_level, bytes: key_bytes, len: expected_len })
277    }

308    /// Verifies a signature on a message
309    ///
310    /// # Arguments
311    ///
312    /// * `message` - The message to verify
313    /// * `signature` - The signature to verify
314    /// * `context` - Optional context string (max 255 bytes, typically empty)
315    ///
316    /// # Returns
317    ///
318    /// Returns `Ok(true)` if the signature is valid, `Ok(false)` otherwise.
319    ///
320    /// `context` is the FIPS 205 context string; pass `&[]` for domain-neutral
321    /// verification. Matches the signature-API shape of ML-DSA and FN-DSA.
322    ///
323    /// # Errors
324    /// Returns an error if the context is too long (>255 bytes) or the
325    /// key/signature is malformed.
326    #[instrument(level = "debug", skip(self, message, signature, context), fields(security_level = ?self.security_level, message_len = message.len(), signature_len = signature.len(), context_len = context.len()))]
327    pub fn verify(
328        &self,
329        message: &[u8],
330        signature: &[u8],
331        context: &[u8],
332    ) -> Result<bool, SlhDsaError> {
333        // SLH-DSA verify hashes the entire message before traversing the
334        // hyper-tree; an attacker who can submit arbitrary bytes through
335        // any verify entry point can force unbounded hashing work. The
336        // bound here mirrors the one in `sign()`.
337        // collapse into the generic
338        // verification-failure variant. `MessageTooLong` on verify
339        // leaked the configured cap to a probing attacker.
340        if let Err(e) = crate::primitives::resource_limits::validate_signature_size(message.len()) {
341            tracing::debug!(error = ?e, msg_len = message.len(), "SLH-DSA verify rejected: message exceeds resource limit");
342            return Err(SlhDsaError::VerificationFailed);
343        }
344
345        let ctx = context;
346
347        // Validate context length. Pattern-6 (matches the ML-DSA
348        // verify-side collapse): the FIPS 205 / 204 cap is structural,
349        // but a distinct `ContextTooLong` variant would let a probing
350        // attacker fingerprint the cap. Fold into the generic
351        // `VerificationFailed`.
352        if ctx.len() > 255 {
353            tracing::debug!(
354                ctx_len = ctx.len(),
355                "SLH-DSA verify rejected: context > 255 bytes (FIPS 205 §3.3)"
356            );
357            return Err(SlhDsaError::VerificationFailed);
358        }
359
360        let is_valid = match self.security_level {
361            SlhDsaSecurityLevel::Shake128s => {
362                let pk_bytes: [u8; shake_128s::PK_LEN] =
363                    self.as_bytes().try_into().map_err(|_e| SlhDsaError::VerificationFailed)?;
364                let pk = shake_128s::PublicKey::try_from_bytes(&pk_bytes)
365                    .map_err(|_e| SlhDsaError::VerificationFailed)?;
366                let sig_bytes: [u8; shake_128s::SIG_LEN] =
367                    signature.try_into().map_err(|_e| SlhDsaError::VerificationFailed)?;
368                pk.verify(message, &sig_bytes, ctx)
369            }
370            SlhDsaSecurityLevel::Shake192s => {
371                let pk_bytes: [u8; shake_192s::PK_LEN] =
372                    self.as_bytes().try_into().map_err(|_e| SlhDsaError::VerificationFailed)?;
373                let pk = shake_192s::PublicKey::try_from_bytes(&pk_bytes)
374                    .map_err(|_e| SlhDsaError::VerificationFailed)?;
375                let sig_bytes: [u8; shake_192s::SIG_LEN] =
376                    signature.try_into().map_err(|_e| SlhDsaError::VerificationFailed)?;
377                pk.verify(message, &sig_bytes, ctx)
378            }
379            SlhDsaSecurityLevel::Shake256s => {
380                let pk_bytes: [u8; shake_256s::PK_LEN] =
381                    self.as_bytes().try_into().map_err(|_e| SlhDsaError::VerificationFailed)?;
382                let pk = shake_256s::PublicKey::try_from_bytes(&pk_bytes)
383                    .map_err(|_e| SlhDsaError::VerificationFailed)?;
384                let sig_bytes: [u8; shake_256s::SIG_LEN] =
385                    signature.try_into().map_err(|_e| SlhDsaError::VerificationFailed)?;
386                pk.verify(message, &sig_bytes, ctx)
387            }
388        };
389
390        Ok(is_valid)
391    }
392}

398/// SLH-DSA signing key (secret key)
399///
400/// This is a wrapper around the audited fips205 crate's private key.
401/// Secret keys are zeroized on drop to prevent memory leaks.
402///
403/// # Security
404///
405/// - Does not implement `Clone` to prevent unzeroized copies
406/// - Implements `ConstantTimeEq` for timing-safe comparisons
407/// - Zeroized on drop via custom `Drop` implementation
408pub struct SigningKey {
409    /// The security level used
410    security_level: SlhDsaSecurityLevel,
411
412    /// The underlying secret key bytes (zeroized on drop). Sized to the
413    /// largest variant (Shake256s = 128 bytes); trailing `SK_LEN - len`
414    /// bytes are always zero. See [`VerifyingKey`]'s "Memory layout" docs
415    /// for the rationale (same trade-off applies here).
416    bytes: [u8; shake_256s::SK_LEN],
417
418    /// The actual length of the secret key (per `security_level`).
419    len: usize,
420
421    /// The verifying key (public key)
422    verifying_key: VerifyingKey,
423}
424
425impl ConstantTimeEq for SigningKey {
426    fn ct_eq(&self, other: &Self) -> Choice {
427        // Compare security level in constant time
428        let level_eq = (self.security_level as u8).ct_eq(&(other.security_level as u8));
429        // Compare length in constant time
430        let len_eq = self.len.ct_eq(&other.len);
431        // Compare bytes in constant time (only up to the actual length)
432        let bytes_eq = self.bytes[..self.len].ct_eq(&other.bytes[..other.len]);
433        level_eq & len_eq & bytes_eq
434    }
435}

437// `PartialEq`/`Eq` are intentionally NOT implemented on SLH-DSA `SigningKey`.
438// See invariants I-5/I-6 in `docs/SECRET_TYPE_INVARIANTS.md`. Use
439// `ConstantTimeEq::ct_eq` for comparisons.
440
441impl SigningKey {
442    /// Generates a new signing key with the specified security level
443    ///
444    /// Uses the audited fips205 crate's `try_keygen()` function.
445    /// After key generation, a FIPS 140-3 Pairwise Consistency Test (PCT)
446    /// is performed to verify the keypair is valid.
447    ///
448    /// # Arguments
449    ///
450    /// * `security_level` - The security level to use
451    ///
452    /// # Returns
453    ///
454    /// Returns a tuple of (signing_key, verifying_key)
455    ///
456    /// # Errors
457    ///
458    /// Returns `SlhDsaError::RngError` if random number generation fails or PCT fails
459    #[instrument(level = "debug", fields(security_level = ?security_level, nist_level = security_level.nist_level()))]
460    pub fn generate(
461        security_level: SlhDsaSecurityLevel,
462    ) -> Result<(Self, VerifyingKey), SlhDsaError> {
463        let (signing_key, verifying_key) = match security_level {
464            SlhDsaSecurityLevel::Shake128s => {
465                let (pk, sk) = shake_128s::try_keygen().map_err(|_e| SlhDsaError::RngError)?;
466                let verifying_key = VerifyingKey {
467                    security_level,
468                    bytes: {
469                        let mut b = [0u8; shake_256s::PK_LEN];
470                        let pk_bytes = pk.into_bytes();
471                        b[..pk_bytes.len()].copy_from_slice(&pk_bytes);
472                        b
473                    },
474                    len: shake_128s::PK_LEN,
475                };
476                let signing_key = SigningKey {
477                    security_level,
478                    bytes: {
479                        let mut b = [0u8; shake_256s::SK_LEN];
480                        let sk_bytes = sk.into_bytes();
481                        b[..sk_bytes.len()].copy_from_slice(&sk_bytes);
482                        b
483                    },
484                    len: shake_128s::SK_LEN,
485                    verifying_key: verifying_key.clone(),
486                };
487                (signing_key, verifying_key)
488            }
489            SlhDsaSecurityLevel::Shake192s => {
490                let (pk, sk) = shake_192s::try_keygen().map_err(|_e| SlhDsaError::RngError)?;
491                let verifying_key = VerifyingKey {
492                    security_level,
493                    bytes: {
494                        let mut b = [0u8; shake_256s::PK_LEN];
495                        let pk_bytes = pk.into_bytes();
496                        b[..pk_bytes.len()].copy_from_slice(&pk_bytes);
497                        b
498                    },
499                    len: shake_192s::PK_LEN,
500                };
501                let signing_key = SigningKey {
502                    security_level,
503                    bytes: {
504                        let mut b = [0u8; shake_256s::SK_LEN];
505                        let sk_bytes = sk.into_bytes();
506                        b[..sk_bytes.len()].copy_from_slice(&sk_bytes);
507                        b
508                    },
509                    len: shake_192s::SK_LEN,
510                    verifying_key: verifying_key.clone(),
511                };
512                (signing_key, verifying_key)
513            }
514            SlhDsaSecurityLevel::Shake256s => {
515                let (pk, sk) = shake_256s::try_keygen().map_err(|_e| SlhDsaError::RngError)?;
516                let verifying_key = VerifyingKey {
517                    security_level,
518                    bytes: {
519                        let mut b = [0u8; shake_256s::PK_LEN];
520                        let pk_bytes = pk.into_bytes();
521                        b[..pk_bytes.len()].copy_from_slice(&pk_bytes);
522                        b
523                    },
524                    len: shake_256s::PK_LEN,
525                };
526                let signing_key = SigningKey {
527                    security_level,
528                    bytes: {
529                        let mut b = [0u8; shake_256s::SK_LEN];
530                        let sk_bytes = sk.into_bytes();
531                        b[..sk_bytes.len()].copy_from_slice(&sk_bytes);
532                        b
533                    },
534                    len: shake_256s::SK_LEN,
535                    verifying_key: verifying_key.clone(),
536                };
537                (signing_key, verifying_key)
538            }
539        };
540
541        // FIPS 140-3 Pairwise Consistency Test (PCT)
542        // Sign and verify a test message to ensure the keypair is consistent.
543        // PCT failure now maps to the dedicated
544        // `PctFailed` variant. The previous `RngError` mapping let a
545        // FIPS error-state monitor mis-categorize a corrupted-keypair
546        // event as transient entropy failure and likely retry — wrong
547        // semantics for both FIPS and operational triage.
548        crate::primitives::pct::pct_slh_dsa(&verifying_key, &signing_key)
549            .map_err(|_e| SlhDsaError::PctFailed)?;
550
551        Ok((signing_key, verifying_key))
552    }
553
554    /// Creates a new signing key from bytes
555    ///
556    /// # Errors
557    /// Returns an error if the key length is incorrect or the key is malformed.
558    pub fn new(security_level: SlhDsaSecurityLevel, bytes: &[u8]) -> Result<Self, SlhDsaError> {
559        let expected_len = security_level.secret_key_size();
560        if bytes.len() != expected_len {
561            return Err(SlhDsaError::InvalidSecretKey);
562        }
563
564        let mut key_bytes = Zeroizing::new([0u8; shake_256s::SK_LEN]);
565        key_bytes[..expected_len].copy_from_slice(bytes);
566
567        match security_level {
568            SlhDsaSecurityLevel::Shake128s => {
569                if bytes.len() != shake_128s::SK_LEN {
570                    return Err(SlhDsaError::InvalidSecretKey);
571                }
572                let mut sk_bytes: Zeroizing<[u8; shake_128s::SK_LEN]> =
573                    Zeroizing::new([0u8; shake_128s::SK_LEN]);
574                sk_bytes.copy_from_slice(bytes);
575                let sk = shake_128s::PrivateKey::try_from_bytes(&sk_bytes)
576                    .map_err(|_e| SlhDsaError::InvalidSecretKey)?;
577                let pk_bytes = sk.get_public_key().into_bytes();
578                let verifying_key = VerifyingKey {
579                    security_level,
580                    bytes: {
581                        let mut b = [0u8; shake_256s::PK_LEN];
582                        b[..pk_bytes.len()].copy_from_slice(&pk_bytes);
583                        b
584                    },
585                    len: shake_128s::PK_LEN,
586                };
587                Ok(SigningKey {
588                    security_level,
589                    bytes: *key_bytes,
590                    len: expected_len,
591                    verifying_key,
592                })
593            }
594            SlhDsaSecurityLevel::Shake192s => {
595                if bytes.len() != shake_192s::SK_LEN {
596                    return Err(SlhDsaError::InvalidSecretKey);
597                }
598                let mut sk_bytes: Zeroizing<[u8; shake_192s::SK_LEN]> =
599                    Zeroizing::new([0u8; shake_192s::SK_LEN]);
600                sk_bytes.copy_from_slice(bytes);
601                let sk = shake_192s::PrivateKey::try_from_bytes(&sk_bytes)
602                    .map_err(|_e| SlhDsaError::InvalidSecretKey)?;
603                let pk_bytes = sk.get_public_key().into_bytes();
604                let verifying_key = VerifyingKey {
605                    security_level,
606                    bytes: {
607                        let mut b = [0u8; shake_256s::PK_LEN];
608                        b[..pk_bytes.len()].copy_from_slice(&pk_bytes);
609                        b
610                    },
611                    len: shake_192s::PK_LEN,
612                };
613                Ok(SigningKey {
614                    security_level,
615                    bytes: *key_bytes,
616                    len: expected_len,
617                    verifying_key,
618                })
619            }
620            SlhDsaSecurityLevel::Shake256s => {
621                if bytes.len() != shake_256s::SK_LEN {
622                    return Err(SlhDsaError::InvalidSecretKey);
623                }
624                let mut sk_bytes: Zeroizing<[u8; shake_256s::SK_LEN]> =
625                    Zeroizing::new([0u8; shake_256s::SK_LEN]);
626                sk_bytes.copy_from_slice(bytes);
627                let sk = shake_256s::PrivateKey::try_from_bytes(&sk_bytes)
628                    .map_err(|_e| SlhDsaError::InvalidSecretKey)?;
629                let pk_bytes = sk.get_public_key().into_bytes();
630                let verifying_key = VerifyingKey {
631                    security_level,
632                    bytes: {
633                        let mut b = [0u8; shake_256s::PK_LEN];
634                        b[..pk_bytes.len()].copy_from_slice(&pk_bytes);
635                        b
636                    },
637                    len: shake_256s::PK_LEN,
638                };
639                Ok(SigningKey {
640                    security_level,
641                    bytes: *key_bytes,
642                    len: expected_len,
643                    verifying_key,
644                })
645            }
646        }
647    }

689    /// Signs a message using this signing key
690    ///
691    /// Uses the audited fips205 crate's `try_sign()` function with hedging enabled
692    /// for better security against side-channel attacks.
693    ///
694    /// # Arguments
695    ///
696    /// * `message` - The message to sign
697    /// * `context` - Optional context string (max 255 bytes, typically empty)
698    ///
699    /// # Returns
700    ///
701    /// Returns the signature as a byte vector
702    ///
703    /// `context` is the FIPS 205 context string; pass `&[]` for domain-neutral
704    /// signatures. Matches the signature-API shape of ML-DSA and FN-DSA.
705    ///
706    /// # Errors
707    ///
708    /// Returns `SlhDsaError::RngError` if random number generation fails.
709    ///
710    /// Returns `SlhDsaError::SigningFailed` (Pattern-6 opaque) if:
711    /// - the message exceeds the configured resource limit,
712    /// - `context.len() > 255` bytes (FIPS 205 §3.3 cap; collapsed
713    ///   from the previous `ContextTooLong` variant),
714    /// - signing fails at the upstream `fips205` crate.
715    #[instrument(level = "debug", skip(self, message, context), fields(security_level = ?self.security_level, message_len = message.len(), context_len = context.len()))]
716    pub fn sign(&self, message: &[u8], context: &[u8]) -> Result<Vec<u8>, SlhDsaError> {
717        // DoS bound: SLH-DSA signing cost scales with message length.
718        // collapse the resource-cap rejection
719        // to the new `SigningFailed` variant. Cap probing was the same
720        // leak closed on the verify-side; this completes
721        // the sign-side symmetry. Trace captures the actual cause for
722        // operator diagnostics.
723        crate::primitives::resource_limits::validate_signature_size(message.len()).map_err(
724            |e| {
725                tracing::debug!(error = ?e, msg_len = message.len(), "SLH-DSA sign rejected: message exceeds resource limit");
726                SlhDsaError::SigningFailed
727            },
728        )?;
729
730        let ctx = context;
731
732        // Validate context length. Pattern-6: collapse to
733        // `SigningFailed` rather than the distinguishable
734        // `ContextTooLong` so the FIPS 205 cap can't be probed via
735        // error variant. The cause is in the tracing event.
736        if ctx.len() > 255 {
737            tracing::debug!(
738                ctx_len = ctx.len(),
739                "SLH-DSA sign rejected: context > 255 bytes (FIPS 205 §3.3)"
740            );
741            return Err(SlhDsaError::SigningFailed);
742        }
743
744        match self.security_level {
745            SlhDsaSecurityLevel::Shake128s => {
746                if self.expose_secret().len() != shake_128s::SK_LEN {
747                    return Err(SlhDsaError::InvalidSecretKey);
748                }
749                let mut sk_bytes: Zeroizing<[u8; shake_128s::SK_LEN]> =
750                    Zeroizing::new([0u8; shake_128s::SK_LEN]);
751                sk_bytes.copy_from_slice(self.expose_secret());
752                let sk = shake_128s::PrivateKey::try_from_bytes(&sk_bytes)
753                    .map_err(|_e| SlhDsaError::InvalidSecretKey)?;
754                let sig = sk.try_sign(message, ctx, true).map_err(|_e| SlhDsaError::RngError)?;
755                Ok(sig.as_ref().to_vec())
756            }
757            SlhDsaSecurityLevel::Shake192s => {
758                if self.expose_secret().len() != shake_192s::SK_LEN {
759                    return Err(SlhDsaError::InvalidSecretKey);
760                }
761                let mut sk_bytes: Zeroizing<[u8; shake_192s::SK_LEN]> =
762                    Zeroizing::new([0u8; shake_192s::SK_LEN]);
763                sk_bytes.copy_from_slice(self.expose_secret());
764                let sk = shake_192s::PrivateKey::try_from_bytes(&sk_bytes)
765                    .map_err(|_e| SlhDsaError::InvalidSecretKey)?;
766                let sig = sk.try_sign(message, ctx, true).map_err(|_e| SlhDsaError::RngError)?;
767                Ok(sig.as_ref().to_vec())
768            }
769            SlhDsaSecurityLevel::Shake256s => {
770                if self.expose_secret().len() != shake_256s::SK_LEN {
771                    return Err(SlhDsaError::InvalidSecretKey);
772                }
773                let mut sk_bytes: Zeroizing<[u8; shake_256s::SK_LEN]> =
774                    Zeroizing::new([0u8; shake_256s::SK_LEN]);
775                sk_bytes.copy_from_slice(self.expose_secret());
776                let sk = shake_256s::PrivateKey::try_from_bytes(&sk_bytes)
777                    .map_err(|_e| SlhDsaError::InvalidSecretKey)?;
778                let sig = sk.try_sign(message, ctx, true).map_err(|_e| SlhDsaError::RngError)?;
779                Ok(sig.as_ref().to_vec())
780            }
781        }
782    }

784    /// Signs a message and returns the verifying key for convenience
785    ///
786    /// # Errors
787    /// Returns an error if the context is too long or random number generation fails.
788    pub fn sign_with_key(
789        &self,
790        message: &[u8],
791        context: &[u8],
792    ) -> Result<(Vec<u8>, &VerifyingKey), SlhDsaError> {
793        let signature = self.sign(message, context)?;
794        Ok((signature, &self.verifying_key))
795    }
