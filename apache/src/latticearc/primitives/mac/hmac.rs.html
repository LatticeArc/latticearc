6//! HMAC (Hash-based Message Authentication Code)
7//!
8//! This module provides HMAC implementations using the audited `hmac` crate from RustCrypto.
9//!
10//! HMAC is specified in:
11//! - FIPS 198-1: The Keyed-Hash Message Authentication Code (HMAC)
12//! - NIST SP 800-107: Recommendation for Applications Using Approved Hash Algorithms
13//!
14//! The HMAC formula is:
15//! H((K ⊕ opad) || H((K ⊕ ipad) || text))
16//!
17//! Where:
18//! - H is the hash function (SHA-256 in this case)
19//! - K is the secret key (padded or hashed to match block size)
20//! - opad = 0x5c5c...5c (outer padding, repeated block_size times)
21//! - ipad = 0x3636...36 (inner padding, repeated block_size times)
22//! - || denotes concatenation
23//! - ⊕ denotes XOR
24//!
25//! The implementation follows these security requirements:
26//! - Key padding to 64-byte block size for SHA-256
27//! - Constant-time operations via the hmac crate
28//! - Proper handling of keys longer than block size (hashed first)
29//! - No custom HMAC algorithm code - all delegated to audited crate
30

35/// HMAC-SHA256 type alias using the audited hmac crate
36///
37/// This provides the standard HMAC-SHA256 implementation with:
38/// - Proper key padding (64-byte block size for SHA-256)
39/// - Constant-time operations
40/// - FIPS 198-1 compliance
41pub type HmacSha256 = Hmac<Sha256>;

61/// Compute HMAC-SHA256 for given key and data
62///
63/// This function computes the HMAC-SHA256 hash using the formula:
64/// H((K ⊕ opad) || H((K ⊕ ipad) || text))
65///
66/// # Arguments
67/// * `key` - The secret key (any size, will be padded or hashed to block size)
68/// * `data` - The message to authenticate
69///
70/// # Returns
71/// A 32-byte HMAC-SHA256 tag
72///
73/// # Security Requirements
74/// - The key must be cryptographically secure and randomly generated
75/// - Use fresh keys for each context (never reuse keys across applications)
76/// - The key must be kept secret
77/// - Minimum key length: 1 byte (recommended: 32 bytes or more)
78/// - Maximum key length: no limit (will be hashed if longer than block size)
79///
80/// # Example
81/// ```no_run
82/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
83/// use latticearc::primitives::mac::hmac::hmac_sha256;
84///
85/// let key = b"my secret key";
86/// let data = b"message to authenticate";
87///
88/// let tag = hmac_sha256(key, data)?;
89/// assert_eq!(tag.len(), 32);
90/// # Ok(())
91/// # }
92/// ```
93///
94/// # Errors
95/// Returns an error if the key is empty or has an invalid length for HMAC.
96///
97/// # NIST SP 800-107 Compliance
98/// - Uses standard HMAC formula as specified
99/// - Key padding handled by audited hmac crate
100/// - Supports keys of any length (properly hashed if > block size)
101pub fn hmac_sha256(key: &[u8], data: &[u8]) -> Result<[u8; 32]> {
102    // Validate key length (must be at least 1 byte)
103    if key.is_empty() {
104        return Err(LatticeArcError::InvalidInput("HMAC key cannot be empty".to_string()));
105    }
106
107    // Create HMAC instance - the hmac crate handles key padding
108    let mut mac = HmacSha256::new_from_slice(key)
109        .map_err(|_e| LatticeArcError::InvalidInput("Invalid HMAC key length".to_string()))?;
110    mac.update(data);
111    let result = mac.finalize();
112    let result_bytes = result.into_bytes();
113
114    // Safe: HMAC-SHA256 always produces exactly 32 bytes
115    let mut bytes = [0u8; 32];
116    if let Some(src) = result_bytes.get(..32) {
117        bytes.copy_from_slice(src);
118    }
119    Ok(bytes)
120}

122/// Verify HMAC-SHA256 tag using constant-time comparison
123///
124/// This function computes the HMAC-SHA256 tag for the given data and compares it
125/// with the provided tag in constant-time to prevent timing attacks.
126///
127/// # Security Notice
128/// Always use constant-time comparison for tag verification to prevent timing attacks.
129/// Using standard equality comparison (==) on HMAC tags is vulnerable to timing attacks.
130///
131/// # Arguments
132/// * `key` - The secret key
133/// * `data` - The message to verify
134/// * `tag` - The HMAC tag to verify against (must be 32 bytes)
135///
136/// # Returns
137/// `true` if the tag is valid, `false` otherwise
138///
139/// # Example
140/// ```no_run
141/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
142/// use latticearc::primitives::mac::hmac::{hmac_sha256, verify_hmac_sha256};
143///
144/// let key = b"my secret key";
145/// let data = b"message to authenticate";
146///
147/// let tag = hmac_sha256(key, data)?;
148/// let is_valid = verify_hmac_sha256(key, data, &tag);
149/// assert!(is_valid);
150/// # Ok(())
151/// # }
152/// ```
153#[must_use]
154pub fn verify_hmac_sha256(key: &[u8], data: &[u8], tag: &[u8]) -> bool {
155    use subtle::ConstantTimeEq;
156
157    // Always compute MAC to prevent timing side-channels
158    // Use constant-time operations for all comparisons
159    let key_valid = !key.is_empty();
160    let tag_valid: bool = tag.len().ct_eq(&32).into();
161
162    // Compute MAC regardless of validation status (timing-safe)
163    let expected_tag = hmac_sha256(key, data);
164
165    // Constant-time comparison: valid only if key valid AND tag length valid AND MAC matches
166    match expected_tag {
167        Ok(computed_tag) => {
168            let mac_matches = computed_tag.ct_eq(tag).into();
169            key_valid && tag_valid && mac_matches
170        }
171        Err(_) => false,
172    }
173}
