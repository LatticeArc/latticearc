6//! HMAC (Hash-based Message Authentication Code)
7//!
8//! HMAC-SHA256 backed by FIPS 140-3 validated `aws-lc-rs`.
9//!
10//! Standards:
11//! - RFC 2104: HMAC: Keyed-Hashing for Message Authentication
12//! - FIPS 198-1: The Keyed-Hash Message Authentication Code (HMAC)
13//! - NIST SP 800-107: Recommendation for Applications Using Approved Hash Algorithms
14//!
15//! The HMAC formula is:
16//! `H((K ⊕ opad) || H((K ⊕ ipad) || text))`
17//!
18//! All HMAC operations in the crate share a single backend (aws-lc-rs). Key
19//! padding, constant-time tag verification, and FIPS compliance are handled
20//! by the underlying library.
21
22use crate::prelude::error::{LatticeArcError, Result};
23use aws_lc_rs::hmac::{self, HMAC_SHA256};

25/// Compute HMAC-SHA256 for given key and data
26///
27/// This function computes the HMAC-SHA256 hash using the formula:
28/// H((K ⊕ opad) || H((K ⊕ ipad) || text))
29///
30/// # Arguments
31/// * `key` - The secret key (any size, will be padded or hashed to block size)
32/// * `data` - The message to authenticate
33///
34/// # Returns
35/// A 32-byte HMAC-SHA256 tag
36///
37/// # Security Requirements
38/// - The key must be cryptographically secure and randomly generated
39/// - Use fresh keys for each context (never reuse keys across applications)
40/// - The key must be kept secret
41/// - Minimum key length: 1 byte (recommended: 32 bytes or more)
42/// - Maximum key length: no limit (will be hashed if longer than block size)
43///
44/// # Example
45/// ```no_run
46/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
47/// use latticearc::primitives::mac::hmac::hmac_sha256;
48///
49/// let key = b"my secret key";
50/// let data = b"message to authenticate";
51///
52/// let tag = hmac_sha256(key, data)?;
53/// assert_eq!(tag.len(), 32);
54/// # Ok(())
55/// # }
56/// ```
57///
58/// # Errors
59/// Returns an error if the key is empty or has an invalid length for HMAC.
60///
61/// # NIST SP 800-107 Compliance
62/// - Uses standard HMAC formula as specified
63/// - Key padding handled by audited hmac crate
64/// - Supports keys of any length (properly hashed if > block size)
65pub fn hmac_sha256(key: &[u8], data: &[u8]) -> Result<[u8; 32]> {
66    // Validate key length (must be at least 1 byte). aws-lc-rs accepts any
67    // nonzero key length and handles padding/hashing per RFC 2104.
68    if key.is_empty() {
69        return Err(LatticeArcError::InvalidInput("HMAC key cannot be empty".to_string()));
70    }
71
72    let hk = hmac::Key::new(HMAC_SHA256, key);
73    let tag = hmac::sign(&hk, data);
74
75    // HMAC-SHA256 always produces exactly 32 bytes (RFC 2104).
76    // Returning all-zeros on shorter output would be a dangerous silent failure.
77    let src = tag.as_ref().get(..32).ok_or_else(|| LatticeArcError::ValidationError {
78        message: format!("HMAC-SHA256 output is {} bytes, expected 32", tag.as_ref().len()),
79    })?;
80    let mut bytes = [0u8; 32];
81    bytes.copy_from_slice(src);
82    Ok(bytes)
83}

85/// Verify HMAC-SHA256 tag using constant-time comparison
86///
87/// This function computes the HMAC-SHA256 tag for the given data and compares it
88/// with the provided tag in constant-time to prevent timing attacks.
89///
90/// # Security Notice
91/// Always use constant-time comparison for tag verification to prevent timing attacks.
92/// Using standard equality comparison (==) on HMAC tags is vulnerable to timing attacks.
93///
94/// # Arguments
95/// * `key` - The secret key
96/// * `data` - The message to verify
97/// * `tag` - The HMAC tag to verify against (must be 32 bytes)
98///
99/// # Returns
100/// `true` if the tag is valid, `false` otherwise
101///
102/// # Example
103/// ```no_run
104/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
105/// use latticearc::primitives::mac::hmac::{hmac_sha256, verify_hmac_sha256};
106///
107/// let key = b"my secret key";
108/// let data = b"message to authenticate";
109///
110/// let tag = hmac_sha256(key, data)?;
111/// let is_valid = verify_hmac_sha256(key, data, &tag);
112/// assert!(is_valid);
113/// # Ok(())
114/// # }
115/// ```
116#[must_use]
117pub fn verify_hmac_sha256(key: &[u8], data: &[u8], tag: &[u8]) -> bool {
118    use subtle::{Choice, ConstantTimeEq};
119
120    // Always compute MAC to prevent timing side-channels.
121    let key_valid = Choice::from(u8::from(!key.is_empty()));
122    let tag_len_valid = tag.len().ct_eq(&32);
123
124    let mac_matches = match hmac_sha256(key, data) {
125        Ok(computed_tag) => computed_tag.ct_eq(tag),
126        Err(_) => Choice::from(0u8),
127    };
128
129    // Bitwise AND on subtle::Choice: no short-circuit, constant-time combine.
130    bool::from(key_valid & tag_len_valid & mac_matches)
131}

133/// HMAC-SHA256 verifier that caches the `aws-lc-rs` key context.
134///
135/// Use when verifying many tags under the same key (e.g., session MAC
136/// validation, token verification in a loop). The one-shot
137/// [`verify_hmac_sha256`] allocates a fresh `hmac::Key` on every call;
138/// `HmacSha256Verifier` allocates once at construction and reuses the
139/// key context for every `.verify()` call.
140///
141/// Beyond the ergonomic win, the cached-key path also removes aws-lc-rs
142/// FFI allocator churn from the verify hot path — important for the
143/// dudect constant-time gate, whose per-sample resolution is otherwise
144/// swamped by allocator-state variance.
145///
146/// # Example
147///
148/// ```no_run
149/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
150/// use latticearc::primitives::mac::hmac::{hmac_sha256, HmacSha256Verifier};
151///
152/// let key = b"my secret key";
153/// let verifier = HmacSha256Verifier::new(key)?;
154///
155/// let tag = hmac_sha256(key, b"message 1")?;
156/// assert!(verifier.verify(b"message 1", &tag));
157/// assert!(!verifier.verify(b"message 2", &tag));
158/// # Ok(())
159/// # }
160/// ```
161pub struct HmacSha256Verifier {
162    key: hmac::Key,
163}
164
165impl HmacSha256Verifier {
166    /// Construct a verifier bound to `key`.
167    ///
168    /// # Errors
169    /// Returns an error if `key` is empty. aws-lc-rs handles padding/hashing
170    /// for keys longer than the block size.
171    pub fn new(key: &[u8]) -> Result<Self> {
172        if key.is_empty() {
173            return Err(LatticeArcError::InvalidInput("HMAC key cannot be empty".to_string()));
174        }
175        Ok(Self { key: hmac::Key::new(HMAC_SHA256, key) })
176    }
177
178    /// Verify `tag` against `HMAC-SHA256(key, data)` in constant time.
179    ///
180    /// Returns `false` (never errors) on length mismatch or MAC mismatch.
181    /// No FFI allocation on the hot path — the key context was bound at
182    /// construction.
183    #[must_use]
184    pub fn verify(&self, data: &[u8], tag: &[u8]) -> bool {
185        use subtle::{Choice, ConstantTimeEq};
186
187        let tag_len_valid = tag.len().ct_eq(&32);
188
189        let computed = hmac::sign(&self.key, data);
190        let computed_bytes = computed.as_ref();
191        // HMAC-SHA256 output is always 32 bytes by construction; the length
192        // check above guards `tag` so both sides feed subtle equal-length slices
193        // on the accept path and zero-length subtle on the reject path.
194        let mac_matches = match computed_bytes.get(..32) {
195            Some(slice) => slice.ct_eq(tag),
196            None => Choice::from(0u8),
197        };
198
199        bool::from(tag_len_valid & mac_matches)
200    }
201}

206    }
