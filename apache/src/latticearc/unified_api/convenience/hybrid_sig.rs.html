1//! Hybrid signature convenience API (ML-DSA-65 + Ed25519)
2//!
3//! This module wraps [`crate::hybrid::sig_hybrid`] to provide a high-level hybrid
4//! signature API with `SecurityMode` support. The underlying implementation uses
5//! AND-composition: both ML-DSA and Ed25519 must verify for the signature to be valid.
6//!
7//! ## Usage
8//!
9//! ```no_run
10//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
11//! use latticearc::unified_api::{
12//!     generate_hybrid_signing_keypair, sign_hybrid, verify_hybrid_signature,
13//!     SecurityMode,
14//! };
15//!
16//! let (pk, sk) = generate_hybrid_signing_keypair(SecurityMode::Unverified)?;
17//! let signature = sign_hybrid(b"message", &sk, SecurityMode::Unverified)?;
18//! let valid = verify_hybrid_signature(b"message", &signature, &pk, SecurityMode::Unverified)?;
19//! assert!(valid);
20//! # Ok(())
21//! # }
22//! ```
23

33/// Generate a hybrid signing keypair (ML-DSA-65 + Ed25519).
34///
35/// Returns a public key (for verification) and a secret key (for signing).
36/// The keypair combines post-quantum and classical algorithms so that
37/// security holds if *either* algorithm remains secure.
38///
39/// # Errors
40///
41/// Returns an error if:
42/// - The session has expired (when `mode` is `Verified`)
43/// - Key generation fails
44pub fn generate_hybrid_signing_keypair(
45    mode: SecurityMode,
46) -> Result<(HybridSigPublicKey, HybridSigSecretKey)> {
47    mode.validate()?;
48
49    sig_hybrid::generate_keypair().map_err(|e| {
50        CoreError::SignatureFailed(format!("Hybrid signing keypair generation failed: {}", e))
51    })
52}
53
54/// Sign a message using hybrid signatures (ML-DSA-65 + Ed25519).
55///
56/// Both ML-DSA and Ed25519 signatures are generated. An attacker would need
57/// to break *both* algorithms to forge a signature.
58///
59/// # Errors
60///
61/// Returns an error if:
62/// - The session has expired (when `mode` is `Verified`)
63/// - Message size exceeds resource limits
64/// - Signing fails (invalid key material or crypto error)
65pub fn sign_hybrid(
66    message: &[u8],
67    sk: &HybridSigSecretKey,
68    mode: SecurityMode,
69) -> Result<HybridSignature> {
70    mode.validate()?;
71
72    validate_signature_size(message.len())
73        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
74
75    sig_hybrid::sign(sk, message).map_err(hybrid_sig_error_to_core)
76}
77
78/// Verify a hybrid signature (ML-DSA-65 + Ed25519).
79///
80/// Both component signatures must verify for the result to be `Ok(true)`.
81///
82/// # Errors
83///
84/// Returns an error if:
85/// - The session has expired (when `mode` is `Verified`)
86/// - Message size exceeds resource limits
87/// - Key or signature material is malformed
88/// - Either component verification fails
89pub fn verify_hybrid_signature(
90    message: &[u8],
91    signature: &HybridSignature,
92    pk: &HybridSigPublicKey,
93    mode: SecurityMode,
94) -> Result<bool> {
95    mode.validate()?;
96
97    validate_signature_size(message.len())
98        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
99
100    sig_hybrid::verify(pk, message, signature).map_err(hybrid_sig_error_to_core)
101}

103/// Generate a hybrid signing keypair with configuration validation.

190/// Convert `HybridSignatureError` to `CoreError`.
191fn hybrid_sig_error_to_core(e: HybridSignatureError) -> CoreError {
192    match e {
193        HybridSignatureError::MlDsaError(msg) => {
194            CoreError::SignatureFailed(format!("Hybrid ML-DSA error: {}", msg))
195        }
196        HybridSignatureError::Ed25519Error(msg) => {
197            CoreError::SignatureFailed(format!("Hybrid Ed25519 error: {}", msg))
198        }
199        HybridSignatureError::VerificationFailed(_msg) => CoreError::VerificationFailed,
200        HybridSignatureError::InvalidKeyMaterial(msg) => {
201            CoreError::InvalidKey(format!("Hybrid key material error: {}", msg))
202        }
203        HybridSignatureError::CryptoError(msg) => {
204            CoreError::SignatureFailed(format!("Hybrid crypto error: {}", msg))
205        }
206    }
207}
