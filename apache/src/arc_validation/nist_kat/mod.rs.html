11//! NIST Known Answer Test (KAT) Framework
12//!
13//! This module provides a comprehensive framework for validating cryptographic
14//! implementations against official NIST test vectors and RFC specifications.
15//!
16//! ## Supported Standards
17//!
18//! ### NIST Standards
19//! - FIPS 203: ML-KEM (Module-Lattice-Based Key Encapsulation Mechanism)
20//! - FIPS 204: ML-DSA (Module-Lattice-Based Digital Signature Algorithm)
21//! - SP 800-38D: AES-GCM (Galois/Counter Mode)
22//! - FIPS 180-4: SHA-2 Family
23//!
24//! ### RFC Standards
25//! - RFC 5869: HKDF (HMAC-based Key Derivation Function)
26//! - RFC 4231: HMAC Test Vectors
27//! - RFC 8439: ChaCha20-Poly1305 AEAD
28//! - RFC 7748: X25519 Elliptic Curve Diffie-Hellman
29//! - RFC 8032: Ed25519 Digital Signatures
30//!
31//! ## Test Vector Format
32//!
33//! All test vectors are embedded in the source code as hex-encoded strings.
34//! This ensures:
35//! - Reproducibility across all platforms
36//! - No runtime network dependencies
37//! - Cryptographically verifiable test data
38//! - Compliance with FIPS 140-3 requirements
39

53/// Errors from NIST KAT execution
54#[derive(Debug, Error)]
55pub enum NistKatError {
56    /// Test vector validation failed
57    #[error("KAT failed: {algorithm} - {test_name}: {message}")]
58    TestFailed {
59        /// Algorithm name
60        algorithm: String,
61        /// Test name
62        test_name: String,
63        /// Failure message
64        message: String,
65    },
66
67    /// Hex decoding error
68    #[error("Hex decode error: {0}")]
69    HexError(String),
70
71    /// Implementation error
72    #[error("Implementation error: {0}")]
73    ImplementationError(String),
74
75    /// Unsupported algorithm
76    #[error("Unsupported algorithm: {0}")]
77    UnsupportedAlgorithm(String),
78}

80/// Result of running a single KAT test
81#[derive(Debug, Clone)]
82pub struct KatTestResult {
83    /// Test case identifier
84    pub test_case: String,
85    /// Algorithm being tested
86    pub algorithm: String,
87    /// Whether the test passed
88    pub passed: bool,
89    /// Error message if test failed
90    pub error_message: Option<String>,
91    /// Test execution time in microseconds
92    pub execution_time_us: u128,
93}
94
95impl KatTestResult {
96    /// Create a passed test result
97    #[must_use]
98    pub fn passed(test_case: String, algorithm: String, execution_time_us: u128) -> Self {
99        Self { test_case, algorithm, passed: true, error_message: None, execution_time_us }
100    }
101
102    /// Create a failed test result
103    #[must_use]
104    pub fn failed(
105        test_case: String,
106        algorithm: String,
107        error: String,
108        execution_time_us: u128,
109    ) -> Self {
110        Self { test_case, algorithm, passed: false, error_message: Some(error), execution_time_us }
111    }
112}

114/// Helper function to decode hex strings
115///
116/// # Errors
117///
118/// Returns `NistKatError::HexError` if the input string is not valid hex.
119pub fn decode_hex(s: &str) -> Result<Vec<u8>, NistKatError> {
120    hex::decode(s).map_err(|e| NistKatError::HexError(e.to_string()))
