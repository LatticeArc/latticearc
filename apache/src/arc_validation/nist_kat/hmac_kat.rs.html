106/// Run HMAC-SHA256 KAT
107///
108/// # Errors
109///
110/// Returns `NistKatError` if any test vector fails validation.
111pub fn run_hmac_sha256_kat() -> Result<(), NistKatError> {
112    for vector in HMAC_VECTORS {
113        let key = decode_hex(vector.key)?;
114        let message = decode_hex(vector.message)?;
115        let expected_mac = decode_hex(vector.expected_mac_sha256)?;
116
117        let mut mac = HmacSha256::new_from_slice(&key).map_err(|e| {
118            NistKatError::ImplementationError(format!("HMAC creation failed: {:?}", e))
119        })?;
120        mac.update(&message);
121        let result = mac.finalize();
122        let code_bytes = result.into_bytes();
123
124        if code_bytes.as_slice() != expected_mac.as_slice() {
125            return Err(NistKatError::TestFailed {
126                algorithm: "HMAC-SHA256".to_string(),
127                test_name: vector.test_name.to_string(),
128                message: format!(
129                    "MAC mismatch: got {}, expected {}",
130                    hex::encode(code_bytes),
131                    hex::encode(&expected_mac)
132                ),
133            });
134        }
135    }
136    Ok(())
137}

755    // --- Cross-variant consistency: same key/message produces different MACs ---
756
757    #[test]
758    fn test_hmac_variants_produce_different_macs() {
759        let vector = &HMAC_VECTORS[0];
760        let key = decode_hex(vector.key).expect("valid key hex");
761        let message = decode_hex(vector.message).expect("valid message hex");
762
763        let mut mac256 = HmacSha256::new_from_slice(&key).expect("valid key for sha256");
764        mac256.update(&message);
765        let result256 = mac256.finalize().into_bytes();
766
767        let mut mac512 = HmacSha512::new_from_slice(&key).expect("valid key for sha512");
768        mac512.update(&message);
769        let result512 = mac512.finalize().into_bytes();
770
771        // SHA-256 output is 32 bytes, SHA-512 output is 64 bytes
772        assert_ne!(result256.len(), result512.len());
773    }
