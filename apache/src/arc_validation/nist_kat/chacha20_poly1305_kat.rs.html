50/// Run ChaCha20-Poly1305 KAT
51///
52/// # Errors
53///
54/// Returns `NistKatError` if any test vector fails validation.
55pub fn run_chacha20_poly1305_kat() -> Result<(), NistKatError> {
56    for vector in CHACHA20_POLY1305_VECTORS {
57        run_chacha20_poly1305_test(vector)?;
58    }
59    Ok(())
60}

62fn run_chacha20_poly1305_test(vector: &ChaCha20Poly1305TestVector) -> Result<(), NistKatError> {
63    let key = decode_hex(vector.key)?;
64    let nonce = decode_hex(vector.nonce)?;
65    let aad = decode_hex(vector.aad)?;
66    let plaintext = decode_hex(vector.plaintext)?;
67    let expected_ciphertext = decode_hex(vector.expected_ciphertext)?;
68    let expected_tag = decode_hex(vector.expected_tag)?;
69
70    // Create cipher
71    let key_array: [u8; 32] = key
72        .try_into()
73        .map_err(|_err| NistKatError::ImplementationError("Invalid key length".to_string()))?;
74    let cipher = ChaCha20Poly1305::new(&key_array.into());
75
76    // Test encryption
77    let payload = Payload { msg: &plaintext, aad: &aad };
78
79    let ciphertext_with_tag = cipher
80        .encrypt((&nonce[..]).into(), payload)
81        .map_err(|e| NistKatError::ImplementationError(format!("Encryption failed: {:?}", e)))?;
82
83    // Verify ciphertext and tag
84    if ciphertext_with_tag.len() != expected_ciphertext.len() + expected_tag.len() {
85        return Err(NistKatError::TestFailed {
86            algorithm: "ChaCha20-Poly1305".to_string(),
87            test_name: vector.test_name.to_string(),
88            message: format!(
89                "Output length mismatch: got {}, expected {}",
90                ciphertext_with_tag.len(),
91                expected_ciphertext.len() + expected_tag.len()
92            ),
93        });
94    }
95
96    let (ct_part, tag_part) = ciphertext_with_tag.split_at(expected_ciphertext.len());
97
98    if ct_part != expected_ciphertext.as_slice() {
99        return Err(NistKatError::TestFailed {
100            algorithm: "ChaCha20-Poly1305".to_string(),
101            test_name: vector.test_name.to_string(),
102            message: format!(
103                "Ciphertext mismatch: got {}, expected {}",
104                hex::encode(ct_part),
105                hex::encode(&expected_ciphertext)
106            ),
107        });
108    }
109
110    if tag_part != expected_tag.as_slice() {
111        return Err(NistKatError::TestFailed {
112            algorithm: "ChaCha20-Poly1305".to_string(),
113            test_name: vector.test_name.to_string(),
114            message: format!(
115                "Tag mismatch: got {}, expected {}",
116                hex::encode(tag_part),
117                hex::encode(&expected_tag)
118            ),
119        });
120    }
121
122    // Test decryption
123    let payload_dec = Payload { msg: &ciphertext_with_tag, aad: &aad };
124
125    let decrypted = cipher
126        .decrypt((&nonce[..]).into(), payload_dec)
127        .map_err(|e| NistKatError::ImplementationError(format!("Decryption failed: {:?}", e)))?;
128
129    if decrypted != plaintext {
130        return Err(NistKatError::TestFailed {
131            algorithm: "ChaCha20-Poly1305".to_string(),
132            test_name: vector.test_name.to_string(),
133            message: "Decrypted plaintext mismatch".to_string(),
134        });
135    }
136
137    Ok(())
138}

286    fn test_decrypted_plaintext_mismatch_error() {
287        // This is tricky - to trigger lines 125-130, we need:
288        // 1. Encryption to succeed
289        // 2. Ciphertext and tag to match expected values
290        // 3. But decryption to produce different plaintext than expected
291        //
292        // This is actually impossible with correct crypto - if encryption produces
293        // the expected ciphertext, decryption will always produce the expected plaintext.
294        // This code path exists for defense-in-depth but is cryptographically unreachable
295        // with a correct implementation.
296        //
297        // We'll document this as intentionally uncovered - it's a safety check that
298        // can only trigger if the crypto implementation is broken.
299    }
