554/// Decode hex string to bytes, removing whitespace
555/// Returns None if the hex string is invalid
556#[must_use]
557pub fn decode_hex_vector(hex_str: &str) -> Option<Vec<u8>> {
558    let cleaned: String = hex_str.chars().filter(|c| !c.is_whitespace()).collect();
559    hex::decode(&cleaned).ok()
560}
561
562/// Validate that a hex string is well-formed
563#[must_use]
564pub fn is_valid_hex(hex_str: &str) -> bool {
565    let cleaned: String = hex_str.chars().filter(|c| !c.is_whitespace()).collect();
566    cleaned.len().is_multiple_of(2) && cleaned.chars().all(|c| c.is_ascii_hexdigit())
567}
