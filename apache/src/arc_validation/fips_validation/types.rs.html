1//! FIPS validation types and data structures
2
3use chrono::{DateTime, Utc};
4use serde::{Deserialize, Serialize};
5use std::collections::HashMap;
6
7/// FIPS validation scope
8#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
9pub enum ValidationScope {
10    /// Validate individual algorithms only
11    AlgorithmsOnly,
12    /// Validate cryptographic module interfaces
13    ModuleInterfaces,
14    /// Full module validation (algorithms + interfaces + security policy)
15    FullModule,
16}
17
18/// FIPS security levels
19#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Serialize, Deserialize)]
20pub enum FIPSLevel {
21    /// Level 1: Basic security
22    Level1,
23    /// Level 2: Tamper-evident
24    Level2,
25    /// Level 3: Tamper-resistant
26    Level3,
27    /// Level 4: Highest security (tamper-resistant + environmental failure protection)
28    Level4,
29}
30
31/// Validation issue severity
32#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
33pub enum IssueSeverity {
34    /// Critical issue preventing FIPS compliance
35    Critical,
36    /// High severity issue
37    High,
38    /// Medium severity issue
39    Medium,
40    /// Low severity issue
41    Low,
42    /// Informational note
43    Info,
44}
45
46/// Individual validation issue
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct ValidationIssue {
49    /// Issue identifier
50    pub id: String,
51    /// Issue description
52    pub description: String,
53    /// FIPS requirement reference
54    pub requirement_ref: String,
55    /// Issue severity
56    pub severity: IssueSeverity,
57    /// Affected component
58    pub affected_component: String,
59    /// Remediation guidance
60    pub remediation: String,
61    /// Evidence or test case that revealed the issue
62    pub evidence: String,
63}
64
65/// FIPS validation result
66#[derive(Debug, Clone, Serialize, Deserialize)]
67pub struct ValidationResult {
68    /// Validation run identifier
69    pub validation_id: String,
70    /// Validation timestamp
71    pub timestamp: DateTime<Utc>,
72    /// Validation scope used
73    pub scope: ValidationScope,
74    /// Overall validation status
75    pub is_valid: bool,
76    /// Achieved FIPS security level
77    pub level: Option<FIPSLevel>,
78    /// Validation issues found
79    pub issues: Vec<ValidationIssue>,
80    /// Test results for individual requirements
81    pub test_results: HashMap<String, TestResult>,
82    /// Validation metadata
83    pub metadata: HashMap<String, String>,
84}
85
86/// Individual test result
87#[derive(Debug, Clone, Serialize, Deserialize)]
88pub struct TestResult {
89    /// Test identifier
90    pub test_id: String,
91    /// Test passed
92    pub passed: bool,
93    /// Test duration in milliseconds
94    pub duration_ms: u64,
95    /// Test output or evidence
96    pub output: String,
97    /// Error message if test failed
98    pub error_message: Option<String>,
99}
100
101/// FIPS validation certificate
102#[derive(Debug, Clone, Serialize, Deserialize)]
103pub struct ValidationCertificate {
104    /// Certificate identifier
105    pub id: String,
106    /// Module name
107    pub module_name: String,
108    /// Module version
109    pub module_version: String,
110    /// FIPS security level
111    pub security_level: FIPSLevel,
112    /// Validation date
113    pub validation_date: DateTime<Utc>,
114    /// Certificate expiry date
115    pub expiry_date: DateTime<Utc>,
116    /// Validation lab identifier
117    pub lab_id: String,
118    /// Certificate details
119    pub details: HashMap<String, String>,
120}
121
122impl ValidationResult {
123    /// Check if validation passed
124    #[must_use]
125    pub fn is_valid(&self) -> bool {
126        self.is_valid
127    }
128
129    /// Get issues by severity
130    #[must_use]
131    pub fn issues_by_severity(&self, severity: IssueSeverity) -> Vec<&ValidationIssue> {
132        self.issues.iter().filter(|i| i.severity == severity).collect()
133    }
134
135    /// Get critical issues
136    #[must_use]
137    pub fn critical_issues(&self) -> Vec<&ValidationIssue> {
138        self.issues_by_severity(IssueSeverity::Critical)
139    }
140}
