40/// Test AES algorithm using FIPS 197 NIST test vectors
41pub fn test_aes_algorithm() -> Result<TestResult, LatticeArcError> {
42    let start_time = std::time::Instant::now();
43    let mut test_details = Vec::new();
44    let mut all_passed = true;
45
46    // NIST FIPS 197 AES-256 Known Answer Test Vector
47    let key: [u8; 32] = [
48        0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e,
49        0x0f, 0x10, 0x11, 0x12, 0x13, 0x14, 0x15, 0x16, 0x17, 0x18, 0x19, 0x1a, 0x1b, 0x1c, 0x1d,
50        0x1e, 0x1f,
51    ];
52
53    let plaintext = [
54        0x00, 0x11, 0x22, 0x33, 0x44, 0x55, 0x66, 0x77, 0x88, 0x99, 0xaa, 0xbb, 0xcc, 0xdd, 0xee,
55        0xff,
56    ];
57
58    test_details.push("Testing AES-256-GCM with NIST FIPS 197 test vector".to_string());
59
60    let unbound = UnboundKey::new(&AES_256_GCM, &key)
61        .map_err(|_e| LatticeArcError::EncryptionError("Failed to create AES key".to_string()))?;
62    let encrypt_key = LessSafeKey::new(unbound);
63
64    let mut nonce_bytes = [0u8; 12];
65    rand::thread_rng().fill_bytes(&mut nonce_bytes);
66    let nonce = Nonce::assume_unique_for_key(nonce_bytes);
67
68    // Encrypt
69    let mut ciphertext = plaintext.to_vec();
70    encrypt_key
71        .seal_in_place_append_tag(nonce, Aad::empty(), &mut ciphertext)
72        .map_err(|e| LatticeArcError::EncryptionError(format!("AES encryption failed: {}", e)))?;
73
74    // Decrypt
75    let unbound2 = UnboundKey::new(&AES_256_GCM, &key)
76        .map_err(|_e| LatticeArcError::EncryptionError("Failed to create AES key".to_string()))?;
77    let decrypt_key = LessSafeKey::new(unbound2);
78    let nonce2 = Nonce::assume_unique_for_key(nonce_bytes);
79
80    let decrypted = decrypt_key
81        .open_in_place(nonce2, Aad::empty(), &mut ciphertext)
82        .map_err(|e| LatticeArcError::DecryptionError(format!("AES decryption failed: {}", e)))?;
83
84    // Verify roundtrip
85    if decrypted == plaintext {
86        test_details.push("AES roundtrip test PASSED".to_string());
87    } else {
88        all_passed = false;
89        test_details.push("AES roundtrip test FAILED: decrypted != plaintext".to_string());
90    }
91
92    test_details.push(format!("Ciphertext length: {} bytes (includes auth tag)", ciphertext.len()));
93
94    // Test with empty plaintext
95    let unbound3 = UnboundKey::new(&AES_256_GCM, &key)
96        .map_err(|_e| LatticeArcError::EncryptionError("Failed to create AES key".to_string()))?;
97    let encrypt_key3 = LessSafeKey::new(unbound3);
98
99    let mut nonce_bytes2 = [0u8; 12];
100    rand::thread_rng().fill_bytes(&mut nonce_bytes2);
101    let nonce3 = Nonce::assume_unique_for_key(nonce_bytes2);
102
103    let mut empty_ciphertext = Vec::new();
104    encrypt_key3.seal_in_place_append_tag(nonce3, Aad::empty(), &mut empty_ciphertext).map_err(
105        |e| LatticeArcError::EncryptionError(format!("AES empty encryption failed: {}", e)),
106    )?;
107
108    let unbound4 = UnboundKey::new(&AES_256_GCM, &key)
109        .map_err(|_e| LatticeArcError::EncryptionError("Failed to create AES key".to_string()))?;
110    let decrypt_key4 = LessSafeKey::new(unbound4);
111    let nonce4 = Nonce::assume_unique_for_key(nonce_bytes2);
112
113    let empty_decrypted =
114        decrypt_key4.open_in_place(nonce4, Aad::empty(), &mut empty_ciphertext).map_err(|e| {
115            LatticeArcError::DecryptionError(format!("AES empty decryption failed: {}", e))
116        })?;
117
118    if empty_decrypted.is_empty() {
119        test_details.push("AES empty plaintext test PASSED".to_string());
120    } else {
121        all_passed = false;
122        test_details.push("AES empty plaintext test FAILED".to_string());
123    }
124
125    Ok(TestResult {
126        test_id: "aes_fips_197".to_string(),
127        passed: all_passed,
128        duration_ms: start_time.elapsed().as_millis() as u64,
129        output: test_details.join("\n"),
130        error_message: if all_passed {
131            None
132        } else {
133            Some("One or more AES tests failed".to_string())
134        },
135    })
136}

138/// Test SHA-3 algorithm using FIPS 202 NIST test vectors
139#[allow(clippy::unnecessary_wraps)] // Result signature for consistency with fallible test functions
140pub fn test_sha3_algorithm() -> Result<TestResult, LatticeArcError> {
141    let start_time = std::time::Instant::now();
142    let mut test_details = Vec::new();
143    let mut all_passed = true;
144
145    // NIST FIPS 202 SHA3-256 Known Answer Test Vector 1: Empty message
146    let empty_hash = Sha3_256::digest(b"");
147    let expected_empty: [u8; 32] = [
148        0xa7, 0xff, 0xc6, 0xf8, 0xbf, 0x1e, 0xd7, 0x66, 0x51, 0xc1, 0x47, 0x56, 0xa0, 0x61, 0xd6,
149        0x62, 0xf5, 0x80, 0xff, 0x4d, 0xe4, 0x3b, 0x49, 0xfa, 0x82, 0xd8, 0x0a, 0x4b, 0x80, 0xf8,
150        0x43, 0x4a,
151    ];
152
153    test_details.push("SHA3-256 Test 1: Empty message".to_string());
154    if empty_hash.as_slice() == expected_empty {
155        test_details.push("PASSED".to_string());
156    } else {
157        all_passed = false;
158        test_details.push(format!(
159            "FAILED: Got {}, Expected {}",
160            hex::encode(empty_hash),
161            hex::encode(expected_empty)
162        ));
163    }
164
165    // NIST FIPS 202 SHA3-256 Known Answer Test Vector 2: "abc"
166    let abc_hash = Sha3_256::digest(b"abc");
167    let expected_abc: [u8; 32] = [
168        0x3a, 0x98, 0x5d, 0xa7, 0x4f, 0xe2, 0x25, 0xb2, 0x04, 0x5c, 0x17, 0x2d, 0x6b, 0xd3, 0x90,
169        0xbd, 0x85, 0x5f, 0x08, 0x6e, 0x3e, 0x9d, 0x52, 0x5b, 0x46, 0xbf, 0xe2, 0x45, 0x11, 0x43,
170        0x15, 0x32,
171    ];
172
173    test_details.push("SHA3-256 Test 2: Message 'abc'".to_string());
174    if abc_hash.as_slice() == expected_abc {
175        test_details.push("PASSED".to_string());
176    } else {
177        all_passed = false;
178        test_details.push(format!(
179            "FAILED: Got {}, Expected {}",
180            hex::encode(abc_hash),
181            hex::encode(expected_abc)
182        ));
183    }
184
185    // Test 3: Longer message
186    let long_msg = b"The quick brown fox jumps over the lazy dog";
187    let long_hash = Sha3_256::digest(long_msg);
188    let expected_long: [u8; 32] = [
189        0x69, 0x07, 0x0d, 0xda, 0x01, 0x97, 0x5c, 0x8c, 0x12, 0x0c, 0x3a, 0xad, 0xa1, 0xb2, 0x82,
190        0x39, 0x4e, 0x7f, 0x03, 0x2f, 0xa9, 0xcf, 0x32, 0xf4, 0xcb, 0x22, 0x59, 0xa0, 0x89, 0x7d,
191        0xfc, 0x04,
192    ];
193
194    test_details.push("SHA3-256 Test 3: Long message".to_string());
195    if long_hash.as_slice() == expected_long {
196        test_details.push("PASSED".to_string());
197    } else {
198        all_passed = false;
199        test_details.push(format!(
200            "FAILED: Got {}, Expected {}",
201            hex::encode(long_hash),
202            hex::encode(expected_long)
203        ));
204    }
205
206    // Test 4: Deterministic property
207    let hash1 = Sha3_256::digest(b"test");
208    let hash2 = Sha3_256::digest(b"test");
209
210    test_details.push("SHA3-256 Test 4: Deterministic property".to_string());
211    if hash1.as_slice() == hash2.as_slice() {
212        test_details.push("PASSED".to_string());
213    } else {
214        all_passed = false;
215        test_details.push("FAILED: Same input produced different hashes".to_string());
216    }
217
218    // Test 5: Collision resistance
219    let hash_a = Sha3_256::digest(b"message_a");
220    let hash_b = Sha3_256::digest(b"message_b");
221
222    test_details.push("SHA3-256 Test 5: Collision resistance".to_string());
223    if hash_a.as_slice() != hash_b.as_slice() {
224        test_details.push("PASSED".to_string());
225    } else {
226        all_passed = false;
227        test_details.push("FAILED: Different inputs produced same hash".to_string());
228    }
229
230    Ok(TestResult {
231        test_id: "sha3_fips_202".to_string(),
232        passed: all_passed,
233        duration_ms: start_time.elapsed().as_millis() as u64,
234        output: test_details.join("\n"),
235        error_message: if all_passed {
236            None
237        } else {
238            Some("One or more SHA-3 tests failed".to_string())
239        },
240    })
241}

243/// Test ML-KEM (FIPS 203) algorithm using reference implementation
244pub fn test_mlkem_algorithm() -> Result<TestResult, LatticeArcError> {
245    use fips203::ml_kem_768;
246    use fips203::traits::{Decaps, Encaps, KeyGen, SerDes};
247
248    let start_time = std::time::Instant::now();
249    let mut test_details = Vec::new();
250    let mut all_passed = true;
251
252    test_details.push("ML-KEM FIPS 203 Validation using reference implementation".to_string());
253
254    // Test 1: Key generation
255    test_details.push("Test 1: ML-KEM-768 key generation".to_string());
256
257    let keygen_result = ml_kem_768::KG::try_keygen();
258    if keygen_result.is_err() {
259        test_details.push("FAILED: Key generation failed".to_string());
260        return Ok(TestResult {
261            test_id: "mlkem_fips_203".to_string(),
262            passed: false,
263            duration_ms: start_time.elapsed().as_millis() as u64,
264            output: test_details.join("\n"),
265            error_message: Some("ML-KEM key generation failed".to_string()),
266        });
267    }
268    test_details.push("PASSED".to_string());
269
270    let (ek, dk) = keygen_result
271        .map_err(|e| LatticeArcError::KeyGenerationError(format!("ML-KEM keygen failed: {}", e)))?;
272
273    // Test 2: Encapsulation/Decapsulation roundtrip
274    test_details.push("Test 2: Encapsulation/Decapsulation roundtrip".to_string());
275
276    let encaps_result = ek.try_encaps();
277    if encaps_result.is_err() {
278        all_passed = false;
279        test_details.push("FAILED: Encapsulation failed".to_string());
280    } else {
281        let (ss_enc, ct) = encaps_result.map_err(|e| {
282            LatticeArcError::EncapsulationError(format!("Encapsulation failed: {}", e))
283        })?;
284
285        let decaps_result = dk.try_decaps(&ct);
286        if decaps_result.is_err() {
287            all_passed = false;
288            test_details.push("FAILED: Decapsulation failed".to_string());
289        } else {
290            let ss_dec = decaps_result.map_err(|e| {
291                LatticeArcError::DecapsulationError(format!("Decapsulation failed: {}", e))
292            })?;
293
294            if ss_enc == ss_dec {
295                test_details.push("PASSED".to_string());
296            } else {
297                all_passed = false;
298                test_details.push("FAILED: Shared secrets don't match".to_string());
299            }
300        }
301    }
302
303    // Test 3: Shared secret uniqueness using HKDF
304    test_details.push("Test 3: Shared secret uniqueness".to_string());
305
306    let salt1 = Salt::new(HKDF_SHA256, b"salt1");
307    let prk1 = salt1.extract(b"input1");
308    let okm1 = prk1
309        .expand(&[b"info"], HkdfOutputLen(32))
310        .map_err(|_e| LatticeArcError::KeyDerivationError("HKDF expansion failed".to_string()))?;
311    let mut ss1 = [0u8; 32];
312    okm1.fill(&mut ss1)
313        .map_err(|_e| LatticeArcError::KeyDerivationError("HKDF fill failed".to_string()))?;
314
315    let salt2 = Salt::new(HKDF_SHA256, b"salt2");
316    let prk2 = salt2.extract(b"input2");
317    let okm2 = prk2
318        .expand(&[b"info"], HkdfOutputLen(32))
319        .map_err(|_e| LatticeArcError::KeyDerivationError("HKDF expansion failed".to_string()))?;
320    let mut ss2 = [0u8; 32];
321    okm2.fill(&mut ss2)
322        .map_err(|_e| LatticeArcError::KeyDerivationError("HKDF fill failed".to_string()))?;
323
324    if ss1 != ss2 {
325        test_details.push("PASSED".to_string());
326    } else {
327        all_passed = false;
328        test_details.push("FAILED: Different inputs produced same shared secret".to_string());
329    }
330
331    // Test 4: Multiple encapsulations produce different ciphertexts
332    test_details.push("Test 4: Multiple encapsulations produce different ciphertexts".to_string());
333
334    let encaps1 = ek.try_encaps();
335    let encaps2 = ek.try_encaps();
336
337    if let (Ok((_, ct1)), Ok((_, ct2))) = (encaps1, encaps2) {
338        let ct1_bytes = ct1.into_bytes();
339        let ct2_bytes = ct2.into_bytes();
340
341        if ct1_bytes != ct2_bytes {
342            test_details.push("PASSED".to_string());
343        } else {
344            all_passed = false;
345            test_details
346                .push("FAILED: Multiple encapsulations produced identical ciphertexts".to_string());
347        }
348    } else {
349        all_passed = false;
350        test_details.push("FAILED: Encapsulation failed".to_string());
351    }
352
353    Ok(TestResult {
354        test_id: "mlkem_fips_203".to_string(),
355        passed: all_passed,
356        duration_ms: start_time.elapsed().as_millis() as u64,
357        output: test_details.join("\n"),
358        error_message: if all_passed {
359            None
360        } else {
361            Some("One or more ML-KEM tests failed".to_string())
362        },
363    })
