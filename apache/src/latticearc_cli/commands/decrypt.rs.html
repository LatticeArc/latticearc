1//! Decrypt command — authenticated decryption of files.
2//!
3//! Reverses the encryption performed by the `encrypt` command. Reads the
4//! JSON-formatted encrypted file, validates the authentication tag, and
5//! recovers the original plaintext.
6//!
7//! **Integrity check:** AES-256-GCM decryption fails if the ciphertext has
8//! been modified in any way (the 16-byte authentication tag won't match).
9//! This protects against tampering and truncation attacks.
10//!
11//! **Key types accepted:**
12//! - `symmetric` — for AES-256-GCM decryption
13//! - `public` — rejected with a clear error message
14//! - `secret` — for hybrid or PQ-only decryption (auto-detected from scheme)
15
16use anyhow::{Context, Result, bail};
17use clap::Args;
18use std::path::PathBuf;
19
20use crate::keyfile::{KeyFile, KeyType};
21
22/// Arguments for the `decrypt` subcommand.
23#[derive(Args)]
24pub(crate) struct DecryptArgs {
25    /// Input file containing encrypted JSON (reads from stdin if omitted).
26    #[arg(short, long)]
27    pub input: Option<PathBuf>,
28    /// Output file for decrypted data (writes to stdout if omitted).
29    #[arg(short, long)]
30    pub output: Option<PathBuf>,
31    /// Key file path (symmetric key or hybrid secret key).
32    #[arg(short, long)]
33    pub key: PathBuf,
34    /// Overwrite the output file if it already exists. Default: false.
35    /// Round-26 audit fix (H12).
36    #[arg(long)]
37    pub force: bool,
38}
39
40/// Execute the decrypt command.
41pub(crate) fn run(args: DecryptArgs) -> Result<()> {
42    let cipher_json = read_input_string(&args.input)?;
43    let key_file = KeyFile::read_from(&args.key)?;
44
45    let encrypted =
46        latticearc::unified_api::serialization::deserialize_encrypted_output(&cipher_json)
47            .map_err(|e| anyhow::anyhow!("Failed to parse encrypted data: {e}"))?;
48
49    let plaintext = match key_file.key_type {
50        KeyType::Symmetric => decrypt_symmetric(&encrypted, &key_file)?,
51        KeyType::Secret => {
52            // Determine if this is a hybrid or PQ-only secret key
53            if encrypted.scheme().requires_pq_key() {
54                decrypt_pq_only(&encrypted, &key_file)?
55            } else {
56                decrypt_hybrid(&encrypted, &key_file)?
57            }
58        }
59        KeyType::Public => {
60            bail!("Cannot decrypt with a public key. Provide the secret key.");
61        }
62        _ => anyhow::bail!("Unsupported KeyType variant"),
63    };
64
65    write_output(&args.output, &plaintext, args.force)
66}

68fn decrypt_symmetric(
69    encrypted: &latticearc::EncryptedOutput,
70    key_file: &KeyFile,
71) -> Result<zeroize::Zeroizing<Vec<u8>>> {
72    let key_bytes = key_file.key_bytes()?;
73    if key_bytes.len() != 32 {
74        bail!("AES-256 requires a 32-byte key, got {} bytes", key_bytes.len());
75    }
76
77    latticearc::decrypt(
78        encrypted,
79        latticearc::DecryptKey::Symmetric(&key_bytes),
80        latticearc::CryptoConfig::new(),
81    )
82    // Round-26 audit fix (M19): uniform "Decryption failed: <scheme>"
83    // prefix across all three branches so scripted callers cannot
84    // distinguish symmetric vs hybrid vs PQ-only failure paths from
85    // stderr.
86    .map_err(|e| anyhow::anyhow!("Decryption failed (symmetric): {e}"))
87}

89fn decrypt_hybrid(
90    encrypted: &latticearc::EncryptedOutput,
91    key_file: &KeyFile,
92) -> Result<zeroize::Zeroizing<Vec<u8>>> {
93    // Reconstruct HybridKemSecretKey from the PortableKey.
94    // ML-KEM public key is extracted from the secret key file's metadata
95    // (stored at keygen time, following PKCS#12 bundling pattern).
96    let hybrid_sk = key_file
97        .portable_key()
98        .to_hybrid_secret_key()
99        .map_err(|e| anyhow::anyhow!("Failed to reconstruct hybrid secret key: {e}"))?;
100
101    latticearc::decrypt(
102        encrypted,
103        latticearc::DecryptKey::Hybrid(&hybrid_sk),
104        latticearc::CryptoConfig::new(),
105    )
106    .map_err(|e| anyhow::anyhow!("Decryption failed (hybrid): {e}"))
107}

109fn decrypt_pq_only(
110    encrypted: &latticearc::EncryptedOutput,
111    key_file: &KeyFile,
112) -> Result<zeroize::Zeroizing<Vec<u8>>> {
113    let level = super::common::resolve_ml_kem_level(key_file.portable_key().algorithm())
114        .map_err(|e| anyhow::anyhow!("{e}"))?;
115
116    let sk_bytes = key_file.key_bytes()?;
117
118    // PqOnlySecretKey now requires the recipient PK so the HKDF info on
119    // decrypt can bind to identity (HPKE / RFC 9180 §5.1). The PK is
120    // stored in the key file's metadata at keygen time under the same
121    // `ml_kem_pk` slot used by the hybrid path.
122    let pk_b64 =
123        key_file.portable_key().metadata().get("ml_kem_pk").and_then(|v| v.as_str()).ok_or_else(
124            || {
125                anyhow::anyhow!(
126                    "PQ-only secret key file is missing the 'ml_kem_pk' metadata field; \
127                 regenerate the keypair with the latest CLI."
128                )
129            },
130        )?;
131    let pk_bytes = base64::Engine::decode(&base64::engine::general_purpose::STANDARD, pk_b64)
132        .map_err(|e| anyhow::anyhow!("Invalid 'ml_kem_pk' base64 in metadata: {e}"))?;
133
134    let pq_sk = latticearc::PqOnlySecretKey::from_bytes(level, &sk_bytes, &pk_bytes)
135        .map_err(|e| anyhow::anyhow!("Invalid PQ-only secret key: {e}"))?;
136
137    latticearc::decrypt(
138        encrypted,
139        latticearc::DecryptKey::PqOnly(&pq_sk),
140        latticearc::CryptoConfig::new().crypto_mode(latticearc::CryptoMode::PqOnly),
141    )
142    .map_err(|e| anyhow::anyhow!("Decryption failed (pq-only): {e}"))
143}

145fn read_input_string(path: &Option<PathBuf>) -> Result<String> {
146    // Round-9 audit fix #3: route through the shared helper.
147    super::common::read_file_or_stdin_string(
148        path.as_deref(),
149        super::common::CLI_MAX_DECRYPTION_INPUT_BYTES,
150        "decrypt",
151    )
152}

154fn write_output(path: &Option<PathBuf>, data: &[u8], force: bool) -> Result<()> {
155    if let Some(p) = path {
156        // Atomic write with 0o600. `tempfile::NamedTempFile` already
157        // creates files with mode 0o600 and `persist()` preserves that;
158        // `.secret_mode()` is retained as defense-in-depth in case
159        // tempfile's default ever changes.
160        // Round-26 audit fix (H12): only overwrite when --force is
161        // passed.
162        latticearc::unified_api::atomic_write::AtomicWrite::new(data)
163            .secret_mode()
164            .overwrite_existing(force)
165            .write(p)
166            .with_context(|| {
167                format!("Failed to write {} (use --force to overwrite)", p.display())
168            })?;
169        // Path on stderr would leak through process accounting and log
170        // aggregation; route to tracing::debug! instead.
171        tracing::debug!(path = %p.display(), "decrypted data written");
172    } else {
173        // Stdout-as-default-target hazard: writing decrypted plaintext
174        // to a TTY exposes it to recorded shell sessions, screen
175        // sharing, scrollback, and any logging proxy. When the user is
176        // interactive, surface a one-line warning to stderr. Pipelines
177        // (`| tee`, file redirection) where stdout is not a TTY get no
178        // warning.
179        use std::io::IsTerminal;
180        if std::io::stdout().is_terminal() {
181            eprintln!(
182                "warning: decrypted plaintext is being written to a TTY. \
183                 Pass --output <file> to write to disk with 0600 permissions, \
184                 or pipe stdout to avoid leaving secrets in scrollback."
185            );
186        }
187        // Try to print as UTF-8, fall back to hex. The hex encoding is
188        // wrapped in `Zeroizing<String>` so a derived plaintext copy
189        // doesn't outlive the source `Zeroizing<Vec<u8>>` — otherwise
190        // the encoded string would linger on the heap until allocator
191        // reclaim.
192        if let Ok(s) = std::str::from_utf8(data) {
193            print!("{s}");
194        } else {
195            let encoded = zeroize::Zeroizing::new(hex::encode(data));
196            print!("{}", encoded.as_str());
197        }
198    }
199    Ok(())
200}
