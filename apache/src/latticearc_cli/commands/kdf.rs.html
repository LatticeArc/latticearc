1//! KDF command — derive cryptographic keys from input material.
2//!
3//! Key Derivation Functions (KDFs) stretch or transform passwords, shared
4//! secrets, or raw key material into one or more cryptographic keys of a
5//! desired length.
6//!
7//! **Supported algorithms:**
8//!
9//! | Algorithm        | Standard   | Use case                                  |
10//! |------------------|------------|-------------------------------------------|
11//! | HKDF-SHA256      | SP 800-56C | Deriving keys from high-entropy material.  |
12//! | PBKDF2-HMAC-SHA256 | SP 800-132 | Deriving keys from passwords (slow hash). |
13//!
14//! **When to use which:**
15//! - **HKDF** is for *non-password* input (DH shared secrets, random bytes).
16//!   It's fast, deterministic, and domain-separable via `--info`.
17//! - **PBKDF2** is for *passwords*. It's deliberately slow (600,000 iterations
18//!   by default) to resist brute-force attacks on weak passwords.
19//!
20//! **Output formats:** `hex` (default, lowercase) or `base64`.
21//!
22//! **HKDF output limit:** max 8,160 bytes (255 × 32 for SHA-256, per RFC 5869).
23
24use anyhow::{Context, Result, bail};
25use clap::{Args, ValueEnum};
26
27/// KDF algorithm.
28#[derive(Debug, Clone, ValueEnum)]
29pub(crate) enum KdfAlgorithm {
30    /// HKDF-SHA256 (SP 800-56C). Requires --salt and --info.
31    Hkdf,
32    /// PBKDF2 (SP 800-132). Requires --salt and --iterations. PRF
33    /// configurable via --prf (default hmac-sha256).
34    Pbkdf2,
35}
36
37/// PBKDF2 PRF choice.
38#[derive(Debug, Clone, Copy, ValueEnum)]
39pub(crate) enum PbkdfPrf {
40    /// HMAC-SHA256 (OWASP 2023 floor: 600,000 iterations).
41    HmacSha256,
42    /// HMAC-SHA512 (OWASP 2023 floor: 210,000 iterations).
43    HmacSha512,
44}
45
46impl PbkdfPrf {
47    fn to_lib(self) -> latticearc::primitives::kdf::PrfType {
48        match self {
49            Self::HmacSha256 => latticearc::primitives::kdf::PrfType::HmacSha256,
50            Self::HmacSha512 => latticearc::primitives::kdf::PrfType::HmacSha512,
51        }
52    }
53
54    /// OWASP 2023 minimum iteration count for this PRF. Delegates to
55    /// the library's canonical `Pbkdf2Params::min_iterations` so the
56    /// CLI and library stay in lockstep on the per-PRF floor.
57    fn min_iterations(self) -> u32 {
58        latticearc::primitives::kdf::Pbkdf2Params::min_iterations(self.to_lib())
59    }
60}

62/// Arguments for the `kdf` subcommand.
63#[derive(Args)]
64pub(crate) struct KdfArgs {
65    /// KDF algorithm.
66    #[arg(short, long, value_enum)]
67    pub algorithm: KdfAlgorithm,
68    /// Input key material (hex-encoded) or password (for PBKDF2).
69    ///
70    /// **SECURITY:** Passing a password as a CLI argument is visible in
71    /// `ps`, `/proc/<pid>/cmdline`, shell history, and kernel audit
72    /// logs. For PBKDF2 use cases, prefer one of:
73    /// - Read from `LATTICEARC_KDF_INPUT` env var (set then `unset`).
74    /// - Pipe the password on stdin and pass `--input-stdin`.
75    ///
76    /// `--input` is supported for HKDF (where the input is hex-encoded
77    /// non-secret material) and for backwards-compat with KAT replay
78    /// scripts.
79    #[arg(short, long, conflicts_with = "input_stdin")]
80    pub input: Option<String>,
81    /// Read the input from stdin. The whole stream is consumed (up
82    /// to a 1 MiB cap), and a single trailing `\n` or `\r\n` is
83    /// stripped — so `echo -n password` and `printf '%s' password`
84    /// both work, and a multi-line piped input is taken verbatim
85    /// (with the trailing newline removed). Use instead of `--input`
86    /// when the input is a secret (PBKDF2 password) and you don't
87    /// want it in `ps` / shell history.
88    ///
89    /// **Behaviour change in 0.8.x**: prior to this
90    /// flag read only the first line via `read_line`; it now reads
91    /// the whole stream up to 1 MiB. Multi-line piped input that
92    /// silently produced a different KDF output before now produces
93    /// the same output if the trailing-newline pattern matches.
94    #[arg(long, conflicts_with = "input")]
95    pub input_stdin: bool,
96    /// Salt (hex-encoded).
97    #[arg(short, long)]
98    pub salt: String,
99    /// Output key length in bytes.
100    #[arg(short, long, default_value = "32")]
101    pub length: usize,
102    /// Info string for HKDF (optional).
103    #[arg(long)]
104    pub info: Option<String>,
105    /// Iteration count for PBKDF2 (default 600000, OWASP 2023 minimum
106    /// for HMAC-SHA256). Values below the per-PRF OWASP floor are
107    /// rejected unless `--allow-weak-iterations` is also passed.
108    /// Per-PRF floors: HMAC-SHA256 → 600,000; HMAC-SHA512 → 210,000.
109    #[arg(long, default_value = "600000")]
110    pub iterations: u32,
111    /// PRF for PBKDF2 (default: hmac-sha256).
112    /// HMAC-SHA512 was reachable from the library API but unreachable
113    /// from the CLI before this flag.
114    #[arg(long, value_enum, default_value = "hmac-sha256")]
115    pub prf: PbkdfPrf,
116    /// Bypass the OWASP-2023 PBKDF2 iteration floor (600,000). Reserved
117    /// for KAT replay and reproducibility against legacy fixtures.
118    /// Production use is unsupported.
119    #[arg(long)]
120    pub allow_weak_iterations: bool,
121    /// Allow `--input <secret>` with `--algorithm pbkdf2`. Reserved for
122    /// KAT replay and reproducibility — production use is unsupported
123    /// because argv-passed secrets are visible to `ps`, kernel audit
124    /// logs, and shell history.
125    #[arg(long)]
126    pub allow_argv_secret: bool,
127    /// Output format: hex (default) or base64.
128    #[arg(short, long, default_value = "hex")]
129    pub format: super::hash::OutputFormat,
130}

132/// Execute the kdf command.
133pub(crate) fn run(args: KdfArgs) -> Result<()> {
134    if args.salt.is_empty() {
135        bail!(
136            "--salt must not be empty. PBKDF2 requires ≥16 bytes (NIST SP 800-132 §5.1); \
137             HKDF accepts an empty salt but a fixed application-specific salt is preferred."
138        );
139    }
140    let salt = hex::decode(&args.salt).context("Invalid hex in --salt")?;
141    // the previous version's error message
142    // claimed "PBKDF2 requires ≥16 bytes" but only `is_empty()` was
143    // checked. A 1-byte salt (`--salt aa`) silently passed. Enforce
144    // the SP 800-132 §5.1 minimum here for PBKDF2 — HKDF tolerates
145    // shorter salts but the CLI surface is shared, so apply the
146    // tighter rule.
147    if matches!(args.algorithm, KdfAlgorithm::Pbkdf2) && salt.len() < 16 {
148        bail!(
149            "--salt is {} byte(s); PBKDF2 requires ≥16 bytes (NIST SP 800-132 §5.1).",
150            salt.len()
151        );
152    }
153
154    // PBKDF2 with `--input` puts the password on argv (visible in `ps`,
155    // `/proc/<pid>/cmdline`, shell history). The `--input` flag's docstring
156    // calls this out, but argv routing is not algorithm-aware in clap, so
157    // a user can still combine `--algorithm pbkdf2 --input <password>`
158    // without the warning binding. Reject unless explicitly opted in via
159    // `--allow-argv-secret` (KAT replay only).
160    if matches!(args.algorithm, KdfAlgorithm::Pbkdf2)
161        && args.input.is_some()
162        && !args.allow_argv_secret
163    {
164        bail!(
165            "Refusing to read a PBKDF2 password from --input because the value is visible in \
166             `ps`, `/proc/<pid>/cmdline`, and shell history. Use --input-stdin (recommended) \
167             or LATTICEARC_KDF_INPUT. Pass --allow-argv-secret to bypass for KAT replay only."
168        );
169    }
170
171    let resolved_input = resolve_input(&args)?;
172
173    let derived = match args.algorithm {
174        KdfAlgorithm::Hkdf => derive_hkdf_with_input(&args, &salt, &resolved_input)?,
175        KdfAlgorithm::Pbkdf2 => derive_pbkdf2_with_input(&args, &salt, &resolved_input)?,
176    };
177
178    // Encoded output stays in `Zeroizing<String>` so password-derived key
179    // material can't linger on the heap until allocator reclaim.
180    let encoded: zeroize::Zeroizing<String> = match args.format {
181        super::hash::OutputFormat::Hex => zeroize::Zeroizing::new(hex::encode(derived.as_slice())),
182        super::hash::OutputFormat::Base64 => {
183            use base64::Engine;
184            zeroize::Zeroizing::new(
185                base64::engine::general_purpose::STANDARD.encode(derived.as_slice()),
186            )
187        }
188    };
189
190    // TTY guard mirrors `commands/decrypt.rs` —
191    // derived key bytes shouldn't land in interactive scrollback by
192    // default. `print!` (no trailing newline) matches the
193    // length-bound the user sees in scripts; the warning is on stderr
194    // so pipelines aren't polluted.
195    use std::io::IsTerminal;
196    if std::io::stdout().is_terminal() {
197        eprintln!(
198            "warning: derived KDF output is being written to a TTY. \
199             Pipe stdout to a file or downstream tool to avoid leaving \
200             key material in shell scrollback."
201        );
202    }
203    print!("{}", encoded.as_str());
204    Ok(())
205}

207/// Resolve the input string from one of:
208/// - `--input <value>` (CLI argument; visible in `ps`/history — flagged
209///   in the arg's docstring and acceptable for HKDF / KAT replay).
210/// - `--input-stdin` (read one line from stdin, no trailing newline;
211///   the recommended path for PBKDF2 passwords).
212/// - `LATTICEARC_KDF_INPUT` env var (recommended for scripted PBKDF2;
213///   caller must `unset` immediately afterwards — see SECURITY.md).
214///
215/// Exactly one source must be provided. The result is `Zeroizing` so
216/// password material is wiped at function exit.
217///
218/// # Precedence intentionally differs from `keyfile::resolve_passphrase`
219///
220/// `keyfile::resolve_passphrase` checks `LATTICEARC_PASSPHRASE` FIRST
221/// (env > prompt). Here we check `--input` first, then `--input-stdin`,
222/// then the env var. The asymmetry is intentional, not a bug:
223///
224/// - Passphrase: there is only ONE possible explicit source (the prompt
225///   — there's no `--passphrase <value>` flag, deliberately, because
226///   that would put the secret on the command line). The env var is
227///   THE escape hatch for non-interactive use, so env-first is right.
228/// - KDF input: there are THREE possible explicit sources, and CLI
229///   ergonomics demand that an explicit `--input` or `--input-stdin`
230///   take precedence over a stale env var that may have leaked in from
231///   a parent shell. Otherwise `LATTICEARC_KDF_INPUT=foo
232///   latticearc-cli kdf --input bar …` would silently use `foo`.
233fn resolve_input(args: &KdfArgs) -> Result<zeroize::Zeroizing<String>> {
234    if let Some(s) = args.input.as_deref() {
235        return Ok(zeroize::Zeroizing::new(s.to_string()));
236    }
237    if args.input_stdin {
238        use std::io::Read;
239        // Cap the read at 1 MiB. `read_line` is unbounded by default —
240        // piping `/dev/urandom` would OOM the process. Other CLI input
241        // paths route through `read_file_with_cap` /
242        // `read_stdin_with_limit`; mirror that discipline here.
243        const KDF_INPUT_MAX: u64 = 1024 * 1024;
244        let mut buf_bytes = Vec::new();
245        let mut limited = std::io::stdin().take(KDF_INPUT_MAX.saturating_add(1));
246        limited.read_to_end(&mut buf_bytes).context("Failed to read --input-stdin")?;
247        if buf_bytes.len() as u64 > KDF_INPUT_MAX {
248            bail!(
249                "--input-stdin exceeded {} bytes; refusing to derive a KDF input from an unbounded stream",
250                KDF_INPUT_MAX
251            );
252        }
253        let mut buf =
254            String::from_utf8(buf_bytes).context("--input-stdin contains invalid UTF-8")?;
255        // Strip a single trailing newline (LF or CRLF) — common when
256        // the user does `echo password | latticearc-cli kdf …`.
257        if buf.ends_with('\n') {
258            buf.pop();
259            if buf.ends_with('\r') {
260                buf.pop();
261            }
262        }
263        if buf.is_empty() {
264            bail!("--input-stdin received an empty line");
265        }
266        return Ok(zeroize::Zeroizing::new(buf));
267    }
268    if let Ok(env_val) = std::env::var("LATTICEARC_KDF_INPUT") {
269        if env_val.is_empty() {
270            bail!("LATTICEARC_KDF_INPUT is set but empty");
271        }
272        // emit a TTY warning when the env-var
273        // path is used. Env vars are visible via `/proc/<pid>/environ`
274        // for any process the user can read (typically the same UID).
275        // `keyfile.rs::resolve_passphrase` already warns for the
276        // analogous LATTICEARC_PASSPHRASE; the kdf path was missing
277        // the parallel warning.
278        // Mirrors `keyfile.rs::resolve_passphrase`: emit a tracing
279        // warning on EVERY env-var read (so audit pipelines that
280        // scrape `warn` events see every use), and additionally
281        // print to stderr on a TTY so an interactive operator gets
282        // immediate visible feedback even with no tracing
283        // subscriber wired up.
284        use std::io::IsTerminal;
285        let is_tty = std::io::stdin().is_terminal();
286        if is_tty {
287            eprintln!(
288                "warning: reading KDF input from LATTICEARC_KDF_INPUT. \
289                 Env vars are readable via /proc/<pid>/environ by processes \
290                 owned by the same user. Prefer --input-stdin for genuinely \
291                 secret input, and `unset LATTICEARC_KDF_INPUT` immediately \
292                 after this invocation."
293            );
294            tracing::warn!(
295                tty = true,
296                "LATTICEARC_KDF_INPUT used on interactive TTY (likely accidental)"
297            );
298        } else {
299            tracing::warn!(
300                tty = false,
301                "LATTICEARC_KDF_INPUT used non-interactively; ensure the \
302                 wrapping script unsets it immediately after the command \
303                 exits"
304            );
305        }
306        return Ok(zeroize::Zeroizing::new(env_val));
307    }
308    bail!(
309        "kdf requires one of: --input <value>, --input-stdin, or \
310         LATTICEARC_KDF_INPUT in the environment. For PBKDF2 passwords, \
311         prefer --input-stdin so the password is not visible in `ps` / \
312         `/proc/<pid>/cmdline` / shell history. Recommended pattern:\n\
313         \n\
314            read -rs PASS && printf '%s' \"$PASS\" | latticearc-cli kdf --input-stdin …\n\
315         \n\
316         `read -rs` reads silently into a shell variable, and `printf '%s'` \
317         pipes the value without writing it to the shell history."
318    )
319}

321/// Upper bound on KDF output length applied by the CLI to both HKDF and
322/// PBKDF2. Set to HKDF-SHA256's algorithmic maximum (255 × 32 = 8160
323/// bytes) so the same constant is correct for both algorithms — PBKDF2
324/// has no algorithmic ceiling but allowing arbitrary `--length` is a
325/// trivial self-DoS (e.g. `--length 1073741824` allocates 1 GiB and
326/// then runs 600k iterations per block).
327///
328/// previously this constant was 8192 and used
329/// only by the PBKDF2 path; HKDF independently hardcoded 8160 in its
330/// `bail!` check. The constants silently desynced. Both paths now
331/// reference this single source of truth.
332const CLI_MAX_KDF_OUTPUT_LEN: usize = 8160;
333
334fn derive_hkdf_with_input(
335    args: &KdfArgs,
336    salt: &[u8],
337    input: &str,
338) -> Result<zeroize::Zeroizing<Vec<u8>>> {
339    let ikm =
340        zeroize::Zeroizing::new(hex::decode(input).context("Invalid hex in --input for HKDF")?);
341    let info = args.info.as_deref().unwrap_or("");
342
343    if args.length == 0 || args.length > CLI_MAX_KDF_OUTPUT_LEN {
344        bail!("Output length must be 1..={CLI_MAX_KDF_OUTPUT_LEN} bytes");
345    }
346
347    let derived = latticearc::derive_key_with_info(
348        &ikm,
349        salt,
350        args.length,
351        info.as_bytes(),
352        latticearc::SecurityMode::Unverified,
353    )
354    .map_err(|e| anyhow::anyhow!("HKDF derivation failed: {e}"))?;
355    // `derive_key_with_info` now returns Zeroizing<Vec<u8>> directly.
356    Ok(derived)
357}

359// the previous hardcoded
360// `OWASP_PBKDF2_MIN_ITERATIONS = 600_000` was the floor for HMAC-SHA256
361// only, regardless of the PRF actually selected. The CLI now looks up
362// the per-PRF floor via `PbkdfPrf::min_iterations()` so HMAC-SHA512
363// callers get the correct 210,000 floor.
364
365fn derive_pbkdf2_with_input(
366    args: &KdfArgs,
367    salt: &[u8],
368    input: &str,
369) -> Result<zeroize::Zeroizing<Vec<u8>>> {
370    let password = zeroize::Zeroizing::new(input.as_bytes().to_vec());
371
372    if args.length == 0 || args.length > CLI_MAX_KDF_OUTPUT_LEN {
373        bail!(
374            "Output length must be 1..={CLI_MAX_KDF_OUTPUT_LEN} bytes (PBKDF2 has no \
375             algorithmic ceiling but the CLI caps to prevent self-DoS)."
376        );
377    }
378
379    let owasp_min = args.prf.min_iterations();
380    if args.iterations < owasp_min && !args.allow_weak_iterations {
381        bail!(
382            "PBKDF2 iteration count {iters} is below the OWASP 2023 minimum of {min} \
383             for the selected PRF ({prf:?}). Pass --allow-weak-iterations to bypass \
384             (KAT replay only — not for production).",
385            iters = args.iterations,
386            min = owasp_min,
387            prf = args.prf,
388        );
389    }
390    if args.iterations < owasp_min {
391        // emit BOTH eprintln! (so an interactive operator
392        // sees the warning even with no tracing subscriber configured)
393        // AND tracing::warn! (so audit-log pipelines that scrape `warn`
394        // events from JSON-formatted tracing output capture it). Stderr
395        // alone is lost when the CLI is invoked from a script that
396        // redirects 2>/dev/null; tracing alone is silent on a default
397        // CLI install with no subscriber. Both is the safe default.
398        eprintln!(
399            "warning: PBKDF2 iteration count {iters} is below the OWASP 2023 \
400             minimum ({min}) for {prf:?}; --allow-weak-iterations was passed.",
401            iters = args.iterations,
402            min = owasp_min,
403            prf = args.prf,
404        );
405        tracing::warn!(
406            iterations = args.iterations,
407            owasp_minimum = owasp_min,
408            prf = ?args.prf,
409            "PBKDF2 iteration count below OWASP 2023 minimum; --allow-weak-iterations bypassed"
410        );
411    }
412
413    let params = latticearc::primitives::kdf::Pbkdf2Params::with_salt(salt)
414        .iterations(args.iterations)
415        .key_length(args.length)
416        .prf(args.prf.to_lib());
417
418    // The library `pbkdf2()` rejects iteration counts below the per-PRF
419    // OWASP 2023 floor (600 k for HMAC-SHA256, 210 k for HMAC-SHA512)
420    // — that's the right policy for typical callers. The CLI's
421    // `--allow-weak-iterations` flag exists specifically for KAT
422    // replay, where lower counts are required to match published test
423    // vectors (RFC 6070 uses iterations as low as 1). When the flag
424    // is set, we route through `pbkdf2_kat` so the legitimate KAT
425    // path actually works; the operator already saw the
426    // "KAT replay only — not for production" warning above.
427    let result = if args.allow_weak_iterations && args.iterations < owasp_min {
428        latticearc::primitives::kdf::pbkdf2_kat(&password, &params)
429            .map_err(|e| anyhow::anyhow!("PBKDF2 derivation failed: {e}"))?
430    } else {
431        latticearc::primitives::kdf::pbkdf2(&password, &params)
432            .map_err(|e| anyhow::anyhow!("PBKDF2 derivation failed: {e}"))?
433    };
434
435    Ok(zeroize::Zeroizing::new(result.expose_secret().to_vec()))
436}
