1//! Verify command — check digital signatures.
2//!
3//! Supports two signature formats:
4//!
5//! **SignedData (recommended):** Produced by `sign --public-key`. Contains scheme
6//! metadata, timestamp, and public key — the library's `verify()` handles everything.
7//! No `--algorithm` or `--key` flags needed (both are embedded in the signature).
8//!
9//! **Legacy JSON:** Produced by `sign --algorithm`. Requires `--key` (public key)
10//! and optionally `--algorithm` (auto-detected from the signature file).
11
12use anyhow::{Context, Result, bail};
13use clap::{Args, ValueEnum};
14use std::path::PathBuf;
15
16use crate::keyfile::{KeyFile, KeyType};

18/// Verification algorithm (legacy mode only).
19#[derive(Debug, Clone, Copy, PartialEq, Eq, ValueEnum)]
20pub(crate) enum VerifyAlgorithm {
21    /// ML-DSA-65 (FIPS 204).
22    MlDsa65,
23    /// ML-DSA-44 (FIPS 204).
24    MlDsa44,
25    /// ML-DSA-87 (FIPS 204).
26    MlDsa87,
27    /// SLH-DSA-SHAKE-128s (FIPS 205).
28    SlhDsa,
29    /// FN-DSA-512 (FIPS 206).
30    FnDsa,
31    /// Ed25519.
32    Ed25519,
33    /// Hybrid ML-DSA-65 + Ed25519.
34    Hybrid,
35}

37/// Arguments for the `verify` subcommand.
38#[derive(Args)]
39pub(crate) struct VerifyArgs {
40    /// Verification algorithm (legacy mode). Auto-detected from signature file if omitted.
41    #[arg(short, long, value_enum)]
42    pub algorithm: Option<VerifyAlgorithm>,
43    /// Use case for configuration (optional).
44    #[arg(long, value_parser = super::common::parse_use_case,
45          value_name = "USE_CASE")]
46    pub use_case: Option<latticearc::types::types::UseCase>,
47    /// Security level override (default: high).
48    #[arg(long, value_parser = super::common::parse_security_level,
49          value_name = "LEVEL")]
50    pub security_level: Option<latticearc::types::types::SecurityLevel>,
51    /// Compliance mode (default, fips, cnsa-2.0).
52    #[arg(long, value_parser = super::common::parse_compliance,
53          value_name = "MODE")]
54    pub compliance: Option<latticearc::types::types::ComplianceMode>,
55    /// Input file that was signed (reads from stdin if omitted).
56    ///
57    /// Symmetric with `sign --input <stdin>`: a CI pipeline that
58    /// hashes a stream and signs it should be able to verify the
59    /// same way without staging to a temp file.
60    #[arg(short, long)]
61    pub input: Option<PathBuf>,
62    /// Signature file (SignedData JSON or legacy JSON).
63    #[arg(short, long)]
64    pub signature: PathBuf,
65    /// Public key file. Required for legacy format. Optional for
66    /// SignedData — but **WARNING**: omitting `--key` for a SignedData
67    /// envelope trusts the public key embedded in the envelope
68    /// itself, which an attacker who delivered the envelope chose.
69    /// Pass `--key` to enforce a specific trust anchor; the verifier
70    /// will reject the signature if the operator's `--key` does not
71    /// match the embedded public key.
72    #[arg(short, long)]
73    pub key: Option<PathBuf>,
74}

76/// Execute the verify command.
77///
78/// Returns `Ok(true)` for VALID, `Ok(false)` for INVALID (forgery /
79/// tampering), or `Err(_)` for operational errors (missing file, bad
80/// key, etc.). `main` translates the boolean into the documented
81/// 0/1/≥2 exit-code contract via `std::process::exit(1)` AFTER all
82/// destructors run — exiting from inside `verify::run` would skip
83/// Drop on local `KeyFile`/`Vec<u8>` state, which is currently benign
84/// (PK is non-secret) but would silently regress if the same pattern
85/// were copied to `sign` or `decrypt`.
86pub(crate) fn run(args: VerifyArgs) -> Result<bool> {
87    let input_data = read_verify_input(&args)?;
88    // open-once with `take(limit + 1)` instead of stat-then-read.
89    // A `metadata().len()` + `read_to_string` sequence is TOCTOU: a
90    // path swap between the size check and the read can let a
91    // multi-gig file slip past the cap. `metadata().len()` also
92    // returns 0 for `/dev/zero` / FIFOs, so a subsequent read of an
93    // unbounded special file would have no upper bound.
94    let sig_bytes = super::common::read_file_with_cap(
95        &args.signature,
96        super::common::CLI_MAX_SIGNATURE_INPUT_BYTES,
97        "verify",
98    )?;
99    let sig_json = String::from_utf8(sig_bytes)
100        .with_context(|| format!("{} is not valid UTF-8", args.signature.display()))?;
101
102    // Detect whether the file is in the SignedData envelope shape
103    // *before* committing to a verifier. The shape check is a
104    // structural pre-pass: if the JSON has the unified-API SignedData
105    // fields (`metadata.signature`, `metadata.signature_algorithm`,
106    // `scheme`), require `deserialize_signed_data` to succeed; any
107    // deserialization failure collapses to `print_invalid()`. Without
108    // this gate, an `if let Ok(...)` shape silently falls through to
109    // the legacy auto-detect branch and emits a distinguishable
110    // "Signature file missing 'algorithm' field" — letting an
111    // attacker tell apart "I tampered with metadata.signature_algorithm"
112    // from "wrong signature" by stderr text (Pattern-6).
113    let looks_like_signed_data = serde_json::from_str::<serde_json::Value>(&sig_json)
114        .ok()
115        .and_then(|v| {
116            v.get("metadata")?.get("signature")?;
117            v.get("metadata")?.get("signature_algorithm")?;
118            v.get("scheme")?;
119            Some(())
120        })
121        .is_some();
122    if looks_like_signed_data {
123        // Deserialize INSIDE the structural gate (not before it), so a
124        // hoist or reorder can't silently fall through to the legacy
125        // auto-detect branch and reopen the Pattern-6 distinguisher
126        // (legacy branch emitted "Signature file missing 'algorithm'
127        // field" — distinguishable from a crypto-reject error).
128        let signed =
129            match latticearc::unified_api::serialization::deserialize_signed_data(&sig_json) {
130                Ok(signed) => signed,
131                Err(e) => {
132                    // `%e` Display, not `?e` Debug — Display shows the
133                    // top-level deserialize error message (typed,
134                    // bounded), while Debug walks the source chain
135                    // which can contain attacker-controlled JSON
136                    // snippets via the underlying `serde_json::Error`.
137                    tracing::debug!(
138                        error = %e,
139                        "verify (SignedData shape detected, parse rejected)"
140                    );
141                    return Ok(print_invalid());
142                }
143            };
144        // Verify the signed data matches the input file. `signed.data`
145        // currently carries public message material, so the comparison
146        // doesn't need to be CT in the threat-model sense. A `ct_eq`
147        // wrapper here would be cargo-cult: a length-equality branch
148        // before it would make the CT comparison dead anyway. If a
149        // future caller routes secret bytes through this path, the
150        // responsibility is to handle CT at THAT site, not to pretend
151        // this site is CT when it can't be.
152        //
153        // A mismatch is attacker-controllable (the SignedData envelope
154        // is loaded from a file the attacker may have written) and so
155        // must collapse to `print_invalid()` — emitting a
156        // distinguishable "Signature was created over different data"
157        // error would let an attacker tell apart "I tampered with
158        // signed.data" from "I tampered with the signature bytes" by
159        // branching on stderr text (Pattern-6). Debug-log the original
160        // detail so operators investigating a legitimate user mistake
161        // (signed the wrong file) can recover the context with
162        // `RUST_LOG=debug`.
163        if signed.data != input_data {
164            tracing::debug!(
165                input = %input_label(&args),
166                "verify (SignedData path) rejected: embedded data does not match input"
167            );
168            return Ok(print_invalid());
169        }
170
171        // SECURITY: if the operator passed `--key`, ENFORCE that the
172        // embedded `signed.metadata.public_key` matches. Without this
173        // check, a SignedData envelope crafted with an attacker's own
174        // (key, sig, data) triple verifies against ANY operator-
175        // trusted `--key` because `latticearc::verify` uses the
176        // embedded public key. The user-facing contract "verify this
177        // signature against my trusted key" silently becomes "verify
178        // this signature against the key it carries".
179        //
180        // Mismatch is attacker-controllable and collapses to
181        // `print_invalid()` for Pattern-6 indistinguishability with
182        // crypto reject. Plain `==` is sufficient: both sides are
183        // public keys, not secret material, and the operator's key is
184        // fixed across the call so per-invocation timing cannot be
185        // amplified across runs.
186        //
187        // `--key` for SignedData remains optional: omitting it falls
188        // through to the embedded-key trust shape, appropriate when
189        // the operator has not asserted a specific trust anchor.
190        if let Some(key_path) = args.key.as_ref() {
191            let pk_bytes = load_operator_public_key(key_path)?;
192            if *pk_bytes != signed.metadata.public_key {
193                tracing::debug!(
194                    "verify (SignedData path) rejected: --key bytes do not match embedded public_key"
195                );
196                return Ok(print_invalid());
197            }
198        }
199
200        // Operator-side `--algorithm` cross-check: if the operator
201        // asserted an algorithm via `--algorithm`, the envelope's
202        // scheme MUST match. This mirrors `verify_legacy` Phase 1b's
203        // operator-picked algorithm vs key cross-check, but on the
204        // SignedData path the comparison is between the operator's
205        // assertion and the envelope's declared scheme.
206        //
207        // Strip the optional `pq-` prefix on the envelope side: the
208        // unified sign API emits both forms ("ml-dsa-65" and
209        // "pq-ml-dsa-65") for the same scheme depending on policy,
210        // but `algorithm_name` canonicalises to the un-prefixed form.
211        //
212        // SECURITY: the envelope's `signature_algorithm` field is
213        // attacker-controllable (the envelope is loaded from a file
214        // the attacker may have written). The bail message therefore
215        // does NOT interpolate it — an attacker writing
216        // `signature_algorithm: "X".repeat(1_000_000)` would otherwise
217        // produce a 1 MiB stderr line on operator misuse. Operators
218        // who need the envelope-side detail can recover it via
219        // `RUST_LOG=debug`. Mismatch bubbles as helpful `Err` (exit
220        // ≥2) — operator misuse, not an attacker action.
221        if let Some(cli_alg) = args.algorithm {
222            let envelope_alg = signed
223                .metadata
224                .signature_algorithm
225                .strip_prefix("pq-")
226                .unwrap_or(&signed.metadata.signature_algorithm);
227            let operator_alg = algorithm_name(cli_alg);
228            if envelope_alg != operator_alg {
229                tracing::debug!(
230                    envelope_scheme = %signed.metadata.signature_algorithm,
231                    operator_assertion = operator_alg,
232                    "verify (SignedData) rejected: --algorithm vs envelope scheme mismatch"
233                );
234                bail!(
235                    "Algorithm mismatch: --algorithm asserts '{operator_alg}' but the \
236                     SignedData envelope declares a different scheme. Either omit \
237                     --algorithm to verify against the embedded scheme, or pass \
238                     --algorithm matching the envelope. Set RUST_LOG=debug for the \
239                     envelope's declared scheme."
240                );
241            }
242        }
243
244        let config =
245            super::common::build_config(args.use_case, args.security_level, &args.compliance);
246        // A `latticearc::verify(...)` Err (e.g. malformed public key)
247        // collapses to the same user-visible outcome as `Ok(false)` so
248        // an attacker can't distinguish the structural / crypto /
249        // library-error paths via stderr text or exit-code class. See
250        // `print_invalid()` for the centralised reject contract.
251        let valid = match latticearc::verify(&signed, config) {
252            Ok(v) => v,
253            Err(e) => {
254                // `%e` Display — see deserialize_signed_data branch above
255                // for the rationale on attacker-content avoidance.
256                tracing::debug!(error = %e, "verify (SignedData path) rejected");
257                return Ok(print_invalid());
258            }
259        };
260
261        // return the result; `main` does the exit-code translation
262        // after destructors so the 0/1/≥2 contract is preserved at
263        // the boundary, not here.
264        //
265        // Stream choice: VALID on stdout, INVALID on stderr. The
266        // asymmetry is intentional — the machine-readable verdict is
267        // the exit code (Ok(true)/false), which IS symmetric. Verdict
268        // text on either stream is operator status only. Tests that
269        // need stream-swap protection should use `run_ok_combined`.
270        if valid {
271            println!("Signature is VALID. (scheme: {})", signed.scheme);
272            return Ok(true);
273        }
274        return Ok(print_invalid());
275    }
276
277    // Fall back to legacy format (requires --key)
278    let key_path = args.key.as_ref().ok_or_else(|| {
279        anyhow::anyhow!(
280            "Public key file (--key) is required for legacy signature format.\n\
281             Signatures created with 'sign --public-key' embed the public key \
282             and don't need --key."
283        )
284    })?;
285
286    verify_legacy(&args, &sig_json, key_path, input_data)
287}

289/// Centralised reject for any path that must be indistinguishable
290/// from a cryptographic-verification failure: emits the same INVALID
291/// stderr line and returns `false` (which `run`'s callers translate
292/// to exit code 1, the "invalid" class — distinct from the ≥2 "error"
293/// class).
294///
295/// # SECURITY: indistinguishability contract
296///
297/// Every reject path triggered by **attacker-controllable** signature-
298/// file content MUST funnel through `print_invalid()` (or, in
299/// `verify_legacy`, through the closure that maps `Err(_)` →
300/// `print_invalid()` via `unwrap_or_else`). The distinguishers this
301/// contract closes are concrete and have been observed in audit:
302///
303/// 1. **Stderr text** — a stricter message ("Invalid base64 in
304///    signature", "Unknown algorithm", "Verification failed",
305///    "Signature was created over different data than the input",
306///    "Hybrid verification failed: <upstream detail>") lets an
307///    attacker probe which validation step rejected by reading the
308///    process's stderr stream.
309/// 2. **Exit-code class** — bubbling an `Err(_)` out of `run` lands
310///    at exit ≥2 ("operational error"), distinct from the `Ok(false)`
311///    → exit 1 ("invalid signature") class. An attacker can
312///    distinguish the two with `[ $? -eq 1 ]` vs `[ $? -ge 2 ]`.
313/// 3. **Echo amplification** — interpolating an attacker-controlled
314///    field (`{other}` from the `algorithm` JSON value, `{e}` from a
315///    structured upstream error) into the error string both reveals
316///    the field's content AND amplifies its length (attacker writes
317///    1 MiB → log line is 1 MiB).
318///
319/// The "operator-side" errors in `verify_legacy` (missing `--key`,
320/// wrong key type, encrypted key without passphrase, operator-picked
321/// `--algorithm` mismatching the key) are NOT reachable by an
322/// attacker who can only tamper with the signature file, so they keep
323/// helpful error messages and surface as exit ≥2. See `verify_legacy`
324/// docs for the boundary.
325///
326/// # Tracing observability boundary (workspace-wide)
327///
328/// `tracing::debug!` calls used as the safe alternative below ARE
329/// observable on stderr when the operator (or a CI runner) sets
330/// `RUST_LOG=debug`. The default subscriber's filter ships at
331/// `INFO`/`WARN`, so production deployments don't expose this surface
332/// — but the contract isn't "this never reaches stderr", it's "this
333/// never reaches stderr at default log levels". Treat that as a
334/// reasonable trade-off: operator-side debugging is the legitimate
335/// reason debug logs exist, and an attacker who can flip the
336/// operator's `RUST_LOG` already has more powerful capabilities than
337/// reading verify error text.
338///
339/// **What's safe to put inside `tracing::debug!` calls:**
340/// - Errors via `error = %e` (Display) — fine. anyhow's Display
341///   shows ONLY the outermost `.context(...)` message, which is
342///   static text we author ourselves ("Failed to parse signature
343///   JSON", "Invalid base64 in signature", etc.). The attacker-
344///   controlled detail lives in the source chain, which Display
345///   does not walk.
346/// - The chosen algorithm name — fine; not secret, comes from a
347///   small fixed set, useful for debugging mismatches.
348/// - A short reject reason string (e.g. "alg mismatch", "data
349///   mismatch") — fine.
350///
351/// **What's NOT safe inside `tracing::debug!` calls:**
352/// - Raw signature bytes, key bytes, or any base64 thereof — even at
353///   `RUST_LOG=debug`, plaintext key material in logs is a separate
354///   incident class.
355/// - `error = ?e` (Debug) where `e` is `anyhow::Error` or any error
356///   whose source chain reaches `serde_json::Error` /
357///   `base64::DecodeError`. Debug walks the full source chain, and
358///   those underlying errors carry attacker-controlled JSON snippets
359///   and byte offsets / values from the input. Use `%e` (Display)
360///   instead — Display shows only the outermost message, which is
361///   static text we author. This contract applies workspace-wide to
362///   every reject-collapse `tracing::debug!` call that handles an
363///   error potentially wrapping attacker-controlled parser output.
364/// - Attacker-controlled fields verbatim (the `algorithm` JSON
365///   value, arbitrary error strings from the parsed signature).
366///
367/// **Narrow exception — `?e` IS safe when:**
368/// - `e` has a concrete typed `Debug` impl whose source chain does
369///   NOT reach `anyhow::Error` / `serde_json::Error` /
370///   `base64::DecodeError`. Closure parameters typed `&dyn
371///   std::fmt::Debug` whose producers are bounded internal crate
372///   enums fall into this category. The ONLY currently-sanctioned
373///   site is the `log_reject` closure inside
374///   `latticearc::primitives::sig::ml_dsa::SigningKey::sign_message`
375///   (search for "tracing-observability contract" in that file).
376///   New uses of this exception MUST cite the rationale at the
377///   call site AND be added to this allowlist; do not extend the
378///   exception by analogy to similar-looking sites.
379///
380/// # When adding a new reject path
381///
382/// If the path can be triggered by the content of `--signature` (or,
383/// for SignedData, by the embedded `signed.data` / `signed.metadata`),
384/// it MUST route through `print_invalid()`; preserve any operator-
385/// facing detail in `tracing::debug` instead of `eprintln!` / `bail!`
386/// / `anyhow!` — and follow the safe-content rules above.
387fn print_invalid() -> bool {
388    eprintln!("Signature is INVALID.");
389    false
390}

410/// Load the operator's `--key` file and require it to be a public key.
411///
412/// **Scope: this helper does ONLY the read + key-type check.** It
413/// does NOT validate the key's algorithm against any specific
414/// operator-asserted algorithm — that cross-check is the caller's
415/// responsibility (see `verify_legacy` Phase 1b for the canonical
416/// pattern, and the SignedData `--key` byte-equality block in `run`
417/// for the alternative shape). A new caller that copies this helper
418/// without also implementing the algorithm check creates an
419/// operator-intent vs envelope-content mismatch — the same class of
420/// signature-verification bypass the SignedData `--key` enforcement
421/// closes.
422///
423/// The `canonical_name()` form of the key-type rejection is used
424/// instead of `{:?}` Debug — the Debug repr of a `#[non_exhaustive]`
425/// enum is variant-fingerprintable across versions.
426fn read_operator_public_key_file(key_path: &std::path::Path) -> Result<KeyFile> {
427    let key_file = KeyFile::read_from(key_path)?;
428    if key_file.key_type != KeyType::Public {
429        bail!(
430            "Expected public key file for verification, got {}",
431            key_file.key_type.canonical_name()
432        );
433    }
434    Ok(key_file)
435}

437/// Load the operator's `--key` file and decode the raw public-key
438/// bytes. Combines `read_operator_public_key_file` + `key_bytes()`
439/// for callers that don't need the `KeyFile` itself.
440fn load_operator_public_key(key_path: &std::path::Path) -> Result<zeroize::Zeroizing<Vec<u8>>> {
441    read_operator_public_key_file(key_path)?.key_bytes()
442}

444/// The six non-hybrid `VerifyAlgorithm` variants, lifted into their
445/// own enum so `verify_standard` can be type-enforced as never
446/// receiving `Hybrid` — without this, an unreachable `bail!("Internal
447/// error...")` arm sits in `verify_standard` waiting for a future
448/// variant addition to silently make it reachable.
449#[derive(Debug, Clone, Copy)]
450enum NonHybridAlgorithm {
451    MlDsa44,
452    MlDsa65,
453    MlDsa87,
454    SlhDsa,
455    FnDsa,
456    Ed25519,
457}
458
459/// Dispatch shape for the verify-side step in `verify_legacy`.
460///
461/// The variant is chosen in Phase 1c (operator-side parsing) so that
462/// hybrid pk-parse failures — which indicate a malformed operator key
463/// file, NOT a tampered signature — surface as helpful `Err` at exit
464/// ≥2 rather than collapsing into `print_invalid()`. By the time the
465/// Phase-2 closure dispatches, the type system guarantees the right
466/// pk shape for the chosen algorithm: `Standard` cannot carry
467/// `Hybrid` (rejected by `NonHybridAlgorithm`) and `Hybrid` carries
468/// the parsed pk directly.
469///
470/// # Field-name asymmetry: `pk_bytes` vs `pk`
471///
472/// The two variants intentionally name their pk field differently
473/// because they hold different things at different stages of parsing:
474/// `Standard` carries raw bytes that the per-algorithm crypto helper
475/// will pass straight to the verify primitive (no operator-side parse
476/// needed); `Hybrid` carries the already-parsed `HybridSigPublicKey`
477/// because the hybrid wire format requires an operator-trusted
478/// byte-layout parse that lives in Phase 1c — failure there must
479/// bubble as helpful `Err`, not collapse to INVALID.
480enum LegacyVerifier<'a> {
481    Standard { algorithm: NonHybridAlgorithm, pk_bytes: &'a [u8] },
482    Hybrid { pk: latticearc::hybrid::sig_hybrid::HybridSigPublicKey },
483}
484
485/// Project a `VerifyAlgorithm` onto its non-hybrid sub-domain. The
486/// `Hybrid` variant has no representation in `NonHybridAlgorithm` and
487/// produces `Err(())`; the caller (`verify_legacy` Phase 1c) routes
488/// that case to `LegacyVerifier::Hybrid` after parsing the operator
489/// pk. Implemented as `TryFrom` rather than a hand-named constructor
490/// to align with the standard library projection-conversion idiom.
491impl TryFrom<VerifyAlgorithm> for NonHybridAlgorithm {
492    type Error = ();
493
494    fn try_from(alg: VerifyAlgorithm) -> Result<Self, Self::Error> {
495        match alg {
496            VerifyAlgorithm::MlDsa44 => Ok(Self::MlDsa44),
497            VerifyAlgorithm::MlDsa65 => Ok(Self::MlDsa65),
498            VerifyAlgorithm::MlDsa87 => Ok(Self::MlDsa87),
499            VerifyAlgorithm::SlhDsa => Ok(Self::SlhDsa),
500            VerifyAlgorithm::FnDsa => Ok(Self::FnDsa),
501            VerifyAlgorithm::Ed25519 => Ok(Self::Ed25519),
502            VerifyAlgorithm::Hybrid => Err(()),
503        }
504    }
505}

507/// Legacy verification path — custom JSON format + primitive-level API.
508///
509/// # Pattern-6 boundary
510///
511/// `verify_legacy` splits its work into two phases with different
512/// error-handling contracts:
513///
514/// 1. **Operator-side validation** (key file existence, key type,
515///    encrypted-key passphrase, operator-picked `--algorithm` vs key
516///    mismatch, hybrid pk byte-layout parse). Errors here surface as
517///    `Err(_)` and become CLI exit code ≥2 ("operational error") with
518///    a helpful stderr message. The threat model treats the operator's
519///    `--key` file as trusted — an attacker cannot reach these paths
520///    by tampering with the signature file.
521///
522/// 2. **Signature-side validation** (algorithm auto-detect from the
523///    signature JSON, base64/JSON parse, missing fields, validate-
524///    algorithm vs auto-detected name, crypto verification). Errors
525///    here are attacker-controllable (the signature file may have been
526///    tampered) and so collapse via `unwrap_or_else` to `false`, which
527///    routes through [`print_invalid()`]. Without the collapse, an
528///    attacker could distinguish "I broke the JSON envelope" from "I
529///    broke the signature" by reading stderr text or exit-code class
530///    (Pattern-6). The original detail is preserved in `tracing::debug`
531///    for operators investigating a legitimate user mistake.
532fn verify_legacy(
533    args: &VerifyArgs,
534    sig_json: &str,
535    key_path: &std::path::Path,
536    data: Vec<u8>,
537) -> Result<bool> {
538    // `data` is threaded in from `run`. Reading input again here
539    // would drain stdin a second time in pipe mode, returning empty
540    // bytes and silently producing INVALID for any legacy-format
541    // signature piped on stdin.
542
543    // ── Phase 1a: read operator key file (helpful errors, exit ≥2) ──
544    let key_file = read_operator_public_key_file(key_path)?;
545    let pk_bytes = key_file.key_bytes()?;
546
547    // ── Phase 1b: operator-picked algorithm vs key (helpful err) ──
548    if let Some(cli_alg) = args.algorithm {
549        key_file.validate_algorithm(algorithm_name(cli_alg))?;
550    }
551
552    // ── Phase 1c: determine algorithm + parse operator-side pk ──
553    //
554    // Algorithm determination has a mixed posture:
555    //   * `--algorithm <X>` came from the operator → trusted.
556    //   * Auto-detect comes from the signature file → attacker-
557    //     controllable. If detection fails or the auto-detected
558    //     algorithm doesn't match the key, collapse to `print_invalid()`
559    //     immediately; both shapes are signature-side rejects.
560    //
561    // Once the algorithm is known, the *pk byte-layout parse* is
562    // operator-side regardless of where the algorithm came from
563    // (`pk_bytes` is the operator's key file). A malformed hybrid pk
564    // is operator misuse, not an attacker action — surface it as a
565    // helpful Err.
566    let algorithm = if let Some(alg) = args.algorithm {
567        alg
568    } else {
569        let Some(detected) = detect_algorithm(sig_json) else {
570            tracing::debug!("verify (legacy auto-detect) rejected: algorithm undetectable");
571            return Ok(print_invalid());
572        };
573        tracing::debug!(
574            algorithm = %algorithm_name(detected),
575            "auto-detected legacy verify algorithm"
576        );
577        if let Err(e) = key_file.validate_algorithm(algorithm_name(detected)) {
578            // `%e` Display, not `?e` Debug — see Phase 2 closure below
579            // for the full rationale on attacker-content avoidance in
580            // tracing output.
581            tracing::debug!(error = %e, "verify (legacy auto-detect) rejected: alg mismatch");
582            return Ok(print_invalid());
583        }
584        detected
585    };
586
587    let verifier = if let Ok(non_hybrid) = NonHybridAlgorithm::try_from(algorithm) {
588        LegacyVerifier::Standard { algorithm: non_hybrid, pk_bytes: &pk_bytes }
589    } else {
590        // Algorithm was Hybrid (the only `Err`-returning case).
591        // pk parse failure here is operator-side (their key file is
592        // malformed) — bubble as helpful Err.
593        let pk = crate::keyfile::parse_hybrid_sign_pk_from_bytes(&pk_bytes)?;
594        LegacyVerifier::Hybrid { pk }
595    };
596
597    // ── Phase 2: signature-side verify (returns plain `bool`) ──
598    //
599    // `verify_standard` and `verify_hybrid` both return `bool` directly
600    // (NOT `Result<bool>`): any error inside them — JSON parse, missing
601    // field, base64 decode, crypto reject — collapses internally to
602    // `false` with a `%e` Display debug-log. The `-> bool` return type
603    // is structural: a future maintainer who tries to bubble an
604    // operator-side error from these functions gets a compile error,
605    // which prevents accidentally widening the Phase-2 collapse to
606    // swallow operator state.
607    let valid = match &verifier {
608        LegacyVerifier::Hybrid { pk } => verify_hybrid(&data, sig_json, pk),
609        LegacyVerifier::Standard { algorithm, pk_bytes } => {
610            verify_standard(&data, sig_json, pk_bytes, *algorithm)
611        }
612    };
613
614    // Route both VALID and INVALID through their respective helpers
615    // so the reject contract (see `print_invalid()`) cannot drift via
616    // an inline `eprintln!`.
617    if valid {
618        println!("Signature is VALID.");
619        Ok(true)
620    } else {
621        Ok(print_invalid())
622    }
623}

625/// Detect algorithm from the "algorithm" field in a legacy .sig.json file.
626///
627/// Returns `None` for any reason the algorithm can't be identified —
628/// JSON parse failure, missing `"algorithm"` field, non-string value,
629/// or an unrecognised name. Returning `Option` (not `Result`) is
630/// load-bearing for the Pattern-6 collapse contract: every "couldn't
631/// detect" cause must be indistinguishable from every other one to
632/// stop attacker-controllable signature-file content from carving out
633/// a stderr-text side channel. The caller (`verify_legacy`) maps
634/// `None` into the same `print_invalid()` outcome as a crypto reject;
635/// operator-facing detail is recovered via `tracing::debug` at
636/// `RUST_LOG=debug`.
637///
638/// In particular, NEVER interpolate the `algorithm` field into an
639/// error string — an attacker can supply an arbitrarily long value,
640/// producing a proportionally large log line and a variant-name
641/// oracle for which validation step rejected.
642fn detect_algorithm(sig_json: &str) -> Option<VerifyAlgorithm> {
643    let sig: serde_json::Value = serde_json::from_str(sig_json).ok()?;
644    let alg_str = sig.get("algorithm").and_then(|v| v.as_str())?;
645    match alg_str {
646        "ml-dsa-65" => Some(VerifyAlgorithm::MlDsa65),
647        "ml-dsa-44" => Some(VerifyAlgorithm::MlDsa44),
648        "ml-dsa-87" => Some(VerifyAlgorithm::MlDsa87),
649        "slh-dsa-shake-128s" => Some(VerifyAlgorithm::SlhDsa),
650        "fn-dsa-512" => Some(VerifyAlgorithm::FnDsa),
651        "ed25519" => Some(VerifyAlgorithm::Ed25519),
652        "hybrid-ml-dsa-65-ed25519" => Some(VerifyAlgorithm::Hybrid),
653        _ => None,
654    }
655}
656
657/// Map verify algorithm enum to canonical key file algorithm name.
658fn algorithm_name(alg: VerifyAlgorithm) -> &'static str {
659    match alg {
660        VerifyAlgorithm::MlDsa65 => "ml-dsa-65",
661        VerifyAlgorithm::MlDsa44 => "ml-dsa-44",
662        VerifyAlgorithm::MlDsa87 => "ml-dsa-87",
663        VerifyAlgorithm::SlhDsa => "slh-dsa-shake-128s",
664        VerifyAlgorithm::FnDsa => "fn-dsa-512",
665        VerifyAlgorithm::Ed25519 => "ed25519",
666        VerifyAlgorithm::Hybrid => "hybrid-ml-dsa-65-ed25519",
667    }
668}

670/// Verify a non-hybrid legacy signature.
671///
672/// Returns a plain `bool` — never `Err`. All signature-side errors
673/// (JSON parse, missing fields, base64 decode, crypto reject from a
674/// `latticearc::verify_*` helper) collapse internally to `false` with
675/// the underlying cause debug-logged via `%e` (Display only — see
676/// `print_invalid()` SECURITY contract).
677///
678/// The `-> bool` return type is structural enforcement of the
679/// Phase-2 contract: a future maintainer who tries to bubble an
680/// operator-side error (e.g. by adding a `?` on a non-signature-side
681/// fallible call) gets a compile error rather than silently widening
682/// the closure-collapse to swallow operator state.
683///
684/// Do NOT re-flatten errors per arm with `.map_err(|_| anyhow!("..."))`
685/// — a per-arm message becomes its own Pattern-6 distinguisher. Let
686/// errors propagate raw to the inner closure boundary; the function-
687/// level `unwrap_or_else` handles indistinguishability uniformly.
688fn verify_standard(
689    data: &[u8],
690    sig_json: &str,
691    pk_bytes: &[u8],
692    algorithm: NonHybridAlgorithm,
693) -> bool {
694    (|| -> Result<bool> {
695        let sig_obj: serde_json::Value =
696            serde_json::from_str(sig_json).context("Failed to parse signature JSON")?;
697
698        use base64::Engine;
699        let sig_b64 = sig_obj
700            .get("signature")
701            .and_then(|v| v.as_str())
702            .context("Missing 'signature' field in signature JSON")?;
703        let sig_bytes = base64::engine::general_purpose::STANDARD
704            .decode(sig_b64)
705            .context("Invalid base64 in signature")?;
706
707        // Exhaustive over `NonHybridAlgorithm`. If a new variant is
708        // added there, this match becomes a compile error rather than
709        // a hidden unreachable arm.
710        let valid = match algorithm {
711            NonHybridAlgorithm::MlDsa65 => latticearc::verify_pq_ml_dsa(
712                data,
713                &sig_bytes,
714                pk_bytes,
715                latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa65,
716                latticearc::SecurityMode::Unverified,
717            )?,
718            NonHybridAlgorithm::MlDsa44 => latticearc::verify_pq_ml_dsa(
719                data,
720                &sig_bytes,
721                pk_bytes,
722                latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa44,
723                latticearc::SecurityMode::Unverified,
724            )?,
725            NonHybridAlgorithm::MlDsa87 => latticearc::verify_pq_ml_dsa(
726                data,
727                &sig_bytes,
728                pk_bytes,
729                latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa87,
730                latticearc::SecurityMode::Unverified,
731            )?,
732            NonHybridAlgorithm::SlhDsa => latticearc::verify_pq_slh_dsa(
733                data,
734                &sig_bytes,
735                pk_bytes,
736                latticearc::primitives::sig::slh_dsa::SlhDsaSecurityLevel::Shake128s,
737                latticearc::SecurityMode::Unverified,
738            )?,
739            NonHybridAlgorithm::FnDsa => latticearc::verify_pq_fn_dsa(
740                data,
741                &sig_bytes,
742                pk_bytes,
743                latticearc::primitives::sig::fndsa::FnDsaSecurityLevel::Level512,
744                latticearc::SecurityMode::Unverified,
745            )?,
746            NonHybridAlgorithm::Ed25519 => latticearc::verify_ed25519(
747                data,
748                &sig_bytes,
749                pk_bytes,
750                latticearc::SecurityMode::Unverified,
751            )?,
752        };
753        Ok(valid)
754    })()
755    .unwrap_or_else(|e| {
756        tracing::debug!(error = %e, "verify (legacy standard path) rejected");
757        false
758    })
759}

761/// Verify a hybrid legacy signature against a parsed hybrid public key.
762///
763/// Returns a plain `bool` — never `Err`. All signature-side errors
764/// (JSON parse, missing fields, base64 decode, crypto reject)
765/// collapse internally to `false` with a `%e` Display debug-log.
766/// Same `-> bool` structural enforcement as `verify_standard`: a
767/// future maintainer cannot bubble an operator-side error out of
768/// this function and accidentally widen the Phase-2 collapse.
769///
770/// The pk is parsed by the caller (`verify_legacy` Phase 1c) so that
771/// pk-parse failures — which indicate a malformed *operator* key file
772/// — surface as helpful `Err` at exit ≥2 there, not collapse into
773/// INVALID here.
774fn verify_hybrid(
775    data: &[u8],
776    sig_json: &str,
777    pk: &latticearc::hybrid::sig_hybrid::HybridSigPublicKey,
778) -> bool {
779    (|| -> Result<bool> {
780        let sig: serde_json::Value =
781            serde_json::from_str(sig_json).context("Failed to parse hybrid signature JSON")?;
782
783        use base64::Engine;
784        let ml_dsa_b64 = sig
785            .get("ml_dsa_sig")
786            .and_then(|v| v.as_str())
787            .context("Missing 'ml_dsa_sig' field in hybrid signature")?;
788        let ml_dsa_sig = base64::engine::general_purpose::STANDARD
789            .decode(ml_dsa_b64)
790            .context("Invalid base64 in ml_dsa_sig")?;
791
792        let ed25519_b64 = sig
793            .get("ed25519_sig")
794            .and_then(|v| v.as_str())
795            .context("Missing 'ed25519_sig' field in hybrid signature")?;
796        let ed25519_sig = base64::engine::general_purpose::STANDARD
797            .decode(ed25519_b64)
798            .context("Invalid base64 in ed25519_sig")?;
799
800        let hybrid_sig =
801            latticearc::hybrid::sig_hybrid::HybridSignature::new(ml_dsa_sig, ed25519_sig);
802
803        // CoreError::VerificationFailed is a routine cryptographic
804        // reject (`Ok(false)`-equivalent). Any other CoreError variant
805        // is propagated as anyhow::Error::new(e) — the typed source
806        // survives for `%e` Display, which on CoreError gives a fixed
807        // variant message without attacker-controlled interpolation.
808        // A structured upstream error like "ML-DSA component verify
809        // failed" vs "Ed25519 component verify failed" would otherwise
810        // tell an attacker which half of the hybrid pair rejected.
811        use latticearc::CoreError;
812        match latticearc::verify_hybrid_signature(
813            data,
814            &hybrid_sig,
815            pk,
816            latticearc::SecurityMode::Unverified,
817        ) {
818            Ok(valid) => Ok(valid),
819            Err(CoreError::VerificationFailed) => Ok(false),
820            Err(e) => Err(anyhow::Error::new(e)),
821        }
822    })()
823    .unwrap_or_else(|e| {
824        tracing::debug!(error = %e, "verify (legacy hybrid path) rejected");
825        false
826    })
