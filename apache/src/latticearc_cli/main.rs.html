4//! # LatticeArc CLI
5//!
6//! Post-quantum cryptography from the command line.
7//!
8//! This binary provides 8 subcommands for everyday cryptographic operations
9//! using NIST-standardized post-quantum algorithms:
10//!
11//! | Command   | Purpose                                        |
12//! |-----------|------------------------------------------------|
13//! | `keygen`  | Generate signing, encryption, or symmetric keys |
14//! | `sign`    | Create a digital signature over a file          |
15//! | `verify`  | Check a signature against a public key          |
16//! | `encrypt` | Encrypt a file (AES-256-GCM or hybrid PQ)       |
17//! | `decrypt` | Decrypt a previously encrypted file              |
18//! | `hash`    | Compute a cryptographic hash (SHA3, SHA2, BLAKE2)|
19//! | `kdf`     | Derive a key from a password or key material     |
20//! | `info`    | Show version, FIPS status, supported algorithms  |
21//!
22//! ## Supported Algorithms
23//!
24//! - **Signatures:** ML-DSA-44/65/87 (FIPS 204), SLH-DSA (FIPS 205),
25//!   FN-DSA-512 (FIPS 206 draft), Ed25519 (RFC 8032), Hybrid ML-DSA+Ed25519
26//! - **Encryption:** AES-256-GCM (FIPS 197), Hybrid ML-KEM+X25519+AES-256-GCM
27//! - **Hashing:** SHA3-256 (FIPS 202), SHA-256/512 (FIPS 180-4), BLAKE2b (RFC 7693)
28//! - **KDF:** HKDF-SHA256 (SP 800-56C), PBKDF2-HMAC-SHA256 (SP 800-132)
29//!
30//! ## Architecture
31//!
32//! ```text
33//! main.rs          Entry point, CLI argument parsing (clap)
34//! ├── keyfile.rs   JSON key file format, serialization, permissions
35//! └── commands/
36//!     ├── keygen   Key generation for all 12 algorithm variants
37//!     ├── sign     Digital signature creation (7 algorithms)
38//!     ├── verify   Signature verification with auto-detection
39//!     ├── encrypt  Authenticated encryption (symmetric + hybrid)
40//!     ├── decrypt  Decryption with integrity verification
41//!     ├── hash     Cryptographic hashing (4 algorithms)
42//!     ├── kdf      Key derivation (HKDF + PBKDF2)
43//!     └── info     Version and capability reporting
44//! ```
45//!
46//! All cryptographic operations are delegated to the `latticearc` library crate.
47//! The CLI handles file I/O, key serialization, argument parsing, and user output.

52use anyhow::Result;
53use clap::{Parser, Subcommand};
54
55/// LatticeArc — Post-quantum cryptography from the command line.
56///
57/// Generate keys, encrypt, decrypt, sign, verify, hash, and derive keys
58/// using NIST-standard post-quantum algorithms (ML-KEM, ML-DSA, SLH-DSA, FN-DSA)
59/// and classical algorithms (AES-256-GCM, Ed25519, X25519).
60#[derive(Parser)]
61#[command(name = "latticearc-cli", version, about, long_about = None)]
62#[command(propagate_version = true)]
63// Round-7 audit fix #15: showing the full help when invoked with no args
64// is more discoverable than the default 2-line "USAGE:" stub for
65// first-time users. Costs nothing for scripted callers (they always
66// supply a subcommand) and pays back the discoverability for
67// `latticearc-cli` typed at a fresh shell.
68#[command(arg_required_else_help = true)]
69struct Cli {
70    /// Subcommand to execute.
71    #[command(subcommand)]
72    command: Commands,
73}
74
75/// Available subcommands.
76#[derive(Subcommand)]
77enum Commands {
78    /// Generate a cryptographic keypair or symmetric key.
79    Keygen(commands::keygen::KeygenArgs),
80    /// Encrypt data using symmetric or hybrid post-quantum encryption.
81    Encrypt(commands::encrypt::EncryptArgs),
82    /// Decrypt data.
83    Decrypt(commands::decrypt::DecryptArgs),
84    /// Sign data with a signing key.
85    Sign(commands::sign::SignArgs),
86    /// Verify a signature.
87    Verify(commands::verify::VerifyArgs),
88    /// Hash data (SHA3-256, SHA-256, SHA-512, BLAKE2b).
89    Hash(commands::hash::HashArgs),
90    /// Derive a key from a password using HKDF or PBKDF2.
91    Kdf(commands::kdf::KdfArgs),
92    /// Show version, supported algorithms, build info, and FIPS status.
93    Info(commands::info::InfoArgs),
94}
95
96fn main() -> Result<()> {
97    let cli = Cli::parse();
98
99    // Wire up the `tracing` subscriber so `tracing::{debug,info,warn,error}!`
100    // events from the library reach stderr. The `tracing-init` Cargo
101    // feature on our `latticearc` dep pulls in the `tracing-subscriber`
102    // and `tracing-appender` crates that back this helper. Subscriber
103    // wiring is the binary's job (see DESIGN_PATTERNS Pattern 6 +
104    // SECURITY.md observability section); a library that calls this
105    // would `panic!` the next downstream consumer.
106    //
107    // Filter defaults to `latticearc=info`; override at runtime with
108    // `RUST_LOG=latticearc=debug` (or `=trace` to see per-stage
109    // crypto-error tags).
110    //
111    // Best-effort: if a subscriber is somehow already installed (e.g., the
112    // CLI is being driven from a test harness that wired its own), we
113    // swallow the error rather than aborting the user's command.
114    let _ = latticearc::unified_api::logging::init_tracing();
115
116    // Initialize LatticeArc (runs FIPS power-up self-tests)
117    latticearc::init().map_err(|e| anyhow::anyhow!("Library initialization failed: {e}"))?;
118
119    match cli.command {
120        Commands::Keygen(args) => commands::keygen::run(args),
121        Commands::Encrypt(args) => commands::encrypt::run(args),
122        Commands::Decrypt(args) => commands::decrypt::run(args),
123        Commands::Sign(args) => commands::sign::run(args),
124        // Round-8 audit fix #5: verify returns Ok(true)=VALID,
125        // Ok(false)=INVALID. Translate INVALID into exit 1 here so
126        // destructors on the per-command state inside `verify::run`
127        // (KeyFile, Vec<u8>, etc.) all get to run before the process
128        // dies. The earlier in-function `process::exit(1)` skipped
129        // them — currently benign because verify's per-command state
130        // only holds public material, but a regression copying the
131        // pattern to sign/decrypt would skip secret zeroization.
132        Commands::Verify(args) => {
133            let valid = commands::verify::run(args)?;
134            if !valid {
135                // `clippy::exit` allowed: deterministic exit 1 is the
136                // documented INVALID-signature signal in our exit-code
137                // contract (0/1/≥2 — see QUICK_REFERENCE.md). Pattern
138                // 12 wants the rationale adjacent to the `#[allow]`;
139                // round-9 audit fix #5 moved it from 11 lines above.
140                #[allow(clippy::exit)]
141                std::process::exit(1);
142            }
143            Ok(())
144        }
145        Commands::Hash(args) => commands::hash::run(args),
146        Commands::Kdf(args) => commands::kdf::run(args),
147        Commands::Info(args) => commands::info::run(args),
148    }
149}
