6//! # LatticeArc TLS
7//!
8//! TLS 1.3 implementation with post-quantum key exchange support.
9//!
10//! This crate provides TLS connectivity with hybrid key exchange (ECDHE + ML-KEM)
11//! for quantum-resistant secure communications.
12//!
13//! ## Features
14//!
15//! - **Hybrid Key Exchange**: X25519MLKEM768 (ECDHE + ML-KEM) for post-quantum security
16//! - **Backward Compatible**: Works with standard TLS 1.3 clients
17//! - **Flexible Modes**: Classic, Hybrid, and PQ-only modes
18//! - **Easy API**: Simple client/server connector functions
19//! - **Comprehensive Error Handling**: Detailed error types, recovery mechanisms, and tracing
20//!
21//! ## Quick Start
22//!
23//! ```no_run
24//! use arc_tls::*;
25//!
26//! # async fn example() -> Result<(), TlsError> {
27//! // Client connection (default: hybrid mode)
28//! let stream = tls_connect("example.com:443", "example.com", &TlsConfig::default()).await?;
29//!
30//! // Server setup
31//! let acceptor = create_server_acceptor(&TlsConfig::default(), "server.crt", "server.key")?;
32//! let listener = tokio::net::TcpListener::bind("0.0.0.0:8443").await.map_err(TlsError::from)?;
33//! let (tcp_stream, _) = listener.accept().await.map_err(TlsError::from)?;
34//! let tls_stream = tls_accept(tcp_stream, &acceptor).await?;
35//! # Ok(())
36//! # }
37//! ```
38//!
39//! ## Error Handling
40//!
41//! The crate provides comprehensive error handling with:
42//! - Detailed error types with error codes
43//! - Error recovery mechanisms (retry, fallback, circuit breaker)
44//! - Structured logging with tracing
45//! - Error context propagation
46//!
47//! ```no_run
48//! use arc_tls::*;
49//! use arc_tls::recovery::{RetryPolicy, retry_with_policy};
50//!
51//! # async fn example() -> Result<(), TlsError> {
52//! let policy = RetryPolicy::default();
53//! let result = retry_with_policy(&policy, || async {
54//!     tls_connect("example.com:443", "example.com", &TlsConfig::default()).await
55//! }, "TLS connection").await?;
56//! # Ok(())
57//! # }
58//! ```
59//!
60//! ## TLS Modes
61//!
62//! ### Hybrid Mode (Default, Recommended)
63//! Uses X25519MLKEM768 combining:
64//! - X25519: Classical ECDH, well-tested
65//! - ML-KEM-768: Post-quantum KEM (NIST FIPS 203)
66//!
67//! Security: Requires breaking BOTH components
68//!
69//! ### Classic Mode
70//! Standard TLS 1.3 with X25519 only. Not PQ secure.
71//!
72//! ### PQ-Only Mode
73//! Post-quantum only with ML-KEM. May have compatibility issues.
74//!
75//!
76//! ## Performance Impact
77//!
78//! Hybrid TLS 1.3 with X25519MLKEM768:
79//! - ClientHello: +1184 bytes (ML-KEM key share)
80//! - ServerHello: +1088 bytes (ML-KEM ciphertext)
81//! - CPU overhead: ~2-3x on key exchange (negligible overall)
82//!
83//! See [rustls performance report](https://rustls.dev/perf/2024-12-17-pq-kx/) for details.
84//!
85//! ## Architecture
86//!
87//! ```text
88//! ┌─────────────────────────────────────────────────────────────────────┐
89//! │                         TLS 1.3 Handshake                           │
90//! ├─────────────────────────────────────────────────────────────────────┤
91//! │                                                                     │
92//! │  Client                                              Server         │
93//! │    │                                                   │            │
94//! │    │  ClientHello                                      │            │
95//! │    │  ├─ supported_versions: TLS 1.3                   │            │
96//! │    │  ├─ key_share: X25519 (32 bytes)                  │            │
97//! │    │  └─ key_share: ML-KEM-768 (1184 bytes) ◄── PQ     │            │
98//! │    │──────────────────────────────────────────────────►│            │
99//! │    │                                                   │            │
100//! │    │                               ServerHello         │            │
101//! │    │               key_share: X25519 (32 bytes) ──┤    │            │
102//! │    │  key_share: ML-KEM-768 ciphertext (1088 bytes) ◄──┤            │
103//! │    │◄──────────────────────────────────────────────────│            │
104//! │    │                                                   │            │
105//! │    │        ┌────────────────────────────────┐         │            │
106//! │    │        │  Hybrid Shared Secret (64 B)   │         │            │
107//! │    │        │  = HKDF(X25519_SS ║ ML-KEM_SS) │         │            │
108//! │    │        └────────────────────────────────┘         │            │
109//! │    │                                                   │            │
110//! │    │  {EncryptedExtensions}                            │            │
111//! │    │  {Certificate}                                    │            │
112//! │    │  {CertificateVerify}                              │            │
113//! │    │  {Finished}                                       │            │
114//! │    │◄──────────────────────────────────────────────────│            │
115//! │    │                                                   │            │
116//! │    │  {Finished}                                       │            │
117//! │    │──────────────────────────────────────────────────►│            │
118//! │    │                                                   │            │
119//! │    │◄═════════════════════════════════════════════════►│            │
120//! │    │            Application Data (encrypted)           │            │
121//! │                                                                     │
122//! └─────────────────────────────────────────────────────────────────────┘
123//!
124//! Security: Attacker must break BOTH X25519 AND ML-KEM to compromise session
125//! ```

572    /// Validate the TLS configuration
573    ///
574    /// Checks for configuration errors that would cause unexpected behavior
575    /// at runtime. Call this after building a configuration to catch issues early.
576    ///
577    /// # Errors
578    ///
579    /// Returns [`TlsError::Config`] if:
580    /// - Protocol version range produces no valid versions (e.g., min > max)
581    /// - Protocol version range excludes all supported versions
582    ///
583    /// # Example
584    /// ```
585    /// use arc_tls::{TlsConfig, TlsError};
586    ///
587    /// let config = TlsConfig::new()
588    ///     .with_min_protocol_version(rustls::ProtocolVersion::TLSv1_2)
589    ///     .with_max_protocol_version(rustls::ProtocolVersion::TLSv1_3);
590    ///
591    /// if let Err(e) = config.validate() {
592    ///     eprintln!("Invalid config: {}", e);
593    /// }
594    /// ```
595    pub fn validate(&self) -> Result<(), TlsError> {
596        // Validate protocol version range
597        if let Some(min_version) = self.min_protocol_version {
598            let versions: Vec<&'static rustls::SupportedProtocolVersion> =
599                if let Some(max_version) = self.max_protocol_version {
600                    rustls::ALL_VERSIONS
601                        .iter()
602                        .filter(|v| {
603                            let v_num: u16 = v.version.into();
604                            let min_num: u16 = min_version.into();
605                            let max_num: u16 = max_version.into();
606                            v_num >= min_num && v_num <= max_num
607                        })
608                        .copied()
609                        .collect()
610                } else {
611                    rustls::ALL_VERSIONS
612                        .iter()
613                        .filter(|v| {
614                            let v_num: u16 = v.version.into();
615                            let min_num: u16 = min_version.into();
616                            v_num >= min_num
617                        })
618                        .copied()
619                        .collect()
620                };
621
622            if versions.is_empty() {
623                return Err(TlsError::Config {
624                    message: format!(
625                        "Protocol version range ({:?} - {:?}) produces no valid versions. \
626                         Supported versions are TLS 1.2 and TLS 1.3.",
627                        min_version, self.max_protocol_version
628                    ),
629                    field: Some("protocol_version".to_string()),
630                    code: ErrorCode::InvalidProtocolVersion,
631                    context: Box::new(ErrorContext {
632                        code: ErrorCode::InvalidProtocolVersion,
633                        phase: OperationPhase::Initialization,
634                        ..Default::default()
635                    }),
636                    recovery: Box::new(RecoveryHint::Reconfigure {
637                        field: "min_protocol_version / max_protocol_version".to_string(),
638                        suggestion: "Use TLSv1_2 or TLSv1_3 as version bounds".to_string(),
639                    }),
640                });
641            }
642        }
643
644        Ok(())
645    }

648impl From<&TlsConfig> for Tls13Config {
649    fn from(config: &TlsConfig) -> Self {
650        let mut tls13_config = match config.mode {
651            TlsMode::Classic => Tls13Config::classic(),
652            TlsMode::Hybrid => Tls13Config::hybrid(),
653            TlsMode::Pq => Tls13Config::pq(),
654        };
655
656        // Apply additional configuration from TlsConfig
657        if !config.alpn_protocols.is_empty() {
658            tls13_config.alpn_protocols = config.alpn_protocols.clone();
659        }
660
661        if let Some(size) = config.max_fragment_size {
662            tls13_config.max_fragment_size = Some(size);
663        }
664
665        if config.enable_early_data {
666            tls13_config.enable_early_data = true;
667            tls13_config.max_early_data_size = config.max_early_data_size;
668        }
669
670        if let Some(min_version) = config.min_protocol_version {
671            let versions: Vec<&'static rustls::SupportedProtocolVersion> =
672                if let Some(max_version) = config.max_protocol_version {
673                    rustls::ALL_VERSIONS
674                        .iter()
675                        .filter(|v| {
676                            let v_num: u16 = v.version.into();
677                            let min_num: u16 = min_version.into();
678                            let max_num: u16 = max_version.into();
679                            v_num >= min_num && v_num <= max_num
680                        })
681                        .copied()
682                        .collect()
683                } else {
684                    rustls::ALL_VERSIONS
685                        .iter()
686                        .filter(|v| {
687                            let v_num: u16 = v.version.into();
688                            let min_num: u16 = min_version.into();
689                            v_num >= min_num
690                        })
691                        .copied()
692                        .collect()
693                };
694
695            if versions.is_empty() {
696                // Log warning: requested protocol versions resulted in empty list
697                // This happens when min/max versions don't overlap with supported versions
698                // Defaulting to TLS 1.3 since this library is TLS 1.3 focused
699                ::tracing::warn!(
700                    min_version = ?min_version,
701                    max_version = ?config.max_protocol_version,
702                    "Requested protocol version range produced empty list. \
703                     Defaulting to TLS 1.3. Consider using TLS 1.2 or TLS 1.3."
704                );
705                tls13_config.protocol_versions = vec![&rustls::version::TLS13];
706            } else {
707                tls13_config.protocol_versions = versions;
708            }
709        }
710
711        // Configure cipher suites if specified
712        if let Some(ref cipher_suites) = config.cipher_suites {
713            tls13_config.cipher_suites = Some(cipher_suites.clone());
714        }
715
716        // Configure key logging if enabled
717        if config.enable_key_logging {
718            tls13_config.key_log = Some(std::sync::Arc::new(rustls::KeyLogFile::new()));
719        }
720
721        // Configure mTLS client authentication
722        tls13_config.client_verification = config.client_verification;
723
724        tls13_config
725    }
726}
