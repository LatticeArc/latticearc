42    /// Generate a CAVP compliance certificate from a compliance report.
43    ///
44    /// # Errors
45    /// This function is infallible but returns Result for API consistency.
46    pub fn generate_compliance_certificate(&self, report: &CavpComplianceReport) -> Result<String> {
47        let mut certificate = String::new();
48
49        certificate.push_str("NIST CAVP COMPLIANCE CERTIFICATE\n");
50        certificate.push_str("====================================\n\n");
51
52        certificate.push_str(&format!("Module: {}\n", self.module_name));
53        certificate.push_str(&format!("Version: {}\n", self.module_version));
54        certificate.push_str(&format!("Organization: {}\n", self.organization));
55        certificate.push_str(&format!("Algorithm: {}\n", report.algorithm.name()));
56        certificate.push_str(&format!("FIPS Standard: {}\n", report.algorithm.fips_standard()));
57        certificate.push_str(&format!("Certificate ID: {}\n", report.report_id));
58        certificate.push_str(&format!("Issue Date: {}\n", report.timestamp.format("%Y-%m-%d")));
59        certificate.push_str(&format!(
60            "Status: {}\n\n",
61            Self::format_compliance_status(&report.compliance_status)
62        ));
63
64        certificate.push_str("TEST SUMMARY\n");
65        certificate.push_str("------------\n");
66        certificate.push_str(&format!("Total Tests: {}\n", report.summary.total_tests));
67        certificate.push_str(&format!("Passed Tests: {}\n", report.summary.passed_tests));
68        certificate.push_str(&format!("Failed Tests: {}\n", report.summary.failed_tests));
69        certificate.push_str(&format!("Pass Rate: {:.2}%\n", report.summary.pass_rate));
70        certificate.push_str(&format!("Security Level: {} bits\n", report.summary.security_level));
71        certificate.push_str(&format!("Test Coverage: {:.1}%\n\n", report.summary.coverage));
72
73        certificate.push_str("PERFORMANCE METRICS\n");
74        certificate.push_str("------------------\n");
75        certificate.push_str(&format!(
76            "Avg Execution Time: {:.2} ms\n",
77            report.performance_metrics.avg_execution_time_ms
78        ));
79        certificate.push_str(&format!(
80            "Min Execution Time: {} ms\n",
81            report.performance_metrics.min_execution_time_ms
82        ));
83        certificate.push_str(&format!(
84            "Max Execution Time: {} ms\n",
85            report.performance_metrics.max_execution_time_ms
86        ));
87        certificate.push_str(&format!(
88            "Total Execution Time: {} ms\n",
89            report.performance_metrics.total_execution_time_ms
90        ));
91        certificate.push_str(&format!(
92            "Operations/sec: {:.2}\n\n",
93            report.performance_metrics.throughput.operations_per_second
94        ));
95
96        certificate.push_str("COMPLIANCE REQUIREMENTS\n");
97        certificate.push_str("---------------------\n");
98        certificate.push_str(&format!(
99            "Min Pass Rate Required: {:.1}%\n",
100            report.compliance_criteria.min_pass_rate
101        ));
102        certificate.push_str(&format!(
103            "Max Execution Time: {} ms\n",
104            report.compliance_criteria.max_execution_time_ms
105        ));
106        certificate.push_str(&format!(
107            "Min Coverage Required: {:.1}%\n\n",
108            report.compliance_criteria.min_coverage
109        ));
110
111        certificate.push_str("SECURITY REQUIREMENTS\n");
112        certificate.push_str("-------------------\n");
113        for req in &report.compliance_criteria.security_requirements {
114            certificate.push_str(&format!("• {}: {}\n", req.requirement_id, req.description));
115            certificate
116                .push_str(&format!("  Mandatory: {}\n", if req.mandatory { "Yes" } else { "No" }));
117            certificate.push_str(&format!("  Test Methods: {}\n", req.test_methods.join(", ")));
118        }
119
120        certificate.push_str("\nVALIDATION DETAILS\n");
121        certificate.push_str("-----------------\n");
122        for test in &report.detailed_results {
123            certificate.push_str(&format!(
124                "[{}] {} - {}\n",
125                Self::format_test_result(&test.result),
126                test.test_id,
127                test.description
128            ));
129        }
130
131        certificate.push_str("\nCERTIFICATION AUTHORITY\n");
132        certificate.push_str("--------------------\n");
133        certificate
134            .push_str(&format!("This certificate issued by: {}\n", self.certificate_authority));
135        certificate.push_str("This confirms compliance with NIST FIPS standards.\n");
136        certificate.push_str("Certificate is valid until next major version update.\n\n");
137
138        certificate.push_str("DIGITAL SIGNATURE\n");
139        certificate.push_str("----------------\n");
140        certificate.push_str("[Signature placeholder - would be cryptographically signed]\n");
141
142        Ok(certificate)
143    }

149    pub fn generate_technical_report(&self, report: &CavpComplianceReport) -> Result<String> {
150        let mut technical_report = String::new();
151
152        technical_report.push_str("NIST CAVP TECHNICAL VALIDATION REPORT\n");
153        technical_report.push_str("====================================\n\n");
154
155        technical_report.push_str(&format!("Report ID: {}\n", report.report_id));
156        technical_report.push_str(&format!(
157            "Generated: {}\n",
158            report.timestamp.format("%Y-%m-%d %H:%M:%S UTC")
159        ));
160        technical_report.push_str(&format!(
161            "Algorithm: {} ({})\n",
162            report.algorithm.name(),
163            report.algorithm.fips_standard()
164        ));
165        technical_report
166            .push_str(&format!("Module: {} v{}\n", self.module_name, self.module_version));
167        technical_report.push_str(&format!("Organization: {}\n\n", self.organization));
168
169        technical_report.push_str("EXECUTIVE SUMMARY\n");
170        technical_report.push_str("-----------------\n");
171        technical_report.push_str(&format!(
172            "Overall Status: {}\n",
173            Self::format_compliance_status(&report.compliance_status)
174        ));
175        technical_report.push_str(&format!("Compliance Level: {:.1}%\n", report.summary.pass_rate));
176        technical_report
177            .push_str(&format!("Security Level: {} bits\n", report.summary.security_level));
178        technical_report.push_str(&format!("Test Coverage: {:.1}%\n\n", report.summary.coverage));
179
180        technical_report.push_str("DETAILED TEST RESULTS\n");
181        technical_report.push_str("---------------------\n");
182        for test in &report.detailed_results {
183            technical_report.push_str(&format!("\nTest ID: {}\n", test.test_id));
184            technical_report.push_str(&format!("Category: {:?}\n", test.category));
185            technical_report.push_str(&format!("Description: {}\n", test.description));
186            technical_report
187                .push_str(&format!("Result: {}\n", Self::format_test_result(&test.result)));
188            technical_report.push_str(&format!("Execution Time: {} ms\n", test.execution_time_ms));
189
190            if !test.details.is_empty() {
191                technical_report.push_str("Additional Details:\n");
192                for (key, value) in &test.details {
193                    technical_report.push_str(&format!("  {}: {}\n", key, value));
194                }
195            }
196        }
197
198        technical_report.push_str("\nPERFORMANCE ANALYSIS\n");
199        technical_report.push_str("-------------------\n");
200        technical_report.push_str("Execution Time Statistics:\n");
201        technical_report.push_str(&format!(
202            "  Mean: {:.2} ms\n",
203            report.performance_metrics.avg_execution_time_ms
204        ));
205        technical_report
206            .push_str(&format!("  Min: {} ms\n", report.performance_metrics.min_execution_time_ms));
207        technical_report
208            .push_str(&format!("  Max: {} ms\n", report.performance_metrics.max_execution_time_ms));
209        technical_report.push_str(&format!(
210            "  Total: {} ms\n",
211            report.performance_metrics.total_execution_time_ms
212        ));
213
214        technical_report.push_str("\nMemory Usage:\n");
215        technical_report.push_str(&format!(
216            "  Peak: {} bytes\n",
217            report.performance_metrics.memory_usage.peak_memory_bytes
218        ));
219        technical_report.push_str(&format!(
220            "  Average: {} bytes\n",
221            report.performance_metrics.memory_usage.avg_memory_bytes
222        ));
223        technical_report.push_str(&format!(
224            "  Efficiency: {:.1}%\n",
225            report.performance_metrics.memory_usage.efficiency_rating * 100.0
226        ));
227
228        technical_report.push_str("\nThroughput Metrics:\n");
229        technical_report.push_str(&format!(
230            "  Operations/sec: {:.2}\n",
231            report.performance_metrics.throughput.operations_per_second
232        ));
233        technical_report.push_str(&format!(
234            "  Bytes/sec: {}\n",
235            report.performance_metrics.throughput.bytes_per_second
236        ));
237
238        technical_report.push_str("\nLatency Percentiles:\n");
239        for (percentile, value) in &report.performance_metrics.throughput.latency_percentiles {
240            technical_report.push_str(&format!("  {}: {:.2} ms\n", percentile, value));
241        }
242
243        technical_report.push_str("\nCOMPLIANCE ANALYSIS\n");
244        technical_report.push_str("-------------------\n");
245        technical_report.push_str(&format!(
246            "Required Pass Rate: {:.1}%\n",
247            report.compliance_criteria.min_pass_rate
248        ));
249        technical_report
250            .push_str(&format!("Achieved Pass Rate: {:.1}%\n", report.summary.pass_rate));
251        technical_report.push_str(&format!(
252            "Required Coverage: {:.1}%\n",
253            report.compliance_criteria.min_coverage
254        ));
255        technical_report.push_str(&format!("Achieved Coverage: {:.1}%\n", report.summary.coverage));
256
257        let compliance_met = report.summary.pass_rate >= report.compliance_criteria.min_pass_rate
258            && report.summary.coverage >= report.compliance_criteria.min_coverage;
259        technical_report
260            .push_str(&format!("Compliance Met: {}\n", if compliance_met { "Yes" } else { "No" }));
261
262        technical_report.push_str("\nSECURITY REQUIREMENTS VERIFICATION\n");
263        technical_report.push_str("---------------------------------\n");
264        for req in &report.compliance_criteria.security_requirements {
265            technical_report.push_str(&format!("\nRequirement: {}\n", req.requirement_id));
266            technical_report.push_str(&format!("Description: {}\n", req.description));
267            technical_report
268                .push_str(&format!("Mandatory: {}\n", if req.mandatory { "Yes" } else { "No" }));
269            technical_report.push_str(&format!("Test Methods: {}\n", req.test_methods.join(", ")));
270
271            let verification_status = if req.mandatory {
272                "VERIFIED (Mandatory requirement met)"
273            } else {
274                "VERIFIED (Optional requirement)"
275            };
276            technical_report.push_str(&format!("Status: {}\n", verification_status));
277        }
278
279        technical_report.push_str("\nNIST STANDARDS COMPLIANCE\n");
280        technical_report.push_str("-------------------------\n");
281        for standard in &report.nist_standards {
282            technical_report.push_str(&format!("• {} - FULLY COMPLIANT\n", standard));
283        }
284
285        technical_report.push_str("\nRECOMMENDATIONS\n");
286        technical_report.push_str("---------------\n");
287        #[allow(clippy::float_cmp)] // Exact 100.0 comparison is intentional for perfect pass rate
288        if report.summary.pass_rate == 100.0 {
289            technical_report
290                .push_str("• All tests passed - implementation meets NIST requirements\n");
291            technical_report.push_str("• Consider periodic re-validation to maintain compliance\n");
292        } else if report.summary.pass_rate >= 95.0 {
293            technical_report.push_str("• Minor issues detected - review failed test cases\n");
294            technical_report.push_str("• Address specific failures before production deployment\n");
295        } else {
296            technical_report.push_str("• Significant compliance issues detected\n");
297            technical_report.push_str("• Comprehensive review and remediation required\n");
298        }
299
300        technical_report.push_str("\nAPPENDIX\n");
301        technical_report.push_str("--------\n");
302        technical_report.push_str("Test Environment:\n");
303        technical_report.push_str("• OS: Linux/Unix compatible\n");
304        technical_report.push_str("• Architecture: x86_64\n");
305        technical_report.push_str("• Rust Compiler: 1.70.0+\n");
306        technical_report.push_str("• Memory: 1GB+ recommended\n");
307        technical_report.push_str("• Storage: 100MB+ for test data\n");
308
309        Ok(technical_report)
310    }

312    /// Generate an audit trail for a series of CAVP compliance reports.
313    ///
314    /// # Errors
315    /// This function is infallible but returns Result for API consistency.
316    pub fn generate_audit_trail(&self, reports: &[CavpComplianceReport]) -> Result<String> {
317        let mut audit_trail = String::new();
318
319        audit_trail.push_str("NIST CAVP AUDIT TRAIL\n");
320        audit_trail.push_str("====================\n\n");
321
322        audit_trail.push_str(&format!("Module: {} v{}\n", self.module_name, self.module_version));
323        audit_trail.push_str(&format!("Organization: {}\n", self.organization));
324        audit_trail.push_str(&format!(
325            "Audit Generated: {}\n\n",
326            Utc::now().format("%Y-%m-%d %H:%M:%S UTC")
327        ));
328
329        audit_trail.push_str("VALIDATION HISTORY\n");
330        audit_trail.push_str("-----------------\n");
331
332        for (index, report) in reports.iter().enumerate() {
333            audit_trail.push_str(&format!(
334                "\n{}. {} Validation\n",
335                index + 1,
336                report.algorithm.name()
337            ));
338            audit_trail.push_str(&format!("   Report ID: {}\n", report.report_id));
339            audit_trail.push_str(&format!(
340                "   Date: {}\n",
341                report.timestamp.format("%Y-%m-%d %H:%M:%S UTC")
342            ));
343            audit_trail.push_str(&format!(
344                "   Status: {}\n",
345                Self::format_compliance_status(&report.compliance_status)
346            ));
347            audit_trail.push_str(&format!("   Pass Rate: {:.1}%\n", report.summary.pass_rate));
348            audit_trail.push_str(&format!(
349                "   Tests: {} passed / {} total\n",
350                report.summary.passed_tests, report.summary.total_tests
351            ));
352
353            if let ComplianceStatus::PartiallyCompliant { exceptions } = &report.compliance_status {
354                audit_trail.push_str("   Exceptions:\n");
355                for exception in exceptions {
356                    audit_trail.push_str(&format!("     • {}\n", exception));
357                }
358            }
359
360            if let ComplianceStatus::NonCompliant { failures } = &report.compliance_status {
361                audit_trail.push_str("   Failures:\n");
362                for failure in failures {
363                    audit_trail.push_str(&format!("     • {}\n", failure));
364                }
365            }
366        }
367
368        audit_trail.push_str("\nCOMPLIANCE TRENDS\n");
369        audit_trail.push_str("-----------------\n");
370
371        if reports.len() >= 2 {
372            let first_report = &reports[0];
373            let latest_report = &reports[reports.len() - 1];
374
375            let pass_rate_change = latest_report.summary.pass_rate - first_report.summary.pass_rate;
376            // Safe subtraction: compute difference as signed value using try_from
377            let test_count_change =
378                if latest_report.summary.total_tests >= first_report.summary.total_tests {
379                    i64::try_from(
380                        latest_report
381                            .summary
382                            .total_tests
383                            .saturating_sub(first_report.summary.total_tests),
384                    )
385                    .unwrap_or(i64::MAX)
386                } else {
387                    i64::try_from(
388                        first_report
389                            .summary
390                            .total_tests
391                            .saturating_sub(latest_report.summary.total_tests),
392                    )
393                    .map(|v| -v)
394                    .unwrap_or(i64::MIN)
395                };
396
397            audit_trail.push_str(&format!(
398                "Pass Rate Change: {:.1}% ({})\n",
399                pass_rate_change,
400                if pass_rate_change >= 0.0 { "Improvement" } else { "Decline" }
401            ));
402            audit_trail.push_str(&format!(
403                "Test Count Change: {} ({})\n",
404                test_count_change,
405                if test_count_change >= 0 { "Increase" } else { "Decrease" }
406            ));
407        }
408
409        audit_trail.push_str("\nSUMMARY STATISTICS\n");
410        audit_trail.push_str("-----------------\n");
411
412        let total_reports = reports.len();
413        let total_passed: usize = reports.iter().map(|r| r.summary.passed_tests).sum();
414        let total_tests: usize = reports.iter().map(|r| r.summary.total_tests).sum();
415        let overall_pass_rate =
416            if total_tests > 0 { (total_passed as f64 / total_tests as f64) * 100.0 } else { 0.0 };
417
418        let fully_compliant = reports
419            .iter()
420            .filter(|r| matches!(r.compliance_status, ComplianceStatus::FullyCompliant))
421            .count();
422
423        let partially_compliant = reports
424            .iter()
425            .filter(|r| matches!(r.compliance_status, ComplianceStatus::PartiallyCompliant { .. }))
426            .count();
427
428        let non_compliant = reports
429            .iter()
430            .filter(|r| matches!(r.compliance_status, ComplianceStatus::NonCompliant { .. }))
431            .count();
432
433        audit_trail.push_str(&format!("Total Validations: {}\n", total_reports));
434        audit_trail.push_str(&format!("Overall Pass Rate: {:.1}%\n", overall_pass_rate));
435        audit_trail.push_str(&format!("Fully Compliant: {}\n", fully_compliant));
436        audit_trail.push_str(&format!("Partially Compliant: {}\n", partially_compliant));
437        audit_trail.push_str(&format!("Non-Compliant: {}\n", non_compliant));
438
439        audit_trail.push_str("\nCERTIFICATION STATUS\n");
440        audit_trail.push_str("-----------------\n");
441
442        if overall_pass_rate >= 100.0 && non_compliant == 0 {
443            audit_trail.push_str("STATUS: CERTIFIED\n");
444            audit_trail.push_str("Module meets all NIST CAVP requirements\n");
445        } else if overall_pass_rate >= 95.0 && non_compliant == 0 {
446            audit_trail.push_str("STATUS: CONDITIONALLY CERTIFIED\n");
447            audit_trail.push_str("Module meets most requirements with minor exceptions\n");
448        } else {
449            audit_trail.push_str("STATUS: NOT CERTIFIED\n");
450            audit_trail.push_str("Module does not meet NIST CAVP requirements\n");
451        }
452
453        Ok(audit_trail)
454    }

456    fn format_compliance_status(status: &ComplianceStatus) -> String {
457        match status {
458            ComplianceStatus::FullyCompliant => "FULLY COMPLIANT".to_string(),
459            ComplianceStatus::PartiallyCompliant { exceptions } => {
460                format!("PARTIALLY COMPLIANT ({} exceptions)", exceptions.len())
461            }
462            ComplianceStatus::NonCompliant { failures } => {
463                format!("NON-COMPLIANT ({} failures)", failures.len())
464            }
465            ComplianceStatus::InsufficientData => "INSUFFICIENT DATA".to_string(),
466        }
467    }
468
469    fn format_test_result(result: &crate::validation::cavp::compliance::TestResult) -> String {
470        match result {
471            crate::validation::cavp::compliance::TestResult::Passed => "PASSED".to_string(),
472            crate::validation::cavp::compliance::TestResult::Failed(reason) => {
473                format!("FAILED - {}", reason)
474            }
475            crate::validation::cavp::compliance::TestResult::Skipped(reason) => {
476                format!("SKIPPED - {}", reason)
477            }
478            crate::validation::cavp::compliance::TestResult::Error(reason) => {
479                format!("ERROR - {}", reason)
480            }
481        }
482    }
