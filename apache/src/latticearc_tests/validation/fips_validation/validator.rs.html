40    /// Validate cryptographic module against FIPS requirements.
41    ///
42    /// # Errors
43    /// Returns an error if algorithm, interface, or security policy validation fails.
44    pub fn validate_module(&self) -> Result<ValidationResult, LatticeArcError> {
45        let validation_id = format!("fips-val-{}", uuid::Uuid::new_v4());
46        let start_time = std::time::Instant::now();
47
48        tracing::info!("Starting FIPS validation: {}", validation_id);
49
50        let mut issues = Vec::new();
51        let mut test_results = HashMap::new();
52
53        // Run validation tests based on scope
54        match self.scope {
55            ValidationScope::AlgorithmsOnly => {
56                Self::validate_algorithms(&mut issues, &mut test_results)?;
57            }
58            ValidationScope::ModuleInterfaces => {
59                Self::validate_algorithms(&mut issues, &mut test_results)?;
60                Self::validate_interfaces(&mut issues, &mut test_results)?;
61            }
62            ValidationScope::FullModule => {
63                Self::validate_algorithms(&mut issues, &mut test_results)?;
64                Self::validate_interfaces(&mut issues, &mut test_results)?;
65                Self::validate_security_policy(&mut issues, &mut test_results)?;
66            }
67        }
68
69        let duration = start_time.elapsed().as_millis() as u64;
70
71        // Determine overall validation status
72        let is_valid = issues.iter().all(|i| i.severity != IssueSeverity::Critical);
73        let level = Self::determine_security_level(&issues, &test_results);
74
75        let result = ValidationResult {
76            validation_id,
77            timestamp: Utc::now(),
78            scope: self.scope,
79            is_valid,
80            level,
81            issues,
82            test_results: test_results.clone(),
83            metadata: HashMap::from([
84                ("validation_duration_ms".to_string(), duration.to_string()),
85                ("tests_run".to_string(), test_results.len().to_string()),
86            ]),
87        };
88
89        tracing::info!(
90            "FIPS validation completed in {}ms, valid: {}, level: {:?}",
91            duration,
92            result.is_valid,
93            result.level
94        );
95
96        Ok(result)
97    }

99    /// Generate validation certificate for successful validation.
100    ///
101    /// # Errors
102    /// Returns an error if validation failed or no security level was achieved.
103    pub fn generate_certificate(
104        &self,
105        result: &ValidationResult,
106    ) -> Result<ValidationCertificate, LatticeArcError> {
107        if !result.is_valid || result.level.is_none() {
108            return Err(LatticeArcError::InvalidInput(
109                "Validation failed or no security level achieved".to_string(),
110            ));
111        }
112
113        let level =
114            result.level.ok_or(LatticeArcError::InvalidInput("No security level".to_string()))?;
115
116        Ok(ValidationCertificate {
117            id: format!("fips-cert-{}", uuid::Uuid::new_v4()),
118            module_name: "LatticeArc Core".to_string(),
119            module_version: env!("CARGO_PKG_VERSION").to_string(),
120            security_level: level,
121            validation_date: result.timestamp,
122            expiry_date: result.timestamp + chrono::Duration::days(365),
123            lab_id: "latticearc-lab".to_string(),
124            details: HashMap::from([
125                ("validation_id".to_string(), result.validation_id.clone()),
126                ("scope".to_string(), format!("{:?}", result.scope)),
127                ("issues_found".to_string(), result.issues.len().to_string()),
128            ]),
129        })
130    }

132    /// Get remediation guidance for validation issues
133    #[must_use]
134    pub fn get_remediation_guidance(&self, result: &ValidationResult) -> Vec<String> {
135        let mut guidance = Vec::new();
136
137        for issue in &result.issues {
138            guidance
139                .push(format!("{} ({}): {}", issue.id, issue.requirement_ref, issue.remediation));
140        }
141
142        if guidance.is_empty() {
143            guidance.push("No remediation required - all tests passed".to_string());
144        }
145
146        guidance
147    }

149    /// Validate cryptographic algorithms
150    fn validate_algorithms(
151        issues: &mut Vec<ValidationIssue>,
152        test_results: &mut HashMap<String, TestResult>,
153    ) -> Result<(), LatticeArcError> {
154        // AES validation
155        let aes_result = test_aes_algorithm()?;
156        test_results.insert("aes_validation".to_string(), aes_result.clone());
157
158        if !aes_result.passed {
159            issues.push(ValidationIssue {
160                id: "FIPS-AES-001".to_string(),
161                description: "AES algorithm validation failed".to_string(),
162                requirement_ref: "FIPS 197".to_string(),
163                severity: IssueSeverity::Critical,
164                affected_component: "AES implementation".to_string(),
165                remediation: "Fix AES implementation to comply with FIPS 197".to_string(),
166                evidence: aes_result.output,
167            });
168        }
169
170        // SHA-3 validation
171        let sha3_result = test_sha3_algorithm()?;
172        test_results.insert("sha3_validation".to_string(), sha3_result.clone());
173
174        if !sha3_result.passed {
175            issues.push(ValidationIssue {
176                id: "FIPS-SHA3-001".to_string(),
177                description: "SHA-3 algorithm validation failed".to_string(),
178                requirement_ref: "FIPS 202".to_string(),
179                severity: IssueSeverity::Critical,
180                affected_component: "SHA-3 implementation".to_string(),
181                remediation: "Fix SHA-3 implementation to comply with FIPS 202".to_string(),
182                evidence: sha3_result.output,
183            });
184        }
185
186        // ML-KEM validation (FIPS 203)
187        let mlkem_result = test_mlkem_algorithm()?;
188        test_results.insert("mlkem_validation".to_string(), mlkem_result.clone());
189
190        if !mlkem_result.passed {
191            issues.push(ValidationIssue {
192                id: "FIPS-MLKEM-001".to_string(),
193                description: "ML-KEM algorithm validation failed".to_string(),
194                requirement_ref: "FIPS 203".to_string(),
195                severity: IssueSeverity::Critical,
196                affected_component: "ML-KEM implementation".to_string(),
197                remediation: "Fix ML-KEM implementation to comply with FIPS 203".to_string(),
198                evidence: mlkem_result.output,
199            });
200        }
201
202        Ok(())
203    }

246    fn validate_security_policy(
247        issues: &mut Vec<ValidationIssue>,
248        test_results: &mut HashMap<String, TestResult>,
249    ) -> Result<(), LatticeArcError> {
250        // Test self-tests

258                requirement_ref: "FIPS 140-3 Section 7".to_string(),
259                severity: IssueSeverity::Critical,
260                affected_component: "Self-test mechanisms".to_string(),
261                remediation: "Implement proper power-up and conditional self-tests".to_string(),
262                evidence: selftest_result.output,
263            });
264        }
265
266        // Test error handling
267        let error_result = test_error_handling()?;
268        test_results.insert("error_handling".to_string(), error_result.clone());

271            issues.push(ValidationIssue {

276                affected_component: "Error handling".to_string(),
277                remediation: "Improve error handling to prevent information leakage".to_string(),
278                evidence: error_result.output,
279            });
280        }
281
282        Ok(())
283    }

285    /// Determine achievable FIPS security level based on test results and issues
286    ///
287    /// FIPS 140-3 Security Level Determination:
288    /// - Level 1: Basic security requirements met, algorithm validation passed
289    /// - Level 2: Tamper-evident features + all Level 1 requirements
290    /// - Level 3: Tamper-resistant features + all Level 2 requirements
291    /// - Level 4: Tamper-resistant + environmental failure protection + all Level 3 requirements
292    fn determine_security_level(
293        issues: &[ValidationIssue],
294        test_results: &HashMap<String, TestResult>,
295    ) -> Option<FIPSLevel> {
296        let has_critical = issues.iter().any(|i| i.severity == IssueSeverity::Critical);
297        let has_high = issues.iter().any(|i| i.severity == IssueSeverity::High);
298        let has_medium = issues.iter().any(|i| i.severity == IssueSeverity::Medium);
299
300        // Algorithm validation tests must all pass for any FIPS level.
301        // Non-algorithm tests (error_handling, key_management, api_interfaces)
302        // generate issues with appropriate severity but don't block Level 1.
303        let algorithm_test_keys = ["aes_validation", "sha3_validation", "mlkem_validation"];
304        let all_algorithm_tests_passed =
305            algorithm_test_keys.iter().all(|key| test_results.get(*key).is_none_or(|r| r.passed));
306
307        // Cannot achieve any FIPS level with critical issues or algorithm failures
308        if has_critical || !all_algorithm_tests_passed {
309            return None;
310        }
311
312        // Software-only modules are always Level 1 per FIPS 140-3.
313        // Level 2+ requires tamper-evident physical security mechanisms
314        // (e.g., HSM, TPM, tamper-evident coatings) which software cannot provide.
315        let _ = (has_high, has_medium); // Acknowledged but irrelevant for sw-only
316        Some(FIPSLevel::Level1)
317    }

383                        eprintln!("    Error: {}", err);

385                }
386            }
387        }

390        assert!(result.level.is_some());
