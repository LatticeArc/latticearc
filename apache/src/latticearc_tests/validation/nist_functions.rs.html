17//! # NIST SP 800-106 Randomized Hashing for Digital Signatures

27/// Randomized hashing modes
28#[derive(Clone, Debug, PartialEq)]
29pub enum RandomizedHashMode {
30    /// No randomization (standard hashing)
31    None,
32    /// Random salt prepended to message
33    SaltPrefix,
34    /// Random salt appended to message
35    SaltSuffix,
36    /// Random salt inserted at multiple positions
37    SaltDistributed,
38}

40/// Randomized hash configuration
41#[derive(Clone, Debug)]
42pub struct RandomizedHashConfig {
43    /// Hash algorithm to use
44    pub algorithm: String,
45    /// Randomization mode
46    pub mode: RandomizedHashMode,
47    /// Salt length in bytes
48    pub salt_length: usize,
49    /// Number of salt insertions (for distributed mode)
50    pub salt_insertions: usize,
51}

53impl Default for RandomizedHashConfig {
54    fn default() -> Self {
55        Self {
56            algorithm: "SHA-256".to_string(),
57            mode: RandomizedHashMode::SaltPrefix,
58            salt_length: 32,
59            salt_insertions: 3,
60        }
61    }

64/// Randomized hash generator
65pub struct RandomizedHasher {
66    config: RandomizedHashConfig,
67}

69impl RandomizedHasher {
70    /// Create a new randomized hasher
71    #[must_use]
72    pub fn new(config: RandomizedHashConfig) -> Self {
73        Self { config }
74    }
75
76    /// Create hasher with default configuration
77    #[must_use]
78    pub fn default() -> Self {
79        Self::new(RandomizedHashConfig::default())
80    }

82    /// Generate randomized hash of a message.
83    ///
84    /// # Errors
85    /// Returns an error if salt generation fails or the hash algorithm is unsupported.
86    pub fn hash(&self, message: &[u8]) -> Result<RandomizedHash> {
87        let salt = self.generate_salt()?;
88        let randomized_message = self.apply_randomization(message, &salt)?;
89
90        let hash = self.compute_hash(&randomized_message)?;
91
92        Ok(RandomizedHash {
93            hash,
94            salt,
95            algorithm: self.config.algorithm.clone(),
96            mode: self.config.mode.clone(),
97        })
98    }
99
100    /// Verify a randomized hash.
101    ///
102    /// # Errors
103    /// Returns an error if the hash algorithm is unsupported or hash computation fails.
104    pub fn verify(&self, message: &[u8], randomized_hash: &RandomizedHash) -> Result<bool> {
105        if randomized_hash.algorithm != self.config.algorithm
106            || randomized_hash.mode != self.config.mode
107        {
108            return Ok(false);
109        }
110
111        // Use the salt from the randomized_hash to re-compute
112        let randomized_message = self.apply_randomization(message, &randomized_hash.salt)?;
113        let computed_hash = self.compute_hash(&randomized_message)?;
114
115        Ok(computed_hash == randomized_hash.hash)
116    }
117
118    /// Compute hash for the given data using the configured algorithm
119    fn compute_hash(&self, data: &[u8]) -> Result<Vec<u8>> {
120        match self.config.algorithm.as_str() {
121            "SHA-256" => {
122                let mut hasher = Sha256::new();
123                hasher.update(data);
124                Ok(hasher.finalize().to_vec())
125            }
126            "SHA-384" => {
127                let mut hasher = Sha384::new();
128                hasher.update(data);
129                Ok(hasher.finalize().to_vec())
130            }
131            "SHA-512" => {
132                let mut hasher = Sha512::new();
133                hasher.update(data);
134                Ok(hasher.finalize().to_vec())
135            }
136            _ => Err(LatticeArcError::InvalidKey(format!(
137                "Unsupported hash algorithm: {}",
138                self.config.algorithm
139            ))),
140        }
141    }
142
143    /// Generate random salt
144    fn generate_salt(&self) -> Result<Vec<u8>> {
145        let mut salt = vec![0u8; self.config.salt_length];
146        if !salt.is_empty() {
147            rand::thread_rng().fill_bytes(&mut salt);
148
149            // Ensure salt is not all zeros (basic sanity check)
150            if salt.iter().all(|&x| x == 0)
151                && let Some(first_byte) = salt.get_mut(0)
152            {
153                *first_byte = 1; // Force at least one non-zero byte
154            }
155        }
156
157        Ok(salt)
158    }
159
160    /// Apply randomization to message based on mode
161    fn apply_randomization(&self, message: &[u8], salt: &[u8]) -> Result<Vec<u8>> {
162        match self.config.mode {
163            RandomizedHashMode::None => Ok(message.to_vec()),
164            RandomizedHashMode::SaltPrefix => {
165                let mut result = Vec::with_capacity(salt.len() + message.len());
166                result.extend_from_slice(salt);
167                result.extend_from_slice(message);
168                Ok(result)
169            }
170            RandomizedHashMode::SaltSuffix => {
171                let mut result = Vec::with_capacity(message.len() + salt.len());
172                result.extend_from_slice(message);
173                result.extend_from_slice(salt);
174                Ok(result)
175            }
176            RandomizedHashMode::SaltDistributed => self.apply_distributed_salt(message, salt),
177        }
178    }
179
180    /// Apply distributed salt randomization
181    fn apply_distributed_salt(&self, message: &[u8], salt: &[u8]) -> Result<Vec<u8>> {
182        if message.is_empty() {
183            return Ok(salt.to_vec());
184        }
185
186        let mut result = Vec::new();
187        let salt_chunks = self.split_salt_into_chunks(salt)?;
188
189        // Distribute salt chunks throughout the message
190        let message_parts = self.split_message_into_parts(message);
191
192        for (i, part) in message_parts.iter().enumerate() {
193            result.extend_from_slice(part);
194
195            // Insert salt chunk if available
196            if let Some(salt_chunk) = salt_chunks.get(i) {
197                result.extend_from_slice(salt_chunk);
198            }
199        }
200
201        Ok(result)
202    }
203
204    /// Split salt into chunks for distributed mode
205    fn split_salt_into_chunks(&self, salt: &[u8]) -> Result<Vec<Vec<u8>>> {
206        let chunk_size = std::cmp::max(1, salt.len() / self.config.salt_insertions);
207        let mut chunks = Vec::new();
208
209        for i in 0..self.config.salt_insertions {
210            let start = i * chunk_size;
211            let end = std::cmp::min(start + chunk_size, salt.len());
212            if start < salt.len() {
213                chunks.push(salt[start..end].to_vec());
214            }
215        }
216
217        // Add any remaining salt to the last chunk
218        if chunks.len() < self.config.salt_insertions && !salt.is_empty() {
219            let remaining_start = chunks.len() * chunk_size;
220            if remaining_start < salt.len() {
221                chunks.push(salt[remaining_start..].to_vec());
222            }
223        }
224
225        Ok(chunks)
226    }
227
228    /// Split message into parts for distributed salt insertion
229    fn split_message_into_parts(&self, message: &[u8]) -> Vec<Vec<u8>> {
230        if message.is_empty() {
231            return vec![vec![]];
232        }
233
234        let num_parts = self.config.salt_insertions + 1;
235        let part_size = message.len() / num_parts;
236        let mut parts = Vec::new();
237        let mut start = 0;
238
239        for i in 0..num_parts {
240            let end = if i == num_parts - 1 { message.len() } else { start + part_size };
241
242            parts.push(message[start..end].to_vec());
243            start = end;
244        }
245
246        parts
247    }
248}

250/// Randomized hash result
251#[derive(Clone, Debug, PartialEq)]
252pub struct RandomizedHash {
253    /// The computed hash value
254    pub hash: Vec<u8>,
255    /// The salt used for randomization
256    pub salt: Vec<u8>,
257    /// Hash algorithm used
258    pub algorithm: String,
259    /// Randomization mode used
260    pub mode: RandomizedHashMode,
261}
262
263impl RandomizedHash {
264    /// Get hash as hex string
265    #[must_use]
266    pub fn hash_hex(&self) -> String {
267        hex::encode(&self.hash)
268    }
269
270    /// Get salt as hex string
271    #[must_use]
272    pub fn salt_hex(&self) -> String {
273        hex::encode(&self.salt)
274    }
275}

277/// Convenience functions for randomized hashing
278pub struct RandomizedHashing;
279
280impl RandomizedHashing {
281    /// Create a randomized hash with default settings.
282    ///
283    /// # Errors
284    /// Returns an error if salt generation fails or hash computation fails.
285    pub fn hash_message(message: &[u8]) -> Result<RandomizedHash> {
286        let hasher = RandomizedHasher::default();
287        hasher.hash(message)
288    }
289
290    /// Create a randomized hash with custom configuration.
291    ///
292    /// # Errors
293    /// Returns an error if salt generation fails or the hash algorithm is unsupported.
294    pub fn hash_message_with_config(
295        message: &[u8],
296        config: RandomizedHashConfig,
297    ) -> Result<RandomizedHash> {
298        let hasher = RandomizedHasher::new(config);
299        hasher.hash(message)
300    }
301
302    /// Verify a randomized hash.
303    ///
304    /// # Errors
305    /// Returns an error if the hash algorithm is unsupported or hash computation fails.
306    pub fn verify_hash(message: &[u8], hash: &RandomizedHash) -> Result<bool> {
307        let config = RandomizedHashConfig {
308            algorithm: hash.algorithm.clone(),
309            mode: hash.mode.clone(),
310            salt_length: hash.salt.len(),
311            salt_insertions: 3, // Default value
312        };
313
314        let hasher = RandomizedHasher::new(config);
315        hasher.verify(message, hash)
316    }
317
318    /// Get recommended configuration for different security levels
319    #[must_use]
320    pub fn recommended_config(security_level: usize) -> RandomizedHashConfig {
321        match security_level {
322            128 => RandomizedHashConfig {
323                algorithm: "SHA-256".to_string(),
324                mode: RandomizedHashMode::SaltPrefix,
325                salt_length: 16,
326                salt_insertions: 2,
327            },
328            192 => RandomizedHashConfig {
329                algorithm: "SHA-384".to_string(),
330                mode: RandomizedHashMode::SaltDistributed,
331                salt_length: 24,
332                salt_insertions: 3,
333            },
334            256 => RandomizedHashConfig {
335                algorithm: "SHA-512".to_string(),
336                mode: RandomizedHashMode::SaltDistributed,
337                salt_length: 32,
338                salt_insertions: 4,
339            },
340            _ => RandomizedHashConfig::default(),
341        }
342    }

351    fn test_randomized_hash_basic() {
352        let message = b"Hello, randomized hashing!";
353        let hash1 = RandomizedHashing::hash_message(message).unwrap();
354        let hash2 = RandomizedHashing::hash_message(message).unwrap();
355
356        // Different salts should produce different hashes
357        assert_ne!(hash1.hash, hash2.hash);
358        assert_ne!(hash1.salt, hash2.salt);
359
360        // But verification should work
361        assert!(RandomizedHashing::verify_hash(message, &hash1).unwrap());
362        assert!(RandomizedHashing::verify_hash(message, &hash2).unwrap());
363    }
