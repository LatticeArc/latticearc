1//! # arc-tests
2//!
3//! Comprehensive test suite for LatticeArc providing:
4//!
5//! - **Regression Tests**: Prevent reintroduction of fixed bugs
6//! - **API Stability Tests**: Ensure backward compatibility across versions
7//! - **Concurrency Tests**: Verify thread-safe operation
8//!
9//! ## Running Tests
10//!
11//! ```bash
12//! # Run all tests
13//! cargo test -p arc-tests
14//!
15//! # Run specific category
16//! cargo test -p arc-tests regression
17//! cargo test -p arc-tests api_stability
18//! cargo test -p arc-tests concurrency
19//! ```
20//!
21//! ## Adding Regression Tests
22//!
23//! Each regression test should:
24//! 1. Be in its own file: `src/regression/issue_NNN_description.rs`
25//! 2. Include a doc comment linking to the original issue
26//! 3. Reproduce the exact bug condition
27//! 4. Verify the fix works
28//!
29//! ## Test Naming Convention
30//!
31//! - Regression: `regression_issue_NNN_description`
32//! - API Stability: `api_stability_<aspect>_<test>`
33//! - Concurrency: `concurrent_<operation>_<scenario>`

43/// Shared test utilities
44pub mod utils {
45    use rand::rngs::OsRng;
46
47    /// Get a cryptographically secure RNG for tests
48    #[must_use]
49    pub fn test_rng() -> OsRng {
50        OsRng
51    }
52
53    /// Assert two byte slices are equal with descriptive message
54    ///
55    /// # Panics
56    ///
57    /// Panics if `left` and `right` are not equal, displaying the provided `context`.
58    pub fn assert_bytes_eq(left: &[u8], right: &[u8], context: &str) {
59        assert_eq!(left, right, "{}: byte slices differ at first mismatch", context);
60    }
61
62    /// Assert two byte slices are NOT equal
63    ///
64    /// # Panics
65    ///
66    /// Panics if `left` and `right` are equal, displaying the provided `context`.
67    pub fn assert_bytes_ne(left: &[u8], right: &[u8], context: &str) {
68        assert_ne!(left, right, "{}: byte slices should differ", context);
69    }
70}

73mod tests {

76    #[test]
77    fn test_rng_returns_osrng() {
78        use rand::RngCore;
79        let mut rng = utils::test_rng();
80        let mut buf = [0u8; 16];
81        rng.fill_bytes(&mut buf);
82        // Should produce non-zero random bytes
83        assert!(buf.iter().any(|&b| b != 0), "RNG should produce non-zero output");
84    }

86    #[test]

89        utils::assert_bytes_eq(&[], &[], "empty slices");
90        utils::assert_bytes_eq(&[1, 2, 3], &[1, 2, 3], "numeric slices");
91    }
