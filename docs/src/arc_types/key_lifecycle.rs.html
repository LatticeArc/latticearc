1//! SP 800-57 Key Lifecycle Management
2//!
3//! This module implements formal key lifecycle management per NIST SP 800-57
4//! requirements, including state transitions, custodianship, and audit trails.
5//!
6//! # Key States (SP 800-57 Section 3)
7//!
8//! - **Generation**: Key material is being generated
9//! - **Active**: Key is ready for use
10//! - **Rotating**: Key rotation in progress (overlap period)
11//! - **Retired**: Key scheduled for retirement
12//! - **Destroyed**: Key material zeroized
13//!
14//! # Example
15//!
16//! ```
17//! use arc_types::key_lifecycle::{KeyLifecycleRecord, KeyLifecycleState};
18//!
19//! let mut record = KeyLifecycleRecord::new(
20//!     "key-123".to_string(),
21//!     "ML-KEM-768".to_string(),
22//!     3,   // security level
23//!     365, // rotation interval (days)
24//!     30,  // overlap period (days)
25//! );
26//!
27//! // Activate the key
28//! record.transition(
29//!     KeyLifecycleState::Active,
30//!     "alice".to_string(),
31//!     "Key generation complete".to_string(),
32//!     Some("approval-123".to_string()),
33//! ).expect("Valid transition");
34//!
35//! assert!(record.is_valid_for_use());
36//! ```
37

41/// SP 800-57 Section 3: Key Lifecycle States
42#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
43#[cfg_attr(kani, derive(kani::Arbitrary))]
44pub enum KeyLifecycleState {
45    /// Key material generation
46    Generation,
47    /// Key activation - ready for use
48    Active,
49    /// Key rotation in progress (overlap period)
50    Rotating,
51    /// Key scheduled for retirement
52    Retired,
53    /// Key destruction - material zeroized
54    Destroyed,
55}
56
57/// SP 800-57 Key State Transitions (formally verifiable)
58pub struct KeyStateMachine;
59
60impl KeyStateMachine {
61    /// Verify state transition is valid per SP 800-57
62    ///
63    /// Valid transitions:
64    /// - None -> Generation (initial state)
65    /// - Generation -> Active (initialization complete)
66    /// - Active -> Rotating (rotation initiated)
67    /// - Active -> Retired (direct retirement)
68    /// - Rotating -> Retired (rotation complete)
69    /// - Retired -> Destroyed (cleanup)
70    #[must_use]
71    pub fn is_valid_transition(from: Option<KeyLifecycleState>, to: KeyLifecycleState) -> bool {
72        match (from, to) {
73            // Generation is always valid initial state
74            (None, KeyLifecycleState::Generation) => true,
75
76            // Generation -> Active (initialization complete)
77            (Some(KeyLifecycleState::Generation), KeyLifecycleState::Active) => true,
78
79            // Active -> Rotating (rotation initiated)
80            (Some(KeyLifecycleState::Active), KeyLifecycleState::Rotating) => true,
81
82            // Rotating -> Retired (rotation complete)
83            (Some(KeyLifecycleState::Rotating), KeyLifecycleState::Retired) => true,
84
85            // Active -> Retired (direct retirement)
86            (Some(KeyLifecycleState::Active), KeyLifecycleState::Retired) => true,
87
88            // Retired -> Destroyed (cleanup)
89            (Some(KeyLifecycleState::Retired), KeyLifecycleState::Destroyed) => true,
90
91            // All other transitions are invalid
92            _ => false,
93        }
94    }
95
96    /// Get allowed next states from current state
97    #[must_use]
98    pub fn allowed_next_states(current: KeyLifecycleState) -> Vec<KeyLifecycleState> {
99        match current {
100            KeyLifecycleState::Generation => vec![KeyLifecycleState::Active],
101            KeyLifecycleState::Active => {
102                vec![KeyLifecycleState::Rotating, KeyLifecycleState::Retired]
103            }
104            KeyLifecycleState::Rotating => vec![KeyLifecycleState::Retired],
105            KeyLifecycleState::Retired => vec![KeyLifecycleState::Destroyed],
106            KeyLifecycleState::Destroyed => vec![],
107        }
108    }
109}

139/// Key lifecycle record with audit trail

141pub struct KeyLifecycleRecord {

143    pub key_id: String,
144    /// Algorithm/key type (e.g., "ML-KEM-768", "ML-DSA-65")
145    pub key_type: String,
146    /// Security level (1-5)

148
149    // State management

151    pub current_state: KeyLifecycleState,
152    /// History of state transitions
153    pub state_history: Vec<StateTransition>,
154
155    // Custodianship
156    /// ID of the key generator
157    pub generator: Option<String>,
158    /// IDs of approvers
159    pub approvers: Vec<String>,
160    /// ID of the destroyer
161    pub destroyer: Option<String>,
162
163    // Timing

166    /// When the key was activated
167    pub activated_at: Option<chrono::DateTime<chrono::Utc>>,
168    /// When key rotation was initiated
169    pub rotated_at: Option<chrono::DateTime<chrono::Utc>>,
170    /// When the key was retired
171    pub retired_at: Option<chrono::DateTime<chrono::Utc>>,
172    /// When the key was destroyed
173    pub destroyed_at: Option<chrono::DateTime<chrono::Utc>>,
174
175    // SP 800-57 requirements
176    /// How often the key should be rotated (days)
177    pub rotation_interval_days: u32,
178    /// Overlap period during rotation (days)
179    pub overlap_period_days: u32,
180}

182/// Record of a state transition
183#[derive(Debug, Clone, Serialize, Deserialize)]
184pub struct StateTransition {
185    /// Previous state (None if initial)
186    pub from_state: Option<KeyLifecycleState>,
187    /// New state
188    pub to_state: KeyLifecycleState,
189    /// When the transition occurred
190    pub timestamp: chrono::DateTime<chrono::Utc>,
191    /// ID of the custodian who performed the transition
192    pub custodian_id: String,
193    /// Reason for the transition
194    pub justification: String,
195    /// Approval reference (if applicable)
196    pub approval_id: Option<String>,
197}
198
199impl KeyLifecycleRecord {
200    /// Create new key lifecycle record
201    ///
202    /// # Arguments
203    ///
204    /// * `key_id` - Unique identifier for the key
205    /// * `key_type` - Algorithm/key type (e.g., "ML-KEM-768")
206    /// * `security_level` - Security level (1-5)
207    /// * `rotation_interval_days` - How often to rotate the key
208    /// * `overlap_period_days` - Overlap period during rotation
209    #[must_use]
210    pub fn new(
211        key_id: String,
212        key_type: String,
213        security_level: u32,
214        rotation_interval_days: u32,
215        overlap_period_days: u32,
216    ) -> Self {
217        Self {
218            key_id,
219            key_type,
220            security_level,
221            current_state: KeyLifecycleState::Generation,
222            state_history: Vec::new(),
223            generator: None,
224            approvers: Vec::new(),
225            destroyer: None,
226            generated_at: chrono::Utc::now(),
227            activated_at: None,
228            rotated_at: None,
229            retired_at: None,
230            destroyed_at: None,
231            rotation_interval_days,
232            overlap_period_days,
233        }
234    }
235
236    /// Transition key to new state with custodianship tracking
237    ///
238    /// # Arguments
239    ///
240    /// * `to_state` - Target state
241    /// * `custodian_id` - ID of the custodian performing the transition
242    /// * `justification` - Reason for the transition
243    /// * `approval_id` - Optional approval reference
244    ///
245    /// # Errors
246    ///
247    /// Returns `TypeError::InvalidStateTransition` if the transition is invalid
248    pub fn transition(
249        &mut self,
250        to_state: KeyLifecycleState,
251        custodian_id: String,
252        justification: String,
253        approval_id: Option<String>,
254    ) -> Result<()> {
255        if !KeyStateMachine::is_valid_transition(Some(self.current_state), to_state) {
256            return Err(TypeError::InvalidStateTransition {
257                from: self.current_state,
258                to: to_state,
259            });
260        }
261        let transition = StateTransition {
262            from_state: Some(self.current_state),
263            to_state,
264            timestamp: chrono::Utc::now(),
265            custodian_id: custodian_id.clone(),
266            justification,
267            approval_id,
268        };
269
270        self.state_history.push(transition);
271        self.current_state = to_state;
272
273        // Update timestamps
274        match to_state {
275            KeyLifecycleState::Active => self.activated_at = Some(chrono::Utc::now()),
276            KeyLifecycleState::Rotating => self.rotated_at = Some(chrono::Utc::now()),
277            KeyLifecycleState::Retired => self.retired_at = Some(chrono::Utc::now()),
278            KeyLifecycleState::Destroyed => self.destroyed_at = Some(chrono::Utc::now()),
279            _ => {}
280        }
281
282        // Update custodianship
283        // Generator is the custodian who completes key generation (moves to Active)
284        // Check the last transition we just added to see if it was from Generation
285        if let Some(last_transition) = self.state_history.last()
286            && last_transition.from_state == Some(KeyLifecycleState::Generation)
287            && to_state == KeyLifecycleState::Active
288        {
289            self.generator = Some(custodian_id.clone());
290        }
291        if to_state == KeyLifecycleState::Destroyed {
292            self.destroyer = Some(custodian_id);
293        }
294
295        Ok(())
296    }
297
298    /// Check if key is due for rotation per SP 800-57
299    #[must_use]
300    pub fn requires_rotation(&self) -> bool {
301        if let Some(activated_at) = self.activated_at {
302            // Use signed_duration_since for safe duration calculation
303            let duration = chrono::Utc::now().signed_duration_since(activated_at);
304            let age_days_i64 = duration.num_days();
305            // Convert safely: negative ages (future activation) treated as 0
306            // Ages larger than u32::MAX are capped (would be ~11.7M years)
307            let age_days = u32::try_from(age_days_i64).unwrap_or(0);
308            age_days >= self.rotation_interval_days
309        } else {
310            false
311        }
312    }
313
314    /// Get key age in days since activation
315    #[must_use]
316    pub fn age_days(&self) -> Option<u32> {
317        self.activated_at.map(|activated| {
318            // Use signed_duration_since for safe duration calculation
319            let duration = chrono::Utc::now().signed_duration_since(activated);
320            let days_i64 = duration.num_days();
321            // Convert safely: negative ages (future activation) treated as 0
322            u32::try_from(days_i64).unwrap_or(0)
323        })
324    }
325
326    /// Check if key is in valid state for use
327    #[must_use]
328    pub fn is_valid_for_use(&self) -> bool {
329        matches!(self.current_state, KeyLifecycleState::Active | KeyLifecycleState::Rotating)
330    }
331
332    /// Get the number of state transitions
333    #[must_use]
334    pub fn transition_count(&self) -> usize {
335        self.state_history.len()
336    }
337
338    /// Add an approver to the key
339    pub fn add_approver(&mut self, approver_id: impl Into<String>) {
340        let approver_id = approver_id.into();
341        if !self.approvers.contains(&approver_id) {
342            self.approvers.push(approver_id);
343        }
344    }
345}

349mod kani_proofs {

352    // --- Existing proofs ---
353
354    #[kani::proof]
355    fn key_state_machine_destroyed_cannot_transition() {
356        let to: KeyLifecycleState = kani::any();
357
358        // Property: Destroyed keys cannot transition to any state
359        let is_valid = KeyStateMachine::is_valid_transition(Some(KeyLifecycleState::Destroyed), to);
360        kani::assert(!is_valid, "Destroyed keys should not transition");
361    }
362
363    #[kani::proof]
364    fn key_state_machine_no_backward_to_generation() {
365        let from: KeyLifecycleState = kani::any();
366
367        // Property: Cannot go back to Generation from any state
368        kani::assume(from != KeyLifecycleState::Generation);
369        let is_valid =
370            KeyStateMachine::is_valid_transition(Some(from), KeyLifecycleState::Generation);
371        kani::assert(!is_valid, "Cannot transition back to Generation");
372    }
373
374    // --- New proofs ---
375
376    /// Proves that the only valid initial state is Generation.
377    /// Security property: keys must begin in the Generation state.
378    #[kani::proof]
379    fn key_state_machine_only_generation_from_none() {
380        let to: KeyLifecycleState = kani::any();
381        kani::assume(to != KeyLifecycleState::Generation);
382
383        let is_valid = KeyStateMachine::is_valid_transition(None, to);
384        kani::assert(!is_valid, "Only Generation is valid from initial state (None)");
385    }
386
387    /// Proves that `is_valid_transition` matches the SP 800-57 transition
388    /// specification for all state pairs. This is an independent encoding of
389    /// the allowed transitions â€” if `is_valid_transition` has a bug, this
390    /// proof will catch the discrepancy.
391    ///
392    /// Note: `allowed_next_states()` returns `Vec`, which Kani cannot
393    /// efficiently verify (unbounded loop unwinding on heap iteration).
394    /// Unit tests verify `allowed_next_states()` separately.
395    #[kani::proof]
396    fn key_state_machine_transitions_match_spec() {
397        let from: KeyLifecycleState = kani::any();
398        let to: KeyLifecycleState = kani::any();
399
400        let transition_valid = KeyStateMachine::is_valid_transition(Some(from), to);
401
402        // Independent specification of SP 800-57 allowed transitions
403        let spec_allows = match from {
404            KeyLifecycleState::Generation => to == KeyLifecycleState::Active,
405            KeyLifecycleState::Active => {
406                to == KeyLifecycleState::Rotating || to == KeyLifecycleState::Retired
407            }
408            KeyLifecycleState::Rotating => to == KeyLifecycleState::Retired,
409            KeyLifecycleState::Retired => to == KeyLifecycleState::Destroyed,
410            KeyLifecycleState::Destroyed => false,
411        };
412
413        kani::assert(
414            transition_valid == spec_allows,
415            "is_valid_transition must match SP 800-57 specification",
416        );
417    }
418
419    /// Proves that retired keys can only transition to Destroyed.
420    /// Security property: once retired, the only path is secure destruction.
421    #[kani::proof]
422    fn key_state_machine_retired_only_to_destroyed() {
423        let to: KeyLifecycleState = kani::any();
424        kani::assume(to != KeyLifecycleState::Destroyed);
425
426        let is_valid = KeyStateMachine::is_valid_transition(Some(KeyLifecycleState::Retired), to);
427        kani::assert(!is_valid, "Retired keys can only transition to Destroyed");
428    }
