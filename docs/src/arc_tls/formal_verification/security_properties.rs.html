4//! TLS security property definitions and verification.
5//!
6//! Defines the core security properties that a TLS implementation must guarantee:
7//! - **Confidentiality**: Data encrypted in transit
8//! - **Authentication**: Server (and optionally client) identity verified
9//! - **Forward secrecy**: Past sessions protected even if long-term keys compromised
10//! - **Integrity**: Data not modified in transit
11//! - **PQ resistance**: Quantum-safe key exchange (when hybrid/PQ mode enabled)
12
13use crate::{ClientVerificationMode, TlsConfig, TlsMode};
14
15/// A verifiable security property of a TLS configuration.
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17pub enum SecurityProperty {
18    /// Data is encrypted during transit using authenticated encryption (AES-GCM or ChaCha20-Poly1305).
19    Confidentiality,
20    /// Server identity is verified via X.509 certificate chain.
21    ServerAuthentication,
22    /// Client identity is verified via mTLS.
23    ClientAuthentication,
24    /// Ephemeral key exchange protects past sessions from future key compromise.
25    ForwardSecrecy,
26    /// AEAD ciphers provide integrity guarantees on all application data.
27    Integrity,
28    /// Key exchange is resistant to quantum computing attacks.
29    QuantumResistance,
30}
31
32/// Result of checking a single security property.
33#[derive(Debug, Clone)]
34pub struct PropertyCheck {
35    /// The property being checked.
36    pub property: SecurityProperty,
37    /// Whether the property is satisfied.
38    pub satisfied: bool,
39    /// Explanation of the check result.
40    pub reason: String,
41}
42
43/// Security property checker for TLS configurations.
44pub struct SecurityProperties {
45    /// Properties to check.
46    required: Vec<SecurityProperty>,
47}
48
49impl SecurityProperties {
50    /// Create a checker for the standard TLS security properties.
51    ///
52    /// Standard checks: Confidentiality, ServerAuthentication, ForwardSecrecy, Integrity.
53    #[must_use]
54    pub fn new() -> Self {
55        Self {
56            required: vec![
57                SecurityProperty::Confidentiality,
58                SecurityProperty::ServerAuthentication,
59                SecurityProperty::ForwardSecrecy,
60                SecurityProperty::Integrity,
61            ],
62        }
63    }
64
65    /// Create a checker requiring all security properties including PQ resistance.
66    #[must_use]
67    pub fn full() -> Self {
68        Self {
69            required: vec![
70                SecurityProperty::Confidentiality,
71                SecurityProperty::ServerAuthentication,
72                SecurityProperty::ClientAuthentication,
73                SecurityProperty::ForwardSecrecy,
74                SecurityProperty::Integrity,
75                SecurityProperty::QuantumResistance,
76            ],
77        }
78    }
79
80    /// Add a required property.
81    #[must_use]
82    pub fn require(mut self, property: SecurityProperty) -> Self {
83        if !self.required.contains(&property) {
84            self.required.push(property);
85        }
86        self
87    }
88
89    /// Check all required properties against a TLS configuration.
90    #[must_use]
91    pub fn verify(&self, config: &TlsConfig) -> Vec<PropertyCheck> {
92        self.required.iter().map(|prop| self.check_property(*prop, config)).collect()
93    }
94
95    /// Check whether all required properties are satisfied.
96    #[must_use]
97    pub fn all_satisfied(&self, config: &TlsConfig) -> bool {
98        self.verify(config).iter().all(|c| c.satisfied)
99    }
100
101    #[allow(clippy::unused_self)] // Method on struct for API consistency
102    fn check_property(&self, property: SecurityProperty, config: &TlsConfig) -> PropertyCheck {
103        match property {
104            SecurityProperty::Confidentiality => PropertyCheck {
105                property,
106                satisfied: true, // TLS 1.3 always uses AEAD
107                reason: "TLS 1.3 mandates authenticated encryption (AES-GCM/ChaCha20-Poly1305)"
108                    .to_string(),
109            },
110            SecurityProperty::ServerAuthentication => PropertyCheck {
111                property,
112                satisfied: true, // rustls always verifies server certificates
113                reason: "rustls verifies server certificate chain against system root store"
114                    .to_string(),
115            },
116            SecurityProperty::ClientAuthentication => {
117                let satisfied = config.client_verification != ClientVerificationMode::None;
118                PropertyCheck {
119                    property,
120                    satisfied,
121                    reason: if satisfied {
122                        format!("Client verification mode: {:?}", config.client_verification)
123                    } else {
124                        "Client authentication not configured (mTLS disabled)".to_string()
125                    },
126                }
127            }
128            SecurityProperty::ForwardSecrecy => PropertyCheck {
129                property,
130                satisfied: true, // TLS 1.3 mandates ephemeral key exchange
131                reason: "TLS 1.3 uses ephemeral key exchange (X25519/ML-KEM) for all sessions"
132                    .to_string(),
133            },
134            SecurityProperty::Integrity => PropertyCheck {
135                property,
136                satisfied: true, // AEAD provides integrity
137                reason: "AEAD cipher suites provide authenticated encryption with integrity"
138                    .to_string(),
139            },
140            SecurityProperty::QuantumResistance => {
141                let satisfied = config.mode != TlsMode::Classic;
142                PropertyCheck {
143                    property,
144                    satisfied,
145                    reason: if satisfied {
146                        format!("{:?} mode uses ML-KEM for quantum resistance", config.mode)
147                    } else {
148                        "Classic mode uses only classical key exchange (X25519)".to_string()
149                    },
150                }
151            }
152        }
153    }
154}
155
156impl Default for SecurityProperties {
157    fn default() -> Self {
158        Self::new()
159    }
160}

162#[cfg(test)]
163#[allow(clippy::expect_used)]
164mod tests {
165    use super::*;
166
167    #[test]
168    fn test_standard_properties_satisfied_by_default() {
169        let checker = SecurityProperties::new();
170        let config = TlsConfig::new();
171        assert!(checker.all_satisfied(&config));
172    }
173
174    #[test]
175    fn test_full_properties_fail_without_mtls() {
176        let checker = SecurityProperties::full();
177        let config = TlsConfig::new(); // no client auth
178        assert!(!checker.all_satisfied(&config));
179        let checks = checker.verify(&config);
180        let client_auth =
181            checks.iter().find(|c| c.property == SecurityProperty::ClientAuthentication);
182        assert!(client_auth.is_some());
183        assert!(!client_auth.expect("checked above").satisfied);
184    }
185
186    #[test]
187    fn test_quantum_resistance_fails_classic_mode() {
188        let checker = SecurityProperties::new().require(SecurityProperty::QuantumResistance);
189        let config = TlsConfig { mode: TlsMode::Classic, ..TlsConfig::default() };
190        assert!(!checker.all_satisfied(&config));
191    }
192
193    #[test]
194    fn test_quantum_resistance_passes_hybrid_mode() {
195        let checker = SecurityProperties::new().require(SecurityProperty::QuantumResistance);
196        let config = TlsConfig::new(); // default is Hybrid
197        assert!(checker.all_satisfied(&config));
198    }
199
200    #[test]
201    fn test_verify_returns_all_checks() {
202        let checker = SecurityProperties::new();
203        let config = TlsConfig::new();
204        let checks = checker.verify(&config);
205        assert_eq!(checks.len(), 4);
206    }
207
208    #[test]
209    fn test_require_adds_property() {
210        let checker = SecurityProperties::new().require(SecurityProperty::QuantumResistance);
211        let config = TlsConfig::new();
212        let checks = checker.verify(&config);
213        assert_eq!(checks.len(), 5);
214    }
215
216    #[test]
217    fn test_require_deduplicates() {
218        let checker = SecurityProperties::new().require(SecurityProperty::Confidentiality); // already included
219        let config = TlsConfig::new();
220        let checks = checker.verify(&config);
221        assert_eq!(checks.len(), 4); // should not add duplicate
222    }
223
224    #[test]
225    fn test_property_check_debug() {
226        let check = PropertyCheck {
227            property: SecurityProperty::Confidentiality,
228            satisfied: true,
229            reason: "test".to_string(),
