6//! SHA-3 Hash Functions
7//!
8//! This module provides SHA-3 implementations (SHA3-256, SHA3-384, SHA3-512).
9
10use sha3::{Digest, Sha3_256, Sha3_384, Sha3_512};
11
12/// SHA3-256 hash function
13#[must_use]
14pub fn sha3_256(data: &[u8]) -> [u8; 32] {
15    let mut hasher = Sha3_256::new();
16    hasher.update(data);
17    hasher.finalize().into()
18}
19
20/// SHA3-384 hash function
21#[must_use]
22pub fn sha3_384(data: &[u8]) -> [u8; 48] {
23    let mut hasher = Sha3_384::new();
24    hasher.update(data);
25    hasher.finalize().into()
26}
27
28/// SHA3-512 hash function
29#[must_use]
30pub fn sha3_512(data: &[u8]) -> [u8; 64] {
31    let mut hasher = Sha3_512::new();
32    hasher.update(data);
33    hasher.finalize().into()
34}
