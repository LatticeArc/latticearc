22/// Result of a KAT execution
23#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
24pub struct KatResult {
25    /// Test case identifier
26    pub test_case: String,
27    /// Whether the test passed
28    pub passed: bool,
29    /// Execution time in nanoseconds
30    pub execution_time_ns: u128,
31    /// Error message if test failed
32    pub error_message: Option<String>,
33}
34
35impl KatResult {
36    /// Create a new successful KAT result
37    #[must_use]
38    pub fn passed(test_case: String, execution_time: Duration) -> Self {
39        Self {
40            test_case,
41            passed: true,
42            execution_time_ns: execution_time.as_nanos(),
43            error_message: None,
44        }
45    }
46
47    /// Create a new failed KAT result
48    #[must_use]
49    pub fn failed(test_case: String, execution_time: Duration, error: String) -> Self {
50        Self {
51            test_case,
52            passed: false,
53            execution_time_ns: execution_time.as_nanos(),
54            error_message: Some(error),
55        }
56    }
57}

239/// Algorithm identifier for KAT testing
240#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
241pub enum AlgorithmType {
242    /// ML-KEM (Key Encapsulation Mechanism)
243    MlKem { variant: String },
244    /// ML-DSA (Digital Signature Algorithm)
245    MlDsa { variant: String },
246    /// SLH-DSA (Stateless Hash-Based Digital Signature Algorithm)
247    SlhDsa { variant: String },
248    /// Hybrid KEM
249    HybridKem,
250    /// AES-GCM (Authenticated Encryption)
251    AesGcm { key_size: usize },
252    /// SHA3 (Hash Function)
253    Sha3 { variant: String },
254    /// Ed25519 (Elliptic Curve Digital Signature Algorithm)
255    Ed25519,
256    /// BLS12-381 (Pairing-based Cryptography)
257    Bls12_381,
258    /// BN254 (Pairing-based Cryptography)
259    Bn254,
260    /// Secp256k1 (Elliptic Curve)
261    Secp256k1,
262}
263
264impl AlgorithmType {
265    /// Get the standard name for the algorithm
266    #[must_use]
267    pub fn name(&self) -> String {
268        match self {
269            AlgorithmType::MlKem { variant } => format!("ML-KEM-{}", variant),
270            AlgorithmType::MlDsa { variant } => format!("ML-DSA-{}", variant),
271            AlgorithmType::SlhDsa { variant } => format!("SLH-DSA-{}", variant),
272            AlgorithmType::HybridKem => "Hybrid-KEM".to_string(),
273            AlgorithmType::AesGcm { key_size } => format!("AES-{}-GCM", key_size * 8),
274            AlgorithmType::Sha3 { variant } => format!("SHA3-{}", variant),
275            AlgorithmType::Ed25519 => "Ed25519".to_string(),
276            AlgorithmType::Bls12_381 => "BLS12-381".to_string(),
277            AlgorithmType::Bn254 => "BN254".to_string(),
278            AlgorithmType::Secp256k1 => "secp256k1".to_string(),
279        }
280    }
281
282    /// Get the security level in bits
283    #[must_use]
284    pub fn security_level(&self) -> usize {
285        match self {
286            AlgorithmType::MlKem { variant } => match variant.as_str() {
287                "512" => 128,
288                "768" => 192,
289                "1024" => 256,
290                _ => 128,
291            },
292            AlgorithmType::MlDsa { variant } => match variant.as_str() {
293                "44" => 128,
294                "65" => 192,
295                "87" => 256,
296                _ => 128,
297            },
298            AlgorithmType::SlhDsa { variant } => match variant.as_str() {
299                "128" => 128,
300                "192" => 192,
301                "256" => 256,
302                _ => 128,
303            },
304            AlgorithmType::HybridKem => 256, // X25519 (128) + ML-KEM-1024 (256)
305            AlgorithmType::AesGcm { key_size } => key_size * 8,
306            AlgorithmType::Sha3 { variant } => variant.parse().unwrap_or(256),
307            AlgorithmType::Ed25519 => 128,
308            AlgorithmType::Bls12_381 => 128,
309            AlgorithmType::Bn254 => 128,
310            AlgorithmType::Secp256k1 => 128,
311        }
312    }
313}

315/// KAT Test Configuration
316#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
317pub struct KatConfig {
318    /// Algorithm to test
319    pub algorithm: AlgorithmType,
320    /// Number of test vectors to run
321    pub test_count: usize,
322    /// Whether to run statistical tests on generated randomness
323    pub run_statistical_tests: bool,
324    /// Timeout for individual test cases
325    pub timeout_per_test: Duration,
326    /// Whether to validate FIPS compliance
327    pub validate_fips: bool,
328}
329
330impl Default for KatConfig {
331    fn default() -> Self {
332        Self {
333            algorithm: AlgorithmType::MlKem { variant: "768".to_string() },
334            test_count: 100,
335            run_statistical_tests: true,
336            timeout_per_test: Duration::from_secs(10),
337            validate_fips: true,
338        }
339    }
340}
