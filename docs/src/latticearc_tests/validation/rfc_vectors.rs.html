1//! RFC Test Vectors
2//!
3//! This module validates LatticeArc's cryptographic implementations against
4//! official IETF RFC test vectors for protocol compliance.
5//!
6//! ## Supported RFCs
7//!
8//! - RFC 7748: X25519 Elliptic Curve Diffie-Hellman
9//! - RFC 8032: Ed25519 Digital Signatures
10//! - RFC 8439: ChaCha20-Poly1305 AEAD
11//! - RFC 5869: HKDF (HMAC-based Key Derivation Function)
12
13#![deny(unsafe_code)]
14#![deny(clippy::unwrap_used)]
15#![deny(clippy::panic)]
16// JUSTIFICATION: Test result counters - overflow is impossible with realistic test counts
17#![allow(clippy::arithmetic_side_effects)]
18// JUSTIFICATION: Test code uses expect() for known-valid test vectors and println! for test output
19#![allow(clippy::expect_used)]
20#![allow(clippy::print_stdout)]
21// JUSTIFICATION: Clone needed for test data manipulation
22#![allow(clippy::redundant_clone)]

24use thiserror::Error;
25
26/// Errors from RFC test vector validation
27#[derive(Debug, Error)]
28pub enum RfcTestError {
29    /// Test vector validation failed
30    #[error("RFC test failed: {rfc} - {test_name}: {message}")]
31    TestFailed {
32        /// RFC number
33        rfc: String,
34        /// Test name
35        test_name: String,
36        /// Failure message
37        message: String,
38    },
39
40    /// Hex decoding error
41    #[error("Hex decode error: {0}")]
42    HexError(String),
43}
44
45/// Result of running RFC tests
46#[derive(Debug, Default)]
47pub struct RfcTestResults {
48    /// Total tests run
49    pub total: usize,
50    /// Tests passed
51    pub passed: usize,
52    /// Tests failed
53    pub failed: usize,
54    /// Failure details
55    pub failures: Vec<String>,
56}
57
58impl RfcTestResults {
59    /// Create new results
60    #[must_use]
61    pub fn new() -> Self {
62        Self::default()
63    }
64
65    /// Record a passed test
66    pub fn add_pass(&mut self) {
67        self.total += 1;
68        self.passed += 1;
69    }
70
71    /// Record a failed test
72    pub fn add_failure(&mut self, message: String) {
73        self.total += 1;
74        self.failed += 1;
75        self.failures.push(message);
76    }
77
78    /// Check if all tests passed
79    #[must_use]
80    pub fn all_passed(&self) -> bool {
81        self.failed == 0
82    }
83}

93    /// RFC 8439 Section 2.4.2: ChaCha20 Block Function Test Vector
94    #[test]
95    fn test_rfc8439_chacha20_block() {
96        use chacha20poly1305::{
97            ChaCha20Poly1305,
98            aead::{Aead, KeyInit, Payload},
99        };
100
101        let mut results = RfcTestResults::new();
102
103        // RFC 8439 Section 2.8.2: Example and Test Vector for AEAD_CHACHA20_POLY1305
104        // Key
105        let key = hex::decode("808182838485868788898a8b8c8d8e8f909192939495969798999a9b9c9d9e9f")
106            .expect("hex decode");
107        let key_array: [u8; 32] = key.try_into().expect("key length");
108
109        // Nonce (96-bit)
110        let nonce = hex::decode("070000004041424344454647").expect("hex decode");
111
112        // AAD
113        let aad = hex::decode("50515253c0c1c2c3c4c5c6c7").expect("hex decode");
114
115        // Plaintext
116        let plaintext = b"Ladies and Gentlemen of the class of '99: If I could offer you only one tip for the future, sunscreen would be it.";
117
118        // Expected ciphertext + tag from RFC 8439
119        let expected_ciphertext = hex::decode(
120            "d31a8d34648e60db7b86afbc53ef7ec2a4aded51296e08fea9e2b5a736ee62d63dbea45e8ca9671282fafb69da92728b1a71de0a9e060b2905d6a5b67ecd3b3692ddbd7f2d778b8c9803aee328091b58fab324e4fad675945585808b4831d7bc3ff4def08e4b7a9de576d26586cec64b6116"
121        ).expect("hex decode");
122
123        let expected_tag = hex::decode("1ae10b594f09e26a7e902ecbd0600691").expect("hex decode");
124
125        let cipher = ChaCha20Poly1305::new(&key_array.into());
126
127        // Encrypt
128        let ciphertext_result =
129            cipher.encrypt((&nonce[..]).into(), Payload { msg: plaintext, aad: &aad });
130
131        match ciphertext_result {
132            Ok(ct) => {
133                // ChaCha20Poly1305 appends the 16-byte tag
134                if ct.len() == expected_ciphertext.len() + expected_tag.len() {
135                    let (ct_part, tag_part) = ct.split_at(expected_ciphertext.len());
136                    if ct_part == expected_ciphertext.as_slice()
137                        && tag_part == expected_tag.as_slice()
138                    {
139                        results.add_pass();
140                    } else {
141                        results.add_failure("RFC 8439: ciphertext or tag mismatch".to_string());
142                    }
143                } else {
144                    results.add_failure(format!(
145                        "RFC 8439: unexpected output length {} vs {}",
146                        ct.len(),
147                        expected_ciphertext.len() + expected_tag.len()
148                    ));
149                }
150            }
151            Err(e) => {
152                results.add_failure(format!("RFC 8439: encryption failed: {e:?}"));
153            }
154        }
155
156        // Decrypt and verify
157        let mut ct_with_tag = expected_ciphertext.clone();
158        ct_with_tag.extend_from_slice(&expected_tag);
159
160        let decrypt_result =
161            cipher.decrypt((&nonce[..]).into(), Payload { msg: &ct_with_tag, aad: &aad });
162
163        match decrypt_result {
164            Ok(pt) => {
165                if pt == plaintext {
166                    results.add_pass();
167                } else {
168                    results.add_failure("RFC 8439: decrypted plaintext mismatch".to_string());
169                }
170            }
171            Err(e) => {
172                results.add_failure(format!("RFC 8439: decryption failed: {e:?}"));
173            }
174        }
175
176        println!("RFC 8439 ChaCha20-Poly1305: {}/{} passed", results.passed, results.total);
177        assert!(results.all_passed(), "RFC 8439 failures: {:?}", results.failures);
178    }

303    /// RFC 7748 Section 6.1: X25519 Test Vectors
304    #[test]
305    fn test_rfc7748_x25519() {
306        use x25519_dalek::{PublicKey, StaticSecret};
307
308        let mut results = RfcTestResults::new();
309
310        // RFC 7748 Section 6.1: First test vector
311        // Alice's private key (scalar)
312        let alice_private =
313            hex::decode("77076d0a7318a57d3c16c17251b26645df4c2f87ebc0992ab177fba51db92c2a")
314                .expect("hex");
315        // Alice's public key
316        let alice_public_expected =
317            hex::decode("8520f0098930a754748b7ddcb43ef75a0dbf3a0d26381af4eba4a98eaa9b4e6a")
318                .expect("hex");
319
320        // Bob's private key
321        let bob_private =
322            hex::decode("5dab087e624a8a4b79e17f8b83800ee66f3bb1292618b6fd1c2f8b27ff88e0eb")
323                .expect("hex");
324        // Bob's public key
325        let bob_public_expected =
326            hex::decode("de9edb7d7b7dc1b4d35b61c2ece435373f8343c85b78674dadfc7e146f882b4f")
327                .expect("hex");
328
329        // Shared secret
330        let shared_secret_expected =
331            hex::decode("4a5d9d5ba4ce2de1728e3bf480350f25e07e21c947d19e3376f09b3c1e161742")
332                .expect("hex");
333
334        // Test Alice's key derivation
335        let alice_sk_bytes: [u8; 32] = alice_private.try_into().expect("sk length");
336        let alice_secret = StaticSecret::from(alice_sk_bytes);
337        let alice_public = PublicKey::from(&alice_secret);
338
339        if alice_public.as_bytes() == alice_public_expected.as_slice() {
340            results.add_pass();
341        } else {
342            results.add_failure("RFC 7748: Alice public key mismatch".to_string());
343        }
344
345        // Test Bob's key derivation
346        let bob_sk_bytes: [u8; 32] = bob_private.try_into().expect("sk length");
347        let bob_secret = StaticSecret::from(bob_sk_bytes);
348        let bob_public = PublicKey::from(&bob_secret);
349
350        if bob_public.as_bytes() == bob_public_expected.as_slice() {
351            results.add_pass();
352        } else {
353            results.add_failure("RFC 7748: Bob public key mismatch".to_string());
354        }
355
356        // Test shared secret (Alice computes with Bob's public key)
357        let shared_alice = alice_secret.diffie_hellman(&bob_public);
358        if shared_alice.as_bytes() == shared_secret_expected.as_slice() {
359            results.add_pass();
360        } else {
361            results.add_failure("RFC 7748: Alice shared secret mismatch".to_string());
362        }
363
364        // Test shared secret (Bob computes with Alice's public key)
365        let shared_bob = bob_secret.diffie_hellman(&alice_public);
366        if shared_bob.as_bytes() == shared_secret_expected.as_slice() {
367            results.add_pass();
368        } else {
369            results.add_failure("RFC 7748: Bob shared secret mismatch".to_string());
370        }
371
372        // Verify both parties compute the same shared secret
373        if shared_alice.as_bytes() == shared_bob.as_bytes() {
374            results.add_pass();
375        } else {
376            results.add_failure("RFC 7748: Shared secrets don't match".to_string());
377        }
378
379        println!("RFC 7748 X25519: {}/{} passed", results.passed, results.total);
380        assert!(results.all_passed(), "RFC 7748 failures: {:?}", results.failures);
381    }
