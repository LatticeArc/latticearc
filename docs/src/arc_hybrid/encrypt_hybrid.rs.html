6//! Hybrid Encryption Module
7//!
8//! This module provides hybrid encryption combining post-quantum (ML-KEM) key
9//! encapsulation with AES-256-GCM symmetric encryption for quantum-resistant
10//! data encryption with classical performance characteristics.
11//!
12//! # Overview
13//!
14//! The hybrid encryption scheme uses:
15//! - **ML-KEM-768** (FIPS 203) for post-quantum key encapsulation
16//! - **AES-256-GCM** for authenticated symmetric encryption
17//! - **HKDF-SHA256** for key derivation with domain separation
18//!
19//! # Security Properties
20//!
21//! - IND-CCA2 security from ML-KEM
22//! - Authenticated encryption with associated data (AEAD)
23//! - Domain separation via HPKE-style key derivation
24//!
25//! # Example
26//!
27//! ```rust,ignore
28//! use arc_hybrid::encrypt_hybrid::{encrypt, decrypt, HybridEncryptionContext};
29//! use rand::rngs::OsRng;
30//!
31//! let mut rng = OsRng;
32//! let plaintext = b"Secret message";
33//! let context = HybridEncryptionContext::default();
34//!
35//! // Encrypt with ML-KEM public key
36//! let ciphertext = encrypt(&mut rng, &ml_kem_pk, plaintext, Some(&context))?;
37//!
38//! // Decrypt with ML-KEM secret key
39//! let decrypted = decrypt(&ml_kem_sk, &ciphertext, Some(&context))?;
40//! ```

50/// Error types for hybrid encryption operations.
51///
52/// This enum captures all possible error conditions that can occur during
53/// hybrid encryption and decryption operations.
54#[derive(Debug, Clone, PartialEq, Eq, Error)]
55pub enum HybridEncryptionError {
56    /// Error during key encapsulation mechanism operations.
57    #[error("KEM error: {0}")]
58    KemError(String),
59    /// Error during symmetric encryption.
60    #[error("Encryption error: {0}")]
61    EncryptionError(String),
62    /// Error during symmetric decryption or authentication failure.
63    #[error("Decryption error: {0}")]
64    DecryptionError(String),
65    /// Error during key derivation function operations.
66    #[error("Key derivation error: {0}")]
67    KdfError(String),
68    /// Invalid input parameters provided to the operation.
69    #[error("Invalid input: {0}")]
70    InvalidInput(String),
71    /// Key length mismatch error.
72    #[error("Key length error: expected {expected}, got {actual}")]
73    KeyLengthError {
74        /// Expected key length in bytes.
75        expected: usize,
76        /// Actual key length provided.
77        actual: usize,
78    },
79}

81/// Hybrid ciphertext containing both KEM and symmetric encryption components.
82///
83/// This structure holds all the data needed to decrypt a hybrid-encrypted message:
84/// - The KEM ciphertext for key decapsulation
85/// - The symmetric ciphertext containing the encrypted message
86/// - The nonce used for AES-GCM encryption
87/// - The authentication tag for integrity verification
88#[derive(Debug, Clone)]
89pub struct HybridCiphertext {
90    /// ML-KEM ciphertext for key decapsulation (1088 bytes for ML-KEM-768).
91    pub kem_ciphertext: Vec<u8>,
92    /// X25519 ephemeral public key for ECDH (32 bytes). Empty for legacy ML-KEM-only ciphertexts.
93    pub ecdh_ephemeral_pk: Vec<u8>,
94    /// AES-256-GCM encrypted message data.
95    pub symmetric_ciphertext: Vec<u8>,
96    /// 12-byte nonce used for AES-GCM encryption.
97    pub nonce: Vec<u8>,
98    /// 16-byte AES-GCM authentication tag.
99    pub tag: Vec<u8>,
100}

102/// HPKE-style context information for hybrid encryption.
103///
104/// This structure provides domain separation and additional authenticated data
105/// for the key derivation and encryption operations, following RFC 9180 (HPKE).
106#[derive(Debug, Clone)]
107pub struct HybridEncryptionContext {
108    /// Application-specific info string for key derivation domain separation.
109    pub info: Vec<u8>,
110    /// Additional authenticated data (AAD) for AEAD encryption.
111    pub aad: Vec<u8>,
112}

125        self.0

129/// HPKE-style key derivation for hybrid encryption.
130///
131/// # Errors
132///
133/// Returns an error if the shared secret is not exactly 32 bytes,
134/// or if HKDF expansion fails.
135pub fn derive_encryption_key(
136    shared_secret: &[u8],
137    context: &HybridEncryptionContext,
138) -> Result<[u8; 32], HybridEncryptionError> {
139    if shared_secret.len() != 32 && shared_secret.len() != 64 {
140        return Err(HybridEncryptionError::KdfError(
141            "Shared secret must be 32 bytes (ML-KEM) or 64 bytes (hybrid)".to_string(),
142        ));
143    }
144
145    // Create info for domain separation
146    let mut info = Vec::new();
147    info.extend_from_slice(&context.info);
148    info.extend_from_slice(b"||");
149    info.extend_from_slice(&context.aad);
150
151    // Use HKDF-SHA256 for key derivation via aws-lc-rs
152    let salt = Salt::new(HKDF_SHA256, &[]);
153    let prk = salt.extract(shared_secret);
154    let info_refs: [&[u8]; 1] = [&info];
155    let okm = prk
156        .expand(&info_refs, HkdfOutputLen(32))
157        .map_err(|_e| HybridEncryptionError::KdfError("HKDF expansion failed".to_string()))?;
158
159    let mut key = [0u8; 32];
160    okm.fill(&mut key)
161        .map_err(|_e| HybridEncryptionError::KdfError("HKDF fill failed".to_string()))?;
162
163    Ok(key)
164}

166/// Hybrid encryption using ML-KEM + AES-256-GCM with HPKE-style key derivation.
167///
168/// # Errors
169///
170/// Returns an error if:
171/// - The ML-KEM public key is not 1184 bytes (ML-KEM-768)
172/// - ML-KEM encapsulation fails
173/// - Key derivation fails
174/// - AES-GCM encryption fails
175pub fn encrypt<R: rand::Rng + rand::CryptoRng>(
176    rng: &mut R,
177    ml_kem_pk: &[u8],
178    plaintext: &[u8],
179    context: Option<&HybridEncryptionContext>,
180) -> Result<HybridCiphertext, HybridEncryptionError> {
181    let default_ctx = HybridEncryptionContext::default();
182    let ctx = context.unwrap_or(&default_ctx);
183
184    // Validate inputs
185    if ml_kem_pk.len() != 1184 {
186        return Err(HybridEncryptionError::InvalidInput(
187            "ML-KEM-768 public key must be 1184 bytes".to_string(),
188        ));
189    }
190
191    // ML-KEM encapsulation
192    let ml_kem_pk_struct = MlKemPublicKey::new(MlKemSecurityLevel::MlKem768, ml_kem_pk.to_vec())
193        .map_err(|e| HybridEncryptionError::KemError(format!("{:?}", e)))?;
194    let (shared_secret, kem_ct_struct) = MlKem::encapsulate(rng, &ml_kem_pk_struct)
195        .map_err(|e| HybridEncryptionError::KemError(format!("{:?}", e)))?;
196    let kem_ct = kem_ct_struct.into_bytes();
197
198    // Derive encryption key using HPKE-style KDF
199    let encryption_key = derive_encryption_key(shared_secret.as_bytes(), ctx)?;
200
201    // Generate random nonce for AES-GCM
202    let mut nonce_bytes = [0u8; 12];
203    rng.fill_bytes(&mut nonce_bytes);
204    let nonce = Nonce::assume_unique_for_key(nonce_bytes);
205
206    // Use AES-256-GCM for authenticated encryption with AAD via aws-lc-rs
207    let unbound_key = UnboundKey::new(&AES_256_GCM, &encryption_key).map_err(|_e| {
208        HybridEncryptionError::EncryptionError("Failed to create AES key".to_string())
209    })?;
210    let aes_key = LessSafeKey::new(unbound_key);
211
212    // Encrypt in-place: plaintext becomes ciphertext + tag
213    let mut in_out = plaintext.to_vec();
214    let aad = Aad::from(&ctx.aad[..]);
215    aes_key.seal_in_place_append_tag(nonce, aad, &mut in_out).map_err(|_e| {
216        HybridEncryptionError::EncryptionError("AES-GCM encryption failed".to_string())
217    })?;
218
219    // AES-GCM tag is the last 16 bytes
220    let tag_len = 16;
221    let ct_len = in_out.len();
222    if ct_len < tag_len {
223        return Err(HybridEncryptionError::EncryptionError(
224            "Ciphertext too short for tag".to_string(),
225        ));
226    }
227
228    // Use checked subtraction - the check above guarantees this won't underflow
229    let split_pos = ct_len.checked_sub(tag_len).ok_or_else(|| {
230        HybridEncryptionError::EncryptionError("Ciphertext length calculation overflow".to_string())
231    })?;
232    let (ciphertext, tag) = in_out.split_at(split_pos);
233
234    Ok(HybridCiphertext {
235        kem_ciphertext: kem_ct,
236        ecdh_ephemeral_pk: vec![], // Legacy ML-KEM-only path — no ECDH
237        symmetric_ciphertext: ciphertext.to_vec(),
238        nonce: nonce_bytes.to_vec(),
239        tag: tag.to_vec(),
240    })
241}

243/// Hybrid decryption using ML-KEM + AES-256-GCM with HPKE-style key derivation.
244///
245/// # Errors
246///
247/// Returns an error if:
248/// - The ML-KEM secret key is not 2400 bytes (ML-KEM-768)
249/// - The ciphertext components have invalid lengths
250/// - ML-KEM decapsulation fails
251/// - Key derivation fails
252/// - AES-GCM decryption or authentication fails
253pub fn decrypt(
254    ml_kem_sk: &[u8],
255    ciphertext: &HybridCiphertext,
256    context: Option<&HybridEncryptionContext>,
257) -> Result<Vec<u8>, HybridEncryptionError> {
258    let default_ctx = HybridEncryptionContext::default();
259    let ctx = context.unwrap_or(&default_ctx);
260
261    // Validate inputs
262    if ml_kem_sk.len() != 2400 {
263        return Err(HybridEncryptionError::InvalidInput(
264            "ML-KEM-768 secret key must be 2400 bytes".to_string(),
265        ));
266    }
267    if ciphertext.kem_ciphertext.len() != 1088 {
268        return Err(HybridEncryptionError::InvalidInput(
269            "ML-KEM-768 ciphertext must be 1088 bytes".to_string(),
270        ));
271    }
272    if ciphertext.nonce.len() != 12 {
273        return Err(HybridEncryptionError::InvalidInput("Nonce must be 12 bytes".to_string()));
274    }
275    if ciphertext.tag.len() != 16 {
276        return Err(HybridEncryptionError::InvalidInput("Tag must be 16 bytes".to_string()));
277    }
278
279    // ML-KEM decapsulation
280    let ml_kem_sk_struct = MlKemSecretKey::new(MlKemSecurityLevel::MlKem768, ml_kem_sk.to_vec())
281        .map_err(|e| HybridEncryptionError::DecryptionError(format!("{:?}", e)))?;
282    let ml_kem_ct_struct =
283        MlKemCiphertext::new(MlKemSecurityLevel::MlKem768, ciphertext.kem_ciphertext.clone())
284            .map_err(|e| HybridEncryptionError::DecryptionError(format!("{:?}", e)))?;
285    let shared_secret = MlKem::decapsulate(&ml_kem_sk_struct, &ml_kem_ct_struct)
286        .map_err(|e| HybridEncryptionError::DecryptionError(format!("{:?}", e)))?;
287
288    // Derive encryption key using HPKE-style KDF
289    let encryption_key = derive_encryption_key(shared_secret.as_bytes(), ctx)?;
290
291    // Setup AES-256-GCM via aws-lc-rs
292    let nonce_bytes: [u8; 12] =
293        ciphertext.nonce.as_slice().try_into().map_err(|_e| {
294            HybridEncryptionError::DecryptionError("Invalid nonce length".to_string())
295        })?;
296    let nonce = Nonce::assume_unique_for_key(nonce_bytes);
297
298    let unbound_key = UnboundKey::new(&AES_256_GCM, &encryption_key).map_err(|_e| {
299        HybridEncryptionError::DecryptionError("Failed to create AES key".to_string())
300    })?;
301    let aes_key = LessSafeKey::new(unbound_key);
302
303    // Combine ciphertext and tag for decryption
304    let mut in_out: Vec<u8> =
305        ciphertext.symmetric_ciphertext.iter().chain(ciphertext.tag.iter()).copied().collect();
306
307    // Decrypt in-place with AAD
308    let aad = Aad::from(&ctx.aad[..]);
309    let plaintext = aes_key.open_in_place(nonce, aad, &mut in_out).map_err(|_e| {
310        HybridEncryptionError::DecryptionError(
311            "AES-GCM decryption/authentication failed".to_string(),
312        )
313    })?;
314
315    Ok(plaintext.to_vec())
316}

318/// True hybrid encryption using ML-KEM-768 + X25519 + AES-256-GCM.
319///
320/// This function performs real hybrid key encapsulation (combining post-quantum
321/// ML-KEM with classical X25519 ECDH via HKDF) before AES-256-GCM encryption.
322/// Security holds if *either* ML-KEM or X25519 remains secure.
323///
324/// # Errors
325///
326/// Returns an error if:
327/// - Hybrid KEM encapsulation fails
328/// - Key derivation fails
329/// - AES-GCM encryption fails
330pub fn encrypt_hybrid<R: rand::Rng + rand::CryptoRng>(
331    rng: &mut R,
332    hybrid_pk: &HybridPublicKey,
333    plaintext: &[u8],
334    context: Option<&HybridEncryptionContext>,
335) -> Result<HybridCiphertext, HybridEncryptionError> {
336    let default_ctx = HybridEncryptionContext::default();
337    let ctx = context.unwrap_or(&default_ctx);
338
339    // Hybrid KEM encapsulation (ML-KEM-768 + X25519 ECDH + HKDF)
340    let encapsulated = kem_hybrid::encapsulate(rng, hybrid_pk)
341        .map_err(|e| HybridEncryptionError::KemError(format!("{}", e)))?;
342
343    // Derive AES-256 encryption key from 64-byte hybrid shared secret
344    let encryption_key = derive_encryption_key(&encapsulated.shared_secret, ctx)?;
345
346    // Generate random nonce for AES-GCM
347    let mut nonce_bytes = [0u8; 12];
348    rng.fill_bytes(&mut nonce_bytes);
349    let nonce = Nonce::assume_unique_for_key(nonce_bytes);
350
351    // AES-256-GCM authenticated encryption
352    let unbound_key = UnboundKey::new(&AES_256_GCM, &encryption_key).map_err(|_e| {
353        HybridEncryptionError::EncryptionError("Failed to create AES key".to_string())
354    })?;
355    let aes_key = LessSafeKey::new(unbound_key);
356
357    let mut in_out = plaintext.to_vec();
358    let aad = Aad::from(&ctx.aad[..]);
359    aes_key.seal_in_place_append_tag(nonce, aad, &mut in_out).map_err(|_e| {
360        HybridEncryptionError::EncryptionError("AES-GCM encryption failed".to_string())
361    })?;
362
363    let tag_len = 16;
364    let ct_len = in_out.len();
365    let split_pos = ct_len.checked_sub(tag_len).ok_or_else(|| {
366        HybridEncryptionError::EncryptionError("Ciphertext length calculation overflow".to_string())
367    })?;
368    let (ciphertext, tag) = in_out.split_at(split_pos);
369
370    Ok(HybridCiphertext {
371        kem_ciphertext: encapsulated.ml_kem_ct.clone(),
372        ecdh_ephemeral_pk: encapsulated.ecdh_pk.clone(),
373        symmetric_ciphertext: ciphertext.to_vec(),
374        nonce: nonce_bytes.to_vec(),
375        tag: tag.to_vec(),
376    })
377}

379/// True hybrid decryption using ML-KEM-768 + X25519 + AES-256-GCM.
380///
381/// This function performs real hybrid key decapsulation (ML-KEM decapsulation +
382/// X25519 ECDH agreement, combined via HKDF) before AES-256-GCM decryption.
383///
384/// # Errors
385///
386/// Returns an error if:
387/// - The ciphertext components have invalid lengths
388/// - Hybrid KEM decapsulation fails
389/// - Key derivation fails
390/// - AES-GCM decryption or authentication fails
391pub fn decrypt_hybrid(
392    hybrid_sk: &HybridSecretKey,
393    ciphertext: &HybridCiphertext,
394    context: Option<&HybridEncryptionContext>,
395) -> Result<Vec<u8>, HybridEncryptionError> {
396    let default_ctx = HybridEncryptionContext::default();
397    let ctx = context.unwrap_or(&default_ctx);
398
399    // Validate ciphertext structure
400    if ciphertext.kem_ciphertext.len() != 1088 {
401        return Err(HybridEncryptionError::InvalidInput(
402            "ML-KEM-768 ciphertext must be 1088 bytes".to_string(),
403        ));
404    }
405    if ciphertext.ecdh_ephemeral_pk.len() != 32 {
406        return Err(HybridEncryptionError::InvalidInput(
407            "X25519 ephemeral public key must be 32 bytes".to_string(),
408        ));
409    }
410    if ciphertext.nonce.len() != 12 {
411        return Err(HybridEncryptionError::InvalidInput("Nonce must be 12 bytes".to_string()));
412    }
413    if ciphertext.tag.len() != 16 {
414        return Err(HybridEncryptionError::InvalidInput("Tag must be 16 bytes".to_string()));
415    }
416
417    // Reconstruct EncapsulatedKey for kem_hybrid::decapsulate
418    let encapsulated = EncapsulatedKey {
419        ml_kem_ct: ciphertext.kem_ciphertext.clone(),
420        ecdh_pk: ciphertext.ecdh_ephemeral_pk.clone(),
421        shared_secret: zeroize::Zeroizing::new(vec![]), // placeholder — decapsulate recovers this
422    };
423
424    // Hybrid KEM decapsulation (ML-KEM + X25519 ECDH + HKDF)
425    let shared_secret = kem_hybrid::decapsulate(hybrid_sk, &encapsulated)
426        .map_err(|e| HybridEncryptionError::DecryptionError(format!("{}", e)))?;
427
428    // Derive AES-256 encryption key from 64-byte hybrid shared secret
429    let encryption_key = derive_encryption_key(&shared_secret, ctx)?;
430
431    // AES-256-GCM authenticated decryption
432    let nonce_bytes: [u8; 12] =
433        ciphertext.nonce.as_slice().try_into().map_err(|_e| {
434            HybridEncryptionError::DecryptionError("Invalid nonce length".to_string())
435        })?;
436    let nonce = Nonce::assume_unique_for_key(nonce_bytes);
437
438    let unbound_key = UnboundKey::new(&AES_256_GCM, &encryption_key).map_err(|_e| {
439        HybridEncryptionError::DecryptionError("Failed to create AES key".to_string())
440    })?;
441    let aes_key = LessSafeKey::new(unbound_key);
442
443    let mut in_out: Vec<u8> =
444        ciphertext.symmetric_ciphertext.iter().chain(ciphertext.tag.iter()).copied().collect();
445
446    let aad = Aad::from(&ctx.aad[..]);
447    let plaintext = aes_key.open_in_place(nonce, aad, &mut in_out).map_err(|_e| {
448        HybridEncryptionError::DecryptionError(
449            "AES-GCM decryption/authentication failed".to_string(),
450        )
451    })?;
452
453    Ok(plaintext.to_vec())
454}
