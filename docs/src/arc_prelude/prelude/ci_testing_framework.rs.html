41        tracing::info!("Running Prelude CI Test Suite");
42
43        let mut report = PreludeCiReport::default();
44

46        tracing::info!("Running Unit Tests");
47        let unit_tests_passed = Self::run_unit_tests().is_ok();
48        report.unit_tests_passed = unit_tests_passed;
49        if unit_tests_passed {
50            tracing::info!("Unit tests passed");
51        } else {

53        }
54
55        // 2. Property Tests
56        tracing::info!("Running Property Tests");
57        let property_tests_passed = Self::run_property_tests();
58        report.property_tests_passed = property_tests_passed;
59        if property_tests_passed {
60            tracing::info!("Property tests passed");
61        } else {
62            tracing::error!("Property tests failed");
63        }
64
65        // 3. Memory Safety Tests
66        tracing::info!("Running Memory Safety Tests");
67        let memory_safety_passed = Self::run_memory_safety_tests().is_ok();
68        report.memory_safety_passed = memory_safety_passed;
69        if memory_safety_passed {
70            tracing::info!("Memory safety tests passed");
71        } else {
72            tracing::error!("Memory safety tests failed");
73        }
74
75        tracing::info!("Prelude CI Test Suite Completed");
76
77        Ok(report)
78    }

108    /// Test hex encoding/decoding.
109    fn test_hex_functions() -> Result<(), LatticeArcError> {
110        let test_data = vec![0, 1, 255, 127, 64];
111
112        // Test encoding
113        let encoded = hex::encode(&test_data);
114        if encoded != "0001ff7f40" {
115            return Err(LatticeArcError::ValidationError {
116                message: format!("Expected '0001ff7f40', got '{}'", encoded),
117            });
118        }
119
120        // Test decoding
121        let decoded = hex::decode(&encoded)?;
122        if decoded != test_data {
123            return Err(LatticeArcError::ValidationError {
124                message: "Decoded data does not match original".to_string(),
125            });
126        }
127
128        for _ in 0..10 {
129            let mut rng = OsRng;
130            let mut data = vec![0u8; 32];
131            rng.fill_bytes(&mut data);
132            let encoded = hex::encode(&data);
133            let decoded = hex::decode(&encoded)?;
134            if data != decoded {
135                return Err(LatticeArcError::ValidationError {
136                    message: "Hex round-trip failed".to_string(),
137                });
138            }
139        }
140
141        Ok(())
142    }

144    /// Test UUID functions.
145    fn test_uuid_functions() -> Result<(), LatticeArcError> {
146        for _ in 0..10 {
147            let uuid = uuid::Uuid::new_v4();
148
149            // Basic validation
150            if uuid.is_nil() {
151                return Err(LatticeArcError::ValidationError {
152                    message: "UUID should not be nil".to_string(),
153                });
154            }
155            if uuid.get_version_num() != 4 {
156                return Err(LatticeArcError::ValidationError {
157                    message: format!("UUID version should be 4, got {}", uuid.get_version_num()),
158                });
159            }
160
161            // String format validation
162            let uuid_str = uuid.to_string();
163            if uuid_str.len() != 36 {
164                return Err(LatticeArcError::ValidationError {
165                    message: format!("UUID string should be 36 chars, got {}", uuid_str.len()),
166                });
167            }
168
169            // Parsing validation
170            let parsed = uuid::Uuid::parse_str(&uuid_str)?;
171            if parsed != uuid {
172                return Err(LatticeArcError::ValidationError {
173                    message: "Parsed UUID should match original".to_string(),
174                });
175            }
176        }
177
178        Ok(())
179    }

181    /// Test domain constants.
182    fn test_domain_constants() -> Result<(), LatticeArcError> {
183        // Test all domain constants are non-empty
184        if domains::HYBRID_KEM.is_empty() {
185            return Err(LatticeArcError::ValidationError {
186                message: "HYBRID_KEM should not be empty".to_string(),
187            });
188        }
189        if domains::CASCADE_OUTER.is_empty() {
190            return Err(LatticeArcError::ValidationError {
191                message: "CASCADE_OUTER should not be empty".to_string(),
192            });
193        }
194        if domains::CASCADE_INNER.is_empty() {
195            return Err(LatticeArcError::ValidationError {
196                message: "CASCADE_INNER should not be empty".to_string(),
197            });
198        }
199        if domains::SIGNATURE_BIND.is_empty() {
200            return Err(LatticeArcError::ValidationError {
201                message: "SIGNATURE_BIND should not be empty".to_string(),
202            });
203        }
204
205        // Test they all contain version identifier
206        let domain_list = [
207            domains::HYBRID_KEM,
208            domains::CASCADE_OUTER,
209            domains::CASCADE_INNER,
210            domains::SIGNATURE_BIND,
211        ];
212        for domain in &domain_list {
213            if !(*domain).windows(12).any(|w| w == b"LatticeArc-v") {
214                return Err(LatticeArcError::ValidationError {
215                    message: "Domain constant should contain version identifier".to_string(),
216                });
217            }
218        }
219
220        // Test uniqueness
221        for (i, &domain1) in domain_list.iter().enumerate() {
222            for (j, &domain2) in domain_list.iter().enumerate() {
223                if i != j && domain1 == domain2 {
224                    return Err(LatticeArcError::ValidationError {
225                        message: "Domain constants should be unique".to_string(),
226                    });
227                }
228            }
229        }
230
231        Ok(())
232    }

234    /// Test error handling.
235    fn test_error_handling() -> Result<(), LatticeArcError> {
236        use crate::prelude::error::{get_error_severity, is_recoverable_error};
237
238        let test_errors = vec![
239            LatticeArcError::InvalidInput("test".to_string()),
240            LatticeArcError::NetworkError("connection failed".to_string()),
241            LatticeArcError::IoError("file error".to_string()),
242        ];
243
244        for error in test_errors {
245            // Test serialization
246            let json = serde_json::to_string(&error)?;
247            let deserialized: LatticeArcError = serde_json::from_str(&json)?;
248            if error != deserialized {
249                return Err(LatticeArcError::ValidationError {
250                    message: "Deserialized error should match original".to_string(),
251                });
252            }
253
254            // Test severity and recovery
255            let _severity = get_error_severity(&error);
256            let _recoverable = is_recoverable_error(&error);
257        }
258
259        Ok(())
260    }

386    /// Check if all critical tests passed.
387    ///
388    /// Returns true if unit tests, property tests, and memory safety tests
389    /// all passed, and there are no critical side-channel vulnerabilities.
390    #[must_use]
391    pub fn all_critical_tests_passed(&self) -> bool {
392        self.unit_tests_passed
393            && self.property_tests_passed
394            && self.memory_safety_passed
395            && self
396                .side_channel_assessments
397                .iter()
398                .filter(|a| {
399                    matches!(a.severity, crate::prelude::side_channel_analysis::Severity::Critical)
400                })
401                .count()
402                == 0
403    }
