21/// Comprehensive error type for all `LatticeArc` operations
22///
23/// This enum covers all possible error conditions that can occur during
24/// cryptographic operations, key management, serialization, and I/O.
25#[derive(Debug, Error, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
26pub enum LatticeArcError {
27    /// Encryption operation failed
28    #[error("Encryption error: {0}")]
29    EncryptionError(String),
30    /// Decryption operation failed
31    #[error("Decryption error: {0}")]
32    DecryptionError(String),
33    /// Key generation failed
34    #[error("Key generation error: {0}")]
35    KeyGenerationError(String),
36    /// Invalid or corrupted key
37    #[error("Invalid key: {0}")]
38    InvalidKey(String),
39    /// KEM encapsulation failed
40    #[error("KEM encapsulation error: {0}")]
41    EncapsulationError(String),
42    /// KEM decapsulation failed
43    #[error("KEM decapsulation error: {0}")]
44    DecapsulationError(String),
45    /// Digital signature operation failed
46    #[error("Signing error: {0}")]
47    SigningError(String),
48    /// Authentication failed
49    #[error("Authentication error: {0}")]
50    AuthenticationError(String),
51    /// Signature verification failed
52    #[error("Signature verification failed")]
53    VerificationError,
54    /// Invalid signature
55    #[error("Invalid signature: {0}")]
56    InvalidSignature(String),
57
58    /// Invalid signature length
59    #[error("Invalid signature length: expected {expected}, got {got}")]
60    InvalidSignatureLength {
61        /// Expected length
62        expected: usize,
63        /// Actual length
64        got: usize,
65    },
66
67    /// Signature verification error
68    #[error("Signature verification error: {0}")]
69    SignatureVerificationError(String),
70
71    /// Invalid key length
72    #[error("Invalid key length: expected {expected}, actual {actual}")]
73    InvalidKeyLength {
74        /// Expected length
75        expected: usize,
76        /// Actual length
77        actual: usize,
78    },
79    /// Serialization/deserialization failed
80    #[error("Serialization error: {0}")]
81    SerializationError(String),
82    /// Deserialization failed
83    #[error("Deserialization error: {0}")]
84    DeserializationError(String),
85    /// I/O operation failed
86    #[error("I/O error: {0}")]
87    IoError(String),
88    /// Random number generation failed
89    #[error("Random number generation failed")]
90    RandomError,
91    /// Unsupported protocol version
92    #[error("Unsupported version: {0}")]
93    UnsupportedVersion(u8),
94    /// Invalid envelope format
95    #[error("Invalid envelope: {0}")]
96    InvalidEnvelope(String),
97    /// Invalid format
98    #[error("Invalid format: {0}")]
99    InvalidFormat(String),
100    /// Invalid configuration
101    #[error("Invalid configuration: {0}")]
102    InvalidConfiguration(String),
103    /// Invalid data
104    #[error("Invalid data: {0}")]
105    InvalidData(String),
106    /// Invalid input
107    #[error("Invalid input: {0}")]
108    InvalidInput(String),
109    /// Service unavailable
110    #[error("Service unavailable: {0}")]
111    ServiceUnavailable(String),
112    /// Security violation
113    #[error("Security violation: {0}")]
114    SecurityViolation(String),
115    /// Policy violation
116    #[error("Policy violation: {0}")]
117    PolicyViolation(String),
118    /// Compliance violation
119    #[error("Compliance violation: {0}")]
120    ComplianceViolation(String),
121    /// Invalid parameter
122    #[error("Invalid parameter: {0}")]
123    InvalidParameter(String),
124    /// Not implemented
125    #[error("Not implemented: {0}")]
126    NotImplemented(String),
127    /// CPU feature not available
128    #[error("CPU feature not available: {0}")]
129    CpuFeatureNotAvailable(String),
130    /// Memory allocation failed
131    #[error("Memory error: {0}")]
132    MemoryError(String),
133    /// Circuit breaker is open
134    #[error("Circuit breaker is open")]
135    CircuitBreakerOpen,
136    /// System resources exhausted
137    #[error("System resources exhausted")]
138    ResourceExhausted,
139    /// Required feature not enabled
140    #[error("Feature not enabled: {0}")]
141    FeatureNotEnabled(String),
142    /// Audit logging failed
143    #[error("Audit error: {0}")]
144    AuditError(String),
145    /// HSM operation failed
146    #[error("HSM error: {0}")]
147    HsmError(String),
148    /// PIN verification failed
149    #[error("PIN verification failed")]
150    PinIncorrect,
151    /// PIN account locked due to too many failed attempts
152    #[error("PIN account locked due to too many failed attempts")]
153    PinLocked,
154    /// Cloud KMS operation failed
155    #[error("Cloud KMS error: {0}")]
156    CloudKmsError(String),
157    /// Database operation failed
158    #[error("Database error: {0}")]
159    DatabaseError(String),
160    /// Network operation failed
161    #[error("Network error: {0}")]
162    NetworkError(String),
163    /// TLS operation failed
164    #[error("TLS error: {0}")]
165    TlsError(String),
166    /// Key derivation error
167    #[error("Key derivation error: {0}")]
168    KeyDerivationError(String),
169    /// Formal verification failed
170    #[error("Formal verification failed: {0}")]
171    VerificationFailed(String),
172    /// Fuzzing test failed
173    #[error("Fuzzing error: {0}")]
174    FuzzingError(String),
175    /// Development tool error
176    #[error("Development tool error: {0}")]
177    DevToolError(String),
178    /// Migration operation failed
179    #[error("Migration error: {0}")]
180    MigrationError(String),
181    /// Performance profiling error
182    #[error("Profiling error: {0}")]
183    ProfilingError(String),
184    /// Side channel mitigation failed
185    #[error("Side channel error: {0}")]
186    SideChannelError(String),
187    /// Async operation failed
188    #[error("Async error: {0}")]
189    AsyncError(String),
190    /// WASM-specific error
191    #[error("WASM error: {0}")]
192    WasmError(String),
193    /// Access denied due to insufficient permissions
194    #[error("Access denied: {0}")]
195    AccessDenied(String),
196    /// Unauthorized access
197    #[error("Unauthorized: {0}")]
198    Unauthorized(String),
199    /// Invalid elliptic curve point
200    #[error("Invalid elliptic curve point")]
201    InvalidPoint,
202    /// Resource or permission expired
203    #[error("Expired: {0}")]
204    Expired(String),
205    /// Hardware acceleration error
206    #[error("Hardware error: {0}")]
207    HardwareError(String),
208    /// Invalid operation attempted
209    #[error("Invalid operation: {0}")]
210    InvalidOperation(String),
211    /// Concurrency-related error
212    #[error("Concurrency error: {0}")]
213    ConcurrencyError(String),
214    /// Timeout error
215    #[error("Timeout: {0}")]
216    TimeoutError(String),
217    /// CAVP validation error
218    #[error("Validation error: {message}")]
219    ValidationError {
220        /// Validation error message
221        message: String,
222    },
223
224    // ============================================================================
225    // Zero-Knowledge Proof Errors
226    // ============================================================================
227    /// Zero-knowledge proof error
228    #[error("ZKP error: {0}")]
229    ZkpError(String),
230}
