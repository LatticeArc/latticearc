3//! This module provides functions for error recovery and severity assessment,

13/// Error recovery strategy.
14///
15/// Defines the strategy to use when recovering from an error.
16#[derive(Debug, Clone, PartialEq)]
17pub enum ErrorRecoveryStrategy {
18    /// Retry the operation with exponential backoff.
19    Retry {
20        /// Maximum number of retry attempts.
21        max_attempts: usize,
22        /// Base delay between retries in milliseconds.
23        delay_ms: u64,
24    },
25    /// Fall back to an alternative approach.
26    Fallback {
27        /// Description of the alternative approach.
28        alternative: String,
29    },
30    /// Degrade to reduced functionality.
31    Degrade {
32        /// Description of reduced functionality.
33        reduced_functionality: String,
34    },
35    /// Ignore the error and continue.
36    Ignore,
37    /// Fail immediately without recovery.
38    Fail,
39}
40
41/// Error severity level for NIST compliance.
42///
43/// Used to classify errors by their impact on security and operations.
44#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
45pub enum ErrorSeverity {
46    /// Low severity - minimal impact.
47    Low = 1,
48    /// Medium severity - moderate impact.
49    Medium = 2,
50    /// High severity - significant impact.
51    High = 3,
52    /// Critical severity - severe impact.
53    Critical = 4,
54}

56/// Attempt error recovery based on error type.
57///
58/// Uses wildcard match intentionally: new error variants default to
59/// no recovery strategy (None) until explicitly handled.
60#[must_use]
61#[allow(clippy::wildcard_enum_match_arm)]
62pub fn attempt_error_recovery(error: &LatticeArcError) -> Option<ErrorRecoveryStrategy> {
63    match error {
64        LatticeArcError::NetworkError(_) => {
65            Some(ErrorRecoveryStrategy::Retry { max_attempts: 3, delay_ms: 1000 })
66        }
67        LatticeArcError::TimeoutError(_) => {
68            Some(ErrorRecoveryStrategy::Retry { max_attempts: 2, delay_ms: 500 })
69        }
70        LatticeArcError::ServiceUnavailable(_) => {
71            Some(ErrorRecoveryStrategy::Retry { max_attempts: 5, delay_ms: 2000 })
72        }
73        LatticeArcError::CircuitBreakerOpen => {
74            Some(ErrorRecoveryStrategy::Retry { max_attempts: 1, delay_ms: 10000 })
75        }
76        LatticeArcError::ResourceExhausted => Some(ErrorRecoveryStrategy::Degrade {
77            reduced_functionality: "Reduced parallelism".to_string(),
78        }),
79        LatticeArcError::HardwareError(_) => {
80            Some(ErrorRecoveryStrategy::Fallback { alternative: "Software fallback".to_string() })
81        }
82        LatticeArcError::FeatureNotEnabled(_) => Some(ErrorRecoveryStrategy::Fail),
83        _ => None, // No recovery strategy for other errors
84    }
85}
86
87/// Check if error is recoverable.
88///
89/// Returns true if a recovery strategy exists for the given error type.
90#[must_use]
91pub fn is_recoverable_error(error: &LatticeArcError) -> bool {
92    attempt_error_recovery(error).is_some()
93}

95/// Get error severity for compliance reporting.
96///
97/// Uses wildcard match intentionally: new error variants default to
98/// Low severity until explicitly categorized.
99#[must_use]
100#[allow(clippy::wildcard_enum_match_arm)]
101pub fn get_error_severity(error: &LatticeArcError) -> ErrorSeverity {
102    match error {
103        LatticeArcError::EncryptionError(_)
104        | LatticeArcError::DecryptionError(_)
105        | LatticeArcError::KeyGenerationError(_)
106        | LatticeArcError::SigningError(_)
107        | LatticeArcError::VerificationError
108        | LatticeArcError::InvalidSignature(_) => ErrorSeverity::Critical,
109
110        LatticeArcError::AuthenticationError(_)
111        | LatticeArcError::AccessDenied(_)
112        | LatticeArcError::Unauthorized(_)
113        | LatticeArcError::SecurityViolation(_)
114        | LatticeArcError::PolicyViolation(_)
115        | LatticeArcError::ComplianceViolation(_) => ErrorSeverity::High,
116
117        LatticeArcError::NetworkError(_)
118        | LatticeArcError::DatabaseError(_)
119        | LatticeArcError::IoError(_)
120        | LatticeArcError::HardwareError(_)
121        | LatticeArcError::ServiceUnavailable(_) => ErrorSeverity::Medium,
122
123        _ => ErrorSeverity::Low,
124    }
125}

127/// Check if error requires immediate security response.
128///
129/// Returns true for Critical and High severity errors that require
130/// immediate attention from security personnel.
131#[must_use]
132pub fn requires_security_response(error: &LatticeArcError) -> bool {
133    matches!(get_error_severity(error), ErrorSeverity::Critical | ErrorSeverity::High)
