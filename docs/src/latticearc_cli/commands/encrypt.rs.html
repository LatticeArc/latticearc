1//! Encrypt command — authenticated encryption of files.
2//!
3//! Supports three encryption modes:
4//!
5//! - **AES-256-GCM** (SP 800-38D) — symmetric authenticated encryption using a
6//!   shared 32-byte key. Provides confidentiality + integrity. Each encryption
7//!   uses a random 12-byte nonce (never reused).
8//! - **Hybrid** (ML-KEM-768 + X25519 + AES-256-GCM) — post-quantum hybrid
9//!   encryption using a recipient's public key. Generates an ephemeral keypair
10//!   internally.
11//! - **PQ-only** (ML-KEM + AES-256-GCM) — pure post-quantum encryption without
12//!   a classical (X25519) component. Required for CNSA 2.0 compliance.
13//!
14//! **Output format:** The encrypted data is written as a self-contained JSON file
15//! containing the Base64-encoded ciphertext, nonce, and algorithm metadata. This
16//! JSON format ensures the encrypted file can be stored, transmitted, and
17//! decrypted without any external state.
18

25/// True when the key file is a pure-PQ ML-KEM public key (no classical
26/// X25519 component). Hybrid `HybridMlKem*X25519` PKs return false.
27/// Used by both the use-case-driven path and the expert-mode inference
28/// to route pure-PQ keys away from the hybrid splitter (which would
29/// misinterpret the trailing bytes as X25519 and produce garbage
30/// ciphertext). Single source of truth — round-26 simplify pass.
31fn is_pq_only_key(key_file: &KeyFile) -> bool {
32    use latticearc::unified_api::key_format::KeyAlgorithm;
33    matches!(
34        key_file.portable_key().algorithm(),
35        KeyAlgorithm::MlKem512 | KeyAlgorithm::MlKem768 | KeyAlgorithm::MlKem1024
36    )
37}

39/// Encryption mode.
40#[derive(Debug, Clone, Copy, ValueEnum)]
41pub(crate) enum EncryptMode {
42    /// AES-256-GCM symmetric encryption (FIPS validated).
43    Aes256Gcm,
44    /// ChaCha20-Poly1305 symmetric encryption (RFC 8439, non-FIPS).
45    Chacha20Poly1305,
46    /// Hybrid ML-KEM-768 + X25519 + AES-256-GCM (generates ephemeral keypair).
47    Hybrid,
48    /// PQ-only ML-KEM + AES-256-GCM (no classical component, CNSA 2.0).
49    PqOnly,
50}

52/// Arguments for the `encrypt` subcommand.
53#[derive(Args)]
54pub(crate) struct EncryptArgs {
55    /// Encryption mode (expert override). Ignored when --use-case is provided.
56    /// Round-26 audit fix (M20): clap-level `conflicts_with` makes the
57    /// constraint visible in --help instead of just emitting a runtime
58    /// warning.
59    #[arg(short, long, value_enum, conflicts_with = "use_case")]
60    pub mode: Option<EncryptMode>,
61    /// Use case for automatic algorithm selection (recommended).
62    #[arg(long, value_parser = super::common::parse_use_case,
63          value_name = "USE_CASE")]
64    pub use_case: Option<latticearc::types::types::UseCase>,
65    /// Security level override (default: high).
66    #[arg(long, value_parser = super::common::parse_security_level,
67          value_name = "LEVEL")]
68    pub security_level: Option<latticearc::types::types::SecurityLevel>,
69    /// Compliance mode (default, fips, cnsa-2.0).
70    #[arg(long, value_parser = super::common::parse_compliance,
71          value_name = "MODE")]
72    pub compliance: Option<latticearc::types::types::ComplianceMode>,
73    /// Input file to encrypt (reads from stdin if omitted).
74    #[arg(short, long)]
75    pub input: Option<PathBuf>,
76    /// Output file for encrypted data (writes to stdout if omitted).
77    #[arg(short, long)]
78    pub output: Option<PathBuf>,
79    /// Key file path (symmetric key for AES, public key for hybrid).
80    #[arg(short, long)]
81    pub key: PathBuf,
82    /// Overwrite the output file if it already exists. Default: false
83    /// (refuses to clobber). Round-26 audit fix (H12).
84    #[arg(long)]
85    pub force: bool,
86}

88/// Execute the encrypt command.
89pub(crate) fn run(args: EncryptArgs) -> Result<()> {
90    let plaintext = read_input(&args.input)?;
91    let key_file = KeyFile::read_from(&args.key)?;
92
93    // Use-case-driven path
94    if args.use_case.is_some() {
95        if args.mode.is_some() {
96            eprintln!(
97                "Warning: --mode is ignored when --use-case is provided. \
98                 The library selects the optimal encryption scheme automatically. \
99                 For PQ-only with a use case, pass --use-case <X> on its own \
100                 with a PQ-only key (the library auto-detects CryptoMode from the key type)."
101            );
102        }
103        let json_output = encrypt_with_config(&plaintext, &key_file, &args)?;
104        return write_output(&args.output, &json_output, args.force);
105    }
106
107    // Expert path (or default). When `--mode` is omitted, infer it from
108    // the key file's type — a hybrid PK should default to hybrid, a
109    // symmetric key to AES-256-GCM. Without this inference (round-7
110    // audit fix #12), passing a hybrid PK without `--mode hybrid`
111    // produced "Expected symmetric key file, got Public" — accurate
112    // but surprising for the common case.
113    //
114    // Round-26 audit fix (H13): branch on the key's algorithm so
115    // pure-PQ ML-KEM PKs are routed to `PqOnly` and hybrid
116    // `HybridMlKem*X25519` PKs to `Hybrid`. Previously we forced
117    // `Hybrid` for any `KeyType::Public`, which made the hybrid splitter
118    // treat the trailing 32 bytes of an ML-KEM-768 PK (1184 bytes) as
119    // X25519 → garbage ciphertext. The use-case path already handled
120    // this correctly via `is_pq_only`; the expert path was the gap.
121    let mode = match args.mode {
122        Some(m) => m,
123        None => match key_file.key_type {
124            KeyType::Public => {
125                if is_pq_only_key(&key_file) {
126                    EncryptMode::PqOnly
127                } else {
128                    EncryptMode::Hybrid
129                }
130            }
131            _ => EncryptMode::Aes256Gcm,
132        },
133    };
134    let json_output = match mode {
135        EncryptMode::Aes256Gcm => encrypt_symmetric(&plaintext, &key_file)?,
136        EncryptMode::Chacha20Poly1305 => encrypt_chacha20(&plaintext, &key_file)?,
137        EncryptMode::Hybrid => encrypt_hybrid(&plaintext, &key_file)?,
138        EncryptMode::PqOnly => encrypt_pq_only_mode(&plaintext, &key_file, &args)?,
139    };
140
141    write_output(&args.output, &json_output, args.force)
142}

144/// Encrypt using the library's unified API with use-case-driven config.
145fn encrypt_with_config(plaintext: &[u8], key_file: &KeyFile, args: &EncryptArgs) -> Result<String> {
146    let config = super::common::build_config(args.use_case, args.security_level, &args.compliance);
147
148    // Determine key type from the key file and encrypt
149    match key_file.key_type {
150        KeyType::Symmetric => {
151            let key_bytes = key_file.key_bytes()?;
152            let config = config.force_scheme(latticearc::CryptoScheme::Symmetric);
153            let encrypted = latticearc::encrypt(
154                plaintext,
155                latticearc::EncryptKey::Symmetric(&key_bytes),
156                config,
157            )
158            .map_err(|e| anyhow::anyhow!("Encryption failed: {e}"))?;
159            latticearc::unified_api::serialization::serialize_encrypted_output(&encrypted)
160                .map_err(|e| anyhow::anyhow!("Serialization failed: {e}"))
161        }
162        KeyType::Public => {
163            // Auto-route based on the key's algorithm: pure-PQ ML-KEM keys
164            // must go through the PqOnly path; hybrid keys stay on Hybrid.
165            // Without this split, `encrypt --use-case X --key pure_pq.pub.json`
166            // would try to parse a pure-PQ key as hybrid and fail with a
167            // length mismatch — the same bug class as the signing-path bug
168            // fixed alongside this change.
169            if is_pq_only_key(key_file) {
170                // Delegate to the dedicated PQ-only path so CryptoMode and
171                // key-parsing are consistent with the algorithm.
172                return encrypt_pq_only_mode(plaintext, key_file, args);
173            }
174
175            let alg = key_file.portable_key().algorithm();
176            let pk_bytes = key_file.key_bytes()?;
177            let pk = crate::keyfile::parse_hybrid_kem_pk_from_bytes(&pk_bytes, alg)?;
178            let encrypted =
179                latticearc::encrypt(plaintext, latticearc::EncryptKey::Hybrid(&pk), config)
180                    .map_err(|e| anyhow::anyhow!("Encryption failed: {e}"))?;
181            latticearc::unified_api::serialization::serialize_encrypted_output(&encrypted)
182                .map_err(|e| anyhow::anyhow!("Serialization failed: {e}"))
183        }
184        _ => bail!(
185            "Expected symmetric or public key file for encryption, got {:?}",
186            key_file.key_type
187        ),
188    }
189}

191fn encrypt_symmetric(plaintext: &[u8], key_file: &KeyFile) -> Result<String> {
192    if key_file.key_type != KeyType::Symmetric {
193        bail!("Expected symmetric key file, got {:?}", key_file.key_type);
194    }
195
196    let key_bytes = key_file.key_bytes()?;
197    if key_bytes.len() != 32 {
198        bail!("AES-256 requires a 32-byte key, got {} bytes", key_bytes.len());
199    }
200
201    let encrypted = latticearc::encrypt(
202        plaintext,
203        latticearc::EncryptKey::Symmetric(&key_bytes),
204        latticearc::CryptoConfig::new().force_scheme(latticearc::CryptoScheme::Symmetric),
205    )
206    .map_err(|e| anyhow::anyhow!("Encryption failed: {e}"))?;
207
208    latticearc::unified_api::serialization::serialize_encrypted_output(&encrypted)
209        .map_err(|e| anyhow::anyhow!("Serialization failed: {e}"))
210}

212fn encrypt_chacha20(plaintext: &[u8], key_file: &KeyFile) -> Result<String> {

221        latticearc::CryptoConfig::new().force_scheme(latticearc::CryptoScheme::SymmetricChaCha20),

223    .map_err(|e| anyhow::anyhow!("ChaCha20-Poly1305 encryption failed: {e}"))?;
224
225    latticearc::unified_api::serialization::serialize_encrypted_output(&encrypted)
226        .map_err(|e| anyhow::anyhow!("Serialization failed: {e}"))

229fn encrypt_hybrid(plaintext: &[u8], key_file: &KeyFile) -> Result<String> {
230    if key_file.key_type != KeyType::Public {
231        bail!("Expected public key file for hybrid encryption, got {:?}", key_file.key_type);
232    }
233
234    let pk_bytes = key_file.key_bytes()?;
235    // The legacy `--algorithm hybrid` path historically wrote ML-KEM-768
236    // public keys, so the algorithm is fixed here. If the key file's
237    // algorithm tag is one of the typed hybrid variants, prefer that;
238    // otherwise fall back to the historic 768.
239    let alg = key_file.portable_key().algorithm();
240    let resolved_alg = if matches!(
241        alg,
242        latticearc::unified_api::key_format::KeyAlgorithm::HybridMlKem512X25519
243            | latticearc::unified_api::key_format::KeyAlgorithm::HybridMlKem768X25519
244            | latticearc::unified_api::key_format::KeyAlgorithm::HybridMlKem1024X25519
245    ) {
246        alg
247    } else {
248        latticearc::unified_api::key_format::KeyAlgorithm::HybridMlKem768X25519
249    };
250    let pk = crate::keyfile::parse_hybrid_kem_pk_from_bytes(&pk_bytes, resolved_alg)?;
251
252    let encrypted = latticearc::encrypt(
253        plaintext,
254        latticearc::EncryptKey::Hybrid(&pk),
255        latticearc::CryptoConfig::new(),
256    )
257    .map_err(|e| anyhow::anyhow!("Hybrid encryption failed: {e}"))?;
258
259    latticearc::unified_api::serialization::serialize_encrypted_output(&encrypted)
260        .map_err(|e| anyhow::anyhow!("Serialization failed: {e}"))
261}

263fn encrypt_pq_only_mode(
264    plaintext: &[u8],
265    key_file: &KeyFile,
266    args: &EncryptArgs,
267) -> Result<String> {
268    if key_file.key_type != KeyType::Public {
269        bail!("Expected public key file for PQ-only encryption, got {:?}", key_file.key_type);
270    }
271
272    let pk_bytes = key_file.key_bytes()?;
273    let level =
274        super::common::resolve_ml_kem_level(key_file.portable_key().algorithm()).map_err(|e| {
275            anyhow::anyhow!("{e}. Generate one with: latticearc-cli keygen --algorithm ml-kem768")
276        })?;
277
278    let pq_pk = latticearc::PqOnlyPublicKey::from_bytes(level, &pk_bytes)
279        .map_err(|e| anyhow::anyhow!("Invalid PQ-only public key: {e}"))?;
280
281    let security_level = super::common::ml_kem_to_security_level(level);
282    let mut config = latticearc::CryptoConfig::new()
283        .crypto_mode(latticearc::CryptoMode::PqOnly)
284        .security_level(args.security_level.unwrap_or(security_level));
285    if let Some(ref compliance) = args.compliance {
286        config = config.compliance(compliance.clone());
287    }
288
289    let encrypted = latticearc::encrypt(plaintext, latticearc::EncryptKey::PqOnly(&pq_pk), config)
290        .map_err(|e| anyhow::anyhow!("PQ-only encryption failed: {e}"))?;
291
292    latticearc::unified_api::serialization::serialize_encrypted_output(&encrypted)
293        .map_err(|e| anyhow::anyhow!("Serialization failed: {e}"))
294}

296fn read_input(path: &Option<PathBuf>) -> Result<Vec<u8>> {
297    // Round-9 audit fix #3: route through the shared helper.
298    super::common::read_file_or_stdin(
299        path.as_deref(),

301        "encrypt",
302    )
303}

305fn write_output(path: &Option<PathBuf>, data: &str, force: bool) -> Result<()> {
306    match path {
307        Some(p) => {
308            // Atomic write — closes the partial-file confidentiality
309            // window for decrypted material at rest. Same helper the
310            // keyfile writer uses (round-7 audit fix #18).
311            // Round-26 audit fix (H12): only overwrite when --force is
312            // passed; otherwise refuse to clobber, matching keygen's
313            // default-safe behavior.
314            // LINT-OK: public-write-ciphertext (encrypted data is not secret)
315            latticearc::unified_api::atomic_write::AtomicWrite::new(data.as_bytes())
316                .overwrite_existing(force)
317                .write(p)
318                .with_context(|| {
319                    format!("Failed to write {} (use --force to overwrite)", p.display())
320                })?;
321            eprintln!("Encrypted data written to: {}", p.display());
322        }
323        None => {
324            // `print!` (not `println!`) — byte-exact stdout for
325            // pipelines that hash or chain into other tools (round-7
326            // audit fix #14). Callers that want a trailing newline can
327            // redirect through `cat` or append themselves.
328            print!("{data}");
329        }
330    }
331    Ok(())
