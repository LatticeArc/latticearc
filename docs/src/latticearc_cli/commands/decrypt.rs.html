1//! Decrypt command — authenticated decryption of files.
2//!
3//! Reverses the encryption performed by the `encrypt` command. Reads the
4//! JSON-formatted encrypted file, validates the authentication tag, and
5//! recovers the original plaintext.
6//!
7//! **Integrity check:** AES-256-GCM decryption fails if the ciphertext has
8//! been modified in any way (the 16-byte authentication tag won't match).
9//! This protects against tampering and truncation attacks.
10//!
11//! **Key types accepted:**
12//! - `symmetric` — for AES-256-GCM decryption
13//! - `public` — rejected with a clear error message
14//! - `secret` — for hybrid or PQ-only decryption (auto-detected from scheme)
15
16use anyhow::{Context, Result, bail};
17use clap::Args;
18use std::path::PathBuf;
19
20use crate::keyfile::{KeyFile, KeyType};
21
22/// Arguments for the `decrypt` subcommand.
23#[derive(Args)]
24pub(crate) struct DecryptArgs {
25    /// Input file containing encrypted JSON (reads from stdin if omitted).
26    #[arg(short, long)]
27    pub input: Option<PathBuf>,
28    /// Output file for decrypted data (writes to stdout if omitted).
29    #[arg(short, long)]
30    pub output: Option<PathBuf>,
31    /// Key file path (symmetric key or hybrid secret key).
32    #[arg(short, long)]
33    pub key: PathBuf,
34    /// Overwrite the output file if it already exists. Default: false.
35    #[arg(long)]
36    pub force: bool,
37}
38
39/// Execute the decrypt command.
40pub(crate) fn run(args: DecryptArgs) -> Result<()> {
41    let cipher_json = read_input_string(&args.input)?;
42    let key_file = KeyFile::read_from(&args.key)?;
43
44    let encrypted =
45        latticearc::unified_api::serialization::deserialize_encrypted_output(&cipher_json)
46            .map_err(|e| anyhow::anyhow!("Failed to parse encrypted data: {e}"))?;
47
48    let plaintext = match key_file.key_type {
49        KeyType::Symmetric => decrypt_symmetric(&encrypted, &key_file)?,
50        KeyType::Secret => {
51            // Determine if this is a hybrid or PQ-only secret key
52            if encrypted.scheme().requires_pq_key() {
53                decrypt_pq_only(&encrypted, &key_file)?
54            } else {
55                decrypt_hybrid(&encrypted, &key_file)?
56            }
57        }
58        KeyType::Public => {
59            bail!("Cannot decrypt with a public key. Provide the secret key.");
60        }
61        _ => anyhow::bail!("Unsupported KeyType variant"),
62    };
63
64    write_output(&args.output, &plaintext, args.force)
65}

67fn decrypt_symmetric(
68    encrypted: &latticearc::EncryptedOutput,
69    key_file: &KeyFile,
70) -> Result<zeroize::Zeroizing<Vec<u8>>> {
71    let key_bytes = key_file.key_bytes()?;
72    if key_bytes.len() != 32 {
73        bail!("AES-256 requires a 32-byte key, got {} bytes", key_bytes.len());
74    }
75
76    latticearc::decrypt(
77        encrypted,
78        latticearc::DecryptKey::Symmetric(&key_bytes),
79        latticearc::CryptoConfig::new(),
80    )
81    // the round-26 M19 fix kept a `(symmetric)`
82    // parenthetical and `{e}` interpolation that re-introduced the
83    // oracle the comment claimed to close — scripted callers could
84    // distinguish symmetric vs hybrid vs PQ-only branches from stderr,
85    // and the upstream `e` leaked inner library wording. Now bare
86    // "Decryption failed", uniform across all three branches; per-stage
87    // cause is preserved via `tracing::debug!` for operator debugging.
88    .map_err(|e| {
89        tracing::debug!(error = %e, scheme = "symmetric", "decrypt failed");
90        anyhow::anyhow!("Decryption failed")
91    })
92}

94fn decrypt_hybrid(
95    encrypted: &latticearc::EncryptedOutput,
96    key_file: &KeyFile,
97) -> Result<zeroize::Zeroizing<Vec<u8>>> {
98    // Reconstruct HybridKemSecretKey from the PortableKey.
99    // ML-KEM public key is extracted from the secret key file's metadata
100    // (stored at keygen time, following PKCS#12 bundling pattern).
101    let hybrid_sk = key_file
102        .portable_key()
103        .to_hybrid_secret_key()
104        .map_err(|e| anyhow::anyhow!("Failed to reconstruct hybrid secret key: {e}"))?;
105
106    latticearc::decrypt(
107        encrypted,
108        latticearc::DecryptKey::Hybrid(&hybrid_sk),
109        latticearc::CryptoConfig::new(),
110    )
111    .map_err(|e| {
112        // bare "Decryption failed"; cause to tracing.
113        tracing::debug!(error = %e, scheme = "hybrid", "decrypt failed");
114        anyhow::anyhow!("Decryption failed")
115    })
116}

118fn decrypt_pq_only(
119    encrypted: &latticearc::EncryptedOutput,
120    key_file: &KeyFile,
121) -> Result<zeroize::Zeroizing<Vec<u8>>> {
122    let level = super::common::resolve_ml_kem_level(key_file.portable_key().algorithm())
123        .map_err(|e| anyhow::anyhow!("{e}"))?;
124
125    let sk_bytes = key_file.key_bytes()?;
126
127    // derive the recipient PK from the SK's embedded PK
128    // (FIPS 203 §6.1) instead of trusting the unauthenticated
129    // `ml_kem_pk` metadata field. A file-write attacker who could
130    // previously swap that metadata to break the HPKE channel binding
131    // can no longer affect the derivation — the PK comes from inside
132    // the SK blob itself.
133    let pq_sk = latticearc::PqOnlySecretKey::from_sk_bytes(level, &sk_bytes)
134        .map_err(|e| anyhow::anyhow!("Invalid PQ-only secret key: {e}"))?;
135
136    latticearc::decrypt(
137        encrypted,
138        latticearc::DecryptKey::PqOnly(&pq_sk),
139        latticearc::CryptoConfig::new().crypto_mode(latticearc::CryptoMode::PqOnly),
140    )
141    .map_err(|e| {
142        // bare "Decryption failed"; cause to tracing.
143        tracing::debug!(error = %e, scheme = "pq-only", "decrypt failed");
144        anyhow::anyhow!("Decryption failed")
145    })
146}

148fn read_input_string(path: &Option<PathBuf>) -> Result<String> {
149    // route through the shared helper.
150    super::common::read_file_or_stdin_string(
151        path.as_deref(),
152        super::common::CLI_MAX_DECRYPTION_INPUT_BYTES,
153        "decrypt",
154    )
155}

157fn write_output(path: &Option<PathBuf>, data: &[u8], force: bool) -> Result<()> {
158    if let Some(p) = path {
159        // Atomic write with 0o600. `tempfile::NamedTempFile` already
160        // creates files with mode 0o600 and `persist()` preserves that;
161        // `.secret_mode()` is retained as defense-in-depth in case
162        // tempfile's default ever changes.
163        // only overwrite when --force is
164        // passed.
165        latticearc::unified_api::atomic_write::AtomicWrite::new(data)
166            .secret_mode()
167            .overwrite_existing(force)
168            .write(p)
169            .with_context(|| {
170                format!("Failed to write {} (use --force to overwrite)", p.display())
171            })?;
172        // Path on stderr would leak through process accounting and log
173        // aggregation; route to tracing::debug! instead.
174        tracing::debug!(path = %p.display(), "decrypted data written");
175    } else {
176        // Stdout-as-default-target hazard: writing decrypted plaintext
177        // to a TTY exposes it to recorded shell sessions, screen
178        // sharing, scrollback, and any logging proxy. When the user is
179        // interactive, surface a one-line warning to stderr. Pipelines
180        // (`| tee`, file redirection) where stdout is not a TTY get no
181        // warning.
182        use std::io::IsTerminal;
183        if std::io::stdout().is_terminal() {
184            eprintln!(
185                "warning: decrypted plaintext is being written to a TTY. \
186                 Pass --output <file> to write to disk with 0600 permissions, \
187                 or pipe stdout to avoid leaving secrets in scrollback."
188            );
189        }
190        // Try to print as UTF-8, fall back to hex. The hex encoding is
191        // wrapped in `Zeroizing<String>` so a derived plaintext copy
192        // doesn't outlive the source `Zeroizing<Vec<u8>>` — otherwise
193        // the encoded string would linger on the heap until allocator
194        // reclaim.
195        if let Ok(s) = std::str::from_utf8(data) {
196            print!("{s}");
197        } else {
198            let encoded = zeroize::Zeroizing::new(hex::encode(data));
199            print!("{}", encoded.as_str());
200        }
201    }
202    Ok(())
203}
