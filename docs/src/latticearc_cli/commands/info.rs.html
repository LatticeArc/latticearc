1//! Info command — display version, FIPS status, and supported algorithms.
2//!
3//! Shows a human-readable summary of the CLI and library versions, whether
4//! the FIPS 140-3 backend (aws-lc-rs) is linked and operational, and a
5//! complete list of every algorithm available across all command categories
6//! (encryption, signatures, hashing, key derivation).
7//!
8//! **Self-tests:** The output includes a `Self-tests passed` line that
9//! reports whether the library's power-on self-tests (KAT vectors) passed.
10//! This is a FIPS 140-3 requirement for validated modules.
11
12use anyhow::Result;
13use clap::Args;
14
15/// Arguments for the `info` subcommand.
16#[derive(Args)]
17pub(crate) struct InfoArgs;
18
19/// Execute the info command.
20///
21/// Returns `Result<()>` for consistency with other command handlers,
22/// even though this function is infallible.
23#[allow(clippy::unnecessary_wraps)]
24pub(crate) fn run(_args: InfoArgs) -> Result<()> {
25    println!("LatticeArc CLI v{}", env!("CARGO_PKG_VERSION"));
26    println!("Library:  latticearc v{}", latticearc::VERSION);
27    println!();
28    // cross-reference `self_tests_passed()`
29    // before printing the "validated backend" claim. The compile-time
30    // `fips_available()` flag is necessary but not sufficient — a
31    // FIPS-validated module is only operational while POST has
32    // succeeded. Reading the runtime POST state once and reusing it
33    // makes both lines self-consistent.
34    let post_ok = latticearc::unified_api::self_tests_passed();
35    println!(
36        "FIPS 140-3 backend: {}",
37        if latticearc::fips_available() && post_ok {
38            // distinguish "validated backend"
39            // (aws-lc-rs is FIPS 140-3 validated as a stand-alone
40            // module) from "validated module" (this binary linking
41            // aws-lc-rs is NOT itself certified). The previous
42            // "available (aws-lc-rs)" wording read like a FIPS-
43            // certified-product claim to operators glancing at
44            // `--info`.
45            "validated backend (aws-lc-rs); this binary links the validated backend but is NOT itself a FIPS 140-3 certified module — see docs/NIST_COMPLIANCE.md and docs/FIPS_SECURITY_POLICY.md"
46        } else if latticearc::fips_available() {
47            "validated backend present but POST has not passed; module is non-operational"
48        } else {
49            "not available"
50        }
51    );
52    println!("Self-tests passed:  {}", post_ok);
53    println!();
54    println!("Supported Algorithms:");
55    println!("  Encryption:");
56    println!("    AES-256-GCM             (FIPS validated via aws-lc-rs)");
57    println!("    ChaCha20-Poly1305       (RFC 8439, non-FIPS)");
58    println!("    ML-KEM-512/768/1024     (FIPS 203, key encapsulation)");
59    println!("    Hybrid ML-KEM+X25519    (ML-KEM + ECDH + HKDF + AES-256-GCM)");
60    println!();
61    println!("  Signatures:");
62    println!("    ML-DSA-44/65/87         (FIPS 204, lattice-based)");
63    println!("    SLH-DSA-SHAKE-128s      (FIPS 205, hash-based, stateless)");
64    println!("    FN-DSA-512              (FIPS 206, compact lattice)");
65    println!("    Ed25519                 (RFC 8032, classical)");
66    println!("    Hybrid ML-DSA+Ed25519   (combined PQ + classical)");
67    println!();
68    println!("  Hashing:");
69    println!("    SHA3-256                (FIPS 202, default)");
70    println!("    SHA-256                 (FIPS 180-4)");
71    println!("    SHA-512                 (FIPS 180-4)");
72    println!("    BLAKE2b-256             (RFC 7693)");
73    println!();
74    println!("  Key Derivation:");
75    println!("    HKDF-SHA256             (SP 800-56C)");
76    println!("    PBKDF2-HMAC-SHA256      (SP 800-132)");
77    Ok(())
78}
