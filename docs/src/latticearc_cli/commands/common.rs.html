47/// Parse a use case from a CLI string.
48pub(crate) fn parse_use_case(s: &str) -> Result<UseCase, String> {
49    USE_CASES.iter().find(|(name, _)| *name == s).map(|(_, uc)| *uc).ok_or_else(|| {
50        let valid: Vec<&str> = USE_CASES.iter().map(|(n, _)| *n).collect();
51        format!("Unknown use case '{s}'. Valid values:\n  {}", valid.join(", "))
52    })
53}

57    SECURITY_LEVELS.iter().find(|(name, _)| *name == s).map(|(_, sl)| *sl).ok_or_else(|| {
58        let valid: Vec<&str> = SECURITY_LEVELS.iter().map(|(n, _)| *n).collect();
59        format!("Unknown security level '{s}'. Valid: {}", valid.join(", "))
60    })

73/// Resolve ML-KEM security level from a key file's algorithm.
74///
75/// Maps `KeyAlgorithm::MlKem512/768/1024` to the corresponding `MlKemSecurityLevel`.
76/// Returns an error for non-ML-KEM algorithms.
77pub(crate) fn resolve_ml_kem_level(
78    algorithm: latticearc::unified_api::key_format::KeyAlgorithm,
79) -> Result<latticearc::primitives::kem::ml_kem::MlKemSecurityLevel, String> {
80    use latticearc::primitives::kem::ml_kem::MlKemSecurityLevel;
81    use latticearc::unified_api::key_format::KeyAlgorithm;
82    match algorithm {
83        KeyAlgorithm::MlKem512 => Ok(MlKemSecurityLevel::MlKem512),
84        KeyAlgorithm::MlKem768 => Ok(MlKemSecurityLevel::MlKem768),
85        KeyAlgorithm::MlKem1024 => Ok(MlKemSecurityLevel::MlKem1024),
86        _ => Err(format!(
87            "PQ-only operation requires an ML-KEM key (ml-kem512/768/1024), got {:?}",
88            algorithm
89        )),
90    }
91}

102/// Build a `CryptoConfig` from optional CLI values.
103pub(crate) fn build_config<'a>(
104    use_case: Option<UseCase>,
105    security_level: Option<SecurityLevel>,
106    compliance: &Option<ComplianceMode>,
107) -> latticearc::CryptoConfig<'a> {
108    let mut config = latticearc::CryptoConfig::new();
109
110    if let Some(uc) = use_case {
111        config = config.use_case(uc);
112    }
113
114    if let Some(sl) = security_level {
115        config = config.security_level(sl);
116    }
117
118    if let Some(cm) = compliance {
119        config = config.compliance(cm.clone());
120    }
121
122    config
123}

125/// Maximum input size for symmetric / KEM-bound encrypt commands. Matches
126/// `latticearc::primitives::resource_limits` defaults (100 MiB plaintext).
127pub(crate) const CLI_MAX_ENCRYPTION_INPUT_BYTES: u64 = 100 * 1024 * 1024;
128
129/// Maximum input size for decrypt commands. Decrypt reads a JSON envelope
130/// containing the ciphertext (base64-encoded), so the wire size can be ~1.5x
131/// the plaintext limit. Cap at 200 MiB to give the library room to enforce
132/// its 100 MiB plaintext limit on the decoded bytes.
133pub(crate) const CLI_MAX_DECRYPTION_INPUT_BYTES: u64 = 200 * 1024 * 1024;
134
135/// Maximum input size for sign / verify commands.
136///
137/// The CLI pre-check exists purely for OOM protection — it is NOT a duplicate
138/// of the library's semantic limits. The library's unified `sign_with_key`
139/// path independently enforces its own `validate_signature_size` bound (64
140/// KiB by default), while legacy primitive paths (`sign_ed25519`,
141/// `sign_pq_ml_dsa`, etc.) accept arbitrarily large inputs. The CLI aligns
142/// with the encryption limit so that any legitimate operation succeeds and
143/// only truly runaway inputs are blocked at the CLI boundary.
144pub(crate) const CLI_MAX_SIGNATURE_INPUT_BYTES: u64 = 100 * 1024 * 1024;
145
146/// Maximum input size for hash commands. Hashing has no library-level limit,
147/// but a 1 GiB ceiling protects against accidental OOM if the user pipes
148/// `/dev/urandom` or a multi-gigabyte file. Adjust if you need to hash
149/// larger blobs (use streaming instead).
150pub(crate) const CLI_MAX_HASH_INPUT_BYTES: u64 = 1024 * 1024 * 1024;

152/// read a file with a hard size cap by opening
153/// once and using a `Take` adapter. The previous
154/// `enforce_input_size_limit` + `std::fs::read` pattern was removed
155/// in this round — it stat'd via `metadata().len()`, which returns 0
156/// for `/dev/zero`, FIFOs, `/proc/*`, etc., letting the subsequent
157/// read consume unbounded bytes from those special files. The
158/// stat-then-open pattern was also a TOCTOU surface (round-26 H15
159/// fixed the same shape in `keyfile.rs`). This open-once helper is
160/// the single canonical entry point for capped CLI reads — used by
161/// every command that takes a file path (encrypt / sign / hash /
162/// verify / kdf).
163///
164/// this helper does not stat first, so the
165/// previous "non-ENOENT silent bypass" failure mode is gone by
166/// construction — `File::open` surfaces every error path uniformly.
167///
168/// Returns the file contents (≤ `limit_bytes`) or a user-facing error
169/// referencing `operation`.
170pub(crate) fn read_file_with_cap(
171    path: &std::path::Path,
172    limit_bytes: u64,
173    operation: &str,
174) -> anyhow::Result<Vec<u8>> {
175    use anyhow::Context;
176    use std::io::Read;
177    let file =
178        std::fs::File::open(path).with_context(|| format!("Failed to open {}", path.display()))?;
179    let mut buf = Vec::new();
180    let mut limited = file.take(limit_bytes.saturating_add(1));
181    limited.read_to_end(&mut buf).with_context(|| format!("Failed to read {}", path.display()))?;
182    if buf.len() as u64 > limit_bytes {
183        anyhow::bail!(
184            "Input from {} exceeded {limit_bytes} bytes; the {operation} command rejects \
185             inputs larger than this. Split the file or use a streaming workflow.",
186            path.display(),
187        );
188    }
189    Ok(buf)
190}

192/// Read bytes from stdin, capped at `limit_bytes`.
193///
194/// Since stdin cannot be stat-ed, the cap is enforced by a bounded reader
195/// (`take(limit + 1)`) so we can detect overflow by observing whether the
196/// reader returned more than `limit_bytes`. Returns a user-facing error
197/// mentioning `operation` on overflow.
198pub(crate) fn read_stdin_with_limit(limit_bytes: u64, operation: &str) -> anyhow::Result<Vec<u8>> {
199    use anyhow::Context;
200    use std::io::Read;
201    let mut buf = Vec::new();
202    let mut limited = std::io::stdin().take(limit_bytes.saturating_add(1));
203    limited.read_to_end(&mut buf).context("Failed to read from stdin")?;
204    if buf.len() as u64 > limit_bytes {
205        anyhow::bail!(
206            "Stdin input exceeded {limit_bytes} bytes; the {operation} command rejects \
207             inputs larger than this. Split the file or use a streaming workflow."
208        );
209    }
210    Ok(buf)
211}

213/// Read input bytes from a file path or stdin (when `path` is `None`),
214/// applying the same `limit_bytes` cap to both sources.
215///
216/// Centralises the duplicated `if let Some(path) = … else stdin`
217/// pattern that previously lived in encrypt / decrypt / sign / verify /
218/// hash. Round-8 audit fix #13.
219///
220/// # Errors
221///
222/// Returns the underlying I/O error or the size-cap rejection from
223/// [`enforce_input_size_limit`] / [`read_stdin_with_limit`].
224pub(crate) fn read_file_or_stdin(
225    path: Option<&std::path::Path>,
226    limit_bytes: u64,
227    operation: &str,
228) -> anyhow::Result<Vec<u8>> {
229    if let Some(p) = path {
230        // use the open-once `read_file_with_cap` helper
231        // instead of `enforce_input_size_limit` + `std::fs::read`.
232        // The old shape relied on `metadata().len()` which returns 0
233        // for `/dev/zero`, FIFOs, and `/proc/*` — letting the
234        // subsequent read consume unbounded bytes. This helper is
235        // used by encrypt / sign / hash / verify --input, so the gap
236        // affected every CLI command that takes a file path.
237        read_file_with_cap(p, limit_bytes, operation)
238    } else {
239        read_stdin_with_limit(limit_bytes, operation)
240    }
241}

243/// Like [`read_file_or_stdin`] but returns a UTF-8 `String`. Used by
244/// commands that operate on JSON envelopes (e.g. `decrypt` parses an
245/// `EncryptedOutput` JSON envelope before any cryptographic work).
246///
247/// # Errors
248///
249/// Returns the underlying I/O error, the size-cap rejection, or a
250/// UTF-8 validation error if the input is not valid UTF-8 text.
251pub(crate) fn read_file_or_stdin_string(
252    path: Option<&std::path::Path>,
253    limit_bytes: u64,
254    operation: &str,
255) -> anyhow::Result<String> {
256    use anyhow::Context;
257    if let Some(p) = path {
258        // same open-once pattern as
259        // `read_file_or_stdin`. UTF-8 validation happens after the
260        // capped read so a multi-gig non-UTF-8 file can't OOM the
261        // process before the validity check.
262        let bytes = read_file_with_cap(p, limit_bytes, operation)?;
263        String::from_utf8(bytes).with_context(|| format!("{} is not valid UTF-8", p.display()))
264    } else {
265        read_stdin_string_with_limit(limit_bytes, operation)
266    }
267}

270///
271/// Used by commands that operate on JSON envelopes (decrypt, verify) where
272/// the input must already be text.
273pub(crate) fn read_stdin_string_with_limit(
274    limit_bytes: u64,
275    operation: &str,
276) -> anyhow::Result<String> {
277    use anyhow::Context;
278    use std::io::Read;
279    let mut buf = String::new();

282    if buf.len() as u64 > limit_bytes {
283        anyhow::bail!(
284            "Stdin input exceeded {limit_bytes} bytes; the {operation} command rejects \
285             inputs larger than this. Split the file or use a streaming workflow."
286        );
287    }
288    Ok(buf)
289}

291/// Infer a `SecurityLevel` from a key file's algorithm.
292///
293/// Used by `sign`/`verify` when the caller loads an existing key file and
294/// hasn't explicitly chosen a scheme. The library's signature-scheme selector
295/// maps `SecurityLevel::{Standard, High, Maximum}` to
296/// `hybrid-ml-dsa-{44, 65, 87}-ed25519` respectively (see
297/// `latticearc::unified_api::selector::SimpleSchemeSelector::select_signature_scheme`),
298/// so passing the inferred level produces a config that will select the same
299/// scheme the key was generated under.
300///
301/// Returns `None` for non-signing algorithms (KEMs, symmetric) — callers
302/// should fall back to their own defaults or surface an error.
303pub(crate) fn infer_signature_security_level(
304    algorithm: latticearc::unified_api::key_format::KeyAlgorithm,
305) -> Option<SecurityLevel> {
306    use latticearc::unified_api::key_format::KeyAlgorithm;
307    match algorithm {
308        // Hybrid signing schemes — map to the security level the selector
309        // resolves back to the same scheme.
310        KeyAlgorithm::HybridMlDsa44Ed25519 => Some(SecurityLevel::Standard),
311        KeyAlgorithm::HybridMlDsa65Ed25519 => Some(SecurityLevel::High),
312        KeyAlgorithm::HybridMlDsa87Ed25519 => Some(SecurityLevel::Maximum),
313        // PQ-only ML-DSA — security level is the same, but the caller MUST
314        // also force PQ-only mode (see `infer_crypto_mode`). Otherwise the
315        // default `CryptoMode::Hybrid` will resolve to a hybrid scheme and
316        // reject the pure-PQ key as a length mismatch.
317        KeyAlgorithm::MlDsa44 => Some(SecurityLevel::Standard),
318        KeyAlgorithm::MlDsa65 => Some(SecurityLevel::High),
319        KeyAlgorithm::MlDsa87 => Some(SecurityLevel::Maximum),
320        _ => None,
321    }
322}

324/// Infer the `CryptoMode` from a key file's algorithm.
325///
326/// Pure-PQ algorithms (ML-DSA-44/65/87) require `CryptoMode::PqOnly`;
327/// hybrid algorithms (hybrid-ml-dsa-*-ed25519) require `CryptoMode::Hybrid`.
328/// Returning the correct mode here prevents the library from trying to parse
329/// a pure-PQ key as a hybrid key (or vice versa) and failing with a length
330/// mismatch.
331///
332/// Returns `None` for non-signing algorithms — callers fall back to the
333/// library default (hybrid).
334pub(crate) fn infer_crypto_mode(
335    algorithm: latticearc::unified_api::key_format::KeyAlgorithm,
336) -> Option<latticearc::CryptoMode> {
337    use latticearc::unified_api::key_format::KeyAlgorithm;
338    match algorithm {
339        KeyAlgorithm::HybridMlDsa44Ed25519
340        | KeyAlgorithm::HybridMlDsa65Ed25519
341        | KeyAlgorithm::HybridMlDsa87Ed25519 => Some(latticearc::CryptoMode::Hybrid),
342        KeyAlgorithm::MlDsa44 | KeyAlgorithm::MlDsa65 | KeyAlgorithm::MlDsa87 => {
343            Some(latticearc::CryptoMode::PqOnly)
344        }
345        _ => None,
346    }
347}

349/// Build a `CryptoConfig` for a sign/verify operation, filling in defaults
350/// from the key file's algorithm when the user hasn't specified a scheme.
351///
352/// Precedence (highest wins):
353///   1. Explicit `--use-case` or `--security-level`
354///   2. Level inferred from the key file's algorithm
355///   3. Library default
356pub(crate) fn build_signing_config<'a>(
357    use_case: Option<UseCase>,
358    security_level: Option<SecurityLevel>,
359    compliance: &Option<ComplianceMode>,
360    key_algorithm: latticearc::unified_api::key_format::KeyAlgorithm,
361) -> latticearc::CryptoConfig<'a> {
362    let user_chose_scheme = use_case.is_some() || security_level.is_some();
363
364    let effective_level = security_level.or_else(|| {
365        if use_case.is_some() { None } else { infer_signature_security_level(key_algorithm) }
366    });
367
368    let mut config = build_config(use_case, effective_level, compliance);
369
370    // Only infer mode when the user didn't pick a scheme explicitly — if they
371    // did, respect their choice (they may want to coerce a key through a
372    // different scheme for testing / migration).
373    if !user_chose_scheme && let Some(mode) = infer_crypto_mode(key_algorithm) {
374        config = config.crypto_mode(mode);
375    }
376
377    config
