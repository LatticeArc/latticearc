1//! Shared CLI types for use-case-driven operations.
2//!
3//! Maps CLI string values directly to library enums — no wrapper types.
4//! The library's `UseCase`, `SecurityLevel`, and `ComplianceMode` are used
5//! directly in clap args via custom `value_parser` functions.
6
7use latticearc::types::types::{ComplianceMode, SecurityLevel, UseCase};
8
9/// All valid use case CLI values. Single source of truth for help text and parsing.
10const USE_CASES: &[(&str, UseCase)] = &[

12    ("email-encryption", UseCase::EmailEncryption),
13    ("vpn-tunnel", UseCase::VpnTunnel),
14    ("api-security", UseCase::ApiSecurity),
15    ("file-storage", UseCase::FileStorage),
16    ("database-encryption", UseCase::DatabaseEncryption),
17    ("cloud-storage", UseCase::CloudStorage),
18    ("backup-archive", UseCase::BackupArchive),
19    ("config-secrets", UseCase::ConfigSecrets),
20    ("authentication", UseCase::Authentication),
21    ("session-token", UseCase::SessionToken),
22    ("digital-certificate", UseCase::DigitalCertificate),
23    ("key-exchange", UseCase::KeyExchange),
24    ("financial-transactions", UseCase::FinancialTransactions),
25    ("legal-documents", UseCase::LegalDocuments),
26    ("blockchain-transaction", UseCase::BlockchainTransaction),
27    ("healthcare-records", UseCase::HealthcareRecords),
28    ("government-classified", UseCase::GovernmentClassified),
29    ("payment-card", UseCase::PaymentCard),
30    ("iot-device", UseCase::IoTDevice),
31    ("firmware-signing", UseCase::FirmwareSigning),
32    ("audit-log", UseCase::AuditLog),
33];

35const SECURITY_LEVELS: &[(&str, SecurityLevel)] = &[
36    ("standard", SecurityLevel::Standard),
37    ("high", SecurityLevel::High),
38    ("maximum", SecurityLevel::Maximum),
39    ("quantum", SecurityLevel::Quantum),
40];

42#[allow(dead_code)] // CLI-04: used when --compliance flag is wired to commands
43const COMPLIANCE_MODES: &[(&str, ComplianceMode)] = &[
44    ("default", ComplianceMode::Default),
45    ("fips", ComplianceMode::Fips140_3),
46    ("cnsa-2.0", ComplianceMode::Cnsa2_0),
47];

49/// Parse a use case from a CLI string.
50pub(crate) fn parse_use_case(s: &str) -> Result<UseCase, String> {
51    USE_CASES.iter().find(|(name, _)| *name == s).map(|(_, uc)| *uc).ok_or_else(|| {
52        let valid: Vec<&str> = USE_CASES.iter().map(|(n, _)| *n).collect();
53        format!("Unknown use case '{s}'. Valid values:\n  {}", valid.join(", "))
54    })
55}
56
57/// Parse a security level from a CLI string.
58pub(crate) fn parse_security_level(s: &str) -> Result<SecurityLevel, String> {
59    SECURITY_LEVELS.iter().find(|(name, _)| *name == s).map(|(_, sl)| *sl).ok_or_else(|| {
60        let valid: Vec<&str> = SECURITY_LEVELS.iter().map(|(n, _)| *n).collect();
61        format!("Unknown security level '{s}'. Valid: {}", valid.join(", "))
62    })
63}
64
65/// Parse a compliance mode from a CLI string.
66#[allow(dead_code)] // CLI-04: used when --compliance flag is wired
67pub(crate) fn parse_compliance(s: &str) -> Result<ComplianceMode, String> {
68    COMPLIANCE_MODES.iter().find(|(name, _)| *name == s).map(|(_, cm)| cm.clone()).ok_or_else(
69        || {
70            let valid: Vec<&str> = COMPLIANCE_MODES.iter().map(|(n, _)| *n).collect();
71            format!("Unknown compliance mode '{s}'. Valid: {}", valid.join(", "))
72        },
73    )
74}

92/// Build a `CryptoConfig` from optional CLI values.
93pub(crate) fn build_config<'a>(
94    use_case: Option<UseCase>,
95    security_level: Option<SecurityLevel>,
96    compliance: &Option<ComplianceMode>,
97) -> latticearc::CryptoConfig<'a> {
98    let mut config = latticearc::CryptoConfig::new();
99
100    if let Some(uc) = use_case {
101        config = config.use_case(uc);
102    }
103
104    if let Some(sl) = security_level {
105        config = config.security_level(sl);
106    }
107
108    if let Some(cm) = compliance {
109        config = config.compliance(cm.clone());
110    }
111
112    config
113}
