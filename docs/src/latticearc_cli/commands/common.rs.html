1//! Shared CLI types for use-case-driven operations.
2//!
3//! Maps CLI string values directly to library enums — no wrapper types.
4//! The library's `UseCase`, `SecurityLevel`, and `ComplianceMode` are used
5//! directly in clap args via custom `value_parser` functions.
6
7use latticearc::types::types::{ComplianceMode, SecurityLevel, UseCase};
8

14    ("api-security", UseCase::ApiSecurity),
15    ("file-storage", UseCase::FileStorage),
16    ("database-encryption", UseCase::DatabaseEncryption),
17    ("cloud-storage", UseCase::CloudStorage),
18    ("backup-archive", UseCase::BackupArchive),
19    ("config-secrets", UseCase::ConfigSecrets),
20    ("authentication", UseCase::Authentication),
21    ("session-token", UseCase::SessionToken),
22    ("digital-certificate", UseCase::DigitalCertificate),
23    ("key-exchange", UseCase::KeyExchange),
24    ("financial-transactions", UseCase::FinancialTransactions),
25    ("legal-documents", UseCase::LegalDocuments),
26    ("blockchain-transaction", UseCase::BlockchainTransaction),
27    ("healthcare-records", UseCase::HealthcareRecords),
28    ("government-classified", UseCase::GovernmentClassified),
29    ("payment-card", UseCase::PaymentCard),
30    ("iot-device", UseCase::IoTDevice),
31    ("firmware-signing", UseCase::FirmwareSigning),

43const COMPLIANCE_MODES: &[(&str, ComplianceMode)] = &[
44    ("default", ComplianceMode::Default),
45    ("fips", ComplianceMode::Fips140_3),
46    ("cnsa-2.0", ComplianceMode::Cnsa2_0),

49/// Parse a use case from a CLI string.
50pub(crate) fn parse_use_case(s: &str) -> Result<UseCase, String> {
51    USE_CASES.iter().find(|(name, _)| *name == s).map(|(_, uc)| *uc).ok_or_else(|| {
52        let valid: Vec<&str> = USE_CASES.iter().map(|(n, _)| *n).collect();
53        format!("Unknown use case '{s}'. Valid values:\n  {}", valid.join(", "))
54    })
55}
56
57/// Parse a security level from a CLI string.
58pub(crate) fn parse_security_level(s: &str) -> Result<SecurityLevel, String> {
59    SECURITY_LEVELS.iter().find(|(name, _)| *name == s).map(|(_, sl)| *sl).ok_or_else(|| {
60        let valid: Vec<&str> = SECURITY_LEVELS.iter().map(|(n, _)| *n).collect();
61        format!("Unknown security level '{s}'. Valid: {}", valid.join(", "))
62    })
63}
64
65/// Parse a compliance mode from a CLI string.
66pub(crate) fn parse_compliance(s: &str) -> Result<ComplianceMode, String> {
67    COMPLIANCE_MODES.iter().find(|(name, _)| *name == s).map(|(_, cm)| cm.clone()).ok_or_else(
68        || {
69            let valid: Vec<&str> = COMPLIANCE_MODES.iter().map(|(n, _)| *n).collect();
70            format!("Unknown compliance mode '{s}'. Valid: {}", valid.join(", "))
71        },
72    )
73}

75/// Resolve ML-KEM security level from a key file's algorithm.
76///
77/// Maps `KeyAlgorithm::MlKem512/768/1024` to the corresponding `MlKemSecurityLevel`.
78/// Returns an error for non-ML-KEM algorithms.
79pub(crate) fn resolve_ml_kem_level(
80    algorithm: latticearc::unified_api::key_format::KeyAlgorithm,
81) -> Result<latticearc::primitives::kem::ml_kem::MlKemSecurityLevel, String> {
82    use latticearc::primitives::kem::ml_kem::MlKemSecurityLevel;
83    use latticearc::unified_api::key_format::KeyAlgorithm;
84    match algorithm {
85        KeyAlgorithm::MlKem512 => Ok(MlKemSecurityLevel::MlKem512),
86        KeyAlgorithm::MlKem768 => Ok(MlKemSecurityLevel::MlKem768),
87        KeyAlgorithm::MlKem1024 => Ok(MlKemSecurityLevel::MlKem1024),
88        _ => Err(format!(
89            "PQ-only operation requires an ML-KEM key (ml-kem512/768/1024), got {:?}",
90            algorithm
91        )),
92    }
93}

95/// Map ML-KEM security level to the corresponding `SecurityLevel`.
96///
97/// Delegates to the library's canonical mapping in the policy engine.
98pub(crate) fn ml_kem_to_security_level(
99    level: latticearc::primitives::kem::ml_kem::MlKemSecurityLevel,
100) -> SecurityLevel {
101    latticearc::unified_api::selector::ml_kem_level_to_security_level(level)
102}

104/// Build a `CryptoConfig` from optional CLI values.
105pub(crate) fn build_config<'a>(
106    use_case: Option<UseCase>,
107    security_level: Option<SecurityLevel>,
108    compliance: &Option<ComplianceMode>,
109) -> latticearc::CryptoConfig<'a> {
110    let mut config = latticearc::CryptoConfig::new();
111
112    if let Some(uc) = use_case {
113        config = config.use_case(uc);
114    }
115
116    if let Some(sl) = security_level {
117        config = config.security_level(sl);
118    }
119
120    if let Some(cm) = compliance {
121        config = config.compliance(cm.clone());
122    }
123
124    config
125}
