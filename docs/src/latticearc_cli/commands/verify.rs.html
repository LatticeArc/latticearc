1//! Verify command — check digital signatures.
2//!
3//! Supports two signature formats:
4//!
5//! **SignedData (recommended):** Produced by `sign --public-key`. Contains scheme
6//! metadata, timestamp, and public key — the library's `verify()` handles everything.
7//! No `--algorithm` or `--key` flags needed (both are embedded in the signature).
8//!
9//! **Legacy JSON:** Produced by `sign --algorithm`. Requires `--key` (public key)
10//! and optionally `--algorithm` (auto-detected from the signature file).
11
12use anyhow::{Context, Result, bail};
13use clap::{Args, ValueEnum};
14use std::path::PathBuf;
15
16use crate::keyfile::{KeyFile, KeyType};

18/// Verification algorithm (legacy mode only).
19#[derive(Debug, Clone, Copy, PartialEq, Eq, ValueEnum)]
20pub(crate) enum VerifyAlgorithm {
21    /// ML-DSA-65 (FIPS 204).
22    MlDsa65,
23    /// ML-DSA-44 (FIPS 204).
24    MlDsa44,
25    /// ML-DSA-87 (FIPS 204).
26    MlDsa87,
27    /// SLH-DSA-SHAKE-128s (FIPS 205).
28    SlhDsa,
29    /// FN-DSA-512 (FIPS 206).
30    FnDsa,
31    /// Ed25519.
32    Ed25519,
33    /// Hybrid ML-DSA-65 + Ed25519.
34    Hybrid,
35}
36
37/// Arguments for the `verify` subcommand.
38#[derive(Args)]
39pub(crate) struct VerifyArgs {
40    /// Verification algorithm (legacy mode). Auto-detected from signature file if omitted.
41    #[arg(short, long, value_enum)]
42    pub algorithm: Option<VerifyAlgorithm>,
43    /// Use case for configuration (optional).
44    #[arg(long, value_parser = super::common::parse_use_case,
45          value_name = "USE_CASE")]
46    pub use_case: Option<latticearc::types::types::UseCase>,
47    /// Security level override (default: high).
48    #[arg(long, value_parser = super::common::parse_security_level,
49          value_name = "LEVEL")]
50    pub security_level: Option<latticearc::types::types::SecurityLevel>,
51    /// Compliance mode (default, fips, cnsa-2.0).
52    #[arg(long, value_parser = super::common::parse_compliance,
53          value_name = "MODE")]
54    pub compliance: Option<latticearc::types::types::ComplianceMode>,
55    /// Input file that was signed (reads from stdin if omitted).
56    ///
57    /// Symmetric with `sign --input <stdin>`: a CI pipeline that
58    /// hashes a stream and signs it should be able to verify the
59    /// same way without staging to a temp file.
60    #[arg(short, long)]
61    pub input: Option<PathBuf>,
62    /// Signature file (SignedData JSON or legacy JSON).
63    #[arg(short, long)]
64    pub signature: PathBuf,
65    /// Public key file (required for legacy format, optional for SignedData).
66    #[arg(short, long)]
67    pub key: Option<PathBuf>,
68}

70/// Execute the verify command.
71///
72/// Returns `Ok(true)` for VALID, `Ok(false)` for INVALID (forgery /
73/// tampering), or `Err(_)` for operational errors (missing file, bad
74/// key, etc.). `main` translates the boolean into the documented
75/// 0/1/≥2 exit-code contract via `std::process::exit(1)` AFTER all
76/// destructors run — exiting from inside `verify::run` would skip
77/// Drop on local `KeyFile`/`Vec<u8>` state, which is currently benign
78/// (PK is non-secret) but would silently regress if the same pattern
79/// were copied to `sign` or `decrypt`.
80pub(crate) fn run(args: VerifyArgs) -> Result<bool> {
81    let input_data = read_verify_input(&args)?;
82    // open-once with `take(limit + 1)` instead of stat-then-read.
83    // A `metadata().len()` + `read_to_string` sequence is TOCTOU: a
84    // path swap between the size check and the read can let a
85    // multi-gig file slip past the cap. `metadata().len()` also
86    // returns 0 for `/dev/zero` / FIFOs, so a subsequent read of an
87    // unbounded special file would have no upper bound.
88    let sig_bytes = super::common::read_file_with_cap(
89        &args.signature,
90        super::common::CLI_MAX_SIGNATURE_INPUT_BYTES,
91        "verify",
92    )?;
93    let sig_json = String::from_utf8(sig_bytes)
94        .with_context(|| format!("{} is not valid UTF-8", args.signature.display()))?;
95
96    // Detect whether the file is in the SignedData envelope shape
97    // *before* committing to a verifier. The shape check is a
98    // structural pre-pass: if the JSON has the unified-API SignedData
99    // fields (`metadata.signature`, `metadata.signature_algorithm`,
100    // `scheme`), require `deserialize_signed_data` to succeed; any
101    // deserialization failure collapses to `print_invalid()`. Without
102    // this gate, an `if let Ok(...)` shape silently falls through to
103    // the legacy auto-detect branch and emits a distinguishable
104    // "Signature file missing 'algorithm' field" — letting an
105    // attacker tell apart "I tampered with metadata.signature_algorithm"
106    // from "wrong signature" by stderr text (Pattern-6).
107    let looks_like_signed_data = serde_json::from_str::<serde_json::Value>(&sig_json)
108        .ok()
109        .and_then(|v| {
110            v.get("metadata")?.get("signature")?;
111            v.get("metadata")?.get("signature_algorithm")?;
112            v.get("scheme")?;
113            Some(())
114        })
115        .is_some();
116    if looks_like_signed_data {
117        // Deserialize INSIDE the structural gate (not before it), so a
118        // hoist or reorder can't silently fall through to the legacy
119        // auto-detect branch and reopen the Pattern-6 distinguisher
120        // (legacy branch emitted "Signature file missing 'algorithm'
121        // field" — distinguishable from a crypto-reject error).
122        let signed =
123            match latticearc::unified_api::serialization::deserialize_signed_data(&sig_json) {
124                Ok(signed) => signed,
125                Err(e) => {
126                    // `%e` Display, not `?e` Debug — Display shows the
127                    // top-level deserialize error message (typed,
128                    // bounded), while Debug walks the source chain
129                    // which can contain attacker-controlled JSON
130                    // snippets via the underlying `serde_json::Error`.
131                    tracing::debug!(
132                        error = %e,
133                        "verify (SignedData shape detected, parse rejected)"
134                    );
135                    return Ok(print_invalid());
136                }
137            };
138        // Verify the signed data matches the input file. `signed.data`
139        // currently carries public message material, so the comparison
140        // doesn't need to be CT in the threat-model sense. A `ct_eq`
141        // wrapper here would be cargo-cult: a length-equality branch
142        // before it would make the CT comparison dead anyway. If a
143        // future caller routes secret bytes through this path, the
144        // responsibility is to handle CT at THAT site, not to pretend
145        // this site is CT when it can't be.
146        //
147        // A mismatch is attacker-controllable (the SignedData envelope
148        // is loaded from a file the attacker may have written) and so
149        // must collapse to `print_invalid()` — emitting a
150        // distinguishable "Signature was created over different data"
151        // error would let an attacker tell apart "I tampered with
152        // signed.data" from "I tampered with the signature bytes" by
153        // branching on stderr text (Pattern-6). Debug-log the original
154        // detail so operators investigating a legitimate user mistake
155        // (signed the wrong file) can recover the context with
156        // `RUST_LOG=debug`.
157        if signed.data != input_data {
158            tracing::debug!(
159                input = %input_label(&args),
160                "verify (SignedData path) rejected: embedded data does not match input"
161            );
162            return Ok(print_invalid());
163        }
164
165        let config =
166            super::common::build_config(args.use_case, args.security_level, &args.compliance);
167        // A `latticearc::verify(...)` Err (e.g. malformed public key)
168        // collapses to the same user-visible outcome as `Ok(false)` so
169        // an attacker can't distinguish the structural / crypto /
170        // library-error paths via stderr text or exit-code class. See
171        // `print_invalid()` for the centralised reject contract.
172        let valid = match latticearc::verify(&signed, config) {
173            Ok(v) => v,
174            Err(e) => {
175                // `%e` Display — see deserialize_signed_data branch above
176                // for the rationale on attacker-content avoidance.
177                tracing::debug!(error = %e, "verify (SignedData path) rejected");
178                return Ok(print_invalid());
179            }
180        };
181
182        // return the result; `main` does the exit-code translation
183        // after destructors so the 0/1/≥2 contract is preserved at
184        // the boundary, not here.
185        //
186        // Stream choice: VALID on stdout, INVALID on stderr. The
187        // asymmetry is intentional — the machine-readable verdict is
188        // the exit code (Ok(true)/false), which IS symmetric. Verdict
189        // text on either stream is operator status only. Tests that
190        // need stream-swap protection should use `run_ok_combined`.
191        if valid {
192            println!("Signature is VALID. (scheme: {})", signed.scheme);
193            return Ok(true);
194        }
195        return Ok(print_invalid());
196    }
197
198    // Fall back to legacy format (requires --key)
199    let key_path = args.key.as_ref().ok_or_else(|| {
200        anyhow::anyhow!(
201            "Public key file (--key) is required for legacy signature format.\n\
202             Signatures created with 'sign --public-key' embed the public key \
203             and don't need --key."
204        )
205    })?;
206
207    verify_legacy(&args, &sig_json, key_path, input_data)
208}

210/// Centralised reject for any path that must be indistinguishable
211/// from a cryptographic-verification failure: emits the same INVALID
212/// stderr line and returns `false` (which `run`'s callers translate
213/// to exit code 1, the "invalid" class — distinct from the ≥2 "error"
214/// class).
215///
216/// # SECURITY: indistinguishability contract
217///
218/// Every reject path triggered by **attacker-controllable** signature-
219/// file content MUST funnel through `print_invalid()` (or, in
220/// `verify_legacy`, through the closure that maps `Err(_)` →
221/// `print_invalid()` via `unwrap_or_else`). The distinguishers this
222/// contract closes are concrete and have been observed in audit:
223///
224/// 1. **Stderr text** — a stricter message ("Invalid base64 in
225///    signature", "Unknown algorithm", "Verification failed",
226///    "Signature was created over different data than the input",
227///    "Hybrid verification failed: <upstream detail>") lets an
228///    attacker probe which validation step rejected by reading the
229///    process's stderr stream.
230/// 2. **Exit-code class** — bubbling an `Err(_)` out of `run` lands
231///    at exit ≥2 ("operational error"), distinct from the `Ok(false)`
232///    → exit 1 ("invalid signature") class. An attacker can
233///    distinguish the two with `[ $? -eq 1 ]` vs `[ $? -ge 2 ]`.
234/// 3. **Echo amplification** — interpolating an attacker-controlled
235///    field (`{other}` from the `algorithm` JSON value, `{e}` from a
236///    structured upstream error) into the error string both reveals
237///    the field's content AND amplifies its length (attacker writes
238///    1 MiB → log line is 1 MiB).
239///
240/// The "operator-side" errors in `verify_legacy` (missing `--key`,
241/// wrong key type, encrypted key without passphrase, operator-picked
242/// `--algorithm` mismatching the key) are NOT reachable by an
243/// attacker who can only tamper with the signature file, so they keep
244/// helpful error messages and surface as exit ≥2. See `verify_legacy`
245/// docs for the boundary.
246///
247/// # Tracing observability boundary
248///
249/// `tracing::debug!` calls used as the safe alternative below ARE
250/// observable on stderr when the operator (or a CI runner) sets
251/// `RUST_LOG=debug`. The default subscriber's filter ships at
252/// `INFO`/`WARN`, so production deployments don't expose this surface
253/// — but the contract isn't "this never reaches stderr", it's "this
254/// never reaches stderr at default log levels". Treat that as a
255/// reasonable trade-off: operator-side debugging is the legitimate
256/// reason debug logs exist, and an attacker who can flip the
257/// operator's `RUST_LOG` already has more powerful capabilities than
258/// reading verify error text.
259///
260/// **What's safe to put inside `tracing::debug!` calls:**
261/// - Errors via `error = %e` (Display) — fine. anyhow's Display
262///   shows ONLY the outermost `.context(...)` message, which is
263///   static text we author ourselves ("Failed to parse signature
264///   JSON", "Invalid base64 in signature", etc.). The attacker-
265///   controlled detail lives in the source chain, which Display
266///   does not walk.
267/// - The chosen algorithm name — fine; not secret, comes from a
268///   small fixed set, useful for debugging mismatches.
269/// - A short reject reason string (e.g. "alg mismatch", "data
270///   mismatch") — fine.
271///
272/// **What's NOT safe inside `tracing::debug!` calls:**
273/// - Raw signature bytes, key bytes, or any base64 thereof — even at
274///   `RUST_LOG=debug`, plaintext key material in logs is a separate
275///   incident class.
276/// - Errors via `error = ?e` (Debug) — Debug walks the full source
277///   chain. For wrapped `serde_json::Error` or `base64::DecodeError`,
278///   the chain includes attacker-controlled JSON snippets and byte
279///   offsets / values from the input. This re-introduces the echo-
280///   amplification distinguisher one log level down. Use `%e`
281///   instead.
282/// - Attacker-controlled fields verbatim (the `algorithm` JSON
283///   value, arbitrary error strings from the parsed signature).
284///
285/// # When adding a new reject path
286///
287/// If the path can be triggered by the content of `--signature` (or,
288/// for SignedData, by the embedded `signed.data` / `signed.metadata`),
289/// it MUST route through `print_invalid()`; preserve any operator-
290/// facing detail in `tracing::debug` instead of `eprintln!` / `bail!`
291/// / `anyhow!` — and follow the safe-content rules above.
292fn print_invalid() -> bool {
293    eprintln!("Signature is INVALID.");
294    false
295}

315/// The six non-hybrid `VerifyAlgorithm` variants, lifted into their
316/// own enum so `verify_standard` can be type-enforced as never
317/// receiving `Hybrid` — without this, an unreachable `bail!("Internal
318/// error...")` arm sits in `verify_standard` waiting for a future
319/// variant addition to silently make it reachable.
320#[derive(Debug, Clone, Copy)]
321enum NonHybridAlgorithm {
322    MlDsa44,
323    MlDsa65,
324    MlDsa87,
325    SlhDsa,
326    FnDsa,
327    Ed25519,
328}
329
330/// Dispatch shape for the verify-side step in `verify_legacy`.
331///
332/// The variant is chosen in Phase 1c (operator-side parsing) so that
333/// hybrid pk-parse failures — which indicate a malformed operator key
334/// file, NOT a tampered signature — surface as helpful `Err` at exit
335/// ≥2 rather than collapsing into `print_invalid()`. By the time the
336/// Phase-2 closure dispatches, the type system guarantees the right
337/// pk shape for the chosen algorithm: `Standard` cannot carry
338/// `Hybrid` (rejected by `NonHybridAlgorithm`) and `Hybrid` carries
339/// the parsed pk directly.
340///
341/// # Field-name asymmetry: `pk_bytes` vs `pk`
342///
343/// The two variants intentionally name their pk field differently
344/// because they hold different things at different stages of parsing:
345/// `Standard` carries raw bytes that the per-algorithm crypto helper
346/// will pass straight to the verify primitive (no operator-side parse
347/// needed); `Hybrid` carries the already-parsed `HybridSigPublicKey`
348/// because the hybrid wire format requires an operator-trusted
349/// byte-layout parse that lives in Phase 1c — failure there must
350/// bubble as helpful `Err`, not collapse to INVALID.
351enum LegacyVerifier<'a> {
352    Standard { algorithm: NonHybridAlgorithm, pk_bytes: &'a [u8] },
353    Hybrid { pk: latticearc::hybrid::sig_hybrid::HybridSigPublicKey },
354}
355
356/// Project a `VerifyAlgorithm` onto its non-hybrid sub-domain. The
357/// `Hybrid` variant has no representation in `NonHybridAlgorithm` and
358/// produces `Err(())`; the caller (`verify_legacy` Phase 1c) routes
359/// that case to `LegacyVerifier::Hybrid` after parsing the operator
360/// pk. Implemented as `TryFrom` rather than a hand-named constructor
361/// to align with the standard library projection-conversion idiom.
362impl TryFrom<VerifyAlgorithm> for NonHybridAlgorithm {
363    type Error = ();
364
365    fn try_from(alg: VerifyAlgorithm) -> Result<Self, Self::Error> {
366        match alg {
367            VerifyAlgorithm::MlDsa44 => Ok(Self::MlDsa44),
368            VerifyAlgorithm::MlDsa65 => Ok(Self::MlDsa65),
369            VerifyAlgorithm::MlDsa87 => Ok(Self::MlDsa87),
370            VerifyAlgorithm::SlhDsa => Ok(Self::SlhDsa),
371            VerifyAlgorithm::FnDsa => Ok(Self::FnDsa),
372            VerifyAlgorithm::Ed25519 => Ok(Self::Ed25519),
373            VerifyAlgorithm::Hybrid => Err(()),
374        }
375    }
376}

378/// Legacy verification path — custom JSON format + primitive-level API.
379///
380/// # Pattern-6 boundary
381///
382/// `verify_legacy` splits its work into two phases with different
383/// error-handling contracts:
384///
385/// 1. **Operator-side validation** (key file existence, key type,
386///    encrypted-key passphrase, operator-picked `--algorithm` vs key
387///    mismatch, hybrid pk byte-layout parse). Errors here surface as
388///    `Err(_)` and become CLI exit code ≥2 ("operational error") with
389///    a helpful stderr message. The threat model treats the operator's
390///    `--key` file as trusted — an attacker cannot reach these paths
391///    by tampering with the signature file.
392///
393/// 2. **Signature-side validation** (algorithm auto-detect from the
394///    signature JSON, base64/JSON parse, missing fields, validate-
395///    algorithm vs auto-detected name, crypto verification). Errors
396///    here are attacker-controllable (the signature file may have been
397///    tampered) and so collapse via `unwrap_or_else` to `false`, which
398///    routes through [`print_invalid()`]. Without the collapse, an
399///    attacker could distinguish "I broke the JSON envelope" from "I
400///    broke the signature" by reading stderr text or exit-code class
401///    (Pattern-6). The original detail is preserved in `tracing::debug`
402///    for operators investigating a legitimate user mistake.
403fn verify_legacy(
404    args: &VerifyArgs,
405    sig_json: &str,
406    key_path: &std::path::Path,
407    data: Vec<u8>,
408) -> Result<bool> {
409    // `data` is threaded in from `run`. Reading input again here
410    // would drain stdin a second time in pipe mode, returning empty
411    // bytes and silently producing INVALID for any legacy-format
412    // signature piped on stdin.
413
414    // ── Phase 1a: read operator key file (helpful errors, exit ≥2) ──
415    let key_file = KeyFile::read_from(key_path)?;
416    if key_file.key_type != KeyType::Public {
417        // `canonical_name()` instead of `{:?}` Debug: the Debug repr
418        // of a `#[non_exhaustive]` enum is variant-fingerprintable
419        // across versions.
420        bail!(
421            "Expected public key file for verification, got {}",
422            key_file.key_type.canonical_name()
423        );
424    }
425    let pk_bytes = key_file.key_bytes()?;
426
427    // ── Phase 1b: operator-picked algorithm vs key (helpful err) ──
428    if let Some(cli_alg) = args.algorithm {
429        key_file.validate_algorithm(algorithm_name(cli_alg))?;
430    }
431
432    // ── Phase 1c: determine algorithm + parse operator-side pk ──
433    //
434    // Algorithm determination has a mixed posture:
435    //   * `--algorithm <X>` came from the operator → trusted.
436    //   * Auto-detect comes from the signature file → attacker-
437    //     controllable. If detection fails or the auto-detected
438    //     algorithm doesn't match the key, collapse to `print_invalid()`
439    //     immediately; both shapes are signature-side rejects.
440    //
441    // Once the algorithm is known, the *pk byte-layout parse* is
442    // operator-side regardless of where the algorithm came from
443    // (`pk_bytes` is the operator's key file). A malformed hybrid pk
444    // is operator misuse, not an attacker action — surface it as a
445    // helpful Err.
446    let algorithm = if let Some(alg) = args.algorithm {
447        alg
448    } else {
449        let Some(detected) = detect_algorithm(sig_json) else {
450            tracing::debug!("verify (legacy auto-detect) rejected: algorithm undetectable");
451            return Ok(print_invalid());
452        };
453        tracing::debug!(
454            algorithm = %algorithm_name(detected),
455            "auto-detected legacy verify algorithm"
456        );
457        if let Err(e) = key_file.validate_algorithm(algorithm_name(detected)) {
458            // `%e` Display, not `?e` Debug — see Phase 2 closure below
459            // for the full rationale on attacker-content avoidance in
460            // tracing output.
461            tracing::debug!(error = %e, "verify (legacy auto-detect) rejected: alg mismatch");
462            return Ok(print_invalid());
463        }
464        detected
465    };
466
467    let verifier = if let Ok(non_hybrid) = NonHybridAlgorithm::try_from(algorithm) {
468        LegacyVerifier::Standard { algorithm: non_hybrid, pk_bytes: &pk_bytes }
469    } else {
470        // Algorithm was Hybrid (the only `Err`-returning case).
471        // pk parse failure here is operator-side (their key file is
472        // malformed) — bubble as helpful Err.
473        let pk = crate::keyfile::parse_hybrid_sign_pk_from_bytes(&pk_bytes)?;
474        LegacyVerifier::Hybrid { pk }
475    };
476
477    // ── Phase 2: signature-side verify (returns plain `bool`) ──
478    //
479    // `verify_standard` and `verify_hybrid` both return `bool` directly
480    // (NOT `Result<bool>`): any error inside them — JSON parse, missing
481    // field, base64 decode, crypto reject — collapses internally to
482    // `false` with a `%e` Display debug-log. The `-> bool` return type
483    // is structural: a future maintainer who tries to bubble an
484    // operator-side error from these functions gets a compile error,
485    // which prevents accidentally widening the Phase-2 collapse to
486    // swallow operator state.
487    let valid = match &verifier {
488        LegacyVerifier::Hybrid { pk } => verify_hybrid(&data, sig_json, pk),
489        LegacyVerifier::Standard { algorithm, pk_bytes } => {
490            verify_standard(&data, sig_json, pk_bytes, *algorithm)
491        }
492    };
493
494    // Route both VALID and INVALID through their respective helpers
495    // so the reject contract (see `print_invalid()`) cannot drift via
496    // an inline `eprintln!`.
497    if valid {
498        println!("Signature is VALID.");
499        Ok(true)
500    } else {
501        Ok(print_invalid())
502    }
503}

505/// Detect algorithm from the "algorithm" field in a legacy .sig.json file.
506///
507/// Returns `None` for any reason the algorithm can't be identified —
508/// JSON parse failure, missing `"algorithm"` field, non-string value,
509/// or an unrecognised name. Returning `Option` (not `Result`) is
510/// load-bearing for the Pattern-6 collapse contract: every "couldn't
511/// detect" cause must be indistinguishable from every other one to
512/// stop attacker-controllable signature-file content from carving out
513/// a stderr-text side channel. The caller (`verify_legacy`) maps
514/// `None` into the same `print_invalid()` outcome as a crypto reject;
515/// operator-facing detail is recovered via `tracing::debug` at
516/// `RUST_LOG=debug`.
517///
518/// In particular, NEVER interpolate the `algorithm` field into an
519/// error string — an attacker can supply an arbitrarily long value,
520/// producing a proportionally large log line and a variant-name
521/// oracle for which validation step rejected.
522fn detect_algorithm(sig_json: &str) -> Option<VerifyAlgorithm> {
523    let sig: serde_json::Value = serde_json::from_str(sig_json).ok()?;
524    let alg_str = sig.get("algorithm").and_then(|v| v.as_str())?;
525    match alg_str {
526        "ml-dsa-65" => Some(VerifyAlgorithm::MlDsa65),
527        "ml-dsa-44" => Some(VerifyAlgorithm::MlDsa44),
528        "ml-dsa-87" => Some(VerifyAlgorithm::MlDsa87),
529        "slh-dsa-shake-128s" => Some(VerifyAlgorithm::SlhDsa),
530        "fn-dsa-512" => Some(VerifyAlgorithm::FnDsa),
531        "ed25519" => Some(VerifyAlgorithm::Ed25519),
532        "hybrid-ml-dsa-65-ed25519" => Some(VerifyAlgorithm::Hybrid),
533        _ => None,
534    }
535}

537/// Map verify algorithm enum to canonical key file algorithm name.
538fn algorithm_name(alg: VerifyAlgorithm) -> &'static str {
539    match alg {
540        VerifyAlgorithm::MlDsa65 => "ml-dsa-65",
541        VerifyAlgorithm::MlDsa44 => "ml-dsa-44",
542        VerifyAlgorithm::MlDsa87 => "ml-dsa-87",
543        VerifyAlgorithm::SlhDsa => "slh-dsa-shake-128s",
544        VerifyAlgorithm::FnDsa => "fn-dsa-512",
545        VerifyAlgorithm::Ed25519 => "ed25519",
546        VerifyAlgorithm::Hybrid => "hybrid-ml-dsa-65-ed25519",
547    }
548}

550/// Verify a non-hybrid legacy signature.
551///
552/// Returns a plain `bool` — never `Err`. All signature-side errors
553/// (JSON parse, missing fields, base64 decode, crypto reject from a
554/// `latticearc::verify_*` helper) collapse internally to `false` with
555/// the underlying cause debug-logged via `%e` (Display only — see
556/// `print_invalid()` SECURITY contract).
557///
558/// The `-> bool` return type is structural enforcement of the
559/// Phase-2 contract: a future maintainer who tries to bubble an
560/// operator-side error (e.g. by adding a `?` on a non-signature-side
561/// fallible call) gets a compile error rather than silently widening
562/// the closure-collapse to swallow operator state.
563///
564/// Do NOT re-flatten errors per arm with `.map_err(|_| anyhow!("..."))`
565/// — a per-arm message becomes its own Pattern-6 distinguisher. Let
566/// errors propagate raw to the inner closure boundary; the function-
567/// level `unwrap_or_else` handles indistinguishability uniformly.
568fn verify_standard(
569    data: &[u8],
570    sig_json: &str,
571    pk_bytes: &[u8],
572    algorithm: NonHybridAlgorithm,
573) -> bool {
574    (|| -> Result<bool> {
575        let sig_obj: serde_json::Value =
576            serde_json::from_str(sig_json).context("Failed to parse signature JSON")?;
577
578        use base64::Engine;
579        let sig_b64 = sig_obj
580            .get("signature")
581            .and_then(|v| v.as_str())
582            .context("Missing 'signature' field in signature JSON")?;
583        let sig_bytes = base64::engine::general_purpose::STANDARD
584            .decode(sig_b64)
585            .context("Invalid base64 in signature")?;
586
587        // Exhaustive over `NonHybridAlgorithm`. If a new variant is
588        // added there, this match becomes a compile error rather than
589        // a hidden unreachable arm.
590        let valid = match algorithm {
591            NonHybridAlgorithm::MlDsa65 => latticearc::verify_pq_ml_dsa(
592                data,
593                &sig_bytes,
594                pk_bytes,
595                latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa65,
596                latticearc::SecurityMode::Unverified,
597            )?,
598            NonHybridAlgorithm::MlDsa44 => latticearc::verify_pq_ml_dsa(
599                data,
600                &sig_bytes,
601                pk_bytes,
602                latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa44,
603                latticearc::SecurityMode::Unverified,
604            )?,
605            NonHybridAlgorithm::MlDsa87 => latticearc::verify_pq_ml_dsa(
606                data,
607                &sig_bytes,
608                pk_bytes,
609                latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa87,
610                latticearc::SecurityMode::Unverified,
611            )?,
612            NonHybridAlgorithm::SlhDsa => latticearc::verify_pq_slh_dsa(
613                data,
614                &sig_bytes,
615                pk_bytes,
616                latticearc::primitives::sig::slh_dsa::SlhDsaSecurityLevel::Shake128s,
617                latticearc::SecurityMode::Unverified,
618            )?,
619            NonHybridAlgorithm::FnDsa => latticearc::verify_pq_fn_dsa(
620                data,
621                &sig_bytes,
622                pk_bytes,
623                latticearc::primitives::sig::fndsa::FnDsaSecurityLevel::Level512,
624                latticearc::SecurityMode::Unverified,
625            )?,
626            NonHybridAlgorithm::Ed25519 => latticearc::verify_ed25519(
627                data,
628                &sig_bytes,
629                pk_bytes,
630                latticearc::SecurityMode::Unverified,
631            )?,
632        };
633        Ok(valid)
634    })()
635    .unwrap_or_else(|e| {
636        tracing::debug!(error = %e, "verify (legacy standard path) rejected");
637        false
638    })
639}

641/// Verify a hybrid legacy signature against a parsed hybrid public key.
642///
643/// Returns a plain `bool` — never `Err`. All signature-side errors
644/// (JSON parse, missing fields, base64 decode, crypto reject)
645/// collapse internally to `false` with a `%e` Display debug-log.
646/// Same `-> bool` structural enforcement as `verify_standard`: a
647/// future maintainer cannot bubble an operator-side error out of
648/// this function and accidentally widen the Phase-2 collapse.
649///
650/// The pk is parsed by the caller (`verify_legacy` Phase 1c) so that
651/// pk-parse failures — which indicate a malformed *operator* key file
652/// — surface as helpful `Err` at exit ≥2 there, not collapse into
653/// INVALID here.
654fn verify_hybrid(
655    data: &[u8],
656    sig_json: &str,
657    pk: &latticearc::hybrid::sig_hybrid::HybridSigPublicKey,
658) -> bool {
659    (|| -> Result<bool> {
660        let sig: serde_json::Value =
661            serde_json::from_str(sig_json).context("Failed to parse hybrid signature JSON")?;
662
663        use base64::Engine;
664        let ml_dsa_b64 = sig
665            .get("ml_dsa_sig")
666            .and_then(|v| v.as_str())
667            .context("Missing 'ml_dsa_sig' field in hybrid signature")?;
668        let ml_dsa_sig = base64::engine::general_purpose::STANDARD
669            .decode(ml_dsa_b64)
670            .context("Invalid base64 in ml_dsa_sig")?;
671
672        let ed25519_b64 = sig
673            .get("ed25519_sig")
674            .and_then(|v| v.as_str())
675            .context("Missing 'ed25519_sig' field in hybrid signature")?;
676        let ed25519_sig = base64::engine::general_purpose::STANDARD
677            .decode(ed25519_b64)
678            .context("Invalid base64 in ed25519_sig")?;
679
680        let hybrid_sig =
681            latticearc::hybrid::sig_hybrid::HybridSignature::new(ml_dsa_sig, ed25519_sig);
682
683        // CoreError::VerificationFailed is a routine cryptographic
684        // reject (`Ok(false)`-equivalent). Any other CoreError variant
685        // is propagated as anyhow::Error::new(e) — the typed source
686        // survives for `%e` Display, which on CoreError gives a fixed
687        // variant message without attacker-controlled interpolation.
688        // A structured upstream error like "ML-DSA component verify
689        // failed" vs "Ed25519 component verify failed" would otherwise
690        // tell an attacker which half of the hybrid pair rejected.
691        use latticearc::CoreError;
692        match latticearc::verify_hybrid_signature(
693            data,
694            &hybrid_sig,
695            pk,
696            latticearc::SecurityMode::Unverified,
697        ) {
698            Ok(valid) => Ok(valid),
699            Err(CoreError::VerificationFailed) => Ok(false),
700            Err(e) => Err(anyhow::Error::new(e)),
701        }
702    })()
703    .unwrap_or_else(|e| {
704        tracing::debug!(error = %e, "verify (legacy hybrid path) rejected");
705        false
706    })
707}
