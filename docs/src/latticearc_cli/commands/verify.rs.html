1//! Verify command — check digital signatures.
2//!
3//! Supports two signature formats:
4//!
5//! **SignedData (recommended):** Produced by `sign --public-key`. Contains scheme
6//! metadata, timestamp, and public key — the library's `verify()` handles everything.
7//! No `--algorithm` or `--key` flags needed (both are embedded in the signature).
8//!
9//! **Legacy JSON:** Produced by `sign --algorithm`. Requires `--key` (public key)
10//! and optionally `--algorithm` (auto-detected from the signature file).
11
12use anyhow::{Context, Result, bail};
13use clap::{Args, ValueEnum};
14use std::path::PathBuf;
15
16use crate::keyfile::{KeyFile, KeyType};
17
18/// Verification algorithm (legacy mode only).
19#[derive(Debug, Clone, ValueEnum)]
20pub(crate) enum VerifyAlgorithm {
21    /// ML-DSA-65 (FIPS 204).
22    MlDsa65,
23    /// ML-DSA-44 (FIPS 204).
24    MlDsa44,
25    /// ML-DSA-87 (FIPS 204).
26    MlDsa87,
27    /// SLH-DSA-SHAKE-128s (FIPS 205).
28    SlhDsa,
29    /// FN-DSA-512 (FIPS 206).
30    FnDsa,
31    /// Ed25519.
32    Ed25519,
33    /// Hybrid ML-DSA-65 + Ed25519.
34    Hybrid,
35}
36
37/// Arguments for the `verify` subcommand.
38#[derive(Args)]
39pub(crate) struct VerifyArgs {
40    /// Verification algorithm (legacy mode). Auto-detected from signature file if omitted.
41    #[arg(short, long, value_enum)]
42    pub algorithm: Option<VerifyAlgorithm>,
43    /// Use case for configuration (optional).
44    #[arg(long, value_parser = super::common::parse_use_case,
45          value_name = "USE_CASE")]
46    pub use_case: Option<latticearc::types::types::UseCase>,
47    /// Security level override (default: high).
48    #[arg(long, value_parser = super::common::parse_security_level,
49          value_name = "LEVEL")]
50    pub security_level: Option<latticearc::types::types::SecurityLevel>,
51    /// Compliance mode (default, fips, cnsa-2.0).
52    #[arg(long, value_parser = super::common::parse_compliance,
53          value_name = "MODE")]
54    pub compliance: Option<latticearc::types::types::ComplianceMode>,
55    /// Input file that was signed (reads from stdin if omitted).
56    ///
57    /// Symmetric with the round-7 `sign --input <stdin>` path: a CI
58    /// pipeline that hashes a stream and signs it should be able to
59    /// verify the same way without staging to a temp file.
60    #[arg(short, long)]
61    pub input: Option<PathBuf>,
62    /// Signature file (SignedData JSON or legacy JSON).
63    #[arg(short, long)]
64    pub signature: PathBuf,
65    /// Public key file (required for legacy format, optional for SignedData).
66    #[arg(short, long)]
67    pub key: Option<PathBuf>,
68}

70/// Execute the verify command.
71//
72// `clippy::exit` is denied workspace-wide on the principle that
73// surfaces returning `Result` should propagate errors through `?`
74// rather than aborting the process unilaterally. The `verify`
75// command needs the inverse: we maintain a documented exit-code
76// contract (0 valid / 1 invalid / ≥2 operational) that scripts rely
77// on to distinguish forgery from a missing key file. anyhow's
78// `bail!` collapses both paths to exit 1, conflating them. The two
79// `process::exit(1)` sites below are the only mechanism that gives
80// us deterministic exit code 1 for the INVALID-signature case
81// while preserving the ≥2 path for everything `?` propagates.
82#[allow(clippy::exit)]
83pub(crate) fn run(args: VerifyArgs) -> Result<()> {
84    let input_data = read_verify_input(&args)?;
85    let sig_json = std::fs::read_to_string(&args.signature)
86        .with_context(|| format!("Failed to read {}", args.signature.display()))?;
87
88    // Try SignedData format first (produced by sign --public-key)
89    if let Ok(signed) = latticearc::unified_api::serialization::deserialize_signed_data(&sig_json) {
90        // Verify the signed data matches the input file
91        if signed.data != input_data {
92            bail!(
93                "Signature was created over different data than the input.\n\
94                 The SignedData envelope contains the original data — it does not match \
95                 the data at {}.",
96                input_label(&args)
97            );
98        }
99
100        let config =
101            super::common::build_config(args.use_case, args.security_level, &args.compliance);
102        let valid = latticearc::verify(&signed, config)
103            .map_err(|e| anyhow::anyhow!("Verification failed: {e}"))?;
104
105        // Round-7 audit fix #8 + #11: explicit exit-code contract.
106        //   0  → signature VALID
107        //   1  → signature INVALID (forgery / tampering)
108        //   ≥2 → operational error (missing file, bad key, etc.)
109        // Convention matches openssl/gpg/ssh-keygen and lets scripts
110        // distinguish "verify failed" from "couldn't verify due to
111        // setup error" — anyhow's default `bail!` collapses both to
112        // exit 1 which conflates the two.
113        if valid {
114            println!("Signature is VALID. (scheme: {})", signed.scheme);
115            return Ok(());
116        }
117        eprintln!("Signature is INVALID.");
118        std::process::exit(1);
119    }
120
121    // Fall back to legacy format (requires --key)
122    let key_path = args.key.as_ref().ok_or_else(|| {
123        anyhow::anyhow!(
124            "Public key file (--key) is required for legacy signature format.\n\
125             Signatures created with 'sign --public-key' embed the public key \
126             and don't need --key."
127        )
128    })?;
129
130    verify_legacy(&args, &sig_json, key_path)
131}

133/// Read the verify input from `--input <path>` or stdin (round-7 fix #10).
134fn read_verify_input(args: &VerifyArgs) -> Result<Vec<u8>> {
135    if let Some(path) = &args.input {
136        super::common::enforce_input_size_limit(
137            path,
138            super::common::CLI_MAX_SIGNATURE_INPUT_BYTES,
139            "verify",
140        )?;
141        std::fs::read(path).with_context(|| format!("Failed to read {}", path.display()))
142    } else {
143        super::common::read_stdin_with_limit(super::common::CLI_MAX_SIGNATURE_INPUT_BYTES, "verify")
144    }
145}

147/// Human-readable label for the input source — used in error messages.
148fn input_label(args: &VerifyArgs) -> String {
149    args.input
150        .as_ref()
151        .map(|p| format!("'{}'", p.display()))
152        .unwrap_or_else(|| "<stdin>".to_string())
153}

155/// Legacy verification path — custom JSON format + primitive-level API.
156//
157// `clippy::exit` allowed here for the same reason as `run` — see the
158// commentary above the `run` function.
159#[allow(clippy::exit)]
160fn verify_legacy(args: &VerifyArgs, sig_json: &str, key_path: &std::path::Path) -> Result<()> {
161    let data = read_verify_input(args)?;
162
163    let key_file = KeyFile::read_from(key_path)?;
164    if key_file.key_type != KeyType::Public {
165        bail!("Expected public key file for verification, got {:?}", key_file.key_type);
166    }
167
168    let algorithm = if let Some(alg) = &args.algorithm {
169        alg.clone()
170    } else {
171        let detected = detect_algorithm(sig_json)?;
172        eprintln!("Auto-detected algorithm: {}", algorithm_name(&detected));
173        detected
174    };
175
176    let expected_alg = algorithm_name(&algorithm);
177    key_file.validate_algorithm(expected_alg)?;
178
179    let pk_bytes = key_file.key_bytes()?;
180
181    let valid = match algorithm {
182        VerifyAlgorithm::Hybrid => verify_hybrid(&data, sig_json, &pk_bytes)?,
183        _ => verify_standard(&data, sig_json, &pk_bytes, &algorithm)?,
184    };
185
186    if valid {
187        println!("Signature is VALID.");
188        Ok(())
189    } else {
190        // Round-7 audit fix #8 + #11: see explanation in the SignedData
191        // path above. Exit 1 reserved for "signature INVALID".
192        eprintln!("Signature is INVALID.");
193        std::process::exit(1);
194    }
195}

197/// Detect algorithm from the "algorithm" field in a legacy .sig.json file.
198fn detect_algorithm(sig_json: &str) -> Result<VerifyAlgorithm> {
199    let sig: serde_json::Value =
200        serde_json::from_str(sig_json).context("Failed to parse signature JSON")?;
201    let alg_str = sig
202        .get("algorithm")
203        .and_then(|v| v.as_str())
204        .context("Signature file missing 'algorithm' field — cannot auto-detect")?;
205    match alg_str {
206        "ml-dsa-65" => Ok(VerifyAlgorithm::MlDsa65),
207        "ml-dsa-44" => Ok(VerifyAlgorithm::MlDsa44),
208        "ml-dsa-87" => Ok(VerifyAlgorithm::MlDsa87),
209        "slh-dsa-shake-128s" => Ok(VerifyAlgorithm::SlhDsa),
210        "fn-dsa-512" => Ok(VerifyAlgorithm::FnDsa),
211        "ed25519" => Ok(VerifyAlgorithm::Ed25519),
212        "hybrid-ml-dsa-65-ed25519" => Ok(VerifyAlgorithm::Hybrid),
213        other => bail!("Unknown algorithm in signature file: '{other}'"),
214    }
215}

217/// Map verify algorithm enum to canonical key file algorithm name.
218fn algorithm_name(alg: &VerifyAlgorithm) -> &'static str {
219    match alg {
220        VerifyAlgorithm::MlDsa65 => "ml-dsa-65",
221        VerifyAlgorithm::MlDsa44 => "ml-dsa-44",
222        VerifyAlgorithm::MlDsa87 => "ml-dsa-87",
223        VerifyAlgorithm::SlhDsa => "slh-dsa-shake-128s",
224        VerifyAlgorithm::FnDsa => "fn-dsa-512",
225        VerifyAlgorithm::Ed25519 => "ed25519",
226        VerifyAlgorithm::Hybrid => "hybrid-ml-dsa-65-ed25519",
227    }
228}

230fn verify_standard(
231    data: &[u8],
232    sig_json: &str,
233    pk_bytes: &[u8],
234    algorithm: &VerifyAlgorithm,
235) -> Result<bool> {
236    let sig_obj: serde_json::Value =
237        serde_json::from_str(sig_json).context("Failed to parse signature JSON")?;
238
239    use base64::Engine;
240    let sig_b64 = sig_obj
241        .get("signature")
242        .and_then(|v| v.as_str())
243        .context("Missing 'signature' field in signature JSON")?;
244    let sig_bytes = base64::engine::general_purpose::STANDARD
245        .decode(sig_b64)
246        .context("Invalid base64 in signature")?;
247
248    match algorithm {
249        VerifyAlgorithm::MlDsa65 => latticearc::verify_pq_ml_dsa(
250            data,
251            &sig_bytes,
252            pk_bytes,
253            latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa65,
254            latticearc::SecurityMode::Unverified,
255        )
256        .map_err(|e| anyhow::anyhow!("Verification failed: {e}")),
257        VerifyAlgorithm::MlDsa44 => latticearc::verify_pq_ml_dsa(
258            data,
259            &sig_bytes,
260            pk_bytes,
261            latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa44,
262            latticearc::SecurityMode::Unverified,
263        )
264        .map_err(|e| anyhow::anyhow!("Verification failed: {e}")),
265        VerifyAlgorithm::MlDsa87 => latticearc::verify_pq_ml_dsa(
266            data,
267            &sig_bytes,
268            pk_bytes,
269            latticearc::primitives::sig::ml_dsa::MlDsaParameterSet::MlDsa87,
270            latticearc::SecurityMode::Unverified,
271        )
272        .map_err(|e| anyhow::anyhow!("Verification failed: {e}")),
273        VerifyAlgorithm::SlhDsa => latticearc::verify_pq_slh_dsa(
274            data,
275            &sig_bytes,
276            pk_bytes,
277            latticearc::primitives::sig::slh_dsa::SlhDsaSecurityLevel::Shake128s,
278            latticearc::SecurityMode::Unverified,
279        )
280        .map_err(|e| anyhow::anyhow!("Verification failed: {e}")),
281        VerifyAlgorithm::FnDsa => latticearc::verify_pq_fn_dsa(
282            data,
283            &sig_bytes,
284            pk_bytes,
285            latticearc::primitives::sig::fndsa::FnDsaSecurityLevel::Level512,
286            latticearc::SecurityMode::Unverified,
287        )
288        .map_err(|e| anyhow::anyhow!("Verification failed: {e}")),
289        VerifyAlgorithm::Ed25519 => latticearc::verify_ed25519(
290            data,
291            &sig_bytes,
292            pk_bytes,
293            latticearc::SecurityMode::Unverified,
294        )
295        .map_err(|e| anyhow::anyhow!("Verification failed: {e}")),
296        VerifyAlgorithm::Hybrid => bail!("Internal error: hybrid handled separately"),
297    }
298}

300fn verify_hybrid(data: &[u8], sig_json: &str, pk_bytes: &[u8]) -> Result<bool> {
301    let sig: serde_json::Value =
302        serde_json::from_str(sig_json).context("Failed to parse hybrid signature JSON")?;
303
304    use base64::Engine;
305    let ml_dsa_b64 = sig
306        .get("ml_dsa_sig")
307        .and_then(|v| v.as_str())
308        .context("Missing 'ml_dsa_sig' field in hybrid signature")?;
309    let ml_dsa_sig = base64::engine::general_purpose::STANDARD
310        .decode(ml_dsa_b64)
311        .context("Invalid base64 in ml_dsa_sig")?;
312
313    let ed25519_b64 = sig
314        .get("ed25519_sig")
315        .and_then(|v| v.as_str())
316        .context("Missing 'ed25519_sig' field in hybrid signature")?;
317    let ed25519_sig = base64::engine::general_purpose::STANDARD
318        .decode(ed25519_b64)
319        .context("Invalid base64 in ed25519_sig")?;
320
321    let pk = crate::keyfile::parse_hybrid_sign_pk_from_bytes(pk_bytes)?;
322    let hybrid_sig = latticearc::hybrid::sig_hybrid::HybridSignature::new(ml_dsa_sig, ed25519_sig);
323
324    latticearc::verify_hybrid_signature(
325        data,
326        &hybrid_sig,
327        &pk,
328        latticearc::SecurityMode::Unverified,
329    )
330    .map_err(|e| anyhow::anyhow!("Hybrid verification failed: {e}"))
