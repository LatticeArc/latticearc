1//! KDF command — derive cryptographic keys from input material.
2//!
3//! Key Derivation Functions (KDFs) stretch or transform passwords, shared
4//! secrets, or raw key material into one or more cryptographic keys of a
5//! desired length.
6//!
7//! **Supported algorithms:**
8//!
9//! | Algorithm        | Standard   | Use case                                  |
10//! |------------------|------------|-------------------------------------------|
11//! | HKDF-SHA256      | SP 800-56C | Deriving keys from high-entropy material.  |
12//! | PBKDF2-HMAC-SHA256 | SP 800-132 | Deriving keys from passwords (slow hash). |
13//!
14//! **When to use which:**
15//! - **HKDF** is for *non-password* input (DH shared secrets, random bytes).
16//!   It's fast, deterministic, and domain-separable via `--info`.
17//! - **PBKDF2** is for *passwords*. It's deliberately slow (600,000 iterations
18//!   by default) to resist brute-force attacks on weak passwords.
19//!
20//! **Output formats:** `hex` (default, lowercase) or `base64`.
21//!
22//! **HKDF output limit:** max 8,160 bytes (255 × 32 for SHA-256, per RFC 5869).
23
24use anyhow::{Context, Result, bail};
25use clap::{Args, ValueEnum};
26
27/// KDF algorithm.
28#[derive(Debug, Clone, ValueEnum)]
29pub(crate) enum KdfAlgorithm {
30    /// HKDF-SHA256 (SP 800-56C). Requires --salt and --info.
31    Hkdf,
32    /// PBKDF2-HMAC-SHA256. Requires --salt and --iterations.
33    Pbkdf2,
34}
35
36/// Arguments for the `kdf` subcommand.
37#[derive(Args)]
38pub(crate) struct KdfArgs {
39    /// KDF algorithm.
40    #[arg(short, long, value_enum)]
41    pub algorithm: KdfAlgorithm,
42    /// Input key material (hex-encoded) or password (for PBKDF2).
43    ///
44    /// **SECURITY:** Passing a password as a CLI argument is visible in
45    /// `ps`, `/proc/<pid>/cmdline`, shell history, and kernel audit
46    /// logs. For PBKDF2 use cases, prefer one of:
47    /// - Read from `LATTICEARC_KDF_INPUT` env var (set then `unset`).
48    /// - Pipe the password on stdin and pass `--input-stdin`.
49    ///
50    /// `--input` is supported for HKDF (where the input is hex-encoded
51    /// non-secret material) and for backwards-compat with KAT replay
52    /// scripts.
53    #[arg(short, long, conflicts_with = "input_stdin")]
54    pub input: Option<String>,
55    /// Read the input from stdin (one line, no trailing newline). Use
56    /// instead of `--input` when the input is a secret (PBKDF2
57    /// password) and you don't want it in `ps` / shell history.
58    #[arg(long, conflicts_with = "input")]
59    pub input_stdin: bool,
60    /// Salt (hex-encoded).
61    #[arg(short, long)]
62    pub salt: String,
63    /// Output key length in bytes.
64    #[arg(short, long, default_value = "32")]
65    pub length: usize,
66    /// Info string for HKDF (optional).
67    #[arg(long)]
68    pub info: Option<String>,
69    /// Iteration count for PBKDF2 (default 600000, OWASP 2023 minimum
70    /// for HMAC-SHA256). Values below 600,000 are rejected unless
71    /// `--allow-weak-iterations` is also passed.
72    #[arg(long, default_value = "600000")]
73    pub iterations: u32,
74    /// Bypass the OWASP-2023 PBKDF2 iteration floor (600,000). Reserved
75    /// for KAT replay and reproducibility against legacy fixtures.
76    /// Production use is unsupported.
77    #[arg(long)]
78    pub allow_weak_iterations: bool,
79    /// Allow `--input <secret>` with `--algorithm pbkdf2`. Reserved for
80    /// KAT replay and reproducibility — production use is unsupported
81    /// because argv-passed secrets are visible to `ps`, kernel audit
82    /// logs, and shell history.
83    #[arg(long)]
84    pub allow_argv_secret: bool,
85    /// Output format: hex (default) or base64.
86    #[arg(short, long, default_value = "hex")]
87    pub format: super::hash::OutputFormat,
88}

90/// Execute the kdf command.
91pub(crate) fn run(args: KdfArgs) -> Result<()> {
92    if args.salt.is_empty() {
93        bail!(
94            "--salt must not be empty. PBKDF2 requires ≥16 bytes (NIST SP 800-132 §5.1); \
95             HKDF accepts an empty salt but a fixed application-specific salt is preferred."
96        );
97    }
98    let salt = hex::decode(&args.salt).context("Invalid hex in --salt")?;
99    // Round-21 audit fix #15: the previous version's error message
100    // claimed "PBKDF2 requires ≥16 bytes" but only `is_empty()` was
101    // checked. A 1-byte salt (`--salt aa`) silently passed. Enforce
102    // the SP 800-132 §5.1 minimum here for PBKDF2 — HKDF tolerates
103    // shorter salts but the CLI surface is shared, so apply the
104    // tighter rule.
105    if matches!(args.algorithm, KdfAlgorithm::Pbkdf2) && salt.len() < 16 {
106        bail!(
107            "--salt is {} byte(s); PBKDF2 requires ≥16 bytes (NIST SP 800-132 §5.1).",
108            salt.len()
109        );
110    }
111
112    // PBKDF2 with `--input` puts the password on argv (visible in `ps`,
113    // `/proc/<pid>/cmdline`, shell history). The `--input` flag's docstring
114    // calls this out, but argv routing is not algorithm-aware in clap, so
115    // a user can still combine `--algorithm pbkdf2 --input <password>`
116    // without the warning binding. Reject unless explicitly opted in via
117    // `--allow-argv-secret` (KAT replay only).
118    if matches!(args.algorithm, KdfAlgorithm::Pbkdf2)
119        && args.input.is_some()
120        && !args.allow_argv_secret
121    {
122        bail!(
123            "Refusing to read a PBKDF2 password from --input because the value is visible in \
124             `ps`, `/proc/<pid>/cmdline`, and shell history. Use --input-stdin (recommended) \
125             or LATTICEARC_KDF_INPUT. Pass --allow-argv-secret to bypass for KAT replay only."
126        );
127    }
128
129    let resolved_input = resolve_input(&args)?;
130
131    let derived = match args.algorithm {
132        KdfAlgorithm::Hkdf => derive_hkdf_with_input(&args, &salt, &resolved_input)?,
133        KdfAlgorithm::Pbkdf2 => derive_pbkdf2_with_input(&args, &salt, &resolved_input)?,
134    };
135
136    // Encoded output stays in `Zeroizing<String>` so password-derived key
137    // material can't linger on the heap until allocator reclaim.
138    let encoded: zeroize::Zeroizing<String> = match args.format {
139        super::hash::OutputFormat::Hex => zeroize::Zeroizing::new(hex::encode(derived.as_slice())),
140        super::hash::OutputFormat::Base64 => {
141            use base64::Engine;
142            zeroize::Zeroizing::new(
143                base64::engine::general_purpose::STANDARD.encode(derived.as_slice()),
144            )
145        }
146    };
147
148    println!("{}", encoded.as_str());
149    Ok(())
150}

152/// Resolve the input string from one of:
153/// - `--input <value>` (CLI argument; visible in `ps`/history — flagged
154///   in the arg's docstring and acceptable for HKDF / KAT replay).
155/// - `--input-stdin` (read one line from stdin, no trailing newline;
156///   the recommended path for PBKDF2 passwords).
157/// - `LATTICEARC_KDF_INPUT` env var (recommended for scripted PBKDF2;
158///   caller must `unset` immediately afterwards — see SECURITY.md).
159///
160/// Exactly one source must be provided. The result is `Zeroizing` so
161/// password material is wiped at function exit.
162///
163/// # Precedence intentionally differs from `keyfile::resolve_passphrase`
164///
165/// `keyfile::resolve_passphrase` checks `LATTICEARC_PASSPHRASE` FIRST
166/// (env > prompt). Here we check `--input` first, then `--input-stdin`,
167/// then the env var. The asymmetry is intentional, not a bug:
168///
169/// - Passphrase: there is only ONE possible explicit source (the prompt
170///   — there's no `--passphrase <value>` flag, deliberately, because
171///   that would put the secret on the command line). The env var is
172///   THE escape hatch for non-interactive use, so env-first is right.
173/// - KDF input: there are THREE possible explicit sources, and CLI
174///   ergonomics demand that an explicit `--input` or `--input-stdin`
175///   take precedence over a stale env var that may have leaked in from
176///   a parent shell. Otherwise `LATTICEARC_KDF_INPUT=foo
177///   latticearc-cli kdf --input bar …` would silently use `foo`.
178///
179/// Round-8 audit fix #7 requires this justification be inline.
180fn resolve_input(args: &KdfArgs) -> Result<zeroize::Zeroizing<String>> {
181    if let Some(s) = args.input.as_deref() {
182        return Ok(zeroize::Zeroizing::new(s.to_string()));
183    }
184    if args.input_stdin {
185        use std::io::BufRead;
186        let stdin = std::io::stdin();
187        let mut buf = String::new();
188        stdin.lock().read_line(&mut buf).context("Failed to read --input-stdin")?;
189        // Strip a single trailing newline (LF or CRLF) — common when
190        // the user does `echo password | latticearc-cli kdf …`.
191        if buf.ends_with('\n') {
192            buf.pop();
193            if buf.ends_with('\r') {
194                buf.pop();
195            }
196        }
197        if buf.is_empty() {
198            bail!("--input-stdin received an empty line");
199        }
200        return Ok(zeroize::Zeroizing::new(buf));
201    }
202    if let Ok(env_val) = std::env::var("LATTICEARC_KDF_INPUT") {
203        if env_val.is_empty() {
204            bail!("LATTICEARC_KDF_INPUT is set but empty");
205        }
206        // Round-20 audit fix #8: emit a TTY warning when the env-var
207        // path is used. Env vars are visible via `/proc/<pid>/environ`
208        // for any process the user can read (typically the same UID).
209        // `keyfile.rs::resolve_passphrase` already warns for the
210        // analogous LATTICEARC_PASSPHRASE; the kdf path was missing
211        // the parallel warning.
212        eprintln!(
213            "warning: reading KDF input from LATTICEARC_KDF_INPUT. \
214             Env vars are readable via /proc/<pid>/environ by processes \
215             owned by the same user. Prefer --input-stdin for genuinely \
216             secret input, and `unset LATTICEARC_KDF_INPUT` immediately \
217             after this invocation."
218        );
219        return Ok(zeroize::Zeroizing::new(env_val));
220    }
221    bail!(
222        "kdf requires one of: --input <value>, --input-stdin, or \
223         LATTICEARC_KDF_INPUT in the environment. For PBKDF2 passwords, \
224         prefer --input-stdin so the password is not visible in `ps` / \
225         `/proc/<pid>/cmdline` / shell history. Recommended pattern:\n\
226         \n\
227            read -rs PASS && printf '%s' \"$PASS\" | latticearc-cli kdf --input-stdin …\n\
228         \n\
229         `read -rs` reads silently into a shell variable, and `printf '%s'` \
230         pipes the value without writing it to the shell history."
231    )
232}

234/// Upper bound on KDF output length applied by the CLI to both HKDF and
235/// PBKDF2. Set to HKDF-SHA256's algorithmic maximum (255 × 32 = 8160
236/// bytes) so the same constant is correct for both algorithms — PBKDF2
237/// has no algorithmic ceiling but allowing arbitrary `--length` is a
238/// trivial self-DoS (e.g. `--length 1073741824` allocates 1 GiB and
239/// then runs 600k iterations per block).
240///
241/// Round-20 audit fix #7: previously this constant was 8192 and used
242/// only by the PBKDF2 path; HKDF independently hardcoded 8160 in its
243/// `bail!` check. The constants silently desynced. Both paths now
244/// reference this single source of truth.
245const CLI_MAX_KDF_OUTPUT_LEN: usize = 8160;
246
247fn derive_hkdf_with_input(
248    args: &KdfArgs,
249    salt: &[u8],
250    input: &str,
251) -> Result<zeroize::Zeroizing<Vec<u8>>> {
252    let ikm =
253        zeroize::Zeroizing::new(hex::decode(input).context("Invalid hex in --input for HKDF")?);
254    let info = args.info.as_deref().unwrap_or("");
255
256    if args.length == 0 || args.length > CLI_MAX_KDF_OUTPUT_LEN {
257        bail!("Output length must be 1..={CLI_MAX_KDF_OUTPUT_LEN} bytes");
258    }
259
260    let derived = latticearc::derive_key_with_info(
261        &ikm,
262        salt,
263        args.length,
264        info.as_bytes(),
265        latticearc::SecurityMode::Unverified,
266    )
267    .map_err(|e| anyhow::anyhow!("HKDF derivation failed: {e}"))?;
268    Ok(zeroize::Zeroizing::new(derived))
269}

271/// OWASP 2023 minimum iteration count for PBKDF2-HMAC-SHA256.
272/// Library `pbkdf2()` already enforces a hard floor of 1,000 (the
273/// SP 800-132 absolute minimum); the CLI raises that to the OWASP
274/// recommendation as the public-facing default to reflect the
275/// password-hashing use case the CLI targets.
276const OWASP_PBKDF2_MIN_ITERATIONS: u32 = 600_000;
277
278fn derive_pbkdf2_with_input(
279    args: &KdfArgs,
280    salt: &[u8],
281    input: &str,
282) -> Result<zeroize::Zeroizing<Vec<u8>>> {
283    let password = zeroize::Zeroizing::new(input.as_bytes().to_vec());
284
285    if args.length == 0 || args.length > CLI_MAX_KDF_OUTPUT_LEN {
286        bail!(
287            "Output length must be 1..={CLI_MAX_KDF_OUTPUT_LEN} bytes (PBKDF2 has no \
288             algorithmic ceiling but the CLI caps to prevent self-DoS)."
289        );
290    }
291
292    if args.iterations < OWASP_PBKDF2_MIN_ITERATIONS && !args.allow_weak_iterations {
293        bail!(
294            "PBKDF2 iteration count {iters} is below the OWASP 2023 minimum of {min}. \
295             Pass --allow-weak-iterations to bypass (KAT replay only — not for production).",
296            iters = args.iterations,
297            min = OWASP_PBKDF2_MIN_ITERATIONS
298        );
299    }
300    if args.iterations < OWASP_PBKDF2_MIN_ITERATIONS {
301        eprintln!(
302            "warning: PBKDF2 iteration count {iters} is below the OWASP 2023 \
303             minimum ({min}); --allow-weak-iterations was passed.",
304            iters = args.iterations,
305            min = OWASP_PBKDF2_MIN_ITERATIONS
306        );
307    }
308
309    let params = latticearc::primitives::kdf::Pbkdf2Params::with_salt(salt)
310        .iterations(args.iterations)
311        .key_length(args.length);
312
313    let result = latticearc::primitives::kdf::pbkdf2(&password, &params)
314        .map_err(|e| anyhow::anyhow!("PBKDF2 derivation failed: {e}"))?;
315
316    Ok(zeroize::Zeroizing::new(result.expose_secret().to_vec()))
317}
