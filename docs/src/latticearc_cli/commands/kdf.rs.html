1//! KDF command — derive cryptographic keys from input material.
2//!
3//! Key Derivation Functions (KDFs) stretch or transform passwords, shared
4//! secrets, or raw key material into one or more cryptographic keys of a
5//! desired length.
6//!
7//! **Supported algorithms:**
8//!
9//! | Algorithm        | Standard   | Use case                                  |
10//! |------------------|------------|-------------------------------------------|
11//! | HKDF-SHA256      | SP 800-56C | Deriving keys from high-entropy material.  |
12//! | PBKDF2-HMAC-SHA256 | SP 800-132 | Deriving keys from passwords (slow hash). |
13//!
14//! **When to use which:**
15//! - **HKDF** is for *non-password* input (DH shared secrets, random bytes).
16//!   It's fast, deterministic, and domain-separable via `--info`.
17//! - **PBKDF2** is for *passwords*. It's deliberately slow (600,000 iterations
18//!   by default) to resist brute-force attacks on weak passwords.
19//!
20//! **Output formats:** `hex` (default, lowercase) or `base64`.
21//!
22//! **HKDF output limit:** max 8,160 bytes (255 × 32 for SHA-256, per RFC 5869).
23
24use anyhow::{Context, Result, bail};
25use clap::{Args, ValueEnum};
26
27/// KDF algorithm.
28#[derive(Debug, Clone, ValueEnum)]
29pub(crate) enum KdfAlgorithm {
30    /// HKDF-SHA256 (SP 800-56C). Requires --salt and --info.
31    Hkdf,
32    /// PBKDF2-HMAC-SHA256. Requires --salt and --iterations.
33    Pbkdf2,
34}
35
36/// Arguments for the `kdf` subcommand.
37#[derive(Args)]
38pub(crate) struct KdfArgs {
39    /// KDF algorithm.
40    #[arg(short, long, value_enum)]
41    pub algorithm: KdfAlgorithm,
42    /// Input key material (hex-encoded) or password (for PBKDF2).
43    ///
44    /// **SECURITY:** Passing a password as a CLI argument is visible in
45    /// `ps`, `/proc/<pid>/cmdline`, shell history, and kernel audit
46    /// logs. For PBKDF2 use cases, prefer one of:
47    /// - Read from `LATTICEARC_KDF_INPUT` env var (set then `unset`).
48    /// - Pipe the password on stdin and pass `--input-stdin`.
49    ///
50    /// `--input` is supported for HKDF (where the input is hex-encoded
51    /// non-secret material) and for backwards-compat with KAT replay
52    /// scripts.
53    #[arg(short, long, conflicts_with = "input_stdin")]
54    pub input: Option<String>,
55    /// Read the input from stdin (one line, no trailing newline). Use
56    /// instead of `--input` when the input is a secret (PBKDF2
57    /// password) and you don't want it in `ps` / shell history.
58    #[arg(long, conflicts_with = "input")]
59    pub input_stdin: bool,
60    /// Salt (hex-encoded).
61    #[arg(short, long)]
62    pub salt: String,
63    /// Output key length in bytes.
64    #[arg(short, long, default_value = "32")]
65    pub length: usize,
66    /// Info string for HKDF (optional).
67    #[arg(long)]
68    pub info: Option<String>,
69    /// Iteration count for PBKDF2 (default 600000, OWASP 2023 minimum
70    /// for HMAC-SHA256). Values below 600,000 are rejected unless
71    /// `--allow-weak-iterations` is also passed.
72    #[arg(long, default_value = "600000")]
73    pub iterations: u32,
74    /// Bypass the OWASP-2023 PBKDF2 iteration floor (600,000). Reserved
75    /// for KAT replay and reproducibility against legacy fixtures.
76    /// Production use is unsupported.
77    #[arg(long)]
78    pub allow_weak_iterations: bool,
79    /// Output format: hex (default) or base64.
80    #[arg(short, long, default_value = "hex")]
81    pub format: super::hash::OutputFormat,
82}
83
84/// Execute the kdf command.
85pub(crate) fn run(args: KdfArgs) -> Result<()> {
86    let salt = hex::decode(&args.salt).context("Invalid hex in --salt")?;
87
88    let resolved_input = resolve_input(&args)?;
89
90    let derived = match args.algorithm {
91        KdfAlgorithm::Hkdf => derive_hkdf_with_input(&args, &salt, &resolved_input)?,
92        KdfAlgorithm::Pbkdf2 => derive_pbkdf2_with_input(&args, &salt, &resolved_input)?,
93    };
94
95    let encoded = match args.format {
96        super::hash::OutputFormat::Hex => hex::encode(&derived),
97        super::hash::OutputFormat::Base64 => {
98            use base64::Engine;
99            base64::engine::general_purpose::STANDARD.encode(&derived)
100        }
101    };
102
103    println!("{encoded}");
104    Ok(())
105}

107/// Resolve the input string from one of:
108/// - `--input <value>` (CLI argument; visible in `ps`/history — flagged
109///   in the arg's docstring and acceptable for HKDF / KAT replay).
110/// - `--input-stdin` (read one line from stdin, no trailing newline;
111///   the recommended path for PBKDF2 passwords).
112/// - `LATTICEARC_KDF_INPUT` env var (recommended for scripted PBKDF2;
113///   caller must `unset` immediately afterwards — see SECURITY.md).
114///
115/// Exactly one source must be provided. The result is `Zeroizing` so
116/// password material is wiped at function exit.
117///
118/// # Precedence intentionally differs from `keyfile::resolve_passphrase`
119///
120/// `keyfile::resolve_passphrase` checks `LATTICEARC_PASSPHRASE` FIRST
121/// (env > prompt). Here we check `--input` first, then `--input-stdin`,
122/// then the env var. The asymmetry is intentional, not a bug:
123///
124/// - Passphrase: there is only ONE possible explicit source (the prompt
125///   — there's no `--passphrase <value>` flag, deliberately, because
126///   that would put the secret on the command line). The env var is
127///   THE escape hatch for non-interactive use, so env-first is right.
128/// - KDF input: there are THREE possible explicit sources, and CLI
129///   ergonomics demand that an explicit `--input` or `--input-stdin`
130///   take precedence over a stale env var that may have leaked in from
131///   a parent shell. Otherwise `LATTICEARC_KDF_INPUT=foo
132///   latticearc-cli kdf --input bar …` would silently use `foo`.
133///
134/// Round-8 audit fix #7 requires this justification be inline.
135fn resolve_input(args: &KdfArgs) -> Result<zeroize::Zeroizing<String>> {
136    if let Some(s) = args.input.as_deref() {
137        return Ok(zeroize::Zeroizing::new(s.to_string()));
138    }
139    if args.input_stdin {
140        use std::io::BufRead;
141        let stdin = std::io::stdin();
142        let mut buf = String::new();
143        stdin.lock().read_line(&mut buf).context("Failed to read --input-stdin")?;
144        // Strip a single trailing newline (LF or CRLF) — common when
145        // the user does `echo password | latticearc-cli kdf …`.
146        if buf.ends_with('\n') {
147            buf.pop();
148            if buf.ends_with('\r') {
149                buf.pop();
150            }
151        }
152        if buf.is_empty() {
153            bail!("--input-stdin received an empty line");
154        }
155        return Ok(zeroize::Zeroizing::new(buf));
156    }
157    if let Ok(env_val) = std::env::var("LATTICEARC_KDF_INPUT") {
158        if env_val.is_empty() {
159            bail!("LATTICEARC_KDF_INPUT is set but empty");
160        }
161        return Ok(zeroize::Zeroizing::new(env_val));
162    }
163    bail!(
164        "kdf requires one of: --input <value>, --input-stdin, or \
165         LATTICEARC_KDF_INPUT in the environment. For PBKDF2 passwords, \
166         prefer --input-stdin (`echo $PASS | latticearc-cli kdf --input-stdin …`) \
167         so the password isn't visible in `ps` / shell history."
168    )
169}

171fn derive_hkdf_with_input(args: &KdfArgs, salt: &[u8], input: &str) -> Result<Vec<u8>> {
172    let ikm =
173        zeroize::Zeroizing::new(hex::decode(input).context("Invalid hex in --input for HKDF")?);
174    let info = args.info.as_deref().unwrap_or("");
175
176    if args.length == 0 || args.length > 8160 {
177        bail!("Output length must be 1..=8160 bytes");
178    }
179
180    latticearc::derive_key_with_info(
181        &ikm,
182        salt,
183        args.length,
184        info.as_bytes(),
185        latticearc::SecurityMode::Unverified,
186    )
187    .map_err(|e| anyhow::anyhow!("HKDF derivation failed: {e}"))
188}

190/// OWASP 2023 minimum iteration count for PBKDF2-HMAC-SHA256.
191/// Library `pbkdf2()` already enforces a hard floor of 1,000 (the
192/// SP 800-132 absolute minimum); the CLI raises that to the OWASP
193/// recommendation as the public-facing default to reflect the
194/// password-hashing use case the CLI targets.
195const OWASP_PBKDF2_MIN_ITERATIONS: u32 = 600_000;
196
197fn derive_pbkdf2_with_input(args: &KdfArgs, salt: &[u8], input: &str) -> Result<Vec<u8>> {
198    let password = zeroize::Zeroizing::new(input.as_bytes().to_vec());
199
200    if args.length == 0 {
201        bail!("Output length must be > 0");
202    }
203
204    if args.iterations < OWASP_PBKDF2_MIN_ITERATIONS && !args.allow_weak_iterations {
205        bail!(
206            "PBKDF2 iteration count {iters} is below the OWASP 2023 minimum of {min}. \
207             Pass --allow-weak-iterations to bypass (KAT replay only — not for production).",
208            iters = args.iterations,
209            min = OWASP_PBKDF2_MIN_ITERATIONS
210        );
211    }
212    if args.iterations < OWASP_PBKDF2_MIN_ITERATIONS {
213        eprintln!(
214            "warning: PBKDF2 iteration count {iters} is below the OWASP 2023 \
215             minimum ({min}); --allow-weak-iterations was passed.",
216            iters = args.iterations,
217            min = OWASP_PBKDF2_MIN_ITERATIONS
218        );
219    }
220
221    let params = latticearc::primitives::kdf::Pbkdf2Params::with_salt(salt)
222        .iterations(args.iterations)
223        .key_length(args.length);
224
225    let result = latticearc::primitives::kdf::pbkdf2(&password, &params)
226        .map_err(|e| anyhow::anyhow!("PBKDF2 derivation failed: {e}"))?;
227
228    Ok(result.key().to_vec())
229}
