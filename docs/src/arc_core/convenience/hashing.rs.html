1//! Hashing, HMAC, and key derivation operations
2//!
3//! This module provides cryptographic hashing, HMAC, and key derivation functions.
4//!
5//! ## Zero Trust Enforcement
6//!
7//! HMAC and key derivation functions use `SecurityMode` to specify verification behavior:
8//! - `SecurityMode::Verified(&session)`: Validates session before operation
9//! - `SecurityMode::Unverified`: Skips session validation
10//!
11//! Hash functions are stateless and don't require a security mode.
12//!
13//! For opt-out scenarios where Zero Trust verification is not required or not possible,
14//! use the `_unverified` variants.

31/// Custom output length type for aws-lc-rs HKDF
32struct HkdfOutputLen(usize);
33
34impl KeyType for HkdfOutputLen {
35    fn len(&self) -> usize {
36        self.0
37    }
38}

40fn hash_parallel(data: &[u8], chunk_size: usize) -> Vec<[u8; 32]> {
41    data.par_chunks(chunk_size)
42        .map(|chunk| {
43            let mut hasher = Sha3_256::new();
44            hasher.update(chunk);
45            hasher.finalize().into()
46        })
47        .collect()
48}
49
50#[inline]
51fn hash_sha3_256(data: &[u8]) -> [u8; 32] {
52    // Use parallel hashing for large data
53    if data.len() > 65536 {
54        let results = hash_parallel(data, 4096);
55        let mut final_hasher = Sha3_256::new();
56        for hash in &results {
57            final_hasher.update(hash);
58        }
59        return final_hasher.finalize().into();
60    }
61
62    let mut hasher = Sha3_256::new();
63    hasher.update(data);
64    hasher.finalize().into()
65}

71/// Internal implementation of HKDF key derivation.
72fn derive_key_hkdf(password: &[u8], salt: &[u8], length: usize) -> Result<Vec<u8>> {
73    let hkdf_salt = Salt::new(HKDF_SHA256, salt);
74    let prk = hkdf_salt.extract(password);
75
76    let okm = prk
77        .expand(&[b"latticearc"], HkdfOutputLen(length))
78        .map_err(|_e| CoreError::KeyDerivationFailed("HKDF expansion failed".to_string()))?;
79
80    let mut output = vec![0u8; length];
81    okm.fill(&mut output)
82        .map_err(|_e| CoreError::KeyDerivationFailed("HKDF fill failed".to_string()))?;
83
84    Ok(output)
85}

87/// Internal implementation of HKDF key derivation with caller-supplied info string.
88///
89/// Identical to [`derive_key_hkdf`] but uses the caller's `info` parameter
90/// instead of the hardcoded `b"latticearc"` context string, allowing domain
91/// separation for different key derivation contexts.
92fn derive_key_hkdf_with_info(
93    password: &[u8],
94    salt: &[u8],
95    length: usize,
96    info: &[u8],
97) -> Result<Vec<u8>> {
98    let hkdf_salt = Salt::new(HKDF_SHA256, salt);
99    let prk = hkdf_salt.extract(password);
100
101    let info_slices: &[&[u8]] = &[info];
102    let okm = prk
103        .expand(info_slices, HkdfOutputLen(length))
104        .map_err(|_e| CoreError::KeyDerivationFailed("HKDF expansion failed".to_string()))?;
105
106    let mut output = vec![0u8; length];
107    okm.fill(&mut output)
108        .map_err(|_e| CoreError::KeyDerivationFailed("HKDF fill failed".to_string()))?;
109
110    Ok(output)
111}

113/// Internal implementation of key derivation with caller-supplied info string.
114fn derive_key_with_info_internal(
115    password: &[u8],
116    salt: &[u8],
117    length: usize,
118    info: &[u8],
119) -> Result<Vec<u8>> {
120    crate::log_crypto_operation_start!(
121        "key_derivation_info",
122        algorithm = "HKDF-SHA256",
123        output_len = length,
124        info_len = info.len()
125    );
126
127    validate_key_derivation_count(1).map_err(|e| {
128        crate::log_crypto_operation_error!("key_derivation_info", e);
129        CoreError::ResourceExceeded(e.to_string())
130    })?;
131
132    if salt.is_empty() {
133        let err = CoreError::InvalidInput("Salt cannot be empty".to_string());
134        crate::log_crypto_operation_error!("key_derivation_info", err);
135        return Err(err);
136    }
137
138    if length == 0 {
139        let err = CoreError::InvalidInput("Length cannot be zero".to_string());
140        crate::log_crypto_operation_error!("key_derivation_info", err);
141        return Err(err);
142    }
143
144    if info.is_empty() {
145        let err = CoreError::InvalidInput("Info string cannot be empty".to_string());
146        crate::log_crypto_operation_error!("key_derivation_info", err);
147        return Err(err);
148    }
149
150    let result = derive_key_hkdf_with_info(password, salt, length, info);
151
152    match &result {
153        Ok(_) => {
154            crate::log_crypto_operation_complete!(
155                "key_derivation_info",
156                algorithm = "HKDF-SHA256",
157                output_len = length
158            );
159            debug!(
160                algorithm = "HKDF-SHA256",
161                output_len = length,
162                info_len = info.len(),
163                "Key derivation with custom info completed"
164            );
165        }
166        Err(e) => {
167            crate::log_crypto_operation_error!("key_derivation_info", e);
168        }
169    }
170
171    result
172}

174/// Internal implementation of key derivation.
175fn derive_key_internal(password: &[u8], salt: &[u8], length: usize) -> Result<Vec<u8>> {
176    crate::log_crypto_operation_start!(
177        "key_derivation",
178        algorithm = "HKDF-SHA256",
179        output_len = length
180    );
181
182    validate_key_derivation_count(1).map_err(|e| {
183        crate::log_crypto_operation_error!("key_derivation", e);
184        CoreError::ResourceExceeded(e.to_string())
185    })?;
186
187    if salt.is_empty() {
188        let err = CoreError::InvalidInput("Salt cannot be empty".to_string());
189        crate::log_crypto_operation_error!("key_derivation", err);
190        return Err(err);
191    }
192
193    if length == 0 {
194        let err = CoreError::InvalidInput("Length cannot be zero".to_string());
195        crate::log_crypto_operation_error!("key_derivation", err);
196        return Err(err);
197    }
198
199    let result = derive_key_hkdf(password, salt, length);
200
201    match &result {
202        Ok(_) => {
203            crate::log_crypto_operation_complete!(
204                "key_derivation",
205                algorithm = "HKDF-SHA256",
206                output_len = length
207            );
208            debug!(algorithm = "HKDF-SHA256", output_len = length, "Key derivation completed");
209        }
210        Err(e) => {
211            crate::log_crypto_operation_error!("key_derivation", e);
212        }
213    }
214
215    result
216}

218/// Internal implementation of HMAC.
219fn hmac_internal(key: &[u8], data: &[u8]) -> Result<Vec<u8>> {
220    crate::log_crypto_operation_start!("hmac", algorithm = "HMAC-SHA256", data_len = data.len());
221
222    if key.is_empty() {
223        let err = CoreError::InvalidInput("HMAC key must not be empty".to_string());
224        crate::log_crypto_operation_error!("hmac", err);
225        return Err(err);
226    }
227
228    let mut mac = <Hmac<Sha256> as hmac::digest::KeyInit>::new_from_slice(key).map_err(|_e| {
229        let err = CoreError::InvalidInput("Invalid HMAC key".to_string());
230        crate::log_crypto_operation_error!("hmac", err);
231        err
232    })?;
233
234    mac.update(data);
235
236    let result = mac.finalize().into_bytes().to_vec();
237    crate::log_crypto_operation_complete!(
238        "hmac",
239        algorithm = "HMAC-SHA256",
240        tag_len = result.len()
241    );
242    debug!(algorithm = "HMAC-SHA256", data_len = data.len(), "HMAC computed");
243
244    Ok(result)
245}

247/// Internal implementation of HMAC verification.
248fn hmac_verify_internal(key: &[u8], data: &[u8], tag: &[u8]) -> Result<bool> {
249    crate::log_crypto_operation_start!(
250        "hmac_verify",
251        algorithm = "HMAC-SHA256",
252        data_len = data.len()
253    );
254
255    if key.is_empty() {
256        let err = CoreError::InvalidInput("HMAC key must not be empty".to_string());
257        crate::log_crypto_operation_error!("hmac_verify", err);
258        return Err(err);
259    }
260
261    if tag.len() != 32 {
262        let err = CoreError::InvalidInput(format!("HMAC tag must be 32 bytes, got {}", tag.len()));
263        crate::log_crypto_operation_error!("hmac_verify", err);
264        return Err(err);
265    }
266
267    let expected = hmac_internal(key, data)?;
268
269    let mut tag_bytes = [0u8; 32];
270    tag_bytes.copy_from_slice(tag);
271    let mut expected_bytes = [0u8; 32];
272    expected_bytes.copy_from_slice(&expected);
273
274    let valid: bool = tag_bytes.ct_eq(&expected_bytes).into();
275
276    crate::log_crypto_operation_complete!("hmac_verify", algorithm = "HMAC-SHA256", valid = valid);
277    debug!(algorithm = "HMAC-SHA256", valid = valid, "HMAC verification completed");
278
279    Ok(valid)
280}

303/// Derive a key from a password and salt using HKDF.
304///
305/// Uses `SecurityMode` to specify verification behavior:
306/// - `SecurityMode::Verified(&session)`: Validates session before derivation
307/// - `SecurityMode::Unverified`: Skips session validation
308///
309/// # Example
310///
311/// ```no_run
312/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
313/// use arc_core::{derive_key, SecurityMode, VerifiedSession};
314/// # let password = b"password";
315/// # let salt = b"salt";
316/// # let pk = [0u8; 32];
317/// # let sk = [0u8; 32];
318///
319/// // With Zero Trust (recommended)
320/// let session = VerifiedSession::establish(&pk, &sk)?;
321/// let key = derive_key(password, salt, 32, SecurityMode::Verified(&session))?;
322///
323/// // Without verification (opt-out)
324/// let key = derive_key(password, salt, 32, SecurityMode::Unverified)?;
325/// # Ok(())
326/// # }
327/// ```
328///
329/// # Errors
330///
331/// Returns an error if:
332/// - The mode is `Verified` and the session has expired (`CoreError::SessionExpired`)
333/// - The resource limit for key derivation operations is exceeded
334/// - The salt is empty
335/// - The requested length is zero
336/// - The HKDF expansion operation fails
337pub fn derive_key(
338    password: &[u8],
339    salt: &[u8],
340    length: usize,
341    mode: SecurityMode,
342) -> Result<Vec<u8>> {
343    mode.validate()?;
344    derive_key_internal(password, salt, length)
345}

347/// Compute HMAC-SHA256 of data.
348///
349/// Uses `SecurityMode` to specify verification behavior:
350/// - `SecurityMode::Verified(&session)`: Validates session before HMAC computation
351/// - `SecurityMode::Unverified`: Skips session validation
352///
353/// # Example
354///
355/// ```no_run
356/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
357/// use arc_core::{hmac, SecurityMode, VerifiedSession};
358/// # let data = b"data";
359/// # let key = b"key";
360/// # let pk = [0u8; 32];
361/// # let sk = [0u8; 32];
362///
363/// // With Zero Trust (recommended)
364/// let session = VerifiedSession::establish(&pk, &sk)?;
365/// let tag = hmac(data, key, SecurityMode::Verified(&session))?;
366///
367/// // Without verification (opt-out)
368/// let tag = hmac(data, key, SecurityMode::Unverified)?;
369/// # Ok(())
370/// # }
371/// ```
372///
373/// # Errors
374///
375/// Returns an error if:
376/// - The mode is `Verified` and the session has expired (`CoreError::SessionExpired`)
377/// - The HMAC key is empty
378#[inline]
379pub fn hmac(data: &[u8], key: &[u8], mode: SecurityMode) -> Result<Vec<u8>> {
380    mode.validate()?;
381    hmac_internal(key, data)
382}

384/// Check HMAC-SHA256 tag in constant time.
385///
386/// Uses `SecurityMode` to specify verification behavior:
387/// - `SecurityMode::Verified(&session)`: Validates session before HMAC verification
388/// - `SecurityMode::Unverified`: Skips session validation
389///
390/// The function name uses "check" rather than "verify" to avoid confusion with
391/// the Zero Trust `_unverified` suffix pattern.
392///
393/// # Example
394///
395/// ```no_run
396/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
397/// use arc_core::{hmac_check, SecurityMode, VerifiedSession};
398/// # let data = b"data";
399/// # let key = b"key";
400/// # let tag = &[0u8; 32];
401/// # let pk = [0u8; 32];
402/// # let sk = [0u8; 32];
403///
404/// // With Zero Trust (recommended)
405/// let session = VerifiedSession::establish(&pk, &sk)?;
406/// let is_valid = hmac_check(data, key, tag, SecurityMode::Verified(&session))?;
407///
408/// // Without verification (opt-out)
409/// let is_valid = hmac_check(data, key, tag, SecurityMode::Unverified)?;
410/// # Ok(())
411/// # }
412/// ```
413///
414/// # Errors
415///
416/// Returns an error if:
417/// - The mode is `Verified` and the session has expired (`CoreError::SessionExpired`)
418/// - The HMAC key is empty
419/// - The tag is not exactly 32 bytes
420#[inline]
421pub fn hmac_check(data: &[u8], key: &[u8], tag: &[u8], mode: SecurityMode) -> Result<bool> {
422    mode.validate()?;
423    hmac_verify_internal(key, data, tag)
424}
