1//! Hybrid signature convenience API (ML-DSA-65 + Ed25519)
2//!
3//! This module wraps [`arc_hybrid::sig_hybrid`] to provide a high-level hybrid
4//! signature API with `SecurityMode` support. The underlying implementation uses
5//! AND-composition: both ML-DSA and Ed25519 must verify for the signature to be valid.
6//!
7//! ## Usage
8//!
9//! ```rust,ignore
10//! use arc_core::{
11//!     generate_hybrid_signing_keypair, sign_hybrid, verify_hybrid_signature,
12//!     SecurityMode,
13//! };
14//!
15//! let (pk, sk) = generate_hybrid_signing_keypair(SecurityMode::Unverified)?;
16//! let signature = sign_hybrid(b"message", &sk, SecurityMode::Unverified)?;
17//! let valid = verify_hybrid_signature(b"message", &signature, &pk, SecurityMode::Unverified)?;
18//! assert!(valid);
19//! ```
20

30/// Generate a hybrid signing keypair (ML-DSA-65 + Ed25519).
31///
32/// Returns a public key (for verification) and a secret key (for signing).
33/// The keypair combines post-quantum and classical algorithms so that
34/// security holds if *either* algorithm remains secure.
35///
36/// # Errors
37///
38/// Returns an error if:
39/// - The session has expired (when `mode` is `Verified`)
40/// - Key generation fails
41pub fn generate_hybrid_signing_keypair(
42    mode: SecurityMode,
43) -> Result<(HybridPublicKey, HybridSecretKey)> {
44    mode.validate()?;
45
46    let mut rng = rand::rngs::OsRng;
47    sig_hybrid::generate_keypair(&mut rng).map_err(|e| {
48        CoreError::SignatureFailed(format!("Hybrid signing keypair generation failed: {}", e))
49    })
50}
51
52/// Sign a message using hybrid signatures (ML-DSA-65 + Ed25519).
53///
54/// Both ML-DSA and Ed25519 signatures are generated. An attacker would need
55/// to break *both* algorithms to forge a signature.
56///
57/// # Errors
58///
59/// Returns an error if:
60/// - The session has expired (when `mode` is `Verified`)
61/// - Message size exceeds resource limits
62/// - Signing fails (invalid key material or crypto error)
63pub fn sign_hybrid(
64    message: &[u8],
65    sk: &HybridSecretKey,
66    mode: SecurityMode,
67) -> Result<HybridSignature> {
68    mode.validate()?;
69
70    validate_signature_size(message.len())
71        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
72
73    sig_hybrid::sign(sk, message).map_err(hybrid_sig_error_to_core)
74}
75
76/// Verify a hybrid signature (ML-DSA-65 + Ed25519).
77///
78/// Both component signatures must verify for the result to be `Ok(true)`.
79///
80/// # Errors
81///
82/// Returns an error if:
83/// - The session has expired (when `mode` is `Verified`)
84/// - Message size exceeds resource limits
85/// - Key or signature material is malformed
86/// - Either component verification fails
87pub fn verify_hybrid_signature(
88    message: &[u8],
89    signature: &HybridSignature,
90    pk: &HybridPublicKey,
91    mode: SecurityMode,
92) -> Result<bool> {
93    mode.validate()?;
94
95    validate_signature_size(message.len())
96        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
97
98    sig_hybrid::verify(pk, message, signature).map_err(hybrid_sig_error_to_core)
99}

186/// Convert `HybridSignatureError` to `CoreError`.
187fn hybrid_sig_error_to_core(e: HybridSignatureError) -> CoreError {
188    match e {
189        HybridSignatureError::MlDsaError(msg) => {
190            CoreError::SignatureFailed(format!("Hybrid ML-DSA error: {}", msg))
191        }
192        HybridSignatureError::Ed25519Error(msg) => {
193            CoreError::SignatureFailed(format!("Hybrid Ed25519 error: {}", msg))
194        }
195        HybridSignatureError::VerificationFailed(_msg) => CoreError::VerificationFailed,
196        HybridSignatureError::InvalidKeyMaterial(msg) => {
197            CoreError::InvalidKey(format!("Hybrid key material error: {}", msg))
198        }
199        HybridSignatureError::CryptoError(msg) => {
200            CoreError::SignatureFailed(format!("Hybrid crypto error: {}", msg))
201        }
202    }
203}
