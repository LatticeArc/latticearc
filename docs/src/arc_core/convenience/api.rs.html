1//! High-level encryption, decryption, signing, and verification API.
2//!
3//! Provides a unified API for cryptographic operations with automatic algorithm
4//! selection based on use case or security level.
5//!
6//! ## Unified API
7//!
8//! All operations use `CryptoConfig` for configuration:
9//!
10//! ```rust,ignore
11//! use latticearc::{encrypt, decrypt, CryptoConfig, UseCase, VerifiedSession};
12//!
13//! // Simple: Use defaults (High security)
14//! let encrypted = encrypt(data, &key, CryptoConfig::new())?;
15//!
16//! // With use case (recommended - library picks optimal algorithm)
17//! let encrypted = encrypt(data, &key, CryptoConfig::new()
18//!     .use_case(UseCase::FileStorage))?;
19//!
20//! // With Zero Trust session
21//! let session = VerifiedSession::establish(&pk, &sk)?;
22//! let encrypted = encrypt(data, &key, CryptoConfig::new()
23//!     .session(&session)
24//!     .use_case(UseCase::FileStorage))?;
25//! ```

67/// Select encryption scheme based on CryptoConfig.
68fn select_encryption_scheme(data: &[u8], options: &CryptoConfig) -> Result<String> {
69    match options.get_selection() {
70        AlgorithmSelection::UseCase(use_case) => {
71            let config = CoreConfig::default();
72            CryptoPolicyEngine::select_encryption_scheme(data, &config, Some(use_case))
73        }
74        AlgorithmSelection::SecurityLevel(level) => {
75            let config = CoreConfig::default().with_security_level(level.clone());
76            CryptoPolicyEngine::select_encryption_scheme(data, &config, None)
77        }
78    }
79}
80
81/// Select signature scheme based on CryptoConfig.
82fn select_signature_scheme(options: &CryptoConfig) -> Result<String> {
83    match options.get_selection() {
84        AlgorithmSelection::UseCase(use_case) => {
85            // For use cases, recommend based on the use case
86            CryptoPolicyEngine::recommend_scheme(use_case, &CoreConfig::default())
87        }
88        AlgorithmSelection::SecurityLevel(level) => {
89            let config = CoreConfig::default().with_security_level(level.clone());
90            CryptoPolicyEngine::select_signature_scheme(&config)
91        }
92    }
93}

99/// Encrypt data with automatic algorithm selection.
100///
101/// This is the single entry point for encryption. Configure algorithm selection
102/// and optional Zero Trust session via the `CryptoConfig` builder.
103///
104/// # Examples
105///
106/// ```rust,ignore
107/// use latticearc::{encrypt, CryptoConfig, UseCase, SecurityLevel, VerifiedSession};
108///
109/// let key = [0u8; 32];
110///
111/// // Simple: Use defaults (High security)
112/// let encrypted = encrypt(data, &key, CryptoConfig::new())?;
113///
114/// // With use case (recommended - library picks optimal algorithm)
115/// let encrypted = encrypt(data, &key, CryptoConfig::new()
116///     .use_case(UseCase::FileStorage))?;
117///
118/// // With security level (manual control)
119/// let encrypted = encrypt(data, &key, CryptoConfig::new()
120///     .security_level(SecurityLevel::Maximum))?;
121///
122/// // With Zero Trust session
123/// let session = VerifiedSession::establish(&pk, &sk)?;
124/// let encrypted = encrypt(data, &key, CryptoConfig::new()
125///     .session(&session)
126///     .use_case(UseCase::FileStorage))?;
127/// ```
128///
129/// # Algorithm Selection
130///
131/// | Use Case | Algorithm |
132/// |----------|-----------|
133/// | `FileStorage` | ML-KEM-1024 + AES-256-GCM |
134/// | `SecureMessaging` | ML-KEM-768 + AES-256-GCM |
135/// | `IoTDevice` | ML-KEM-512 + AES-256-GCM |
136///
137/// | Security Level | Algorithm |
138/// |----------------|-----------|
139/// | `Maximum` | ML-KEM-1024 |
140/// | `High` | ML-KEM-768 |
141/// | `Medium` | ML-KEM-768 |
142/// | `Low` | ML-KEM-512 |
143///
144/// # Errors
145///
146/// Returns an error if:
147/// - Session is set and has expired (`CoreError::SessionExpired`)
148/// - Data size exceeds resource limits
149/// - Key length is invalid for the selected scheme
150/// - Encryption operation fails
151#[allow(deprecated)]
152pub fn encrypt(data: &[u8], key: &[u8], config: CryptoConfig) -> Result<EncryptedData> {
153    config.validate()?;
154
155    let scheme = select_encryption_scheme(data, &config)?;
156
157    validate_encryption_size(data.len()).map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
158
159    crate::log_crypto_operation_start!("encrypt", scheme = %scheme, data_size = data.len());
160
161    // When the selector picks a hybrid/PQ scheme but the caller provides a symmetric
162    // key (≤64 bytes), fall back to AES-256-GCM. The unified API's `&[u8]` key interface
163    // cannot support asymmetric operations (HybridPublicKey/HybridSecretKey are not
164    // serializable to raw bytes). For true PQ+classical hybrid encryption, use
165    // `encrypt_hybrid()` / `decrypt_hybrid()` with typed keys.
166    let (effective_scheme, encrypted) = match scheme.as_str() {
167        "ml-kem-512"
168        | "ml-kem-768"
169        | "ml-kem-1024"
170        | "hybrid-ml-kem-512-aes-256-gcm"
171        | "hybrid-ml-kem-768-aes-256-gcm"
172        | "hybrid-ml-kem-1024-aes-256-gcm" => {
173            if key.len() < 32 {
174                return Err(CoreError::InvalidKeyLength { expected: 32, actual: key.len() });
175            }
176            warn!(
177                "Scheme '{}' selected but unified API received symmetric key. \
178                 Falling back to AES-256-GCM. For true hybrid (PQ+classical) encryption, \
179                 use encrypt_hybrid() with HybridPublicKey.",
180                scheme
181            );
182            ("aes-256-gcm".to_string(), encrypt_aes_gcm_internal(data, key)?)
183        }
184        _ => {
185            if key.len() < 32 {
186                return Err(CoreError::InvalidKeyLength { expected: 32, actual: key.len() });
187            }
188            let ct =
189                if data.is_empty() { data.to_vec() } else { encrypt_aes_gcm_internal(data, key)? };
190            (scheme, ct)
191        }
192    };
193    let scheme = effective_scheme;
194
195    let nonce = encrypted.get(..12).map_or_else(Vec::new, <[u8]>::to_vec);
196    let tag = encrypted
197        .len()
198        .checked_sub(16)
199        .and_then(|start| encrypted.get(start..))
200        .filter(|_| encrypted.len() >= 28)
201        .map_or_else(Vec::new, <[u8]>::to_vec);
202
203    let timestamp = u64::try_from(Utc::now().timestamp()).unwrap_or(0);
204
205    crate::log_crypto_operation_complete!("encrypt", result_size = encrypted.len(), scheme = %scheme);
206
207    Ok(EncryptedData {
208        data: encrypted,
209        metadata: EncryptedMetadata { nonce, tag: Some(tag), key_id: None },
210        scheme,
211        timestamp,
212    })
213}

215/// Decrypt data.
216///
217/// The decryption algorithm is determined by the `encrypted.scheme` field.
218///
219/// # Examples
220///
221/// ```rust,ignore
222/// use latticearc::{decrypt, CryptoConfig, VerifiedSession};
223///
224/// // Simple: No session
225/// let plaintext = decrypt(&encrypted, &key, CryptoConfig::new())?;
226///
227/// // With Zero Trust session
228/// let session = VerifiedSession::establish(&pk, &sk)?;
229/// let plaintext = decrypt(&encrypted, &key, CryptoConfig::new()
230///     .session(&session))?;
231/// ```
232///
233/// # Errors
234///
235/// Returns an error if:
236/// - Session is set and has expired (`CoreError::SessionExpired`)
237/// - Encrypted data size exceeds resource limits
238/// - Key is invalid for the encryption scheme
239/// - Decryption fails
240#[allow(deprecated)]
241pub fn decrypt(encrypted: &EncryptedData, key: &[u8], config: CryptoConfig) -> Result<Vec<u8>> {
242    config.validate()?;
243
244    crate::log_crypto_operation_start!("decrypt", scheme = %encrypted.scheme, data_size = encrypted.data.len());
245
246    if encrypted.data.is_empty() {
247        crate::log_crypto_operation_complete!("decrypt", result_size = 0_usize);
248        return Ok(encrypted.data.clone());
249    }
250
251    validate_decryption_size(encrypted.data.len())
252        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
253
254    let result = match encrypted.scheme.as_str() {
255        // All schemes through the unified API use AES-256-GCM with symmetric keys.
256        // The hybrid/PQ scheme names stored in metadata reflect the *selected* scheme,
257        // but the actual encryption was AES-256-GCM (see encrypt() fallback above).
258        // For true hybrid decryption, use decrypt_hybrid() with HybridSecretKey.
259        "aes-256-gcm"
260        | "chacha20-poly1305"
261        | "ml-kem-512"
262        | "ml-kem-768"
263        | "ml-kem-1024"
264        | "hybrid-ml-kem-512-aes-256-gcm"
265        | "hybrid-ml-kem-768-aes-256-gcm"
266        | "hybrid-ml-kem-1024-aes-256-gcm" => {
267            if key.len() < 32 {
268                return Err(CoreError::InvalidKeyLength { expected: 32, actual: key.len() });
269            }
270            decrypt_aes_gcm_internal(&encrypted.data, key)
271        }
272        _ => {
273            if key.len() < 32 {
274                return Err(CoreError::InvalidKeyLength { expected: 32, actual: key.len() });
275            }
276            decrypt_aes_gcm_internal(&encrypted.data, key)
277        }
278    };
279
280    match result {
281        Ok(plaintext) => {
282            crate::log_crypto_operation_complete!("decrypt", result_size = plaintext.len(), scheme = %encrypted.scheme);
283            Ok(plaintext)
284        }
285        Err(e) => {
286            crate::log_crypto_operation_error!("decrypt", e, scheme = %encrypted.scheme);
287            Err(e)
288        }
289    }
290}

292/// Generate a signing keypair for the scheme selected by config.
293///
294/// The scheme selector auto-picks hybrid (ML-DSA-65 + Ed25519) by default.
295/// Returns `(public_key_bytes, secret_key_bytes, scheme_name)`.
296///
297/// # Unified API
298///
299/// This is the keypair generation companion to [`sign_with_key`]. Together they
300/// form the unified signing API — the scheme selector handles PQ-only, hybrid,
301/// and classical transparently based on `CryptoConfig`.
302///
303/// # Errors
304///
305/// Returns an error if:
306/// - Session is set and has expired (`CoreError::SessionExpired`)
307/// - Key generation fails for the selected scheme
308#[allow(deprecated)]
309pub fn generate_signing_keypair(
310    config: CryptoConfig,
311) -> Result<(Vec<u8>, Zeroizing<Vec<u8>>, String)> {
312    config.validate()?;
313
314    let scheme = select_signature_scheme(&config)?;
315
316    let (public_key, secret_key) = match scheme.as_str() {
317        "ml-dsa-44" | "pq-ml-dsa-44" => {
318            let (pk, sk) = generate_ml_dsa_keypair(MlDsaParameterSet::MLDSA44)?;
319            (pk, sk)
320        }
321        "ml-dsa-65" | "pq-ml-dsa-65" => {
322            let (pk, sk) = generate_ml_dsa_keypair(MlDsaParameterSet::MLDSA65)?;
323            (pk, sk)
324        }
325        "ml-dsa-87" | "pq-ml-dsa-87" => {
326            let (pk, sk) = generate_ml_dsa_keypair(MlDsaParameterSet::MLDSA87)?;
327            (pk, sk)
328        }
329        "slh-dsa-shake-128s" => {
330            let (pk, sk) = generate_slh_dsa_keypair(SlhDsaSecurityLevel::Shake128s)?;
331            (pk, sk)
332        }
333        "slh-dsa-shake-192s" => {
334            let (pk, sk) = generate_slh_dsa_keypair(SlhDsaSecurityLevel::Shake192s)?;
335            (pk, sk)
336        }
337        "slh-dsa-shake-256s" => {
338            let (pk, sk) = generate_slh_dsa_keypair(SlhDsaSecurityLevel::Shake256s)?;
339            (pk, sk)
340        }
341        "fn-dsa" => {
342            let (pk, sk) = generate_fn_dsa_keypair()?;
343            (pk, sk)
344        }
345        "hybrid-ml-dsa-44-ed25519" => {
346            let (pq_pk, pq_sk) = generate_ml_dsa_keypair(MlDsaParameterSet::MLDSA44)?;
347            let (ed_pk, ed_sk) = generate_keypair()?;
348            let combined_pk = [pq_pk, ed_pk].concat();
349            let combined_sk = [pq_sk.as_ref(), ed_sk.as_ref()].concat();
350            (combined_pk, crate::types::PrivateKey::new(combined_sk))
351        }
352        "hybrid-ml-dsa-65-ed25519" | "ml-dsa-65-hybrid-ed25519" => {
353            let (pq_pk, pq_sk) = generate_ml_dsa_keypair(MlDsaParameterSet::MLDSA65)?;
354            let (ed_pk, ed_sk) = generate_keypair()?;
355            let combined_pk = [pq_pk, ed_pk].concat();
356            let combined_sk = [pq_sk.as_ref(), ed_sk.as_ref()].concat();
357            (combined_pk, crate::types::PrivateKey::new(combined_sk))
358        }
359        "hybrid-ml-dsa-87-ed25519" => {
360            let (pq_pk, pq_sk) = generate_ml_dsa_keypair(MlDsaParameterSet::MLDSA87)?;
361            let (ed_pk, ed_sk) = generate_keypair()?;
362            let combined_pk = [pq_pk, ed_pk].concat();
363            let combined_sk = [pq_sk.as_ref(), ed_sk.as_ref()].concat();
364            (combined_pk, crate::types::PrivateKey::new(combined_sk))
365        }
366        _ => {
367            // Fallback: Ed25519
368            let (pk, sk) = generate_keypair()?;
369            (pk, sk)
370        }
371    };
372
373    Ok((public_key, Zeroizing::new(secret_key.as_ref().to_vec()), scheme))
374}

376/// Sign a message using caller-provided keys (unified API).
377///
378/// Use [`generate_signing_keypair`] to create matching keys. The scheme is
379/// determined by `CryptoConfig` — hybrid by default.
380///
381/// # Errors
382///
383/// Returns an error if:
384/// - Session is set and has expired (`CoreError::SessionExpired`)
385/// - Message size exceeds resource limits
386/// - Secret/public key bytes don't match the expected sizes for the scheme
387/// - Signing operation fails
388#[allow(deprecated)]
389#[allow(clippy::arithmetic_side_effects)] // Key size additions use well-defined NIST constants
390pub fn sign_with_key(
391    message: &[u8],
392    secret_key: &[u8],
393    public_key: &[u8],
394    config: CryptoConfig,
395) -> Result<SignedData> {
396    config.validate()?;
397
398    let scheme = select_signature_scheme(&config)?;
399
400    validate_signature_size(message.len())
401        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
402
403    crate::log_crypto_operation_start!("sign_with_key", scheme = %scheme, message_size = message.len());
404
405    let (result_pk, signature) = match scheme.as_str() {
406        "ml-dsa-44" | "pq-ml-dsa-44" => {
407            let sig = sign_pq_ml_dsa_unverified(message, secret_key, MlDsaParameterSet::MLDSA44)?;
408            (public_key.to_vec(), sig)
409        }
410        "ml-dsa-65" | "pq-ml-dsa-65" => {
411            let sig = sign_pq_ml_dsa_unverified(message, secret_key, MlDsaParameterSet::MLDSA65)?;
412            (public_key.to_vec(), sig)
413        }
414        "ml-dsa-87" | "pq-ml-dsa-87" => {
415            let sig = sign_pq_ml_dsa_unverified(message, secret_key, MlDsaParameterSet::MLDSA87)?;
416            (public_key.to_vec(), sig)
417        }
418        "slh-dsa-shake-128s" => {
419            let sig =
420                sign_pq_slh_dsa_unverified(message, secret_key, SlhDsaSecurityLevel::Shake128s)?;
421            (public_key.to_vec(), sig)
422        }
423        "slh-dsa-shake-192s" => {
424            let sig =
425                sign_pq_slh_dsa_unverified(message, secret_key, SlhDsaSecurityLevel::Shake192s)?;
426            (public_key.to_vec(), sig)
427        }
428        "slh-dsa-shake-256s" => {
429            let sig =
430                sign_pq_slh_dsa_unverified(message, secret_key, SlhDsaSecurityLevel::Shake256s)?;
431            (public_key.to_vec(), sig)
432        }
433        "fn-dsa" => {
434            let sig = sign_pq_fn_dsa_unverified(message, secret_key)?;
435            (public_key.to_vec(), sig)
436        }
437        "hybrid-ml-dsa-44-ed25519" => {
438            let pq_sk_len = MlDsaParameterSet::MLDSA44.secret_key_size();
439            let pq_pk_len = MlDsaParameterSet::MLDSA44.public_key_size();
440            const ED25519_SK_LEN: usize = 32;
441
442            if secret_key.len() != pq_sk_len + ED25519_SK_LEN {
443                return Err(CoreError::InvalidKey(format!(
444                    "Hybrid secret key length mismatch: expected {}, got {}",
445                    pq_sk_len + ED25519_SK_LEN,
446                    secret_key.len()
447                )));
448            }
449            if public_key.len() != pq_pk_len + 32 {
450                return Err(CoreError::InvalidKey(format!(
451                    "Hybrid public key length mismatch: expected {}, got {}",
452                    pq_pk_len + 32,
453                    public_key.len()
454                )));
455            }
456
457            let pq_sk = secret_key.get(..pq_sk_len).ok_or_else(|| {
458                CoreError::InvalidKey("Failed to split hybrid secret key".to_string())
459            })?;
460            let ed_sk = secret_key.get(pq_sk_len..).ok_or_else(|| {
461                CoreError::InvalidKey("Failed to split hybrid secret key".to_string())
462            })?;
463
464            let pq_sig = sign_pq_ml_dsa_unverified(message, pq_sk, MlDsaParameterSet::MLDSA44)?;
465            let ed_sig = sign_ed25519_internal(message, ed_sk)?;
466            let combined_sig = [pq_sig, ed_sig].concat();
467            (public_key.to_vec(), combined_sig)
468        }
469        "hybrid-ml-dsa-65-ed25519" | "ml-dsa-65-hybrid-ed25519" => {
470            let pq_sk_len = MlDsaParameterSet::MLDSA65.secret_key_size();
471            let pq_pk_len = MlDsaParameterSet::MLDSA65.public_key_size();
472            const ED25519_SK_LEN: usize = 32;
473
474            if secret_key.len() != pq_sk_len + ED25519_SK_LEN {
475                return Err(CoreError::InvalidKey(format!(
476                    "Hybrid secret key length mismatch: expected {}, got {}",
477                    pq_sk_len + ED25519_SK_LEN,
478                    secret_key.len()
479                )));
480            }
481            if public_key.len() != pq_pk_len + 32 {
482                return Err(CoreError::InvalidKey(format!(
483                    "Hybrid public key length mismatch: expected {}, got {}",
484                    pq_pk_len + 32,
485                    public_key.len()
486                )));
487            }
488
489            let pq_sk = secret_key.get(..pq_sk_len).ok_or_else(|| {
490                CoreError::InvalidKey("Failed to split hybrid secret key".to_string())
491            })?;
492            let ed_sk = secret_key.get(pq_sk_len..).ok_or_else(|| {
493                CoreError::InvalidKey("Failed to split hybrid secret key".to_string())
494            })?;
495
496            let pq_sig = sign_pq_ml_dsa_unverified(message, pq_sk, MlDsaParameterSet::MLDSA65)?;
497            let ed_sig = sign_ed25519_internal(message, ed_sk)?;
498            let combined_sig = [pq_sig, ed_sig].concat();
499            (public_key.to_vec(), combined_sig)
500        }
501        "hybrid-ml-dsa-87-ed25519" => {
502            let pq_sk_len = MlDsaParameterSet::MLDSA87.secret_key_size();
503            let pq_pk_len = MlDsaParameterSet::MLDSA87.public_key_size();
504            const ED25519_SK_LEN: usize = 32;
505
506            if secret_key.len() != pq_sk_len + ED25519_SK_LEN {
507                return Err(CoreError::InvalidKey(format!(
508                    "Hybrid secret key length mismatch: expected {}, got {}",
509                    pq_sk_len + ED25519_SK_LEN,
510                    secret_key.len()
511                )));
512            }
513            if public_key.len() != pq_pk_len + 32 {
514                return Err(CoreError::InvalidKey(format!(
515                    "Hybrid public key length mismatch: expected {}, got {}",
516                    pq_pk_len + 32,
517                    public_key.len()
518                )));
519            }
520
521            let pq_sk = secret_key.get(..pq_sk_len).ok_or_else(|| {
522                CoreError::InvalidKey("Failed to split hybrid secret key".to_string())
523            })?;
524            let ed_sk = secret_key.get(pq_sk_len..).ok_or_else(|| {
525                CoreError::InvalidKey("Failed to split hybrid secret key".to_string())
526            })?;
527
528            let pq_sig = sign_pq_ml_dsa_unverified(message, pq_sk, MlDsaParameterSet::MLDSA87)?;
529            let ed_sig = sign_ed25519_internal(message, ed_sk)?;
530            let combined_sig = [pq_sig, ed_sig].concat();
531            (public_key.to_vec(), combined_sig)
532        }
533        _ => {
534            // Fallback: Ed25519
535            let sig = sign_ed25519_internal(message, secret_key)?;
536            (public_key.to_vec(), sig)
537        }
538    };
539
540    let timestamp = u64::try_from(Utc::now().timestamp()).unwrap_or(0);
541
542    crate::log_crypto_operation_complete!("sign_with_key", signature_size = signature.len(), scheme = %scheme);
543
544    Ok(SignedData {
545        data: message.to_vec(),
546        metadata: SignedMetadata {
547            signature,
548            signature_algorithm: scheme.clone(),
549            public_key: result_pk,
550            key_id: None,
551        },
552        scheme,
553        timestamp,
554    })
555}

557/// Verify a signed message.
558///
559/// The verification algorithm is determined by the `signed.scheme` field.
560///
561/// # Examples
562///
563/// ```rust,ignore
564/// use latticearc::{verify, CryptoConfig, VerifiedSession};
565///
566/// // Simple: No session
567/// let is_valid = verify(&signed, CryptoConfig::new())?;
568///
569/// // With Zero Trust session
570/// let session = VerifiedSession::establish(&pk, &sk)?;
571/// let is_valid = verify(&signed, CryptoConfig::new()
572///     .session(&session))?;
573/// ```
574///
575/// # Errors
576///
577/// Returns an error if:
578/// - Session is set and has expired (`CoreError::SessionExpired`)
579/// - Message size exceeds resource limits
580/// - Public key is invalid
581/// - Signature is malformed or invalid
582#[allow(deprecated)]
583pub fn verify(signed: &SignedData, config: CryptoConfig) -> Result<bool> {
584    config.validate()?;
585
586    crate::log_crypto_operation_start!("verify", scheme = %signed.scheme, message_size = signed.data.len());
587
588    validate_signature_size(signed.data.len())
589        .map_err(|e| CoreError::ResourceExceeded(e.to_string()))?;
590
591    let result = match signed.scheme.as_str() {
592        "ml-dsa-44" | "pq-ml-dsa-44" => verify_pq_ml_dsa_unverified(
593            &signed.data,
594            &signed.metadata.signature,
595            &signed.metadata.public_key,
596            MlDsaParameterSet::MLDSA44,
597        ),
598        "ml-dsa-65" | "pq-ml-dsa-65" => verify_pq_ml_dsa_unverified(
599            &signed.data,
600            &signed.metadata.signature,
601            &signed.metadata.public_key,
602            MlDsaParameterSet::MLDSA65,
603        ),
604        "ml-dsa-87" | "pq-ml-dsa-87" => verify_pq_ml_dsa_unverified(
605            &signed.data,
606            &signed.metadata.signature,
607            &signed.metadata.public_key,
608            MlDsaParameterSet::MLDSA87,
609        ),
610        "slh-dsa-shake-128s" => verify_pq_slh_dsa_unverified(
611            &signed.data,
612            &signed.metadata.signature,
613            &signed.metadata.public_key,
614            SlhDsaSecurityLevel::Shake128s,
615        ),
616        "slh-dsa-shake-192s" => verify_pq_slh_dsa_unverified(
617            &signed.data,
618            &signed.metadata.signature,
619            &signed.metadata.public_key,
620            SlhDsaSecurityLevel::Shake192s,
621        ),
622        "slh-dsa-shake-256s" => verify_pq_slh_dsa_unverified(
623            &signed.data,
624            &signed.metadata.signature,
625            &signed.metadata.public_key,
626            SlhDsaSecurityLevel::Shake256s,
627        ),
628        "fn-dsa" => verify_pq_fn_dsa_unverified(
629            &signed.data,
630            &signed.metadata.signature,
631            &signed.metadata.public_key,
632        ),
633        "hybrid-ml-dsa-44-ed25519" => {
634            const ML_DSA_44_PK_LEN: usize = 1312;
635            const ED25519_PK_LEN: usize = 32;
636
637            let sig_len = signed.metadata.signature.len();
638            if sig_len < 2420 {
639                return Err(CoreError::InvalidInput("Hybrid signature too short".to_string()));
640            }
641
642            let pk_len = signed.metadata.public_key.len();
643            if pk_len != ML_DSA_44_PK_LEN + ED25519_PK_LEN {
644                return Err(CoreError::InvalidInput(format!(
645                    "Invalid hybrid public key length: expected {}, got {}",
646                    ML_DSA_44_PK_LEN + ED25519_PK_LEN,
647                    pk_len
648                )));
649            }
650            let pq_pk = signed.metadata.public_key.get(..ML_DSA_44_PK_LEN).ok_or_else(|| {
651                CoreError::InvalidInput("Invalid hybrid public key format".to_string())
652            })?;
653            let ed_pk = signed.metadata.public_key.get(ML_DSA_44_PK_LEN..).ok_or_else(|| {
654                CoreError::InvalidInput("Invalid hybrid public key format".to_string())
655            })?;
656
657            let pq_sig_len = sig_len.checked_sub(64).ok_or_else(|| {
658                CoreError::InvalidInput(
659                    "Hybrid signature too short for Ed25519 component".to_string(),
660                )
661            })?;
662            let pq_sig = signed.metadata.signature.get(..pq_sig_len).ok_or_else(|| {
663                CoreError::InvalidInput("Invalid hybrid signature format".to_string())
664            })?;
665            let ed_sig = signed.metadata.signature.get(pq_sig_len..).ok_or_else(|| {
666                CoreError::InvalidInput("Invalid hybrid signature format".to_string())
667            })?;
668
669            let pq_valid = verify_pq_ml_dsa_unverified(
670                &signed.data,
671                pq_sig,
672                pq_pk,
673                MlDsaParameterSet::MLDSA44,
674            )?;
675            let ed_valid = verify_ed25519_internal(&signed.data, ed_sig, ed_pk)?;
676            Ok(pq_valid && ed_valid)
677        }
678        "hybrid-ml-dsa-65-ed25519" | "ml-dsa-65-hybrid-ed25519" => {
679            // ML-DSA-65 public key is 1952 bytes, Ed25519 is 32 bytes
680            const ML_DSA_65_PK_LEN: usize = 1952;
681            const ED25519_PK_LEN: usize = 32;
682
683            let sig_len = signed.metadata.signature.len();
684            if sig_len < 3293 {
685                return Err(CoreError::InvalidInput("Hybrid signature too short".to_string()));
686            }
687
688            // Split combined public key: ML-DSA (1952) + Ed25519 (32)
689            let pk_len = signed.metadata.public_key.len();
690            if pk_len != ML_DSA_65_PK_LEN + ED25519_PK_LEN {
691                return Err(CoreError::InvalidInput(format!(
692                    "Invalid hybrid public key length: expected {}, got {}",
693                    ML_DSA_65_PK_LEN + ED25519_PK_LEN,
694                    pk_len
695                )));
696            }
697            let pq_pk = signed.metadata.public_key.get(..ML_DSA_65_PK_LEN).ok_or_else(|| {
698                CoreError::InvalidInput("Invalid hybrid public key format".to_string())
699            })?;
700            let ed_pk = signed.metadata.public_key.get(ML_DSA_65_PK_LEN..).ok_or_else(|| {
701                CoreError::InvalidInput("Invalid hybrid public key format".to_string())
702            })?;
703
704            // Split combined signature: ML-DSA sig + Ed25519 sig (64 bytes)
705            let pq_sig_len = sig_len.checked_sub(64).ok_or_else(|| {
706                CoreError::InvalidInput(
707                    "Hybrid signature too short for Ed25519 component".to_string(),
708                )
709            })?;
710            let pq_sig = signed.metadata.signature.get(..pq_sig_len).ok_or_else(|| {
711                CoreError::InvalidInput("Invalid hybrid signature format".to_string())
712            })?;
713            let ed_sig = signed.metadata.signature.get(pq_sig_len..).ok_or_else(|| {
714                CoreError::InvalidInput("Invalid hybrid signature format".to_string())
715            })?;
716
717            let pq_valid = verify_pq_ml_dsa_unverified(
718                &signed.data,
719                pq_sig,
720                pq_pk,
721                MlDsaParameterSet::MLDSA65,
722            )?;
723            let ed_valid = verify_ed25519_internal(&signed.data, ed_sig, ed_pk)?;
724            Ok(pq_valid && ed_valid)
725        }
726        "hybrid-ml-dsa-87-ed25519" => {
727            // ML-DSA-87 public key is 2592 bytes, Ed25519 is 32 bytes
728            const ML_DSA_87_PK_LEN: usize = 2592;
729            const ED25519_PK_LEN: usize = 32;
730
731            let sig_len = signed.metadata.signature.len();
732            if sig_len < 4627 {
733                // ML-DSA-87 sig (4595) + Ed25519 sig (64) - some overlap
734                return Err(CoreError::InvalidInput("Hybrid signature too short".to_string()));
735            }
736
737            // Split combined public key: ML-DSA (2592) + Ed25519 (32)
738            let pk_len = signed.metadata.public_key.len();
739            if pk_len != ML_DSA_87_PK_LEN + ED25519_PK_LEN {
740                return Err(CoreError::InvalidInput(format!(
741                    "Invalid hybrid public key length: expected {}, got {}",
742                    ML_DSA_87_PK_LEN + ED25519_PK_LEN,
743                    pk_len
744                )));
745            }
746            let pq_pk = signed.metadata.public_key.get(..ML_DSA_87_PK_LEN).ok_or_else(|| {
747                CoreError::InvalidInput("Invalid hybrid public key format".to_string())
748            })?;
749            let ed_pk = signed.metadata.public_key.get(ML_DSA_87_PK_LEN..).ok_or_else(|| {
750                CoreError::InvalidInput("Invalid hybrid public key format".to_string())
751            })?;
752
753            // Split combined signature: ML-DSA sig + Ed25519 sig (64 bytes)
754            let pq_sig_len = sig_len.checked_sub(64).ok_or_else(|| {
755                CoreError::InvalidInput(
756                    "Hybrid signature too short for Ed25519 component".to_string(),
757                )
758            })?;
759            let pq_sig = signed.metadata.signature.get(..pq_sig_len).ok_or_else(|| {
760                CoreError::InvalidInput("Invalid hybrid signature format".to_string())
761            })?;
762            let ed_sig = signed.metadata.signature.get(pq_sig_len..).ok_or_else(|| {
763                CoreError::InvalidInput("Invalid hybrid signature format".to_string())
764            })?;
765
766            let pq_valid = verify_pq_ml_dsa_unverified(
767                &signed.data,
768                pq_sig,
769                pq_pk,
770                MlDsaParameterSet::MLDSA87,
771            )?;
772            let ed_valid = verify_ed25519_internal(&signed.data, ed_sig, ed_pk)?;
773            Ok(pq_valid && ed_valid)
774        }
775        _ => verify_ed25519_internal(
776            &signed.data,
777            &signed.metadata.signature,
778            &signed.metadata.public_key,
779        ),
780    };
781
782    match &result {
783        Ok(valid) => {
784            crate::log_crypto_operation_complete!("verify", valid = *valid, scheme = %signed.scheme);
785        }
786        Err(e) => {
787            crate::log_crypto_operation_error!("verify", e, scheme = %signed.scheme);
788        }
789    }
790    result
791}
