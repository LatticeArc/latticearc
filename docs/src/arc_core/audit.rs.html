1//! # Persistent Audit Storage with Rotation
2//!
3//! Provides tamper-evident audit logging for cryptographic operations.
4//! Events are persisted to disk in JSON Lines format with automatic rotation
5//! based on file size and age.
6//!
7//! ## Security Features
8//!
9//! - **Integrity Verification**: SHA-256 hash chain for tamper detection
10//! - **Automatic Rotation**: Files rotate based on size and age limits
11//! - **Retention Policies**: Configurable retention periods for compliance
12//! - **Thread-Safe**: All operations are safe for concurrent access
13//!
14//! ## Usage
15//!
16//! ```rust,no_run
17//! use arc_core::audit::{AuditConfig, FileAuditStorage, AuditStorage, AuditEvent, AuditEventType, AuditOutcome};
18//! use std::path::PathBuf;
19//! use std::time::Duration;
20//!
21//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
22//! let config = AuditConfig {
23//!     storage_path: PathBuf::from("/var/log/latticearc/audit"),
24//!     max_file_size_bytes: 100 * 1024 * 1024, // 100MB
25//!     max_file_age: Duration::from_secs(24 * 60 * 60), // 24 hours
26//!     retention_days: 90,
27//! };
28//!
29//! let storage = FileAuditStorage::new(config)?;
30//!
31//! // Create and write an audit event
32//! let event = AuditEvent::new(
33//!     AuditEventType::CryptoOperation,
34//!     "encrypt_data",
35//!     AuditOutcome::Success,
36//! );
37//! storage.write(&event)?;
38//! # Ok(())
39//! # }
40//! ```

60/// Audit event for persistent storage.
61///
62/// Each event captures a single auditable action with full context
63/// for compliance and forensic analysis.
64#[derive(Debug, Clone, Serialize, Deserialize)]
65pub struct AuditEvent {

67    pub id: String,
68    /// Timestamp when the event occurred.
69    pub timestamp: DateTime<Utc>,
70    /// Category of the audit event.

73    pub actor: Option<String>,
74    /// Resource being acted upon (optional).
75    pub resource: Option<String>,

77    pub action: String,
78    /// Outcome of the action.
79    pub outcome: AuditOutcome,
80    /// Additional key-value metadata.

82    /// SHA-256 hash for tamper detection (includes previous event hash).
83    pub integrity_hash: String,

386/// Internal state for file rotation tracking.
387struct FileState {
388    /// Current file handle wrapped in a buffered writer.
389    writer: BufWriter<File>,
390    /// Path to the current file (used for logging during rotation).
391    current_path: PathBuf,
392    /// Size of the current file in bytes.
393    current_size: u64,
394    /// Timestamp when the current file was created.
395    created_at: DateTime<Utc>,
396}

398/// File-based audit storage with automatic rotation.
399///
400/// Writes audit events as JSON Lines (one JSON object per line).
401/// Files are rotated when they exceed the configured size or age limits.
402pub struct FileAuditStorage {
403    /// Configuration for this storage instance.
404    config: AuditConfig,
405    /// Current file state (protected by mutex for thread safety).
406    file_state: Mutex<Option<FileState>>,
407    /// Hash of the previous event for chain integrity.
408    previous_hash: RwLock<String>,
409}
410
411impl FileAuditStorage {
412    /// Create a new file-based audit storage.
413    ///
414    /// Creates the storage directory if it doesn't exist.
415    ///
416    /// # Errors
417    ///
418    /// Returns an error if the storage directory cannot be created.
419    pub fn new(config: AuditConfig) -> Result<Arc<Self>> {
420        // Create storage directory if it doesn't exist
421        fs::create_dir_all(&config.storage_path).map_err(|e| {
422            CoreError::AuditError(format!(
423                "Failed to create audit directory '{}': {}",
424                config.storage_path.display(),
425                e
426            ))
427        })?;
428
429        let storage = Arc::new(Self {
430            config,
431            file_state: Mutex::new(None),
432            previous_hash: RwLock::new(String::new()),
433        });
434
435        // Clean up old files based on retention policy
436        storage.cleanup_old_files()?;
437
438        Ok(storage)
439    }
440
441    /// Get the configuration for this storage instance.
442    #[must_use]
443    pub fn config(&self) -> &AuditConfig {
444        &self.config
445    }
446
447    /// Compute the integrity hash for an event.
448    ///
449    /// The hash includes the previous event's hash to create a chain,
450    /// making tampering detectable.
451    fn compute_integrity_hash(event: &AuditEvent, previous_hash: &str) -> String {
452        let mut hasher = Sha256::new();
453
454        // Include previous hash for chain integrity
455        hasher.update(previous_hash.as_bytes());
456
457        // Include all event fields (except the integrity_hash itself)
458        hasher.update(event.id.as_bytes());
459        hasher.update(event.timestamp.to_rfc3339().as_bytes());
460        hasher.update(event.event_type.to_string().as_bytes());
461
462        if let Some(ref actor) = event.actor {
463            hasher.update(actor.as_bytes());
464        }
465        if let Some(ref resource) = event.resource {
466            hasher.update(resource.as_bytes());
467        }
468
469        hasher.update(event.action.as_bytes());
470        hasher.update(event.outcome.to_string().as_bytes());
471
472        // Include metadata in sorted order for deterministic hashing
473        let mut metadata_keys: Vec<&String> = event.metadata.keys().collect();
474        metadata_keys.sort();
475        for key in metadata_keys {
476            hasher.update(key.as_bytes());
477            if let Some(value) = event.metadata.get(key) {
478                hasher.update(value.as_bytes());
479            }
480        }
481
482        hex::encode(hasher.finalize())
483    }
484
485    /// Check if the current file needs rotation.
486    fn needs_rotation(&self, state: &FileState) -> bool {
487        // Check size limit
488        if state.current_size >= self.config.max_file_size_bytes {
489            return true;
490        }
491
492        // Check age limit
493        let age =
494            Utc::now().signed_duration_since(state.created_at).to_std().unwrap_or(Duration::ZERO);
495
496        age >= self.config.max_file_age
497    }
498
499    /// Rotate the current file if needed.
500    fn rotate_if_needed(&self, state: &mut Option<FileState>) -> Result<()> {
501        let should_rotate = state.as_ref().is_some_and(|s| self.needs_rotation(s));
502
503        if should_rotate {
504            // Close current file and create new one
505            if let Some(mut old_state) = state.take() {
506                tracing::info!(
507                    "Rotating audit file: {} (size: {} bytes)",
508                    old_state.current_path.display(),
509                    old_state.current_size
510                );
511                old_state.writer.flush().map_err(|e| {
512                    CoreError::AuditError(format!("Failed to flush audit file: {}", e))
513                })?;
514            }
515        }
516
517        // Create new file if needed
518        if state.is_none() {
519            *state = Some(self.create_new_file()?);
520        }
521
522        Ok(())
523    }
524
525    /// Create a new audit file.
526    fn create_new_file(&self) -> Result<FileState> {
527        let now = Utc::now();
528        let filename = format!("audit-{}.jsonl", now.format("%Y-%m-%dT%H-%M-%S"));
529        let path = self.config.storage_path.join(&filename);
530
531        let file = OpenOptions::new().create(true).append(true).open(&path).map_err(|e| {
532            CoreError::AuditError(format!(
533                "Failed to create audit file '{}': {}",
534                path.display(),
535                e
536            ))
537        })?;
538
539        tracing::debug!("Created new audit file: {}", path.display());
540
541        Ok(FileState {
542            writer: BufWriter::new(file),
543            current_path: path,
544            current_size: 0,
545            created_at: now,
546        })
547    }
548
549    /// Clean up old audit files based on retention policy.
550    fn cleanup_old_files(&self) -> Result<()> {
551        let retention_duration = chrono::Duration::days(i64::from(self.config.retention_days));
552        let cutoff = Utc::now().checked_sub_signed(retention_duration).unwrap_or_else(Utc::now);
553
554        let entries = fs::read_dir(&self.config.storage_path).map_err(|e| {
555            CoreError::AuditError(format!(
556                "Failed to read audit directory '{}': {}",
557                self.config.storage_path.display(),
558                e
559            ))
560        })?;
561
562        for entry in entries {
563            let Ok(entry) = entry else { continue };
564
565            let path = entry.path();
566
567            // Only process .jsonl files
568            if path.extension().and_then(|e| e.to_str()) != Some("jsonl") {
569                continue;
570            }
571
572            // Check file modification time
573            let Ok(metadata) = fs::metadata(&path) else { continue };
574
575            let modified = match metadata.modified() {
576                Ok(t) => DateTime::<Utc>::from(t),
577                Err(_) => continue,
578            };
579
580            if modified < cutoff {
581                // Delete old file
582                if let Err(e) = fs::remove_file(&path) {
583                    tracing::warn!("Failed to remove old audit file '{}': {}", path.display(), e);
584                } else {
585                    tracing::info!("Removed old audit file: {}", path.display());
586                }
587            }
588        }
589
590        Ok(())
591    }
592
593    /// Write an audit event to the current file.
594    fn write_event_to_file(&self, event: &mut AuditEvent) -> Result<()> {
595        let mut file_state = self.file_state.lock();
596
597        // Rotate if needed
598        self.rotate_if_needed(&mut file_state)?;
599
600        let state = file_state
601            .as_mut()
602            .ok_or_else(|| CoreError::AuditError("No active audit file".to_string()))?;
603
604        // Compute integrity hash with chain
605        let previous_hash = self.previous_hash.read().clone();
606        event.integrity_hash = Self::compute_integrity_hash(event, &previous_hash);
607
608        // Update previous hash for next event
609        {
610            let mut prev = self.previous_hash.write();
611            prev.clone_from(&event.integrity_hash);
612        }
613
614        // Serialize event to JSON
615        let json = serde_json::to_string(event).map_err(|e| {
616            CoreError::AuditError(format!("Failed to serialize audit event: {}", e))
617        })?;
618
619        // Write JSON line
620        let line = format!("{}\n", json);
621        let line_bytes = line.as_bytes();
622
623        state
624            .writer
625            .write_all(line_bytes)
626            .map_err(|e| CoreError::AuditError(format!("Failed to write audit event: {}", e)))?;
627
628        // Update size tracking
629        state.current_size = state.current_size.saturating_add(line_bytes.len() as u64);
630
631        Ok(())
632    }
633}

635impl AuditStorage for FileAuditStorage {
636    fn write(&self, event: &AuditEvent) -> Result<()> {
637        let mut event_copy = event.clone();
638        self.write_event_to_file(&mut event_copy)
639    }
640
641    fn flush(&self) -> Result<()> {
642        let mut file_state = self.file_state.lock();
643
644        if let Some(ref mut state) = *file_state {
645            state
646                .writer
647                .flush()
648                .map_err(|e| CoreError::AuditError(format!("Failed to flush audit file: {}", e)))?;
649        }
650
651        Ok(())
652    }
653}

655/// Generate a UUID v4 for event identification.
656fn generate_uuid() -> String {
657    use rand::RngCore;
658
659    let mut bytes = [0u8; 16];
660    rand::rngs::OsRng.fill_bytes(&mut bytes);
661
662    // Set version (4) and variant (RFC 4122) bits
663    bytes[6] = (bytes[6] & 0x0f) | 0x40;
664    bytes[8] = (bytes[8] & 0x3f) | 0x80;
665
666    format!(
667        "{:02x}{:02x}{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}",
668        bytes[0],
669        bytes[1],
670        bytes[2],
671        bytes[3],
672        bytes[4],
673        bytes[5],
674        bytes[6],
675        bytes[7],
676        bytes[8],
677        bytes[9],
678        bytes[10],
679        bytes[11],
680        bytes[12],
681        bytes[13],
682        bytes[14],
683        bytes[15]
684    )
685}
