1//! # Persistent Audit Storage with Rotation
2//!
3//! Provides tamper-evident audit logging for cryptographic operations.
4//! Events are persisted to disk in JSON Lines format with automatic rotation
5//! based on file size and age.
6//!
7//! ## Security Features
8//!
9//! - **Integrity Verification**: SHA-256 hash chain for tamper detection
10//! - **Automatic Rotation**: Files rotate based on size and age limits
11//! - **Retention Policies**: Configurable retention periods for compliance
12//! - **Thread-Safe**: All operations are safe for concurrent access
13//!
14//! ## Usage
15//!
16//! ```rust,ignore
17//! use arc_core::audit::{AuditConfig, FileAuditStorage, AuditStorage, AuditEvent, AuditEventType, AuditOutcome};
18//! use std::path::PathBuf;
19//! use std::time::Duration;
20//!
21//! let config = AuditConfig {
22//!     storage_path: PathBuf::from("/var/log/latticearc/audit"),
23//!     max_file_size_bytes: 100 * 1024 * 1024, // 100MB
24//!     max_file_age: Duration::from_secs(24 * 60 * 60), // 24 hours
25//!     retention_days: 90,
26//! };
27//!
28//! let storage = FileAuditStorage::new(config)?;
29//!
30//! // Create and write an audit event
31//! let event = AuditEvent::new(
32//!     AuditEventType::CryptoOperation,
33//!     "encrypt_data",
34//!     AuditOutcome::Success,
35//! );
36//! storage.write(&event)?;
37//! ```

57/// Audit event for persistent storage.
58///
59/// Each event captures a single auditable action with full context
60/// for compliance and forensic analysis.

65    /// Timestamp when the event occurred.

383/// Internal state for file rotation tracking.
384struct FileState {
385    /// Current file handle wrapped in a buffered writer.
386    writer: BufWriter<File>,
387    /// Path to the current file (used for logging during rotation).
388    current_path: PathBuf,
389    /// Size of the current file in bytes.
390    current_size: u64,
391    /// Timestamp when the current file was created.
392    created_at: DateTime<Utc>,
393}

395/// File-based audit storage with automatic rotation.
396///
397/// Writes audit events as JSON Lines (one JSON object per line).
398/// Files are rotated when they exceed the configured size or age limits.
399pub struct FileAuditStorage {
400    /// Configuration for this storage instance.
401    config: AuditConfig,
402    /// Current file state (protected by mutex for thread safety).
403    file_state: Mutex<Option<FileState>>,
404    /// Hash of the previous event for chain integrity.
405    previous_hash: RwLock<String>,
406}
407
408impl FileAuditStorage {
409    /// Create a new file-based audit storage.
410    ///
411    /// Creates the storage directory if it doesn't exist.
412    ///
413    /// # Errors
414    ///
415    /// Returns an error if the storage directory cannot be created.
416    pub fn new(config: AuditConfig) -> Result<Arc<Self>> {
417        // Create storage directory if it doesn't exist
418        fs::create_dir_all(&config.storage_path).map_err(|e| {
419            CoreError::AuditError(format!(
420                "Failed to create audit directory '{}': {}",
421                config.storage_path.display(),
422                e
423            ))
424        })?;
425
426        let storage = Arc::new(Self {
427            config,
428            file_state: Mutex::new(None),
429            previous_hash: RwLock::new(String::new()),
430        });
431
432        // Clean up old files based on retention policy
433        storage.cleanup_old_files()?;
434
435        Ok(storage)
436    }
437
438    /// Get the configuration for this storage instance.
439    #[must_use]
440    pub fn config(&self) -> &AuditConfig {
441        &self.config
442    }
443
444    /// Compute the integrity hash for an event.
445    ///
446    /// The hash includes the previous event's hash to create a chain,
447    /// making tampering detectable.
448    fn compute_integrity_hash(event: &AuditEvent, previous_hash: &str) -> String {
449        let mut hasher = Sha256::new();
450
451        // Include previous hash for chain integrity
452        hasher.update(previous_hash.as_bytes());
453
454        // Include all event fields (except the integrity_hash itself)
455        hasher.update(event.id.as_bytes());
456        hasher.update(event.timestamp.to_rfc3339().as_bytes());
457        hasher.update(event.event_type.to_string().as_bytes());
458
459        if let Some(ref actor) = event.actor {
460            hasher.update(actor.as_bytes());
461        }
462        if let Some(ref resource) = event.resource {
463            hasher.update(resource.as_bytes());
464        }
465
466        hasher.update(event.action.as_bytes());
467        hasher.update(event.outcome.to_string().as_bytes());
468
469        // Include metadata in sorted order for deterministic hashing
470        let mut metadata_keys: Vec<&String> = event.metadata.keys().collect();
471        metadata_keys.sort();
472        for key in metadata_keys {
473            hasher.update(key.as_bytes());
474            if let Some(value) = event.metadata.get(key) {
475                hasher.update(value.as_bytes());
476            }
477        }
478
479        hex::encode(hasher.finalize())
480    }
481
482    /// Check if the current file needs rotation.
483    fn needs_rotation(&self, state: &FileState) -> bool {
484        // Check size limit
485        if state.current_size >= self.config.max_file_size_bytes {
486            return true;
487        }
488
489        // Check age limit
490        let age =
491            Utc::now().signed_duration_since(state.created_at).to_std().unwrap_or(Duration::ZERO);
492
493        age >= self.config.max_file_age
494    }
495
496    /// Rotate the current file if needed.
497    fn rotate_if_needed(&self, state: &mut Option<FileState>) -> Result<()> {
498        let should_rotate = state.as_ref().is_some_and(|s| self.needs_rotation(s));
499
500        if should_rotate {
501            // Close current file and create new one
502            if let Some(mut old_state) = state.take() {
503                tracing::info!(
504                    "Rotating audit file: {} (size: {} bytes)",
505                    old_state.current_path.display(),
506                    old_state.current_size
507                );
508                old_state.writer.flush().map_err(|e| {
509                    CoreError::AuditError(format!("Failed to flush audit file: {}", e))
510                })?;
511            }
512        }
513
514        // Create new file if needed
515        if state.is_none() {
516            *state = Some(self.create_new_file()?);
517        }
518
519        Ok(())
520    }
521
522    /// Create a new audit file.
523    fn create_new_file(&self) -> Result<FileState> {
524        let now = Utc::now();
525        let filename = format!("audit-{}.jsonl", now.format("%Y-%m-%dT%H-%M-%S"));
526        let path = self.config.storage_path.join(&filename);
527
528        let file = OpenOptions::new().create(true).append(true).open(&path).map_err(|e| {
529            CoreError::AuditError(format!(
530                "Failed to create audit file '{}': {}",
531                path.display(),
532                e
533            ))
534        })?;
535
536        tracing::debug!("Created new audit file: {}", path.display());
537
538        Ok(FileState {
539            writer: BufWriter::new(file),
540            current_path: path,
541            current_size: 0,
542            created_at: now,
543        })
544    }
545
546    /// Clean up old audit files based on retention policy.
547    fn cleanup_old_files(&self) -> Result<()> {
548        let retention_duration = chrono::Duration::days(i64::from(self.config.retention_days));
549        let cutoff = Utc::now().checked_sub_signed(retention_duration).unwrap_or_else(Utc::now);
550
551        let entries = fs::read_dir(&self.config.storage_path).map_err(|e| {
552            CoreError::AuditError(format!(
553                "Failed to read audit directory '{}': {}",
554                self.config.storage_path.display(),
555                e
556            ))
557        })?;
558
559        for entry in entries {
560            let Ok(entry) = entry else { continue };
561
562            let path = entry.path();
563
564            // Only process .jsonl files
565            if path.extension().and_then(|e| e.to_str()) != Some("jsonl") {
566                continue;
567            }
568
569            // Check file modification time
570            let Ok(metadata) = fs::metadata(&path) else { continue };
571
572            let modified = match metadata.modified() {
573                Ok(t) => DateTime::<Utc>::from(t),
574                Err(_) => continue,
575            };
576
577            if modified < cutoff {
578                // Delete old file
579                if let Err(e) = fs::remove_file(&path) {
580                    tracing::warn!("Failed to remove old audit file '{}': {}", path.display(), e);
581                } else {
582                    tracing::info!("Removed old audit file: {}", path.display());
583                }
584            }
585        }
586
587        Ok(())
588    }
589
590    /// Write an audit event to the current file.
591    fn write_event_to_file(&self, event: &mut AuditEvent) -> Result<()> {
592        let mut file_state = self.file_state.lock();
593
594        // Rotate if needed
595        self.rotate_if_needed(&mut file_state)?;
596
597        let state = file_state
598            .as_mut()
599            .ok_or_else(|| CoreError::AuditError("No active audit file".to_string()))?;
600
601        // Compute integrity hash with chain
602        let previous_hash = self.previous_hash.read().clone();
603        event.integrity_hash = Self::compute_integrity_hash(event, &previous_hash);
604
605        // Update previous hash for next event
606        {
607            let mut prev = self.previous_hash.write();
608            prev.clone_from(&event.integrity_hash);
609        }
610
611        // Serialize event to JSON
612        let json = serde_json::to_string(event).map_err(|e| {
613            CoreError::AuditError(format!("Failed to serialize audit event: {}", e))
614        })?;
615
616        // Write JSON line
617        let line = format!("{}\n", json);
618        let line_bytes = line.as_bytes();
619
620        state
621            .writer
622            .write_all(line_bytes)
623            .map_err(|e| CoreError::AuditError(format!("Failed to write audit event: {}", e)))?;
624
625        // Update size tracking
626        state.current_size = state.current_size.saturating_add(line_bytes.len() as u64);
627
628        Ok(())
629    }
630}

632impl AuditStorage for FileAuditStorage {
633    fn write(&self, event: &AuditEvent) -> Result<()> {
634        let mut event_copy = event.clone();
635        self.write_event_to_file(&mut event_copy)
636    }
637
638    fn flush(&self) -> Result<()> {
639        let mut file_state = self.file_state.lock();
640
641        if let Some(ref mut state) = *file_state {
642            state
643                .writer
644                .flush()
645                .map_err(|e| CoreError::AuditError(format!("Failed to flush audit file: {}", e)))?;
646        }
647
648        Ok(())
649    }

652/// Generate a UUID v4 for event identification.
653fn generate_uuid() -> String {
654    use rand::RngCore;
655
656    let mut bytes = [0u8; 16];
657    rand::thread_rng().fill_bytes(&mut bytes);
658
659    // Set version (4) and variant (RFC 4122) bits
660    bytes[6] = (bytes[6] & 0x0f) | 0x40;
661    bytes[8] = (bytes[8] & 0x3f) | 0x80;
662
663    format!(
664        "{:02x}{:02x}{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}",
665        bytes[0],
666        bytes[1],
667        bytes[2],
668        bytes[3],
669        bytes[4],
670        bytes[5],
671        bytes[6],
672        bytes[7],
673        bytes[8],
674        bytes[9],
675        bytes[10],
676        bytes[11],
677        bytes[12],
678        bytes[13],
679        bytes[14],
680        bytes[15]
681    )
682}

820    fn test_integrity_hash_chain() {
821        let event1 =
822            AuditEvent::new(AuditEventType::CryptoOperation, "operation1", AuditOutcome::Success);
823        let event2 =
824            AuditEvent::new(AuditEventType::CryptoOperation, "operation2", AuditOutcome::Success);
825
826        let hash1 = FileAuditStorage::compute_integrity_hash(&event1, "");
827        let hash2 = FileAuditStorage::compute_integrity_hash(&event2, &hash1);
828
829        // Hashes should be different
830        assert_ne!(hash1, hash2);
831
832        // Same event with same previous hash should produce same result
833        let hash2_again = FileAuditStorage::compute_integrity_hash(&event2, &hash1);
834        assert_eq!(hash2, hash2_again);
835
836        // Different previous hash should produce different result
837        let hash2_different = FileAuditStorage::compute_integrity_hash(&event2, "different");
838        assert_ne!(hash2, hash2_different);
839    }
