1//! Parallel Key Generation Tests
2//!
3//! Ensures key generation is safe when called from multiple threads simultaneously.
4
5#[cfg(test)]
6mod tests {
7    use arc_primitives::kem::ml_kem::{MlKem, MlKemSecurityLevel};
8    use rand::rngs::OsRng;
9    use std::collections::HashSet;
10    use std::sync::atomic::{AtomicUsize, Ordering};
11    use std::sync::{Arc, Mutex};
12    use std::thread;
13
14    #[test]
15    fn concurrent_keygen_unique_keys_ml_kem_512() {
16        const NUM_THREADS: usize = 8;
17        const KEYS_PER_THREAD: usize = 5;
18
19        let keys = Arc::new(Mutex::new(Vec::new()));
20
21        let handles: Vec<_> = (0..NUM_THREADS)
22            .map(|_| {
23                let keys = Arc::clone(&keys);
24                thread::spawn(move || {
25                    let mut rng = OsRng;
26                    let mut local_keys = Vec::new();
27
28                    for _ in 0..KEYS_PER_THREAD {
29                        let (pk, _sk) =
30                            MlKem::generate_keypair(&mut rng, MlKemSecurityLevel::MlKem512)
31                                .expect("keygen should succeed");
32                        local_keys.push(pk.to_bytes());
33                    }
34
35                    keys.lock().expect("mutex not poisoned").extend(local_keys);
36                })
37            })
38            .collect();
39
40        for handle in handles {
41            handle.join().expect("thread should not panic");
42        }
43
44        let keys = keys.lock().expect("mutex not poisoned");
45        let unique: HashSet<_> = keys.iter().collect();
46
47        assert_eq!(unique.len(), NUM_THREADS * KEYS_PER_THREAD, "All keys should be unique");
48    }

50    #[test]
51    fn concurrent_keygen_mixed_security_levels() {
52        let levels = [
53            MlKemSecurityLevel::MlKem512,
54            MlKemSecurityLevel::MlKem768,
55            MlKemSecurityLevel::MlKem1024,
56        ];
57
58        let results = Arc::new(Mutex::new(Vec::new()));
59        let results_ref = Arc::clone(&results);
60
61        let handles: Vec<_> = levels
62            .iter()
63            .flat_map(|&level| {
64                let results_inner = Arc::clone(&results_ref);
65                (0..3).map(move |_| {
66                    let results = Arc::clone(&results_inner);
67                    thread::spawn(move || {
68                        let mut rng = OsRng;
69                        let result = MlKem::generate_keypair(&mut rng, level);
70                        results.lock().expect("mutex").push(result.is_ok());
71                    })
72                })
73            })
74            .collect();
75
76        for handle in handles {
77            handle.join().expect("thread should not panic");
78        }
79
80        let results = results.lock().expect("mutex");
81        assert!(results.iter().all(|&ok| ok), "All keygen should succeed");
82        assert_eq!(results.len(), 9, "Should have 9 results (3 levels x 3 threads)");
83    }

85    #[test]
86    fn concurrent_keygen_stress_test() {
87        const NUM_THREADS: usize = 16;
88
89        let success_count = Arc::new(AtomicUsize::new(0));
90
91        let handles: Vec<_> = (0..NUM_THREADS)
92            .map(|i| {
93                let success_count = Arc::clone(&success_count);
94                thread::spawn(move || {
95                    let mut rng = OsRng;
96                    let level = match i % 3 {
97                        0 => MlKemSecurityLevel::MlKem512,
98                        1 => MlKemSecurityLevel::MlKem768,
99                        _ => MlKemSecurityLevel::MlKem1024,
100                    };
101
102                    if MlKem::generate_keypair(&mut rng, level).is_ok() {
103                        success_count.fetch_add(1, Ordering::SeqCst);
104                    }
105                })
106            })
107            .collect();
108
109        for handle in handles {
110            handle.join().expect("thread should not panic");
111        }
112
113        assert_eq!(
114            success_count.load(Ordering::SeqCst),
115            NUM_THREADS,
116            "All concurrent keygen should succeed"
117        );
118    }
