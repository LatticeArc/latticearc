1//! Shared timing-equalizer dummy material for hybrid signature verify.
2//!
3//! Both [`crate::hybrid::sig_hybrid::verify`] (typed-input lower-level
4//! path) and [`crate::unified_api::convenience::api::verify_hybrid_ml_dsa_ed25519`]
5//! (raw-byte unified-API path) need shape-correct stand-in inputs to
6//! substitute when a caller-supplied PK or signature fails the shape
7//! check. Running the verify pipeline against the substitute keeps
8//! wall-clock cost equal between shape-fail and verify-fail, removing
9//! the timing oracle that opaque error returns alone do not close.
10//!
11//! # Two layers of dummy material
12//!
13//! - **Raw bytes** (`pq_pk`, `pq_sig`, `ed_pk`, `ed_sig`): zero-filled
14//!   length-correct buffers. Used by the unified-API path's raw-byte
15//!   interface; the consumer there re-parses internally via
16//!   `verify_pq_ml_dsa_unverified(...).unwrap_or(false)` so a parse
17//!   failure on the dummy collapses to `false` (today the parser is
18//!   length-only so this is unreachable, but the `unwrap_or(false)`
19//!   shape preserves verify-pipeline execution).
20//!
21//! - **Pre-parsed valid material** (`parsed: Option<HybridVerifyDummyParsed>`):
22//!   real ML-DSA keypair + signature generated at init time, plus the
23//!   exact test message that was signed. Used by [`sig_hybrid::verify`]
24//!   when caller bytes fail the shape check OR fail to parse. Ensures
25//!   `MlDsaPublicKey::verify` runs against guaranteed-valid material
26//!   regardless of whether `from_bytes` ever adds content validation
27//!   in a future `fips204` release. Post-85e2bd79e M1 audit fix —
28//!   the previous design held raw bytes only and re-parsed at verify
29//!   time, which would silently lose its equalizer property if
30//!   `from_bytes` started rejecting zero-byte buffers.
31//!
32//! # Init fallibility
33//!
34//! `parsed` is `Option<HybridVerifyDummyParsed>` because keygen + sign
35//! at init time is fallible (RNG failure, FIPS PCT failure, etc.).
36//! When `parsed` is `None` (extremely rare path), the consumer falls
37//! back to today's behavior: parse-fail → verify-skip. The equalizer
38//! is best-effort defense-in-depth; the safety bound depends on
39//! downstream code combining the verify result via bitwise AND with
40//! `pq_shape_ok` so a missed equalization does not affect correctness,
41//! only the strength of the timing-oracle countermeasure.
42

50/// Pre-parsed ML-DSA material for the verify-time equalizer fallback.
51///
52/// Generated lazily once per parameter set on first use of
53/// [`hybrid_verify_dummy_material`]. The keypair is generated via
54/// [`crate::primitives::sig::ml_dsa::generate_keypair`] and the
55/// signature over [`Self::pq_test_message`] via
56/// [`MlDsaSecretKey::sign`]. After init the secret key is dropped —
57/// only the PK and signature need to live `'static` for the
58/// equalizer; the SK has served its purpose by producing the
59/// signature.
60pub struct HybridVerifyDummyParsed {
61    /// Real (validly-derived) ML-DSA public key for this parameter set.
62    pub pq_pk: MlDsaPublicKey,
63    /// Real ML-DSA signature, valid against `pq_pk` over `pq_test_message`.
64    pub pq_sig: MlDsaSignature,
65    /// The fixed test message that `pq_sig` signs. Verifying
66    /// `pq_pk.verify(pq_test_message, pq_sig, &[])` MUST return
67    /// `Ok(true)`. The message is intentionally NOT the caller's
68    /// message — content-dependent verify timing is exactly what the
69    /// equalizer is supposed to mask.
70    pub pq_test_message: Vec<u8>,
71}

73/// Per-parameter-set cached zero-byte buffers + pre-parsed real
74/// material used by both the `sig_hybrid::verify` and
75/// `unified_api::verify_hybrid_ml_dsa_ed25519` timing equalizers.
76///
77/// The Ed25519 dummies are constant-size (32 / 64 bytes) and shared
78/// across parameter sets but live in this struct for locality. No
79/// secret material; safe to keep `'static`. The pre-parsed ML-DSA
80/// material was generated at init from real keygen output — its
81/// public components do not require zeroization.
82pub struct HybridVerifyDummy {
83    /// Raw zero-byte ML-DSA public key, length matches the parameter set.
84    pub pq_pk: Vec<u8>,
85    /// Raw zero-byte ML-DSA signature, length matches the parameter set.
86    pub pq_sig: Vec<u8>,
87    /// Raw zero-byte Ed25519 public key (32 bytes).
88    pub ed_pk: Vec<u8>,
89    /// Raw zero-byte Ed25519 signature (64 bytes).
90    pub ed_sig: Vec<u8>,
91    /// Pre-parsed valid ML-DSA material for fallback when caller's
92    /// bytes fail to parse. `None` if init keygen+sign failed
93    /// (extremely rare — RNG/PCT failure). Consumer must handle
94    /// `None` by falling back to the legacy raw-bytes-only behavior.
95    pub parsed: Option<HybridVerifyDummyParsed>,
96}
97
98/// Lazy-initialized per-parameter-set dummy material.
99///
100/// First call for each `param_set` allocates the zero buffers AND
101/// attempts to generate a real ML-DSA keypair + sign a fixed test
102/// message. The values are cached for the process lifetime. If
103/// keygen or sign fails (extremely rare — RNG/PCT failure), the
104/// `parsed` field stays `None` and the consumer falls back to the
105/// legacy parse-then-verify path. The init closure is structured to
106/// always succeed at constructing the raw-byte fields — only the
107/// `parsed` field is fallible.
108#[must_use]
109pub fn hybrid_verify_dummy_material(param_set: MlDsaParameterSet) -> &'static HybridVerifyDummy {
110    use std::sync::OnceLock;
111
112    static M44: OnceLock<HybridVerifyDummy> = OnceLock::new();
113    static M65: OnceLock<HybridVerifyDummy> = OnceLock::new();
114    static M87: OnceLock<HybridVerifyDummy> = OnceLock::new();
115
116    let cell = match param_set {
117        MlDsaParameterSet::MlDsa44 => &M44,
118        MlDsaParameterSet::MlDsa65 => &M65,
119        MlDsaParameterSet::MlDsa87 => &M87,
120    };
121    cell.get_or_init(|| HybridVerifyDummy {
122        pq_pk: vec![0u8; param_set.public_key_size()],
123        pq_sig: vec![0u8; param_set.signature_size()],
124        ed_pk: vec![0u8; 32],
125        ed_sig: vec![0u8; 64],
126        parsed: try_init_parsed(param_set),
127    })
128}
129
130/// Best-effort init of the pre-parsed equalizer material. Returns
131/// `None` on any failure — keygen RNG hiccup, FIPS PCT failure, sign
132/// failure, etc. The caller (consumer of `HybridVerifyDummy`) must
133/// treat `None` as "equalizer degraded for this parameter set, fall
134/// back to legacy behavior".
135fn try_init_parsed(param_set: MlDsaParameterSet) -> Option<HybridVerifyDummyParsed> {
136    // Fixed test message: not security-sensitive, just a stable byte
137    // string that's signed once at init and verified at every
138    // equalizer-fallback. The label binds verify input to this
139    // module's purpose so the cached signature is unmistakable as
140    // dummy material if it ever leaks via debugging.
141    let pq_test_message: Vec<u8> = b"latticearc/verify-equalizer/dummy-message-v1".to_vec();
142    let (pq_pk, sk): (MlDsaPublicKey, MlDsaSecretKey) =
143        crate::primitives::sig::ml_dsa::generate_keypair(param_set).ok()?;
144    let pq_sig = sk.sign(&pq_test_message, &[]).ok()?;
145    // SK is dropped here — its zeroization is handled by MlDsaSecretKey's
146    // Drop impl (data is wrapped in Zeroizing).
147    drop(sk);
148    Some(HybridVerifyDummyParsed { pq_pk, pq_sig, pq_test_message })
149}

151#[cfg(test)]
152#[allow(clippy::panic)] // Tests panic on missing init material
153mod tests {
154    use super::*;
155
156    /// Locks the equalizer's contract: for every parameter set, the
157    /// pre-parsed material verifies successfully. If a future change
158    /// breaks keygen+sign init, this test catches it. (RNG failure
159    /// is the only path to `None` and is not reachable in normal CI.)
160    #[test]
161    fn parsed_material_verifies_successfully_for_all_parameter_sets() {
162        for param_set in
163            [MlDsaParameterSet::MlDsa44, MlDsaParameterSet::MlDsa65, MlDsaParameterSet::MlDsa87]
164        {
165            let dummy = hybrid_verify_dummy_material(param_set);
166            let Some(parsed) = dummy.parsed.as_ref() else {
167                panic!("init keygen + sign must succeed under test conditions for {param_set:?}");
168            };
169            let result = parsed.pq_pk.verify(&parsed.pq_test_message, &parsed.pq_sig, &[]);
170            assert!(
171                matches!(result, Ok(true)),
172                "pre-parsed equalizer material must self-verify for {:?}",
173                param_set,
174            );
175        }
176    }
177
178    /// The raw-bytes fields preserve their pre-M1 contract: zero-filled,
179    /// length-correct. Consumers that have not migrated to the parsed
180    /// material continue to work.
181    #[test]
182    fn raw_byte_fields_preserve_legacy_contract() {
183        for param_set in
184            [MlDsaParameterSet::MlDsa44, MlDsaParameterSet::MlDsa65, MlDsaParameterSet::MlDsa87]
185        {
186            let dummy = hybrid_verify_dummy_material(param_set);
187            assert_eq!(dummy.pq_pk.len(), param_set.public_key_size());
188            assert_eq!(dummy.pq_sig.len(), param_set.signature_size());
189            assert!(dummy.pq_pk.iter().all(|&b| b == 0));
190            assert!(dummy.pq_sig.iter().all(|&b| b == 0));
191            assert_eq!(dummy.ed_pk.len(), 32);
192            assert_eq!(dummy.ed_sig.len(), 64);
193            assert!(dummy.ed_pk.iter().all(|&b| b == 0));
194            assert!(dummy.ed_sig.iter().all(|&b| b == 0));
195        }
196    }
197}
