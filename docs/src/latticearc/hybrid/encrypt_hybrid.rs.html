6//! Hybrid Encryption Module
7//!
8//! This module provides hybrid encryption combining post-quantum (ML-KEM) key
9//! encapsulation with AES-256-GCM symmetric encryption for quantum-resistant
10//! data encryption with classical performance characteristics.
11//!
12//! # Overview
13//!
14//! The hybrid encryption scheme uses:
15//! - **ML-KEM-768** (FIPS 203) for post-quantum key encapsulation
16//! - **AES-256-GCM** for authenticated symmetric encryption
17//! - **HKDF-SHA256** for key derivation with domain separation
18//!
19//! # Security Properties
20//!
21//! - IND-CCA2 security from ML-KEM
22//! - Authenticated encryption with associated data (AEAD)
23//! - Domain separation via HPKE-style key derivation
24//!
25//! # Example
26//!
27//! ```rust,no_run
28//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
29//! use latticearc::hybrid::encrypt_hybrid::{
30//!     encrypt_hybrid, decrypt_hybrid, HybridEncryptionContext,
31//! };
32//! use latticearc::hybrid::kem_hybrid;
33//! use latticearc::primitives::kem::ml_kem::MlKemSecurityLevel;
34//!
35//! // Generate a hybrid (ML-KEM + X25519) keypair at the desired security level.
36//! let (hybrid_pk, hybrid_sk) =
37//!     kem_hybrid::generate_keypair_with_level(MlKemSecurityLevel::MlKem768)?;
38//!
39//! let plaintext = b"Secret message";
40//! let context = HybridEncryptionContext::default();
41//!
42//! let ciphertext = encrypt_hybrid(&hybrid_pk, plaintext, Some(&context))?;
43//! let decrypted = decrypt_hybrid(&hybrid_sk, &ciphertext, Some(&context))?;
44//! # Ok(())
45//! # }
46//! ```

57/// Error types for hybrid encryption operations.
58///
59/// This enum captures all possible error conditions that can occur during
60/// hybrid encryption and decryption operations.
61#[non_exhaustive]
62#[derive(Debug, Clone, PartialEq, Eq, Error)]
63pub enum HybridEncryptionError {
64    /// Error during key encapsulation mechanism operations.
65    #[error("KEM error: {0}")]
66    KemError(String),
67    /// Error during symmetric encryption.
68    #[error("Encryption error: {0}")]
69    EncryptionError(String),
70    /// Error during symmetric decryption or authentication failure.
71    #[error("Decryption error: {0}")]
72    DecryptionError(String),
73    /// Error during key derivation function operations.
74    #[error("Key derivation error: {0}")]
75    KdfError(String),
76    /// Invalid input parameters provided to the operation.
77    #[error("Invalid input: {0}")]
78    InvalidInput(String),
79    /// Key length mismatch error.
80    #[error("Key length error: expected {expected}, got {actual}")]
81    KeyLengthError {
82        /// Expected key length in bytes.
83        expected: usize,
84        /// Actual key length provided.
85        actual: usize,
86    },
87}

196/// HPKE-style context information for hybrid encryption.
197///
198/// This structure provides domain separation and additional authenticated data
199/// for the key derivation and encryption operations, following RFC 9180 (HPKE).
200#[derive(Debug, Clone)]
201pub struct HybridEncryptionContext {
202    /// Application-specific info string for key derivation domain separation.
203    ///
204    /// Private (Pattern 7): callers must go through [`Self::default`],
205    /// [`Self::with_aad`], or [`Self::with_explicit_info`] so the only way
206    /// to set a non-default `info` is to make a deliberate, grep-able call.
207    /// A `pub` field would let any caller silently collide with another
208    /// protocol's domain-separation label.
209    info: Vec<u8>,
210    /// Additional authenticated data (AAD) for AEAD encryption.
211    ///
212    /// Intentionally `pub` (asymmetric with `info`): AAD is per-message
213    /// application data — transport headers, request IDs, version tags —
214    /// that callers MUST set at every call site. It carries no domain-
215    /// separation role, so the grep-able-audit motivation that privatizes
216    /// `info` does not apply. AAD mismatches surface as ordinary
217    /// authentication failures (no oracle) and are visible to legitimate
218    /// callers by design.
219    ///
220    /// **Length bound:** [`Self::MAX_AAD_LEN`] (64 KiB). The bound is
221    /// enforced at [`encrypt_hybrid`] / [`decrypt_hybrid`] entry, before
222    /// any AEAD or KEM work runs. Real-world AAD payloads (transport
223    /// headers, request IDs, content-type strings) sit well below this
224    /// cap; the limit exists to bound DoS surface, not to constrain
225    /// legitimate callers.
226    pub aad: Vec<u8>,
227}
228
229impl Default for HybridEncryptionContext {
230    fn default() -> Self {
231        Self { info: crate::types::domains::HYBRID_ENCRYPTION_INFO.to_vec(), aad: vec![] }
232    }
233}

235impl HybridEncryptionContext {
236    /// Maximum AAD length accepted by [`encrypt_hybrid`] /
237    /// [`decrypt_hybrid`]. 64 KiB is well above any realistic transport-
238    /// header or request-ID payload while keeping the per-call HKDF
239    /// `info` payload small enough that the length-prefixed concatenation
240    /// stays in a single page.
241    pub const MAX_AAD_LEN: usize = 64 * 1024;
242
243    /// Construct with the canonical `HYBRID_ENCRYPTION_INFO` domain
244    /// separator (recommended) and the supplied AAD.
245    #[must_use]
246    pub fn with_aad(aad: Vec<u8>) -> Self {
247        Self { info: crate::types::domains::HYBRID_ENCRYPTION_INFO.to_vec(), aad }
248    }
249
250    /// Construct with an explicit `info` domain separator. Reserved for
251    /// callers that have registered their own label in
252    /// [`crate::types::domains`] — passing an arbitrary string here risks
253    /// collision with another protocol and breaks Pattern 7.
254    ///
255    /// The `info` argument is `&'static [u8]` deliberately: it forces
256    /// callers to declare the label as a `pub const &[u8]` somewhere
257    /// (typically `domains.rs`), which is the grep-able audit checkpoint.
258    #[must_use]
259    pub fn with_explicit_info(info: &'static [u8], aad: Vec<u8>) -> Self {
260        Self { info: info.to_vec(), aad }
261    }
262
263    /// Read-only access to the `info` field for tests, debug printing, and
264    /// per-call wire framing.
265    #[must_use]
266    pub fn info(&self) -> &[u8] {
267        &self.info
268    }
269}

271/// Public-key + KEM-ciphertext binding for the AEAD-key KDF.
272///
273/// Pattern 7 / HPKE §5.1 require the recipient public key, the
274/// ephemeral public key, and the KEM ciphertext (`enc`) to be bound
275/// into the *final* key-schedule KDF — not just the upstream KEM
276/// combiner — so that key-substitution and KEM-rebinding attacks
277/// against an as-yet-unknown future weakness in either component KEM
278/// cannot produce the same AEAD key under a different `(recipient,
279/// ephemeral, ct)` triple.
280///
281/// The combiner at [`crate::hybrid::kem_hybrid::derive_hybrid_shared_secret`]
282/// already binds `static_pk` and `ephemeral_pk` into the shared
283/// secret, so today the binding is *transitively* present. Re-binding
284/// here is defense-in-depth: a future regression in the combiner that
285/// silently dropped the PK mix would still be caught at this layer.
286///
287/// Borrowed slice fields (no allocation). Empty slices are accepted —
288/// they just contribute zero-length length-prefixed segments — but
289/// production callers (`encrypt_hybrid` / `decrypt_hybrid`) MUST
290/// populate all three.
291#[derive(Debug, Clone, Copy)]
292pub struct DerivationBinding<'a> {
293    /// Recipient's static (long-term) public key. For the hybrid KEM,
294    /// this is the X25519 component of the recipient's `HybridKemPublicKey`.
295    pub recipient_static_pk: &'a [u8],
296    /// Sender's ephemeral public key produced by `kem_hybrid::encapsulate`.
297    /// For the hybrid KEM this is the X25519 ephemeral PK.
298    pub ephemeral_pk: &'a [u8],
299    /// Raw ML-KEM ciphertext (`enc`) produced by `kem_hybrid::encapsulate`.
300    pub kem_ciphertext: &'a [u8],
301}
302
303impl<'a> DerivationBinding<'a> {
304    /// Empty binding (no PK / `enc` mixing). Reserved for tests that
305    /// exercise `derive_encryption_key` in isolation without a real
306    /// KEM round-trip. Production code must use a fully-populated
307    /// binding constructed by `encrypt_hybrid` / `decrypt_hybrid`.
308    #[must_use]
309    pub const fn empty() -> Self {
310        Self { recipient_static_pk: &[], ephemeral_pk: &[], kem_ciphertext: &[] }
311    }
312}

314/// HPKE-style key derivation for hybrid encryption.
315///
316/// Delegates to the primitives HKDF wrapper (`primitives::kdf::hkdf::hkdf`)
317/// which is backed by aws-lc-rs HMAC-SHA256 under the hood, so all callers share
318/// the same FIPS-validated implementation, zeroization guarantees, and
319/// instrumentation.
320///
321/// # Security
322///
323/// The context's AAD is bound in two places: (1) mixed into the HKDF info
324/// parameter here, so different AAD values derive different encryption keys,
325/// and (2) passed to AES-GCM as authenticated data during encrypt/decrypt.
326/// This dual binding ensures AAD provides both key separation and ciphertext
327/// authentication.
328///
329/// `binding` mixes the recipient public key, the ephemeral public key, and
330/// the raw KEM ciphertext into the same HKDF info — see
331/// [`DerivationBinding`] for the rationale (Pattern 7 / HPKE §5.1).
332///
333/// # Errors
334///
335/// Returns an error if the shared secret is not exactly 32 or 64 bytes,
336/// or if HKDF expansion fails.
337pub fn derive_encryption_key(
338    shared_secret: &[u8],
339    context: &HybridEncryptionContext,
340    binding: &DerivationBinding<'_>,
341) -> Result<Zeroizing<[u8; 32]>, HybridEncryptionError> {
342    if shared_secret.len() != 32 && shared_secret.len() != 64 {
343        return Err(HybridEncryptionError::KdfError(
344            "Shared secret must be 32 bytes (ML-KEM) or 64 bytes (hybrid)".to_string(),
345        ));
346    }
347
348    // Reject inputs that would overflow our length prefixes. u32 covers every
349    // realistic field, and the early check keeps the `as u32` conversions
350    // below non-truncating.
351    let oversize = context.info.len() > u32::MAX as usize
352        || context.aad.len() > u32::MAX as usize
353        || binding.recipient_static_pk.len() > u32::MAX as usize
354        || binding.ephemeral_pk.len() > u32::MAX as usize
355        || binding.kem_ciphertext.len() > u32::MAX as usize;
356    if oversize {
357        return Err(HybridEncryptionError::KdfError(
358            "HKDF info / aad / binding field exceeds 2^32 bytes".to_string(),
359        ));
360    }
361
362    // Build a length-prefixed HKDF info payload to prevent canonicalization
363    // collisions between the variable-length fields. Naive concatenation
364    // `f1 || "||" || f2` is ambiguous when source data can contain the
365    // separator bytes; length-prefixing is the standard fix.
366    //
367    // Construction is HPKE-inspired (RFC 9180 §5.1) but **NOT bit-compatible
368    // with HPKE LabeledExtract**: HPKE uses `I2OSP(len, 2)` (2-byte big-
369    // endian) length prefixes capped at 65 535 bytes. We use 4-byte big-
370    // endian (`u32`) prefixes because `HybridEncryptionContext::aad` has a
371    // `MAX_AAD_LEN = 64 KiB` ceiling that already touches HPKE's `u16` cap,
372    // and `info` has no documented ceiling at all. A third party building
373    // an interop derivation MUST mirror this 4-byte prefix layout — they
374    // cannot copy HPKE LabeledExtract directly and expect the same `info`
375    // blob.
376    //
377    // Wire layout (each `len: u32 BE` is followed by `len` bytes):
378    //   [info_len][info]
379    //   [aad_len][aad]
380    //   [recipient_pk_len][recipient_pk]   ── Pattern 7 / channel binding
381    //   [ephemeral_pk_len][ephemeral_pk]   ── ditto
382    //   [kem_ct_len][kem_ct]               ── ditto
383    //
384    // Order is fixed; reordering is a wire-format break.
385    let segments: [&[u8]; 5] = [
386        &context.info,
387        &context.aad,
388        binding.recipient_static_pk,
389        binding.ephemeral_pk,
390        binding.kem_ciphertext,
391    ];
392    let total: usize = segments
393        .iter()
394        .try_fold(0usize, |acc, s| acc.checked_add(4)?.checked_add(s.len()))
395        .ok_or_else(|| {
396            HybridEncryptionError::KdfError("HKDF info payload size overflow".to_string())
397        })?;
398    let mut info = Vec::with_capacity(total);
399    for segment in segments {
400        // Safe: bounds-checked above — `try_from` cannot fail after the
401        // u32::MAX guards, but we use it to silence
402        // `cast_possible_truncation` and prove the non-truncation
403        // invariant to the compiler.
404        let len_u32 = u32::try_from(segment.len()).map_err(|_e| {
405            HybridEncryptionError::KdfError("HKDF info segment exceeds 2^32 bytes".to_string())
406        })?;
407        info.extend_from_slice(&len_u32.to_be_bytes());
408        info.extend_from_slice(segment);
409    }
410
411    // HKDF-SHA256 via primitives wrapper (backed by aws-lc-rs HMAC).
412    // Zero-length salt (`None`) matches HPKE (RFC 9180 §5.1). RFC 5869 §2.2
413    // permits zero salt when IKM is already uniformly random; `shared_secret`
414    // here is a KEM/DH output (32 B ML-KEM or 64 B ML-KEM || ECDH), so
415    // Extract's salt is not doing entropy extraction. Domain separation,
416    // AAD/context binding, and PK / ciphertext binding all live in `info`.
417    let hkdf_result = hkdf(shared_secret, None, Some(&info), 32).map_err(|_e| {
418        log_crypto_operation_error!(op::HYBRID_DERIVE_KEY, "HKDF failed");
419        HybridEncryptionError::KdfError("KDF failed".to_string())
420    })?;
421
422    let mut key = Zeroizing::new([0u8; 32]);
423    key.copy_from_slice(hkdf_result.expose_secret());
424    Ok(key)
425}

427/// True hybrid encryption using ML-KEM-768 + X25519 + AES-256-GCM.
428///
429/// This function performs real hybrid key encapsulation (combining post-quantum
430/// ML-KEM with classical X25519 ECDH via HKDF) before AES-256-GCM encryption.
431/// Security holds if *either* ML-KEM or X25519 remains secure.
432///
433/// # Security
434///
435/// - Hybrid security: breaks only if BOTH ML-KEM and X25519 are compromised
436/// - AES-256-GCM nonce generated internally from OS CSPRNG (SP 800-38D §8.2)
437/// - HKDF dual-PRF combiner with domain separation (SP 800-56C)
438/// - All shared secrets are zeroized after key derivation
439///
440/// # Errors
441///
442/// Returns an error if:
443/// - Hybrid KEM encapsulation fails
444/// - Key derivation fails
445/// - AES-GCM encryption fails
446pub fn encrypt_hybrid(
447    hybrid_pk: &HybridKemPublicKey,
448    plaintext: &[u8],
449    context: Option<&HybridEncryptionContext>,
450) -> Result<HybridCiphertext, HybridEncryptionError> {
451    let default_ctx = HybridEncryptionContext::default();
452    let ctx = context.unwrap_or(&default_ctx);
453
454    // Encrypt-side opacity (defense-in-depth per Pattern 6). Opaque returned
455    // error; tracing::debug! keeps the specific reason for operator debugging.
456    let opaque_kem = || HybridEncryptionError::KemError("encapsulation failed".to_string());
457    let opaque_enc = || HybridEncryptionError::EncryptionError("encryption failed".to_string());
458
459    // DoS bound on AAD: rejected up-front before any KEM / KDF / AEAD
460    // work runs, so an oversized AAD cannot consume cryptographic budget.
461    // See `HybridEncryptionContext::MAX_AAD_LEN`. Round-20 audit fix #4:
462    // collapse to opaque InvalidInput (matching decrypt-side opacity at
463    // `decrypt_hybrid:538`) so the encrypt path doesn't disclose the
464    // observed AAD length and the cap as a probable-bound oracle. Source-
465    // side cause goes to `tracing::debug!` for operator debugging.
466    if ctx.aad.len() > HybridEncryptionContext::MAX_AAD_LEN {
467        log_crypto_operation_error!(op::HYBRID_ENCRYPT, "AAD exceeds MAX_AAD_LEN");
468        return Err(HybridEncryptionError::InvalidInput("invalid input".to_string()));
469    }
470
471    // Hybrid KEM encapsulation (ML-KEM-768 + X25519 ECDH + HKDF)
472    let encapsulated = kem_hybrid::encapsulate(hybrid_pk).map_err(|_e| {
473        log_crypto_operation_error!(op::HYBRID_ENCRYPT, "KEM encapsulation failed");
474        opaque_kem()
475    })?;
476
477    // Derive AES-256 encryption key from 64-byte hybrid shared secret.
478    // Pattern 7 / HPKE: bind recipient PK + ephemeral PK + KEM ciphertext
479    // into the HKDF info, on top of the upstream combiner's binding.
480    let binding = DerivationBinding {
481        recipient_static_pk: hybrid_pk.ecdh_pk(),
482        ephemeral_pk: encapsulated.ecdh_pk(),
483        kem_ciphertext: encapsulated.ml_kem_ct(),
484    };
485    let encryption_key = derive_encryption_key(encapsulated.expose_secret(), ctx, &binding)?;
486
487    // Generate random nonce for AES-GCM via the primitives layer.
488    let nonce_bytes = AesGcm256::generate_nonce();
489
490    // AES-256-GCM via primitives wrapper.
491    let cipher = AesGcm256::new(&*encryption_key).map_err(|_e| {
492        log_crypto_operation_error!(op::HYBRID_ENCRYPT, "AES-256 init failed");
493        opaque_enc()
494    })?;
495    let (ciphertext, tag) =
496        cipher.encrypt(&nonce_bytes, plaintext, Some(&ctx.aad)).map_err(|_e| {
497            log_crypto_operation_error!(op::HYBRID_ENCRYPT, "AES-GCM seal failed");
498            opaque_enc()
499        })?;
500
501    Ok(HybridCiphertext::new(
502        encapsulated.ml_kem_ct().to_vec(),
503        encapsulated.ecdh_pk().to_vec(),
504        ciphertext,
505        nonce_bytes.to_vec(),
506        tag.to_vec(),
507    ))
508}

510/// True hybrid decryption using ML-KEM + X25519 + AES-256-GCM.
511///
512/// This function performs real hybrid key decapsulation (ML-KEM decapsulation +
513/// X25519 ECDH agreement, combined via HKDF) before AES-256-GCM decryption.
514/// The ML-KEM security level is determined by the secret key.
515///
516/// # Errors
517///
518/// Returns an error if:
519/// - The ciphertext components have invalid lengths
520/// - Hybrid KEM decapsulation fails
521/// - Key derivation fails
522/// - AES-GCM decryption or authentication fails
523pub fn decrypt_hybrid(
524    hybrid_sk: &HybridKemSecretKey,
525    ciphertext: &HybridCiphertext,
526    context: Option<&HybridEncryptionContext>,
527) -> Result<Zeroizing<Vec<u8>>, HybridEncryptionError> {
528    let default_ctx = HybridEncryptionContext::default();
529    let ctx = context.unwrap_or(&default_ctx);
530
531    // Pattern 6: every adversary-reachable failure path collapses to the
532    // same opaque returned error. The KEM-ciphertext length check used to
533    // be its own InvalidInput variant naming `expected_ct_size` — that
534    // disclosed the recipient's ML-KEM security level (512 / 768 / 1024)
535    // by error-message inspection. Now folded into the opaque block with
536    // an internal trace under `op::HYBRID_DECRYPT` so operators can
537    // correlate via `correlation_id`. Nonce / tag / ECDH-PK lengths are
538    // also folded for consistency, even though they are public protocol
539    // constants — uniform handling removes the variant-discrimination
540    // oracle entirely.
541    let opaque = || HybridEncryptionError::DecryptionError("decryption failed".to_string());
542
543    // DoS bound on AAD: rejected up-front before any KEM / KDF / AEAD
544    // work runs. Symmetric with the encrypt-side check; folded into the
545    // opaque path so an adversary cannot probe for an AAD-length oracle.
546    if ctx.aad.len() > HybridEncryptionContext::MAX_AAD_LEN {
547        log_crypto_operation_error!(op::HYBRID_DECRYPT, "AAD exceeds MAX_AAD_LEN");
548        return Err(opaque());
549    }
550
551    // Validate ciphertext shape — opaque (no `expected_ct_size` leak) +
552    // per-stage internal trace.
553    let expected_ct_size = hybrid_sk.security_level().ciphertext_size();
554    if ciphertext.kem_ciphertext().len() != expected_ct_size {
555        log_crypto_operation_error!(op::HYBRID_DECRYPT, "KEM ciphertext length mismatch");
556        return Err(opaque());
557    }
558    if ciphertext.ecdh_ephemeral_pk().len() != 32 {
559        log_crypto_operation_error!(op::HYBRID_DECRYPT, "ECDH ephemeral PK length invalid");
560        return Err(opaque());
561    }
562    if ciphertext.nonce().len() != 12 {
563        log_crypto_operation_error!(op::HYBRID_DECRYPT, "AES-GCM nonce length invalid");
564        return Err(opaque());
565    }
566    if ciphertext.tag().len() != 16 {
567        log_crypto_operation_error!(op::HYBRID_DECRYPT, "AES-GCM tag length invalid");
568        return Err(opaque());
569    }
570
571    // Reconstruct EncapsulatedKey for kem_hybrid::decapsulate. The
572    // shared_secret field is a placeholder here — `decapsulate` recovers
573    // the real shared secret from `sk` and the `ml_kem_ct`/`ecdh_pk`
574    // components. Use a zeroed SecretBytes<64> of the correct shape.
575    let encapsulated = EncapsulatedKey::new(
576        ciphertext.kem_ciphertext().to_vec(),
577        ciphertext.ecdh_ephemeral_pk().to_vec(),
578        crate::types::SecretBytes::zero(),
579    );
580
581    // Hybrid KEM decapsulation (ML-KEM + X25519 ECDH + HKDF)
582    let shared_secret = kem_hybrid::decapsulate(hybrid_sk, &encapsulated).map_err(|_e| {
583        log_crypto_operation_error!(op::HYBRID_DECRYPT, "KEM decapsulation failed");
584        opaque()
585    })?;
586
587    // Derive AES-256 encryption key from 64-byte hybrid shared secret.
588    // Pattern 7 / HPKE: bind recipient PK + ephemeral PK + KEM ciphertext
589    // into the HKDF info (must match the encrypt-side binding exactly).
590    let recipient_static_pk = hybrid_sk.ecdh_public_key_bytes();
591    let binding = DerivationBinding {
592        recipient_static_pk: &recipient_static_pk,
593        ephemeral_pk: ciphertext.ecdh_ephemeral_pk(),
594        kem_ciphertext: ciphertext.kem_ciphertext(),
595    };
596    let encryption_key = derive_encryption_key(shared_secret.expose_secret(), ctx, &binding)
597        .map_err(|_e| {
598            log_crypto_operation_error!(op::HYBRID_DECRYPT, "HKDF key derivation failed");
599            opaque()
600        })?;
601
602    let nonce_bytes: [u8; NONCE_LEN] = ciphertext.nonce().try_into().map_err(|_e| {
603        log_crypto_operation_error!(op::HYBRID_DECRYPT, "nonce length != 12");
604        opaque()
605    })?;
606    let tag_bytes: [u8; TAG_LEN] = ciphertext.tag().try_into().map_err(|_e| {
607        log_crypto_operation_error!(op::HYBRID_DECRYPT, "tag length != 16");
608        opaque()
609    })?;
610
611    // AES-256-GCM via primitives wrapper.
612    let cipher = AesGcm256::new(&*encryption_key).map_err(|_e| {
613        log_crypto_operation_error!(op::HYBRID_DECRYPT, "AES-256 init failed");
614        opaque()
615    })?;
616    // `cipher.decrypt` already returns `Zeroizing<Vec<u8>>` — propagate directly.
617    let plaintext = cipher
618        .decrypt(&nonce_bytes, ciphertext.symmetric_ciphertext(), &tag_bytes, Some(&ctx.aad))
619        .map_err(|_aead_err| {
620            log_crypto_operation_error!(op::HYBRID_DECRYPT, "AEAD authentication failed");
621            opaque()
622        })?;
623
624    Ok(plaintext)
625}
