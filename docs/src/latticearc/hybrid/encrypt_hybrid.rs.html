6//! Hybrid Encryption Module
7//!
8//! This module provides hybrid encryption combining post-quantum (ML-KEM) key
9//! encapsulation with AES-256-GCM symmetric encryption for quantum-resistant
10//! data encryption with classical performance characteristics.
11//!
12//! # Overview
13//!
14//! The hybrid encryption scheme uses:
15//! - **ML-KEM-768** (FIPS 203) for post-quantum key encapsulation
16//! - **AES-256-GCM** for authenticated symmetric encryption
17//! - **HKDF-SHA256** for key derivation with domain separation
18//!
19//! # Security Properties
20//!
21//! - IND-CCA2 security from ML-KEM
22//! - Authenticated encryption with associated data (AEAD)
23//! - Domain separation via HPKE-style key derivation
24//!
25//! # Example
26//!
27//! ```rust,no_run
28//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
29//! use latticearc::hybrid::encrypt_hybrid::{
30//!     encrypt_hybrid, decrypt_hybrid, HybridEncryptionContext,
31//! };
32//! use latticearc::hybrid::kem_hybrid;
33//! use latticearc::primitives::kem::ml_kem::MlKemSecurityLevel;
34//!
35//! // Generate a hybrid (ML-KEM + X25519) keypair at the desired security level.
36//! let (hybrid_pk, hybrid_sk) =
37//!     kem_hybrid::generate_keypair_with_level(MlKemSecurityLevel::MlKem768)?;
38//!
39//! let plaintext = b"Secret message";
40//! let context = HybridEncryptionContext::default();
41//!
42//! let ciphertext = encrypt_hybrid(&hybrid_pk, plaintext, Some(&context))?;
43//! let decrypted = decrypt_hybrid(&hybrid_sk, &ciphertext, Some(&context))?;
44//! # Ok(())
45//! # }
46//! ```

55/// Error types for hybrid encryption operations.
56///
57/// This enum captures all possible error conditions that can occur during
58/// hybrid encryption and decryption operations.
59#[non_exhaustive]
60#[derive(Debug, Clone, PartialEq, Eq, Error)]
61pub enum HybridEncryptionError {
62    /// Error during key encapsulation mechanism operations.
63    #[error("KEM error: {0}")]
64    KemError(String),
65    /// Error during symmetric encryption.
66    #[error("Encryption error: {0}")]
67    EncryptionError(String),
68    /// Error during symmetric decryption or authentication failure.
69    #[error("Decryption error: {0}")]
70    DecryptionError(String),
71    /// Error during key derivation function operations.
72    #[error("Key derivation error: {0}")]
73    KdfError(String),
74    /// Invalid input parameters provided to the operation.
75    #[error("Invalid input: {0}")]
76    InvalidInput(String),
77    /// Key length mismatch error.
78    #[error("Key length error: expected {expected}, got {actual}")]
79    KeyLengthError {
80        /// Expected key length in bytes.
81        expected: usize,
82        /// Actual key length provided.
83        actual: usize,
84    },
85}

87/// Hybrid ciphertext containing both KEM and symmetric encryption components.
88///
89/// This structure holds all the data needed to decrypt a hybrid-encrypted message:
90/// - The KEM ciphertext for key decapsulation
91/// - The symmetric ciphertext containing the encrypted message
92/// - The nonce used for AES-GCM encryption
93/// - The authentication tag for integrity verification
94#[derive(Debug, Clone)]
95pub struct HybridCiphertext {

99    ecdh_ephemeral_pk: Vec<u8>,

101    symmetric_ciphertext: Vec<u8>,
102    /// 12-byte nonce used for AES-GCM encryption.

104    /// 16-byte AES-GCM authentication tag.
105    tag: Vec<u8>,
106}

187/// HPKE-style context information for hybrid encryption.
188///
189/// This structure provides domain separation and additional authenticated data
190/// for the key derivation and encryption operations, following RFC 9180 (HPKE).
191#[derive(Debug, Clone)]
192pub struct HybridEncryptionContext {
193    /// Application-specific info string for key derivation domain separation.
194    pub info: Vec<u8>,
195    /// Additional authenticated data (AAD) for AEAD encryption.
196    pub aad: Vec<u8>,
197}

205/// HPKE-style key derivation for hybrid encryption.
206///
207/// Delegates to the primitives HKDF wrapper (`primitives::kdf::hkdf::hkdf`)
208/// which is backed by aws-lc-rs HMAC-SHA256 under the hood, so all callers share
209/// the same FIPS-validated implementation, zeroization guarantees, and
210/// instrumentation.
211///
212/// # Security
213///
214/// The context's AAD is bound in two places: (1) mixed into the HKDF info
215/// parameter here, so different AAD values derive different encryption keys,
216/// and (2) passed to AES-GCM as authenticated data during encrypt/decrypt.
217/// This dual binding ensures AAD provides both key separation and ciphertext
218/// authentication.
219///
220/// # Errors
221///
222/// Returns an error if the shared secret is not exactly 32 or 64 bytes,
223/// or if HKDF expansion fails.
224pub fn derive_encryption_key(
225    shared_secret: &[u8],
226    context: &HybridEncryptionContext,
227) -> Result<Zeroizing<[u8; 32]>, HybridEncryptionError> {
228    if shared_secret.len() != 32 && shared_secret.len() != 64 {
229        return Err(HybridEncryptionError::KdfError(
230            "Shared secret must be 32 bytes (ML-KEM) or 64 bytes (hybrid)".to_string(),
231        ));
232    }
233
234    // Reject inputs that would overflow our length prefixes. u32 covers every
235    // realistic context, and the early check keeps the `as u32` conversions
236    // below non-truncating.
237    if context.info.len() > u32::MAX as usize || context.aad.len() > u32::MAX as usize {
238        return Err(HybridEncryptionError::KdfError(
239            "HKDF info or AAD field exceeds 2^32 bytes".to_string(),
240        ));
241    }
242
243    // Build a length-prefixed HKDF info payload to prevent canonicalization
244    // collisions between the two variable-length fields (info, aad). Naive
245    // concatenation `info || "||" || aad` is ambiguous when the source data
246    // can contain the separator bytes; length-prefixing is the standard fix
247    // (HPKE §5.1, RFC 9180 "LabeledExtract").
248    //
249    // Wire layout: [info_len: u32 BE][info][aad_len: u32 BE][aad]
250    let info_len = context.info.len();
251    let aad_len = context.aad.len();
252    let total = 4usize.saturating_add(info_len).saturating_add(4).saturating_add(aad_len);
253    let mut info = Vec::with_capacity(total);
254    // Safe: bounds-checked above — `try_from` cannot fail after the length
255    // guards, but we use it to silence `cast_possible_truncation` and prove
256    // the non-truncation invariant to the compiler.
257    let info_len_u32 = u32::try_from(info_len).map_err(|_e| {
258        HybridEncryptionError::KdfError("HKDF info field exceeds 2^32 bytes".to_string())
259    })?;
260    let aad_len_u32 = u32::try_from(aad_len).map_err(|_e| {
261        HybridEncryptionError::KdfError("HKDF AAD field exceeds 2^32 bytes".to_string())
262    })?;
263    info.extend_from_slice(&info_len_u32.to_be_bytes());
264    info.extend_from_slice(&context.info);
265    info.extend_from_slice(&aad_len_u32.to_be_bytes());
266    info.extend_from_slice(&context.aad);
267
268    // HKDF-SHA256 via primitives wrapper (backed by aws-lc-rs HMAC).
269    let hkdf_result = hkdf(shared_secret, None, Some(&info), 32)
270        .map_err(|e| HybridEncryptionError::KdfError(format!("HKDF failed: {e}")))?;
271
272    let mut key = Zeroizing::new([0u8; 32]);
273    key.copy_from_slice(hkdf_result.key());
274    Ok(key)
275}

277/// True hybrid encryption using ML-KEM-768 + X25519 + AES-256-GCM.
278///
279/// This function performs real hybrid key encapsulation (combining post-quantum
280/// ML-KEM with classical X25519 ECDH via HKDF) before AES-256-GCM encryption.
281/// Security holds if *either* ML-KEM or X25519 remains secure.
282///
283/// # Security
284///
285/// - Hybrid security: breaks only if BOTH ML-KEM and X25519 are compromised
286/// - AES-256-GCM nonce generated internally from OS CSPRNG (SP 800-38D §8.2)
287/// - HKDF dual-PRF combiner with domain separation (SP 800-56C)
288/// - All shared secrets are zeroized after key derivation
289///
290/// # Errors
291///
292/// Returns an error if:
293/// - Hybrid KEM encapsulation fails
294/// - Key derivation fails
295/// - AES-GCM encryption fails
296pub fn encrypt_hybrid(
297    hybrid_pk: &HybridKemPublicKey,
298    plaintext: &[u8],
299    context: Option<&HybridEncryptionContext>,
300) -> Result<HybridCiphertext, HybridEncryptionError> {
301    let default_ctx = HybridEncryptionContext::default();
302    let ctx = context.unwrap_or(&default_ctx);
303
304    // Hybrid KEM encapsulation (ML-KEM-768 + X25519 ECDH + HKDF)
305    let encapsulated = kem_hybrid::encapsulate(hybrid_pk)
306        .map_err(|e| HybridEncryptionError::KemError(format!("{e}")))?;
307
308    // Derive AES-256 encryption key from 64-byte hybrid shared secret
309    let encryption_key = derive_encryption_key(encapsulated.shared_secret(), ctx)?;
310
311    // Generate random nonce for AES-GCM via the primitives layer.
312    let nonce_bytes = AesGcm256::generate_nonce();
313
314    // AES-256-GCM via primitives wrapper.
315    let cipher = AesGcm256::new(&*encryption_key).map_err(|e| {
316        HybridEncryptionError::EncryptionError(format!("Failed to create AES key: {e}"))
317    })?;
318    let (ciphertext, tag) =
319        cipher.encrypt(&nonce_bytes, plaintext, Some(&ctx.aad)).map_err(|e| {
320            HybridEncryptionError::EncryptionError(format!("AES-GCM encryption failed: {e}"))
321        })?;
322
323    Ok(HybridCiphertext::new(
324        encapsulated.ml_kem_ct().to_vec(),
325        encapsulated.ecdh_pk().to_vec(),
326        ciphertext,
327        nonce_bytes.to_vec(),
328        tag.to_vec(),
329    ))
330}

332/// True hybrid decryption using ML-KEM + X25519 + AES-256-GCM.
333///
334/// This function performs real hybrid key decapsulation (ML-KEM decapsulation +
335/// X25519 ECDH agreement, combined via HKDF) before AES-256-GCM decryption.
336/// The ML-KEM security level is determined by the secret key.
337///
338/// # Errors
339///
340/// Returns an error if:
341/// - The ciphertext components have invalid lengths
342/// - Hybrid KEM decapsulation fails
343/// - Key derivation fails
344/// - AES-GCM decryption or authentication fails
345pub fn decrypt_hybrid(
346    hybrid_sk: &HybridKemSecretKey,
347    ciphertext: &HybridCiphertext,
348    context: Option<&HybridEncryptionContext>,
349) -> Result<Zeroizing<Vec<u8>>, HybridEncryptionError> {
350    let default_ctx = HybridEncryptionContext::default();
351    let ctx = context.unwrap_or(&default_ctx);
352
353    // Validate ciphertext structure against the secret key's security level
354    let expected_ct_size = hybrid_sk.security_level().ciphertext_size();
355    if ciphertext.kem_ciphertext().len() != expected_ct_size {
356        return Err(HybridEncryptionError::InvalidInput(format!(
357            "{} ciphertext must be {} bytes, got {}",
358            hybrid_sk.security_level().name(),
359            expected_ct_size,
360            ciphertext.kem_ciphertext().len()
361        )));
362    }
363    if ciphertext.ecdh_ephemeral_pk().len() != 32 {
364        return Err(HybridEncryptionError::InvalidInput(
365            "X25519 ephemeral public key must be 32 bytes".to_string(),
366        ));
367    }
368    if ciphertext.nonce().len() != 12 {
369        return Err(HybridEncryptionError::InvalidInput("Nonce must be 12 bytes".to_string()));
370    }
371    if ciphertext.tag().len() != 16 {
372        return Err(HybridEncryptionError::InvalidInput("Tag must be 16 bytes".to_string()));
373    }
374
375    // Reconstruct EncapsulatedKey for kem_hybrid::decapsulate
376    let encapsulated = EncapsulatedKey::new(
377        ciphertext.kem_ciphertext().to_vec(),
378        ciphertext.ecdh_ephemeral_pk().to_vec(),
379        Zeroizing::new(vec![]), // placeholder — decapsulate recovers this
380    );
381
382    // Hybrid KEM decapsulation (ML-KEM + X25519 ECDH + HKDF)
383    let shared_secret = kem_hybrid::decapsulate(hybrid_sk, &encapsulated)
384        .map_err(|e| HybridEncryptionError::DecryptionError(format!("{e}")))?;
385
386    // Derive AES-256 encryption key from 64-byte hybrid shared secret
387    let encryption_key = derive_encryption_key(&shared_secret, ctx)?;
388
389    let nonce_bytes: [u8; NONCE_LEN] = ciphertext.nonce().try_into().map_err(|e| {
390        HybridEncryptionError::DecryptionError(format!("Invalid nonce length: {e}"))
391    })?;
392    let tag_bytes: [u8; TAG_LEN] = ciphertext
393        .tag()
394        .try_into()
395        .map_err(|e| HybridEncryptionError::DecryptionError(format!("Invalid tag length: {e}")))?;
396
397    // AES-256-GCM via primitives wrapper.
398    let cipher = AesGcm256::new(&*encryption_key).map_err(|e| {
399        HybridEncryptionError::DecryptionError(format!("Failed to create AES key: {e}"))
400    })?;
401    // `cipher.decrypt` already returns `Zeroizing<Vec<u8>>` — propagate directly.
402    let plaintext = cipher
403        .decrypt(&nonce_bytes, ciphertext.symmetric_ciphertext(), &tag_bytes, Some(&ctx.aad))
404        .map_err(|_aead_err| {
405            // SECURITY: Opaque error per SP 800-38D §5.2.2 — do not propagate
406            // the underlying error to prevent padding/MAC oracle attacks.
407            HybridEncryptionError::DecryptionError("hybrid decryption failed".to_string())
408        })?;
409
410    Ok(plaintext)
411}
