1//! Atomic + permission-restricted file writes.
2//!
3//! Centralises the "write secret material to disk safely" pattern used
4//! by the keyfile writer and the CLI output paths. The implementation:
5//!
6//! 1. Creates a `tempfile::NamedTempFile` in the same directory as the
7//!    target (so the eventual rename is same-filesystem and atomic).
8//! 2. Writes the bytes to the tempfile.
9//! 3. On Unix, sets mode `0o600` BEFORE the rename so there's no window
10//!    where the file is world-readable.
11//! 4. On Windows, the tempfile is created with restrictive ACL by
12//!    `tempfile`'s NTFS-aware path (DACL inheriting only the creator,
13//!    matching the secret-key threat model). The eventual rename
14//!    preserves the ACL.
15//! 5. `persist` performs an atomic rename via `rename(2)` / `MoveFileExW`.
16//!    Either the prior file is replaced wholesale or it stays untouched —
17//!    no partial-write window. The default `persist` REJECTS pre-existing
18//!    targets via `O_EXCL`-equivalent semantics, closing the silent-clobber
19//!    vector. Use [`AtomicWrite::overwrite_existing`] to opt into
20//!    overwrite (matching `std::fs::write` behaviour) when that's the
21//!    desired contract.
22//!
23//! See `docs/DESIGN_PATTERNS.md` Pattern 6 (no symlink-followed writes —
24//! `OpenOptions::create_new(true)` and tempfile+rename both refuse to
25//! follow symlinks, so this helper closes the symlink-clobber vector for
26//! free) and SECURITY.md for the threat model.
27

33use crate::unified_api::error::{CoreError, Result};
34use std::io::Write;
35use std::path::Path;

37/// Builder for an atomic, permission-restricted write.
38///
39/// # Example
40///
41/// ```rust,no_run
42/// # use latticearc::unified_api::atomic_write::AtomicWrite;
43/// # use std::path::Path;
44/// # fn run() -> Result<(), Box<dyn std::error::Error>> {
45/// // Write a secret-key file: refuse to clobber, lock to 0600 on Unix.
46/// AtomicWrite::new(b"secret bytes")
47///     .secret_mode()
48///     .write(Path::new("./identity.sec.json"))?;
49/// # Ok(())
50/// # }
51/// ```
52pub struct AtomicWrite<'a> {
53    bytes: &'a [u8],
54    /// On Unix, set the resulting file's mode to this. None = inherit
55    /// process umask. Set by [`AtomicWrite::secret_mode`] to `0o600`.
56    unix_mode: Option<u32>,
57    /// On Windows, replace the post-rename file's DACL with the
58    /// workspace owner-only policy. Set by
59    /// [`AtomicWrite::secret_mode`]. The Unix counterpart is
60    /// `unix_mode = Some(0o600)`; both are populated by `secret_mode`
61    /// so callers don't have to remember a cross-platform pair.
62    harden_windows_acl: bool,
63    /// If true, an existing target file is overwritten (atomic-rename
64    /// style — the prior file is replaced wholesale, not truncated-then-
65    /// written). If false (default), an existing target causes the
66    /// write to fail with `CoreError::ConfigurationError` carrying a
67    /// "file already exists" message — the caller should report this
68    /// upward and NOT clobber the user's prior key.
69    overwrite_existing: bool,
70}
71
72impl<'a> AtomicWrite<'a> {
73    /// Stage bytes for an atomic write.
74    #[must_use]
75    pub fn new(bytes: &'a [u8]) -> Self {
76        Self { bytes, unix_mode: None, harden_windows_acl: false, overwrite_existing: false }
77    }
78
79    /// Convenience for the most common case: write secret bytes to
80    /// `path` with mode `0o600`, refusing to clobber. Equivalent to
81    /// `AtomicWrite::new(bytes).secret_mode().write(path)`.
82    ///
83    /// # Errors
84    ///
85    /// Same as [`write`](Self::write).
86    pub fn write_secret(bytes: &[u8], path: &Path) -> Result<()> {
87        AtomicWrite::new(bytes).secret_mode().write(path)
88    }
89
90    /// Convenience for "overwrite an existing file atomically" — used
91    /// for non-secret CLI output (encrypted blobs, signatures) where
92    /// the user reasonably expects re-running the command to replace
93    /// prior output. Mode left at process umask.
94    ///
95    /// # Errors
96    ///
97    /// Same as [`write`](Self::write).
98    pub fn write_overwrite(bytes: &[u8], path: &Path) -> Result<()> {
99        AtomicWrite::new(bytes).overwrite_existing(true).write(path)
100    }
101
102    /// Mark this write as a secret-bearing file.
103    ///
104    /// On Unix this sets the mode to `0o600` (owner read+write only)
105    /// atomically before any bytes are written — the file is never
106    /// world-readable, even briefly.
107    ///
108    /// On Windows this sets `harden_windows_acl = true`; after the
109    /// rename completes, [`set_local_admin_dacl`] replaces the file's
110    /// inherited DACL with the workspace's owner-only policy. The
111    /// rename target inherits the parent directory's ACL on creation,
112    /// which on a default user profile typically grants `Users:Read`
113    /// — without this hardening every secret key file would lose its
114    /// confidentiality protection on Windows.
115    #[must_use]
116    pub fn secret_mode(mut self) -> Self {
117        self.unix_mode = Some(0o600);
118        self.harden_windows_acl = true;
119        self
120    }
121
122    /// Set an arbitrary Unix file mode. Convenience for non-secret
123    /// outputs that still want a tighter-than-umask mode.
124    #[must_use]
125    pub fn unix_mode(mut self, mode: u32) -> Self {
126        self.unix_mode = Some(mode);
127        self
128    }
129
130    /// Allow the write to overwrite a pre-existing file at the target
131    /// path. The default refuses to clobber; opt in here only when the
132    /// caller explicitly intends to replace prior content (e.g., a
133    /// `--force` flag was given).
134    #[must_use]
135    pub fn overwrite_existing(mut self, allow: bool) -> Self {
136        self.overwrite_existing = allow;
137        self
138    }
139
140    /// Perform the write to `path`.
141    ///
142    /// # Errors
143    ///
144    /// - `CoreError::ConfigurationError` if the parent directory doesn't
145    ///   exist, the target file exists and `overwrite_existing(true)`
146    ///   wasn't called, or the tempfile cannot be created in the parent
147    ///   directory.
148    /// - `CoreError::Internal` for I/O errors during write or rename.
149    pub fn write(self, path: &Path) -> Result<()> {
150        let parent =
151            path.parent().filter(|p| !p.as_os_str().is_empty()).unwrap_or_else(|| Path::new("."));
152
153        // tempfile::NamedTempFile in the parent dir → same-filesystem
154        // rename. Cross-fs rename would degrade to a non-atomic copy.
155        let mut tmp = tempfile::NamedTempFile::new_in(parent).map_err(|e| {
156            CoreError::ConfigurationError(format!(
157                "failed to create tempfile in {}: {e}",
158                parent.display()
159            ))
160        })?;
161
162        #[cfg(unix)]
163        if let Some(mode) = self.unix_mode {
164            // Set the mode BEFORE writing so there's no window where the
165            // file is world-readable. `as_file()` returns the underlying
166            // `File`; `set_permissions` is a syscall, not a buffered op.
167            use std::os::unix::fs::PermissionsExt;
168            let perms = std::fs::Permissions::from_mode(mode);
169            tmp.as_file()
170                .set_permissions(perms)
171                .map_err(|e| CoreError::Internal(format!("chmod tempfile: {e}")))?;
172        }
173
174        tmp.write_all(self.bytes)
175            .map_err(|e| CoreError::Internal(format!("write tempfile: {e}")))?;
176        tmp.as_file()
177            .sync_all()
178            .map_err(|e| CoreError::Internal(format!("fsync tempfile: {e}")))?;
179
180        // Windows DACL hardening — apply BEFORE the atomic rename so
181        // there's no window where the post-rename file at `path`
182        // exists with the parent dir's permissive default DACL (round-
183        // 39 H2 race). NTFS object security descriptors travel with
184        // the inode/MFT entry, so a same-volume rename is just a
185        // directory-entry update — the hardened DACL we apply to the
186        // tempfile here persists across `tmp.persist()` to the final
187        // path. A failure here is fatal: silently leaving the secret-
188        // key file world-readable is worse than failing the write.
189        if self.harden_windows_acl
190            && let Err(e) = crate::unified_api::set_local_admin_dacl(tmp.path())
191        {
192            // Best-effort secret-bytes scrub before `tmp` drops and
193            // unlinks. The plaintext bytes were already written above;
194            // if we return the error directly, NamedTempFile's drop
195            // just unlinks — which on NTFS releases cluster runs
196            // without zeroizing them, leaving the secret recoverable
197            // from raw disk until those clusters get reallocated.
198            //
199            // Sequence:
200            //   1. seek-and-overwrite with zeros for the active
201            //      length (NTFS resident-data files <~700 B are
202            //      truly zeroed; for cluster-resident files the
203            //      currently-mapped clusters are zeroed before
204            //      release)
205            //   2. sync_all so the zeros hit disk before truncate
206            //   3. set_len(0) releases the now-zeroed clusters
207            // All operations are best-effort (`let _ =`) — we are
208            // already in the error-return path; secondary I/O
209            // failures here cannot improve the user-visible error
210            // and `tmp`'s drop will still unlink either way.
211            use std::io::{Seek, SeekFrom, Write};
212            const ZERO_CHUNK: [u8; 4096] = [0u8; 4096];
213            let f = tmp.as_file_mut();
214            let _ = f.seek(SeekFrom::Start(0));
215            let mut remaining = self.bytes.len();
216            while remaining > 0 {
217                let n = remaining.min(ZERO_CHUNK.len());
218                #[allow(clippy::indexing_slicing)]
219                // SAFETY: `n = remaining.min(ZERO_CHUNK.len())`
220                // ⇒ `n ≤ ZERO_CHUNK.len()` ⇒ `&ZERO_CHUNK[..n]` is
221                // in-bounds. (no possible panic)
222                let chunk = &ZERO_CHUNK[..n];
223                if f.write_all(chunk).is_err() {
224                    break;
225                }
226                remaining = remaining.saturating_sub(n);
227            }
228            let _ = f.sync_all();
229            let _ = f.set_len(0);
230            return Err(CoreError::Internal(format!(
231                "windows DACL hardening on tempfile {} failed: {e}",
232                tmp.path().display()
233            )));
234        }
235
236        // Atomic rename — the choice between `persist` and `persist_noclobber`
237        // is the difference between "atomic but clobber-OK" and "atomic AND
238        // refuse to clobber via link(2)+unlink(2)". The earlier shape used
239        // `path.exists()` + `persist`, which had a TOCTOU window: another
240        // process could create `path` between the check and the rename and
241        // get silently overwritten. `persist_noclobber` collapses both
242        // into a single syscall, so the exclusive-create guarantee is real,
243        // not best-effort.
244        if self.overwrite_existing {
245            tmp.persist(path).map_err(|e| {
246                CoreError::Internal(format!("atomic rename to {}: {e}", path.display()))
247            })?;
248        } else {
249            tmp.persist_noclobber(path).map_err(|e| {
250                // tempfile's `PersistError` carries the source io::Error;
251                // surface AlreadyExists as the user-facing
252                // ConfigurationError ("refusing to overwrite") and wrap any
253                // other I/O failure as Internal.
254                if e.error.kind() == std::io::ErrorKind::AlreadyExists {
255                    CoreError::ConfigurationError(format!(
256                        "Refusing to overwrite existing file: {}. \
257                         Pass --force (or pre-delete the file) if this is intentional.",
258                        path.display()
259                    ))
260                } else {
261                    CoreError::Internal(format!("atomic rename to {}: {e}", path.display()))
262                }
263            })?;
264        }
265
266        // fsync the parent directory so the rename is durable across a
267        // power-loss event. Without this step, on ext4/XFS with the
268        // default `data=ordered` the directory entry can be flushed after
269        // the file inode but before the rename, leaving the file
270        // unrecoverable. Best-effort on non-Unix platforms (Windows
271        // doesn't expose a parent-fsync primitive in the same way).
272        #[cfg(unix)]
273        {
274            if let Ok(dir) = std::fs::File::open(parent) {
275                let _ = dir.sync_all();
276            }
277        }
278
279        // (Windows DACL hardening was applied to the tempfile above,
280        // BEFORE persist, so the rename preserves the hardened DACL.
281        // No post-rename DACL apply is needed — closes round-39 H2.)
282        Ok(())
283    }
