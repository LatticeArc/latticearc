1//! Windows ACL hardening for secret-bearing files and directories.
2//!
3//! On Unix the crate enforces tight `mode(0o600)` (files) /
4//! `mode(0o700)` (directories) atomically at create-time via
5//! `OpenOptionsExt` / `DirBuilderExt`. Windows has no equivalent
6//! convenience; the caller must replace the file/directory's DACL
7//! after creation. This module is the single canonical entry point
8//! for that operation, used by:
9//!
10//! - [`unified_api::audit::FileAuditStorage`] when creating audit logs
11//! - [`unified_api::atomic_write::AtomicWrite::secret_mode`] when
12//!   persisting secret-key files
13//! - `latticearc-cli`'s `keygen` command when creating the output
14//!   directory
15//!
16//! All paths converge on a single SDDL string so the policy is
17//! identical across call sites: protected DACL granting full access
18//! to the file owner (current user), Built-in Administrators, and
19//! `LocalSystem` only — no inheritance from the parent directory.
20//!
21//! The implementation is gated on `cfg(windows)`. On non-Windows
22//! targets the entry points are present but compile to no-ops so the
23//! call sites stay platform-agnostic.
24
25#[cfg(windows)]
26use crate::unified_api::error::CoreError;
27use crate::unified_api::error::Result;
28use std::path::Path;
29
30/// SDDL string applied to secret-bearing files and directories.
31///
32/// - `D:P`     — protected DACL (no inheritance from parent).
33/// - `OICI`    — object-inherit + container-inherit so contents of a
34///   directory inherit the same DACL.
35/// - `FA`      — file-all rights.
36/// - `OW`      — file owner (the current user, who created the file).
37/// - `BA`      — Built-in Administrators (so an admin can recover the
38///   file even if the user account is later disabled — matches the
39///   posture Windows applies to per-user `%USERPROFILE%` files).
40/// - `SY`      — `LocalSystem` (so backup / volume-shadow services
41///   that run as `SYSTEM` can read the files; without this entry,
42///   restic / VSS-based backups silently fail).
43///
44/// No `Everyone`, no `Users`, no `Authenticated Users` entry. A
45/// second user on the same machine cannot read the file even though
46/// the parent directory may grant them traversal.
47#[cfg(windows)]
48const OWNER_ONLY_DACL_SDDL: &str = "D:P(A;OICI;FA;;;OW)(A;OICI;FA;;;BA)(A;OICI;FA;;;SY)";
49
50/// Apply the owner-only DACL to a file or directory at `path`.
51///
52/// On Windows: replaces the object's DACL with [`OWNER_ONLY_DACL_SDDL`]
53/// via `SetNamedSecurityInfoW`. The DACL is `protected` so it does
54/// not inherit any permissive ACEs from the parent (e.g. a
55/// `Users:Read` entry on the user's profile root would otherwise
56/// leak into a fresh secret file).
57///
58/// On non-Windows: no-op. Unix call sites set `mode(0o600)` /
59/// `mode(0o700)` atomically at create-time; this function is only
60/// invoked on the Windows branch.
61///
62/// # Errors
63///
64/// Returns [`CoreError::Internal`] when the SDDL fails to parse
65/// (programmer error — the constant is fixed) or when
66/// `SetNamedSecurityInfoW` rejects the call (typical: target path
67/// does not exist, or the calling process lacks `WRITE_DAC`).
68#[allow(unused_variables)] // path is unused on non-Windows
69pub fn set_owner_only_dacl(path: &Path) -> Result<()> {
70    #[cfg(windows)]
71    {
72        use std::str::FromStr;
73        use windows_permissions::{
74            LocalBox, SecurityDescriptor,
75            constants::{SeObjectType, SecurityInformation},
76            wrappers::SetNamedSecurityInfo,
77        };
78
79        // Parse the SDDL constant once per call. The crate caches the
80        // parsed object via `LocalBox<SecurityDescriptor>`, which is
81        // dropped at the end of this scope; SetNamedSecurityInfoW
82        // copies what it needs internally.
83        let sd: LocalBox<SecurityDescriptor> =
84            LocalBox::<SecurityDescriptor>::from_str(OWNER_ONLY_DACL_SDDL).map_err(|e| {
85                CoreError::Internal(format!(
86                    "win_acl: failed to parse owner-only SDDL ({}): {}",
87                    OWNER_ONLY_DACL_SDDL, e
88                ))
89            })?;
90
91        let dacl = sd.dacl().ok_or_else(|| {
92            CoreError::Internal(
93                "win_acl: parsed SecurityDescriptor had no DACL (SDDL constant is malformed)"
94                    .to_string(),
95            )
96        })?;
97
98        // `Dacl | ProtectedDacl` so the assignment also CLEARS the
99        // inheritance bit — without `ProtectedDacl` the new DACL is
100        // merged with whatever the parent inherits, which can re-add
101        // permissive ACEs.
102        SetNamedSecurityInfo(
103            path.as_os_str(),
104            SeObjectType::SE_FILE_OBJECT,
105            SecurityInformation::Dacl | SecurityInformation::ProtectedDacl,
106            None,
107            None,
108            Some(dacl),
109            None,
110        )
111        .map_err(|e| {
112            CoreError::Internal(format!(
113                "win_acl: SetNamedSecurityInfoW failed for {}: {}",
114                path.display(),
115                e
116            ))
117        })?;
118    }
119    #[cfg(not(windows))]
120    {
121        // Unix call sites set mode(0o600)/mode(0o700) atomically at
122        // create-time. This function is only invoked from the Windows
123        // branch of those call sites, so the non-Windows arm is a
124        // pure no-op rather than a fallback that could mask a missing
125        // mode-call on Unix.
126        let _ = path;
127    }
128    Ok(())
129}
