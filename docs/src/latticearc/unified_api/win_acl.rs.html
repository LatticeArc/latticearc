1//! Windows ACL hardening for secret-bearing files and directories.
2//!
3//! On Unix the crate enforces tight `mode(0o600)` (files) /
4//! `mode(0o700)` (directories) atomically at create-time via
5//! `OpenOptionsExt` / `DirBuilderExt`. Windows has no equivalent
6//! convenience; the caller must replace the file/directory's DACL
7//! after creation. This module is the single canonical entry point
8//! for that operation, used by:
9//!
10//! - [`unified_api::audit::FileAuditStorage`] when creating audit logs
11//! - [`unified_api::atomic_write::AtomicWrite::secret_mode`] when
12//!   persisting secret-key files
13//! - `latticearc-cli`'s `keygen` command when creating the output
14//!   directory
15//!
16//! All paths converge on a single SDDL string so the policy is
17//! identical across call sites: protected DACL granting full access
18//! to **three principals only** — the file owner (current user),
19//! Built-in Administrators (`BA`), and `LocalSystem` (`SY`). No
20//! inheritance from the parent directory; no entry for `Everyone`,
21//! `Users`, or `Authenticated Users`. The function is named
22//! [`set_local_admin_dacl`] (not "owner-only") to avoid promising
23//! more than the policy delivers — the `BA` entry is necessary so
24//! a local administrator can recover the file when the user account
25//! is later disabled, matching the posture Windows applies to
26//! per-user `%USERPROFILE%` files; the `SY` entry is necessary so
27//! VSS / backup services that run as `SYSTEM` can read the file.
28//!
29//! The implementation is gated on `cfg(windows)`. On non-Windows
30//! targets the entry points are present but compile to no-ops so the
31//! call sites stay platform-agnostic.
32
33#[cfg(windows)]
34use crate::unified_api::error::CoreError;
35use crate::unified_api::error::Result;
36use std::path::Path;
37
38/// SDDL string applied to secret-bearing files and directories.
39///
40/// - `D:P`     — protected DACL (no inheritance from parent).
41/// - `OICI`    — object-inherit + container-inherit so contents of a
42///   directory inherit the same DACL.
43/// - `FA`      — file-all rights.
44/// - `OW`      — file owner (the current user, who created the file).
45/// - `BA`      — Built-in Administrators (so an admin can recover the
46///   file even if the user account is later disabled — matches the
47///   posture Windows applies to per-user `%USERPROFILE%` files).
48/// - `SY`      — `LocalSystem` (so backup / volume-shadow services
49///   that run as `SYSTEM` can read the files; without this entry,
50///   restic / VSS-based backups silently fail).
51///
52/// No `Everyone`, no `Users`, no `Authenticated Users` entry. A
53/// second non-admin user on the same machine cannot read the file
54/// even though the parent directory may grant them traversal.
55#[cfg(windows)]
56const LOCAL_ADMIN_DACL_SDDL: &str = "D:P(A;OICI;FA;;;OW)(A;OICI;FA;;;BA)(A;OICI;FA;;;SY)";
57
58/// Apply the protected owner+admin+system DACL to a file or
59/// directory at `path`. Three principals only (Owner / BA / SY).
60///
61/// On Windows: replaces the object's DACL with
62/// [`LOCAL_ADMIN_DACL_SDDL`] via `SetNamedSecurityInfoW`. The DACL
63/// is `protected` so it does not inherit any permissive ACEs from
64/// the parent (e.g. a `Users:Read` entry on the user's profile root
65/// would otherwise leak into a fresh secret file).
66///
67/// **Path-based API note:** `SetNamedSecurityInfoW` resolves `path`
68/// at call time. Callers that have just created a file via
69/// `OpenOptions::create_new(...).open(...)` and then immediately
70/// invoke this helper close the TOCTOU window in practice (the
71/// kernel inode is already published under the path; the only race
72/// is rename-during-call which is fast). For higher-assurance call
73/// sites the handle-based `SetSecurityInfo` would be marginally
74/// safer; this is documented as a known residual path-API exposure.
75///
76/// On non-Windows: no-op. Unix call sites set `mode(0o600)` /
77/// `mode(0o700)` atomically at create-time; this function is only
78/// invoked on the Windows branch.
79///
80/// # Errors
81///
82/// Returns [`CoreError::Internal`] when the SDDL fails to parse
83/// (programmer error — the constant is fixed) or when
84/// `SetNamedSecurityInfoW` rejects the call (typical: target path
85/// does not exist, or the calling process lacks `WRITE_DAC`).
86pub fn set_local_admin_dacl(path: &Path) -> Result<()> {
87    #[cfg(windows)]
88    {
89        use std::str::FromStr;
90        use windows_permissions::{
91            LocalBox, SecurityDescriptor,
92            constants::{SeObjectType, SecurityInformation},
93            wrappers::SetNamedSecurityInfo,
94        };
95
96        // Parse the SDDL constant once per call. The crate caches the
97        // parsed object via `LocalBox<SecurityDescriptor>`, which is
98        // dropped at the end of this scope; SetNamedSecurityInfoW
99        // copies what it needs internally.
100        let sd: LocalBox<SecurityDescriptor> =
101            LocalBox::<SecurityDescriptor>::from_str(LOCAL_ADMIN_DACL_SDDL).map_err(|e| {
102                CoreError::Internal(format!(
103                    "win_acl: failed to parse owner-only SDDL ({}): {}",
104                    LOCAL_ADMIN_DACL_SDDL, e
105                ))
106            })?;
107
108        let dacl = sd.dacl().ok_or_else(|| {
109            CoreError::Internal(
110                "win_acl: parsed SecurityDescriptor had no DACL (SDDL constant is malformed)"
111                    .to_string(),
112            )
113        })?;
114
115        // `Dacl | ProtectedDacl` so the assignment also CLEARS the
116        // inheritance bit — without `ProtectedDacl` the new DACL is
117        // merged with whatever the parent inherits, which can re-add
118        // permissive ACEs.
119        SetNamedSecurityInfo(
120            path.as_os_str(),
121            SeObjectType::SE_FILE_OBJECT,
122            SecurityInformation::Dacl | SecurityInformation::ProtectedDacl,
123            None,
124            None,
125            Some(dacl),
126            None,
127        )
128        .map_err(|e| {
129            CoreError::Internal(format!(
130                "win_acl: SetNamedSecurityInfoW failed for {}: {}",
131                path.display(),
132                e
133            ))
134        })?;
135    }
136    #[cfg(not(windows))]
137    {
138        // Unix call sites set mode(0o600)/mode(0o700) atomically at
139        // create-time. This function is only invoked from the Windows
140        // branch of those call sites, so the non-Windows arm is a
141        // pure no-op rather than a fallback that could mask a missing
142        // mode-call on Unix.
143        let _ = path;
144    }
145    Ok(())
146}
