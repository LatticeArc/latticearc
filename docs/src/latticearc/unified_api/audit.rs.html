1//! # Persistent Audit Storage with Rotation
2//!
3//! Provides tamper-evident audit logging for cryptographic operations.
4//! Events are persisted to disk in JSON Lines format with automatic rotation
5//! based on file size and age.
6//!
7//! ## Security Features
8//!
9//! - **Integrity Verification**: SHA-256 hash chain for tamper detection
10//! - **Automatic Rotation**: Files rotate based on size and age limits
11//! - **Retention Policies**: Configurable retention periods for compliance
12//! - **Thread-Safe**: All operations are safe for concurrent access
13//!
14//! ## Usage
15//!
16//! ```rust,no_run
17//! use latticearc::unified_api::audit::{AuditConfig, FileAuditStorage, AuditStorage, AuditEvent, AuditEventType, AuditOutcome};
18//! use std::path::PathBuf;
19//! use std::time::Duration;
20//!
21//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
22//! let config = AuditConfig {
23//!     storage_path: PathBuf::from("/var/log/latticearc/audit"),
24//!     max_file_size_bytes: 100 * 1024 * 1024, // 100MB
25//!     max_file_age: Duration::from_secs(24 * 60 * 60), // 24 hours
26//!     retention_days: 90,
27//! };
28//!
29//! let storage = FileAuditStorage::new(config)?;
30//!
31//! // Create and write an audit event
32//! let event = AuditEvent::new(
33//!     AuditEventType::CryptoOperation,
34//!     "encrypt_data",
35//!     AuditOutcome::Success,
36//! );
37//! storage.write(&event)?;
38//! # Ok(())
39//! # }
40//! ```

162    /// Maximum byte length per metadata key.

167    /// Add metadata to this event.
168    ///
169    /// Caps:
170    /// - Keys longer than [`MAX_METADATA_KEY_LEN`](Self::MAX_METADATA_KEY_LEN)
171    ///   are truncated.
172    /// - Values longer than [`MAX_METADATA_VALUE_LEN`](Self::MAX_METADATA_VALUE_LEN)
173    ///   are truncated.
174    /// - Beyond [`MAX_METADATA_ENTRIES`](Self::MAX_METADATA_ENTRIES) entries
175    ///   the call is a no-op (audit emission must not abort the operation
176    ///   that triggered it).
177    ///
178    /// These caps prevent a DoS amplification path where attacker-controlled
179    /// strings (e.g. SignedData.scheme, a request header) are routed through
180    /// audit-event metadata: without them, every audit event could carry
181    /// unbounded heap + hashing cost.
182    #[must_use]
183    pub fn with_metadata(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
184        if self.metadata.len() >= Self::MAX_METADATA_ENTRIES {
185            return self;
186        }
187        let mut k = key.into();
188        let mut v = value.into();
189        truncate_utf8_safe(&mut k, Self::MAX_METADATA_KEY_LEN);
190        truncate_utf8_safe(&mut v, Self::MAX_METADATA_VALUE_LEN);
191        self.metadata.insert(k, v); // LINT-OK: canonical-with-metadata-impl
192        self
193    }

250/// Truncate a UTF-8 String to at most `max_bytes` bytes without splitting
251/// a multi-byte code point. `String::truncate` panics on a non-boundary
252/// index — for caller-controlled text we round DOWN to the nearest valid
253/// boundary so attacker-supplied UTF-8 can't crash the audit pipeline.
254fn truncate_utf8_safe(s: &mut String, max_bytes: usize) {
255    if s.len() <= max_bytes {
256        return;
257    }
258    let mut cut = max_bytes;
259    while cut > 0 && !s.is_char_boundary(cut) {
260        cut = cut.saturating_sub(1);
261    }
262    s.truncate(cut);
263}

265/// Sanitize a free-form audit field: returns `None` if the input is
266/// empty after stripping control characters; truncates to `max_bytes`
267/// otherwise. Control-char strip prevents `\n`-laced inputs from
268/// breaking JSONL audit consumers; size cap bounds memory + hash cost.
269fn sanitize_audit_field(input: String, max_bytes: usize) -> Option<String> {
270    let mut s: String = input.chars().filter(|c| !c.is_control()).collect();
271    if s.is_empty() {
272        return None;
273    }
274    truncate_utf8_safe(&mut s, max_bytes);
275    Some(s)
276}
277
278/// Sanitize the `action` verb. Unlike `actor`/`resource`, `action`
279/// is mandatory on every event (the verb that triggered the audit
280/// entry), so an empty result substitutes a `"<empty>"` placeholder
281/// rather than dropping the field — auditors prefer "we know it
282/// happened but the verb was unrepresentable" over a silent gap.
283fn sanitize_action_field(input: String) -> String {
284    let mut s: String = input.chars().filter(|c| !c.is_control()).collect();
285    if s.is_empty() {
286        return "<empty>".to_string();
287    }
288    truncate_utf8_safe(&mut s, AuditEvent::MAX_ACTION_LEN);
289    s
290}

509/// Parse the timestamp embedded in an audit-log filename.
510///
511/// Audit logs are named either:
512/// - `audit-YYYY-MM-DDTHH-MM-SS.jsonl` (legacy second-precision form)
513/// - `audit-YYYY-MM-DDTHH-MM-SS-NNNNNN.jsonl` (current form, with
514///   `NNNNNN` = microseconds since the second, added to defeat
515///   sub-second rotation collisions under `create_new(true)`)
516///
517/// Both shapes parse to the same chronological key — microseconds
518/// are stripped at parse time because retention is second-grained.
519/// Treat the parsed `NaiveDateTime` as UTC since the producer writes
520/// UTC clock readings. Used by the retention sweep so cleanup
521/// decisions don't depend on filesystem `mtime`, which a privileged
522/// attacker can rewrite with `touch`.
523fn parse_audit_filename_timestamp(file_name: &str) -> Option<DateTime<Utc>> {
524    let stem = file_name.strip_prefix("audit-")?.strip_suffix(".jsonl")?;
525    // Try the current micro-precision form first (`SS-NNNNNN`), then
526    // fall back to the legacy second-only form. The micro suffix is
527    // exactly six digits; chrono's `%6f` matches.
528    if let Ok(naive) = chrono::NaiveDateTime::parse_from_str(stem, "%Y-%m-%dT%H-%M-%S-%6f") {
529        return Some(DateTime::<Utc>::from_naive_utc_and_offset(naive, Utc));
530    }
531    let naive = chrono::NaiveDateTime::parse_from_str(stem, "%Y-%m-%dT%H-%M-%S").ok()?;
532    Some(DateTime::<Utc>::from_naive_utc_and_offset(naive, Utc))
533}

616/// Filename for the per-storage genesis anchor.
617///
618/// The genesis anchor is the `previous_hash` value used for the very first
619/// audit event in a storage. It is derived from a domain label, a fresh
620/// random nonce, and the creation timestamp, and is persisted alongside
621/// the audit log so that:
622///
623///   * a truncate-only attack (delete log entries, leave genesis intact)
624///     is detectable: the next event chains from `genesis`, not from the
625///     deleted entry's hash, and any verifier replaying the chain spots
626///     the gap.
627///   * an attacker who deletes both the log and the genesis to fabricate
628///     a "fresh storage" ends up with a different genesis on the next
629///     create — an external pinning store (HSM, write-once log) can spot
630///     the genesis change. Total erasure with no external state is
631///     fundamentally undetectable; this fix closes the
632///     truncate-and-restart-only path the empty-genesis behaviour left
633///     wide open.
634const AUDIT_GENESIS_FILENAME: &str = "genesis";
635
636/// Domain-separation label baked into the genesis hash. Distinct from
637/// every other transcript prefix in the crate so a genesis bytestring
638/// cannot collide with any other artifact.
639const AUDIT_GENESIS_DOMAIN_LABEL: &[u8] = b"latticearc-audit-genesis-v1";

641/// Action verb of the synthetic event written as the first entry of
642/// each rotated audit file. Anchors the new file to the prior file's
643/// final hash; absence (or mismatch) of this anchor at the start of a
644/// non-initial file indicates the chain has been truncated, reordered,
645/// or had files deleted.
646///

699    /// Read the on-disk genesis if it exists, otherwise create one.
700    ///
701    /// The created genesis is `SHA-256(domain_label || nonce || timestamp)`
702    /// hex-encoded, where `nonce` is 32 bytes from the system CSPRNG and
703    /// `timestamp` is the RFC 3339 creation time.
704    fn load_or_create_genesis(storage_path: &std::path::Path) -> Result<String> {
705        let genesis_path = storage_path.join(AUDIT_GENESIS_FILENAME);
706
707        match fs::read_to_string(&genesis_path) {
708            Ok(existing) => {
709                let trimmed = existing.trim().to_string();
710                if trimmed.is_empty() {
711                    return Err(CoreError::AuditError(format!(
712                        "Audit genesis file '{}' exists but is empty; refusing to start \
713                         with an empty chain anchor",
714                        genesis_path.display()
715                    )));
716                }
717                Ok(trimmed)
718            }
719            Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
720                let nonce = crate::primitives::rand::csprng::random_bytes(32);
721                let timestamp = Utc::now().to_rfc3339();
722                let mut buf = Vec::with_capacity(
723                    AUDIT_GENESIS_DOMAIN_LABEL
724                        .len()
725                        .saturating_add(4)
726                        .saturating_add(nonce.len())
727                        .saturating_add(4)
728                        .saturating_add(timestamp.len()),
729                );
730                buf.extend_from_slice(AUDIT_GENESIS_DOMAIN_LABEL);
731                let nonce_len = u32::try_from(nonce.len()).map_err(|_e| {
732                    CoreError::AuditError("Genesis nonce length exceeds u32::MAX".to_string())
733                })?;
734                buf.extend_from_slice(&nonce_len.to_be_bytes());
735                buf.extend_from_slice(&nonce);
736                let ts_len = u32::try_from(timestamp.len()).map_err(|_e| {
737                    CoreError::AuditError("Genesis timestamp length exceeds u32::MAX".to_string())
738                })?;
739                buf.extend_from_slice(&ts_len.to_be_bytes());
740                buf.extend_from_slice(timestamp.as_bytes());
741
742                let digest = crate::primitives::hash::sha2::sha256(&buf).map_err(|e| {
743                    CoreError::AuditError(format!("Failed to hash genesis material: {e}"))
744                })?;
745                let hex = hex::encode(digest);
746
747                // Persist atomically with restrictive permissions on Unix.
748                #[cfg(unix)]
749                {
750                    use std::io::Write as _;
751                    use std::os::unix::fs::OpenOptionsExt;
752                    let mut f = OpenOptions::new()
753                        .create_new(true)
754                        .write(true)
755                        .mode(0o600)
756                        // LINT-OK: mode-on-prev-line — chained above.
757                        .open(&genesis_path)
758                        .map_err(|err| {
759                            CoreError::AuditError(format!(
760                                "Failed to create audit genesis file '{}': {}",
761                                genesis_path.display(),
762                                err
763                            ))
764                        })?;
765                    f.write_all(hex.as_bytes()).map_err(|err| {
766                        CoreError::AuditError(format!(
767                            "Failed to write audit genesis file '{}': {}",
768                            genesis_path.display(),
769                            err
770                        ))
771                    })?;
772                    // fsync the file before declaring the genesis
773                    // committed. A crash mid-flush would otherwise
774                    // leave a zero-byte genesis on disk; the
775                    // `create_new` flag would then refuse every
776                    // subsequent startup until manual `rm`.
777                    f.sync_all().map_err(|err| {
778                        CoreError::AuditError(format!(
779                            "Failed to fsync audit genesis file '{}': {}",
780                            genesis_path.display(),
781                            err
782                        ))
783                    })?;
784                }
785                #[cfg(not(unix))]
786                {
787                    // M9: on Windows, use `OpenOptions` with
788                    // `share_mode(0)` (deny share) and explicit
789                    // `create_new(true)` so the genesis file is
790                    // exclusive to this process while the handle is
791                    // open. Default `fs::write` would inherit the
792                    // process's default DACL — typically world-
793                    // readable on local Windows — exposing the
794                    // chain-integrity HMAC seed to other local users.
795                    // Note: `share_mode(0)` is the closest std-only
796                    // approximation to Unix `0o600` without pulling
797                    // in the full Windows ACL API; a future round
798                    // could tighten this further via the
799                    // `windows-sys` ACL crate if regulators require
800                    // it.
801                    use std::io::Write as _;
802                    use std::os::windows::fs::OpenOptionsExt as _;
803                    // `.mode(0o600)` is `std::os::unix::fs::OpenOptionsExt`
804                    // and is unavailable on this `cfg(not(unix))` branch;
805                    // Windows confidentiality is enforced via the
806                    // `share_mode(0)` exclusive-handle below.
807                    let mut f = OpenOptions::new()
808                        .create_new(true)
809                        .write(true)
810                        .share_mode(0)
811                        // LINT-OK: cfg-not-unix; no Unix `mode` API available
812                        .open(&genesis_path)
813                        .map_err(|err| {
814                            CoreError::AuditError(format!(
815                                "Failed to create audit genesis file '{}': {}",
816                                genesis_path.display(),
817                                err
818                            ))
819                        })?;
820                    f.write_all(hex.as_bytes()).map_err(|err| {
821                        CoreError::AuditError(format!(
822                            "Failed to write audit genesis file '{}': {}",
823                            genesis_path.display(),
824                            err
825                        ))
826                    })?;
827                    f.sync_all().map_err(|err| {
828                        CoreError::AuditError(format!(
829                            "Failed to fsync audit genesis file '{}': {}",
830                            genesis_path.display(),
831                            err
832                        ))
833                    })?;
834                }
835                Ok(hex)
836            }
837            Err(e) => Err(CoreError::AuditError(format!(
838                "Failed to read audit genesis file '{}': {}",
839                genesis_path.display(),
840                e
841            ))),
842        }
843    }

851    /// Compute the integrity hash for an event.
852    ///
853    /// The hash includes the previous event's hash to create a chain,
854    /// making tampering detectable. Routes through the
855    /// [`crate::primitives::hash::sha2::sha256`] wrapper so audit integrity
856    /// uses the same hash call path as the rest of the crate.
857    ///
858    /// made `pub(crate)` so the public
859    /// `verify_chain` helper below can re-use it without duplicating
860    /// the field-encoding rules.
861    ///
862    /// # Errors
863    /// Returns an error if the SHA-256 primitive fails (input exceeds 1 GiB guard).
864    pub(crate) fn compute_integrity_hash(
865        event: &AuditEvent,
866        previous_hash: &str,
867    ) -> Result<String> {
868        // Length-prefix every field so prefix-collision attacks are
869        // impossible (`"ab" + "c"` and `"a" + "bc"` no longer hash to the
870        // same digest). The metadata caps already bound each field, but a
871        // 4-byte BE length per element is the cheap and definitive fix.
872        // (Encoding is BE per the L3 transcript-convention migration in
873        // 85e2bd79e — see `append_lenp_field` doc comment for details.)
874        let mut buf = Vec::new();
875        Self::append_lenp_field(&mut buf, previous_hash.as_bytes())?;
876        Self::append_lenp_field(&mut buf, event.id.as_bytes())?;
877        Self::append_lenp_field(&mut buf, event.timestamp.to_rfc3339().as_bytes())?;
878        Self::append_lenp_field(&mut buf, event.event_type.to_string().as_bytes())?;
879
880        // Optional fields are encoded as a discriminator byte followed by
881        // the length-prefixed bytes; absence vs empty is now distinguishable.
882        match event.actor.as_ref() {
883            Some(a) => {
884                buf.push(1);
885                Self::append_lenp_field(&mut buf, a.as_bytes())?;
886            }
887            None => buf.push(0),
888        }
889        match event.resource.as_ref() {
890            Some(r) => {
891                buf.push(1);
892                Self::append_lenp_field(&mut buf, r.as_bytes())?;
893            }
894            None => buf.push(0),
895        }
896
897        Self::append_lenp_field(&mut buf, event.action.as_bytes())?;
898        Self::append_lenp_field(&mut buf, event.outcome.to_string().as_bytes())?;
899
900        // Metadata: include count, then sorted (key, value) pairs each
901        // with their own length prefix. Overflow at `u32::MAX` (4 G
902        // entries) propagates as `AuditError`, symmetric with the
903        // length-prefix overflow path above. Length prefixes are
904        // big-endian to match the transcript convention used by
905        // `zkp::sigma::compute_challenge` — unifies endianness across
906        // all transcript-style hashing in the crate.
907        let mut metadata_keys: Vec<&String> = event.metadata.keys().collect();
908        metadata_keys.sort();
909        let count = u32::try_from(metadata_keys.len()).map_err(|_e| {
910            CoreError::AuditError("integrity hash metadata count exceeds 2^32".to_string())
911        })?;
912        buf.extend_from_slice(&count.to_be_bytes());
913        for key in metadata_keys {
914            Self::append_lenp_field(&mut buf, key.as_bytes())?;
915            if let Some(value) = event.metadata.get(key) {
916                Self::append_lenp_field(&mut buf, value.as_bytes())?;
917            } else {
918                Self::append_lenp_field(&mut buf, &[])?;
919            }
920        }
921
922        // `sha256` only fails on inputs larger than 1 GiB (resource-limit guard),
923        // which no reasonable audit event approaches.
924        let digest = crate::primitives::hash::sha2::sha256(&buf)
925            .map_err(|e| CoreError::AuditError(format!("integrity hash failed: {}", e)))?;
926        Ok(hex::encode(digest))
927    }
928
929    /// Append `field.len() as u32 BE` followed by `field` to `buf`.
930    ///
931    /// Returns `Err(CoreError::AuditError)` if `field.len()` exceeds
932    /// `u32::MAX` bytes (4 GiB) — symmetric with the overflow handling
933    /// in `zkp::sigma::compute_challenge`. The length is encoded
934    /// big-endian to match the transcript convention used by
935    /// `zkp::sigma::compute_challenge` — LE encoding here would be an
936    /// isolated outlier within the crate's transcript-style hashing.
937    /// The previous saturating-to-`u32::MAX` form was a silent collapse
938    /// that would let two distinct field values share the same length
939    /// prefix; while the SHA-256 backend's 1 GiB cap makes this
940    /// unreachable today, an explicit error preserves the asymmetric
941    /// defensive posture the rest of the crate now uses.
942    fn append_lenp_field(buf: &mut Vec<u8>, field: &[u8]) -> Result<()> {
943        let len = u32::try_from(field.len()).map_err(|_e| {
944            CoreError::AuditError("integrity hash field exceeds 2^32 bytes".to_string())
945        })?;
946        buf.extend_from_slice(&len.to_be_bytes());
947        buf.extend_from_slice(field);
948        Ok(())
949    }

951    /// Check if the current file needs rotation.
952    fn needs_rotation(&self, state: &FileState) -> bool {
953        // Check size limit
954        if state.current_size >= self.config.max_file_size_bytes {
955            return true;
956        }
957
958        // Check age limit
959        let age =
960            Utc::now().signed_duration_since(state.created_at).to_std().unwrap_or(Duration::ZERO);
961
962        age >= self.config.max_file_age
963    }

965    /// Rotate the current file if needed.
966    ///
967    /// On rotation, a `chain-link` event is written as the FIRST entry
968    /// of the new file. The link carries the previous file's name and
969    /// the chain hash at the moment of rotation. Without this anchor,
970    /// `verify_chain` cannot detect a deletion or reorder of files at
971    /// the boundary — the new file's first event would chain from
972    /// in-memory `previous_hash`, but no on-disk record would tie that
973    /// hash to the file it descended from. The anchor's own
974    /// `integrity_hash` covers the metadata, so tampering with the
975    /// recorded `previous_file` / `previous_hash` breaks the chain at
976    /// verification time.
977    fn rotate_if_needed(&self, state: &mut Option<FileState>) -> Result<()> {
978        let should_rotate = state.as_ref().is_some_and(|s| self.needs_rotation(s));
979
980        let mut rotation_link: Option<(String, String)> = None;
981        if should_rotate {
982            // Close current file and create new one. Capture the
983            // outgoing filename and chain hash BEFORE we drop the old
984            // state so the new file's chain-anchor can reference both.
985            if let Some(mut old_state) = state.take() {
986                let old_filename = old_state
987                    .current_path
988                    .file_name()
989                    .and_then(|n| n.to_str())
990                    .unwrap_or("<unknown>")
991                    .to_string();
992                let old_hash = self.previous_hash.read().clone();
993                tracing::info!(
994                    "Rotating audit file: {} (size: {} bytes)",
995                    old_state.current_path.display(),
996                    old_state.current_size
997                );
998                old_state.writer.flush().map_err(|e| {
999                    CoreError::AuditError(format!("Failed to flush audit file: {}", e))
1000                })?;
1001                rotation_link = Some((old_filename, old_hash));
1002            }
1003        }
1004
1005        // Create new file if needed
1006        if state.is_none() {
1007            *state = Some(self.create_new_file()?);
1008        }
1009
1010        // Write chain-anchor as the first entry of the new file when we
1011        // rotated from an existing file. (First-ever-file case has
1012        // `rotation_link = None` and the genesis hash anchors the chain
1013        // implicitly, as documented above.)
1014        if let Some((prev_file, prev_hash)) = rotation_link
1015            && let Some(new_state) = state.as_mut()
1016        {
1017            self.write_chain_anchor(new_state, &prev_file, &prev_hash)?;
1018        }
1019
1020        Ok(())
1021    }

1023    /// Write a chain-anchor event as the first entry of a freshly
1024    /// rotated audit file.
1025    ///
1026    /// The anchor is itself a normal `AuditEvent`, chained from the
1027    /// outgoing file's final hash, with metadata recording that
1028    /// outgoing file's name and hash. Its own `integrity_hash` is
1029    /// recomputed by `verify_chain` and bumped into `previous_hash`
1030    /// like any other event, so the chain is unbroken across the
1031    /// rotation. Tampering with the anchor's metadata invalidates its
1032    /// integrity hash; deletion of either file makes the anchor
1033    /// reference dangling.
1034    fn write_chain_anchor(
1035        &self,
1036        state: &mut FileState,
1037        previous_file: &str,
1038        previous_hash: &str,
1039    ) -> Result<()> {
1040        let mut anchor =
1041            AuditEvent::new(AuditEventType::System, CHAIN_ANCHOR_ACTION, AuditOutcome::Success)
1042                .with_metadata(CHAIN_ANCHOR_PREV_FILE_KEY, previous_file)
1043                .with_metadata(CHAIN_ANCHOR_PREV_HASH_KEY, previous_hash);
1044
1045        // Chain the anchor from the outgoing file's final hash
1046        // (= current `previous_hash`), then advance the in-memory
1047        // chain pointer to the anchor's own hash.
1048        let chain_prev = self.previous_hash.read().clone();
1049        anchor.integrity_hash = Self::compute_integrity_hash(&anchor, &chain_prev)?;
1050        {
1051            let mut prev = self.previous_hash.write();
1052            prev.clone_from(&anchor.integrity_hash);
1053        }
1054
1055        let json = serde_json::to_string(&anchor).map_err(|e| {
1056            CoreError::AuditError(format!("Failed to serialize chain anchor: {}", e))
1057        })?;
1058        let line = format!("{}\n", json);
1059        let line_bytes = line.as_bytes();
1060        state
1061            .writer
1062            .write_all(line_bytes)
1063            .map_err(|e| CoreError::AuditError(format!("Failed to write chain anchor: {}", e)))?;
1064        let line_len_u64 = u64::try_from(line_bytes.len()).unwrap_or(u64::MAX);
1065        state.current_size = state.current_size.saturating_add(line_len_u64);
1066        Ok(())
1067    }

1069    /// Create a new audit file.
1070    ///
1071    /// Filename includes microsecond precision so sub-second rotations
1072    /// (rapid `with_max_file_size` triggering, repeated explicit
1073    /// `flush()` calls, or stress tests) cannot collide. The
1074    /// `create_new(true)` flag below would otherwise error on
1075    /// collisions and leave the rotation loop stuck.
1076    fn create_new_file(&self) -> Result<FileState> {
1077        let now = Utc::now();
1078        let micros = now.timestamp_subsec_micros();
1079        let filename = format!("audit-{}-{:06}.jsonl", now.format("%Y-%m-%dT%H-%M-%S"), micros);
1080        let path = self.config.storage_path.join(&filename);
1081
1082        let file = {
1083            // On Unix, set 0o600 atomically via OpenOptions::mode() so the
1084            // file is never world-readable, even briefly. Audit logs may
1085            // contain operation context (key IDs, paths, actors) and must
1086            // not inherit the default umask the way the previous bare
1087            // `OpenOptions::new().create(...).open(...)` did.
1088            // `create_new(true)` (not `create(true)`): refuse to open a
1089            // pre-existing file. The rotation code generates the
1090            // filename from `Utc::now()` to second precision; if a
1091            // file already exists at that path it means either (a) a
1092            // sub-second rotation race or (b) a stale file from a
1093            // prior process. In either case we MUST NOT silently
1094            // adopt the existing inode and rewrite its DACL — that
1095            // would (a) make the chain hash-input contiguous across
1096            // processes that didn't actually share state, and (b)
1097            // hide the race. Surface the conflict as an error so the
1098            // caller can rotate forward by one second.
1099            #[cfg(unix)]
1100            {
1101                use std::os::unix::fs::OpenOptionsExt;
1102                OpenOptions::new().create_new(true).append(true).mode(0o600).open(&path).map_err(
1103                    |e| {
1104                        CoreError::AuditError(format!(
1105                            "Failed to create audit file '{}': {}",
1106                            path.display(),
1107                            e
1108                        ))
1109                    },
1110                )?
1111            }
1112            #[cfg(not(unix))]
1113            {
1114                let f = {
1115                    // LINT-OK: cfg-not-unix — Windows confidentiality
1116                    // enforced via `set_local_admin_dacl` immediately
1117                    // below (Win32 `OpenOptionsExt` has no `.mode()`).
1118                    OpenOptions::new().create_new(true).append(true).open(&path).map_err(|e| {
1119                        CoreError::AuditError(format!(
1120                            "Failed to create audit file '{}': {}",
1121                            path.display(),
1122                            e
1123                        ))
1124                    })?
1125                };
1126                // Replace the default DACL inherited from the parent
1127                // directory with the owner-only policy applied
1128                // workspace-wide. Symmetric with the Unix `mode(0o600)`
1129                // above. A failure here is fatal — letting an audit
1130                // log inherit a permissive DACL silently would defeat
1131                // the whole point of the rotation.
1132                crate::unified_api::set_local_admin_dacl(&path).map_err(|e| {
1133                    CoreError::AuditError(format!(
1134                        "Failed to harden audit file DACL '{}': {}",
1135                        path.display(),
1136                        e
1137                    ))
1138                })?;
1139                f
1140            }
1141        };
1142
1143        tracing::debug!("Created new audit file: {}", path.display());
1144
1145        Ok(FileState {
1146            writer: BufWriter::new(file),
1147            current_path: path,
1148            current_size: 0,
1149            created_at: now,
1150        })
1151    }

1153    /// Clean up old audit files based on retention policy.
1154    fn cleanup_old_files(&self) -> Result<()> {
1155        // Belt-and-braces: even with `with_retention_days` rejecting
1156        // `0`, a `Default` or struct-literal construction could still
1157        // land here with `retention_days = 0`. Fail closed instead of
1158        // wiping the directory.
1159        if self.config.retention_days == 0 {
1160            return Err(CoreError::AuditError(
1161                "AuditConfig.retention_days = 0 would delete every audit file on \
1162                 cleanup; refusing to proceed (set with_retention_days(>= 1))"
1163                    .to_string(),
1164            ));
1165        }
1166
1167        let retention_duration = chrono::Duration::days(i64::from(self.config.retention_days));
1168        let Some(cutoff) = Utc::now().checked_sub_signed(retention_duration) else {
1169            return Err(CoreError::AuditError(format!(
1170                "Retention period of {} days overflows date arithmetic",
1171                self.config.retention_days
1172            )));
1173        };
1174
1175        let entries = fs::read_dir(&self.config.storage_path).map_err(|e| {
1176            CoreError::AuditError(format!(
1177                "Failed to read audit directory '{}': {}",
1178                self.config.storage_path.display(),
1179                e
1180            ))
1181        })?;
1182
1183        for entry in entries {
1184            let Ok(entry) = entry else { continue };
1185
1186            let path = entry.path();
1187
1188            // Only process .jsonl files
1189            if path.extension().and_then(|e| e.to_str()) != Some("jsonl") {
1190                continue;
1191            }
1192
1193            // Use the timestamp embedded in the filename
1194            // (`audit-YYYY-MM-DDTHH-MM-SS.jsonl`, set by
1195            // `create_new_file`) instead of the filesystem mtime,
1196            // which a privileged attacker can rewrite with `touch`.
1197            // The filename is set at file creation by this process and
1198            // is not modifiable without also changing the file's name,
1199            // which is observable in directory listings and recorded
1200            // in the audit-genesis chain. Files whose names don't
1201            // parse — including operator-imported logs from another
1202            // tool — are skipped rather than deleted.
1203            let Some(file_name) = path.file_name().and_then(|n| n.to_str()) else {
1204                continue;
1205            };
1206            let Some(created_at) = parse_audit_filename_timestamp(file_name) else {
1207                tracing::debug!("Audit cleanup: skipping non-conforming filename '{}'", file_name);
1208                continue;
1209            };

1211            if created_at < cutoff {
1212                // Delete old file
1213                if let Err(e) = fs::remove_file(&path) {
1214                    tracing::warn!("Failed to remove old audit file '{}': {}", path.display(), e);
1215                } else {
1216                    tracing::info!("Removed old audit file: {}", path.display());
1217                }
1218            }
1219        }
1220
1221        Ok(())
1222    }

1224    /// public hash-chain verification.
1225    ///
1226    /// Walks every `audit-*.jsonl` file in the storage directory in
1227    /// filename-timestamp order, recomputes each entry's
1228    /// `integrity_hash` from the persisted genesis, and reports the
1229    /// first divergence (or `Ok(report)` with no mismatches when the
1230    /// chain is intact). Tamper detection previously required
1231    /// hand-writing a verifier in the right encoding — the asymmetry
1232    /// tamper-evident logging exists to remove.
1233    ///
1234    /// The returned [`ChainVerificationReport`] carries the line count
1235    /// inspected and an `Option<Mismatch>` describing the first event
1236    /// whose recomputed hash differed from its persisted
1237    /// `integrity_hash` field.
1238    ///
1239    /// # Errors
1240    /// Returns an error if the storage directory cannot be read, a
1241    /// log file cannot be opened, a line cannot be parsed as JSON, or
1242    /// the hash recomputation fails.
1243    pub fn verify_chain(&self) -> Result<ChainVerificationReport> {
1244        let mut log_files: Vec<PathBuf> = Vec::new();
1245        let entries = fs::read_dir(&self.config.storage_path)
1246            .map_err(|e| CoreError::AuditError(format!("Failed to read audit dir: {}", e)))?;
1247        for entry in entries.flatten() {
1248            let path = entry.path();
1249            let Some(name) = path.file_name().and_then(|n| n.to_str()) else { continue };
1250            // Match `audit-*.jsonl` (case-insensitive on extension;
1251            // audit fix L for case-sensitive .jsonl bug).
1252            let lower = name.to_ascii_lowercase();
1253            if lower.starts_with("audit-") && lower.ends_with(".jsonl") {
1254                log_files.push(path);
1255            }
1256        }
1257        // Filename-timestamp ordering: `audit-YYYYMMDD-HHMMSS-NNN.jsonl`
1258        // is lexicographically chronological. Sort by basename rather
1259        // than the full `PathBuf` — the latter is byte-ordered, which
1260        // on Windows can differ across drive-letter case (`C:\` vs
1261        // `c:\`) or path-separator normalisation, producing an order
1262        // that doesn't match the chronological intent.
1263        log_files.sort_by(|a, b| {
1264            a.file_name()
1265                .unwrap_or_else(|| a.as_os_str())
1266                .cmp(b.file_name().unwrap_or_else(|| b.as_os_str()))
1267        });
1268
1269        // Recompute from genesis. Reads the persisted genesis directly
1270        // rather than the in-memory `previous_hash` so verification
1271        // works on a freshly-mounted storage with no prior writes in
1272        // this process.
1273        let genesis = Self::load_or_create_genesis(&self.config.storage_path)?;
1274        let mut prev_hash = genesis;
1275        let mut events_checked: u64 = 0;
1276        let mut mismatch: Option<ChainMismatch> = None;
1277
1278        // Defense-in-depth pre-decode caps. A persisted JSONL line is at
1279        // worst the product of in-memory caps (action 256 B + actor
1280        // 256 B + resource 1 KiB + 32 metadata × 4 KiB ≈ 130 KiB),
1281        // plus header / chain-marker fields and JSON encoding
1282        // overhead. 1 MiB is generously above that ceiling but small
1283        // enough to defeat a tampered file dropping a 1 GiB action
1284        // string in front of the verifier.
1285        const MAX_LINE_LEN: usize = 1024 * 1024;
1286
1287        // Cross-file chain back-reference state. After processing the
1288        // first file, every subsequent file's first event MUST be a
1289        // `chain-link` anchor whose `previous_file` and `previous_hash`
1290        // metadata match the file we just finished. Without this check
1291        // the chain rotates "blindly" — `verify_chain` would happily
1292        // walk a sequence of files even after one was deleted, because
1293        // each file individually chains via in-memory state alone.
1294        //
1295        // `previous_file_name` is the basename of the prior file in the
1296        // sorted log_files list. `previous_file_final_hash` is the
1297        // chain hash when we exited the prior file's loop.
1298        let mut previous_file_name: Option<String> = None;
1299        let mut previous_file_final_hash: Option<String> = None;
1300
1301        for file in &log_files {
1302            use std::io::BufRead;
1303            let f = File::open(file).map_err(|e| {
1304                CoreError::AuditError(format!("Failed to open {}: {}", file.display(), e))
1305            })?;
1306            let mut reader = std::io::BufReader::new(f);
1307            // Reset per-file: track whether we've seen the first
1308            // non-empty event of this file (used for the cross-file
1309            // anchor check below).
1310            let mut first_event_in_file = true;
1311            // Bounded line-reader. `BufReader::lines()` and the
1312            // obvious `read_until(b'\n', &mut buf)` BOTH allocate the
1313            // whole line BEFORE returning, so a 1 GiB newline-free
1314            // file would OOM the verifier before any post-decode
1315            // `MAX_LINE_LEN` check could fire.
1316            //
1317            // We use `BufRead::fill_buf` / `consume` to scan for
1318            // `b'\n'` in the buffered region and accumulate at most
1319            // `MAX_LINE_LEN + 1` bytes into `line_buf` per line.
1320            // The +1 byte lets us detect the over-cap case
1321            // unambiguously (the cap is exclusive). An earlier
1322            // implementation used per-byte `reader.read()` which was
1323            // correct but did one syscall per byte; this shape keeps
1324            // the bound while letting the kernel/buffer combine reads.
1325            let mut line_buf: Vec<u8> = Vec::new();
1326            let mut line_idx: usize = 0;
1327            loop {
1328                line_buf.clear();
1329                let mut hit_newline = false;
1330                let mut hit_eof = false;
1331                while !hit_newline && !hit_eof {
1332                    let buf = reader.fill_buf().map_err(|e| {
1333                        CoreError::AuditError(format!("Failed to read line: {}", e))
1334                    })?;
1335                    if buf.is_empty() {
1336                        hit_eof = true;
1337                        break;
1338                    }
1339                    // How many bytes can we still accumulate before
1340                    // crossing `MAX_LINE_LEN + 1`? Stop the moment
1341                    // we'd cross — that's the cap-exceeded signal.
1342                    let remaining_room =
1343                        MAX_LINE_LEN.saturating_add(1).saturating_sub(line_buf.len());
1344                    let scan_end = buf.len().min(remaining_room);
1345                    #[allow(clippy::indexing_slicing)]
1346                    // SAFETY: `scan_end = buf.len().min(remaining_room)`
1347                    // ⇒ `scan_end ≤ buf.len()` ⇒ `&buf[..scan_end]` is
1348                    // in-bounds. (no possible panic)
1349                    let scan_slice = &buf[..scan_end];
1350                    if let Some(rel) = scan_slice.iter().position(|&b| b == b'\n') {
1351                        // Found the newline within the room budget —
1352                        // append up to (but excluding) it, consume
1353                        // the newline.
1354                        #[allow(clippy::indexing_slicing)]
1355                        // SAFETY: `rel` came from `scan_slice.iter().position()`
1356                        // ⇒ `rel < scan_slice.len()` ⇒ `&scan_slice[..rel]`
1357                        // is in-bounds. (no possible panic)
1358                        let prefix = &scan_slice[..rel];
1359                        line_buf.extend_from_slice(prefix);
1360                        let consume_n = rel.saturating_add(1);
1361                        reader.consume(consume_n);
1362                        hit_newline = true;
1363                    } else if scan_end < buf.len() {
1364                        // No newline within room budget AND there's
1365                        // more data in the kernel buffer — that means
1366                        // this line exceeds the cap. Append whatever
1367                        // fits in the room budget (may be zero), then
1368                        // BREAK so the over-cap check below converts
1369                        // to Err. Without `break`, the next iteration
1370                        // sees `remaining_room == 0` ⇒ `scan_end == 0`
1371                        // ⇒ no append + no consume ⇒ infinite livelock,
1372                        // giving an attacker with audit-write access
1373                        // a DoS on `verify_chain`.
1374                        line_buf.extend_from_slice(scan_slice);
1375                        reader.consume(scan_end);
1376                        break;
1377                    } else {
1378                        // No newline yet, but the entire buffered
1379                        // chunk fits within room — append, consume,
1380                        // refill.
1381                        line_buf.extend_from_slice(scan_slice);
1382                        reader.consume(scan_end);
1383                    }
1384                }
1385                if hit_eof && line_buf.is_empty() {
1386                    break;
1387                }
1388                if line_buf.len() > MAX_LINE_LEN {
1389                    return Err(CoreError::AuditError(format!(
1390                        "audit line {} of {} exceeds maximum length {}",
1391                        line_idx,
1392                        file.display(),
1393                        MAX_LINE_LEN
1394                    )));
1395                }
1396                let line = match std::str::from_utf8(&line_buf) {
1397                    Ok(s) => s.to_string(),
1398                    Err(e) => {
1399                        return Err(CoreError::AuditError(format!(
1400                            "audit line {} of {} is not valid UTF-8: {}",
1401                            line_idx,
1402                            file.display(),
1403                            e
1404                        )));
1405                    }
1406                };
1407                let saved_idx = line_idx;
1408                line_idx = line_idx.saturating_add(1);
1409                if line.trim().is_empty() {
1410                    continue;
1411                }
1412                let line_idx = saved_idx;
1413                // Pre-pass: extract `integrity_hash` cheaply with
1414                // `serde_json::Value` partial parsing so we can fail
1415                // fast on a tampered chain marker before paying for
1416                // the full `AuditEvent` deserialization (which
1417                // allocates strings for every field). This still
1418                // walks the whole line — `serde_json::from_str` is
1419                // O(n) regardless — but skips the secondary
1420                // allocation of a fully-typed event.
1421                //
1422                // Rejection here cannot leak which structural field
1423                // failed: any pre-pass parse error collapses to a
1424                // single "unparseable" outcome.
1425                let bare: serde_json::Value = serde_json::from_str(&line).map_err(|e| {
1426                    CoreError::AuditError(format!(
1427                        "Failed to parse line {} of {}: {}",
1428                        line_idx,
1429                        file.display(),
1430                        e
1431                    ))
1432                })?;
1433                let _stored_hash_seen =
1434                    bare.get("integrity_hash").and_then(|v| v.as_str()).ok_or_else(|| {
1435                        CoreError::AuditError(format!(
1436                            "Audit line {} of {} missing integrity_hash",
1437                            line_idx,
1438                            file.display(),
1439                        ))
1440                    })?;
1441                // Now do the full typed parse — guaranteed by the
1442                // pre-pass to be a structurally valid event JSON
1443                // with an `integrity_hash` field present.
1444                let event: AuditEvent = serde_json::from_value(bare).map_err(|e| {
1445                    CoreError::AuditError(format!(
1446                        "Failed to typed-parse line {} of {}: {}",
1447                        line_idx,
1448                        file.display(),
1449                        e
1450                    ))
1451                })?;
1452                let stored = event.integrity_hash.clone();
1453                let recomputed = Self::compute_integrity_hash(&event, &prev_hash)?;
1454                events_checked = events_checked.saturating_add(1);
1455                if stored != recomputed {
1456                    mismatch = Some(ChainMismatch {
1457                        file: file.clone(),
1458                        line: line_idx,
1459                        event_id: event.id.clone(),
1460                        stored_hash: stored,
1461                        expected_hash: recomputed,
1462                    });
1463                    break;
1464                }
1465
1466                // Cross-file anchor check: when this is the first
1467                // event of a non-initial file, it must be a chain-link
1468                // anchor whose metadata matches the prior file we
1469                // just verified.
1470                if first_event_in_file {
1471                    if let (Some(prev_name), Some(prev_final_hash)) =
1472                        (previous_file_name.as_deref(), previous_file_final_hash.as_deref())
1473                    {
1474                        // Constant-time compare on the chain hash:
1475                        // it's a hex-encoded SHA-256 that authenticates
1476                        // the prior file's tail state, and an attacker
1477                        // who can time `verify_chain` could otherwise
1478                        // recover it byte-by-byte. The action verb and
1479                        // filename are public values (the rotation
1480                        // produces deterministic timestamped names),
1481                        // so they don't need CT-compare — but we use
1482                        // the same primitive for consistency.
1483                        //
1484                        // The three `ct_eq` results are combined with
1485                        // `subtle::Choice::&` BEFORE the final
1486                        // `.into() -> bool`. A naïve `bool && bool &&
1487                        // bool` shape would short-circuit on the first
1488                        // false result, leaking via timing which of
1489                        // the three checks failed. Combining
1490                        // `Choice<u8>` values keeps the comparison
1491                        // data-independent.
1492                        use subtle::ConstantTimeEq;
1493                        let action_eq =
1494                            event.action.as_bytes().ct_eq(CHAIN_ANCHOR_ACTION.as_bytes());
1495                        let prev_file_eq = event
1496                            .metadata
1497                            .get(CHAIN_ANCHOR_PREV_FILE_KEY)
1498                            .map(|v| v.as_bytes().ct_eq(prev_name.as_bytes()))
1499                            .unwrap_or_else(|| 0u8.ct_eq(&1u8));
1500                        let prev_hash_eq = event
1501                            .metadata
1502                            .get(CHAIN_ANCHOR_PREV_HASH_KEY)
1503                            .map(|v| v.as_bytes().ct_eq(prev_final_hash.as_bytes()))
1504                            .unwrap_or_else(|| 0u8.ct_eq(&1u8));
1505                        let consistent: bool = (action_eq & prev_file_eq & prev_hash_eq).into();
1506                        if !consistent {
1507                            // Surface as a chain mismatch on this line —
1508                            // matches the existing reporting shape so
1509                            // callers don't need a second error
1510                            // variant for "anchor missing/wrong."
1511                            mismatch = Some(ChainMismatch {
1512                                file: file.clone(),
1513                                line: line_idx,
1514                                event_id: event.id.clone(),
1515                                stored_hash: stored.clone(),
1516                                expected_hash: format!(
1517                                    "expected chain-link from previous_file={prev_name:?} \
1518                                     previous_hash={prev_final_hash}"
1519                                ),
1520                            });
1521                            break;
1522                        }
1523                    }
1524                    first_event_in_file = false;
1525                }
1526
1527                prev_hash = recomputed;
1528            }
1529            if mismatch.is_some() {
1530                break;
1531            }
1532            // Capture this file's tail state so the next iteration can
1533            // validate its chain-link anchor.
1534            previous_file_name = file.file_name().and_then(|n| n.to_str()).map(str::to_string);
1535            previous_file_final_hash = Some(prev_hash.clone());
1536        }
1537
1538        Ok(ChainVerificationReport { files_checked: log_files.len(), events_checked, mismatch })
1539    }

1541    /// Write an audit event to the current file.
1542    fn write_event_to_file(&self, event: &mut AuditEvent) -> Result<()> {
1543        let mut file_state = self.file_state.lock();
1544
1545        // Rotate if needed
1546        self.rotate_if_needed(&mut file_state)?;
1547
1548        let state = file_state
1549            .as_mut()
1550            .ok_or_else(|| CoreError::AuditError("No active audit file".to_string()))?;
1551
1552        // Compute integrity hash with chain
1553        let previous_hash = self.previous_hash.read().clone();
1554        event.integrity_hash = Self::compute_integrity_hash(event, &previous_hash)?;
1555
1556        // Update previous hash for next event
1557        {
1558            let mut prev = self.previous_hash.write();
1559            prev.clone_from(&event.integrity_hash);
1560        }
1561
1562        // Serialize event to JSON
1563        let json = serde_json::to_string(event).map_err(|e| {
1564            CoreError::AuditError(format!("Failed to serialize audit event: {}", e))
1565        })?;
1566
1567        // Write JSON line
1568        let line = format!("{}\n", json);
1569        let line_bytes = line.as_bytes();
1570
1571        state
1572            .writer
1573            .write_all(line_bytes)
1574            .map_err(|e| CoreError::AuditError(format!("Failed to write audit event: {}", e)))?;
1575
1576        // Update size tracking. `usize → u64` is widening on 64-bit and
1577        // equal on 32-bit, so the conversion is always lossless — but
1578        // route via `try_from` for consistency with the rest of this
1579        // file and to silence `clippy::cast_possible_truncation`.
1580        let line_len_u64 = u64::try_from(line_bytes.len()).unwrap_or(u64::MAX);
1581        state.current_size = state.current_size.saturating_add(line_len_u64);
1582
1583        Ok(())
1584    }
1585}

2443    /// A single audit JSONL line longer than `MAX_LINE_LEN` bytes with
2444    /// no early newline must return `AuditError` promptly rather than
2445    /// livelocking the inner `BufRead::fill_buf` loop in `verify_chain`.
2446    /// Asserts both correctness AND a wall-clock budget — a previous
2447    /// implementation hit `extend_from_slice(empty)` + `consume(0)`
2448    /// indefinitely on cap exceedance, so a regression that re-opens
2449    /// the same shape would never return without the budget check.
2450    #[test]
2451    fn test_verify_chain_over_cap_line_returns_error_promptly_fails() {
2452        use std::io::Write;
2453        use std::time::{Duration, Instant};
2454
2455        let temp_dir = TempDir::new().unwrap();
2456        let storage_dir = temp_dir.path().join("audit");
2457        std::fs::create_dir_all(&storage_dir).unwrap();
2458
2459        // Write a synthetic JSONL with one line of `MAX_LINE_LEN + 1024`
2460        // bytes (no newline before that point) followed by a newline.
2461        // `1 MiB + 1 KiB` is comfortably over the cap (1 MiB) and small
2462        // enough not to balloon test memory.
2463        let line_len = (1024 * 1024) + 1024;
2464        let file_path = storage_dir.join("audit-test.jsonl");
2465        {
2466            let mut f = std::fs::File::create(&file_path).unwrap();
2467            f.write_all(&vec![b'A'; line_len]).unwrap();
2468            f.write_all(b"\n").unwrap();
2469            f.flush().unwrap();
2470        }
2471
2472        let config = AuditConfig::new(storage_dir.clone());
2473        let storage = FileAuditStorage::new(config).unwrap();
2474        let started = Instant::now();
2475        let result = storage.verify_chain();
2476        let elapsed = started.elapsed();
2477
2478        match result {
2479            Err(CoreError::AuditError(msg)) => {
2480                assert!(
2481                    msg.contains("exceeds maximum length"),
2482                    "expected 'exceeds maximum length' error, got: {msg}"
2483                );
2484            }
2485            other => panic!("expected over-cap AuditError, got {other:?}"),
2486        }
2487        // Pre-fix this test would never return; cap the budget so a
2488        // future regression that re-introduces the livelock fails the
2489        // suite instead of hanging it. 5s is generous for a 1 MiB file
2490        // even on the slowest CI runner.
