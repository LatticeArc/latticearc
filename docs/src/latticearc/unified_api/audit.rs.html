1//! # Persistent Audit Storage with Rotation
2//!
3//! Provides tamper-evident audit logging for cryptographic operations.
4//! Events are persisted to disk in JSON Lines format with automatic rotation
5//! based on file size and age.
6//!
7//! ## Security Features
8//!
9//! - **Integrity Verification**: SHA-256 hash chain for tamper detection
10//! - **Automatic Rotation**: Files rotate based on size and age limits
11//! - **Retention Policies**: Configurable retention periods for compliance
12//! - **Thread-Safe**: All operations are safe for concurrent access
13//!
14//! ## Usage
15//!
16//! ```rust,no_run
17//! use latticearc::unified_api::audit::{AuditConfig, FileAuditStorage, AuditStorage, AuditEvent, AuditEventType, AuditOutcome};
18//! use std::path::PathBuf;
19//! use std::time::Duration;
20//!
21//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
22//! let config = AuditConfig {
23//!     storage_path: PathBuf::from("/var/log/latticearc/audit"),
24//!     max_file_size_bytes: 100 * 1024 * 1024, // 100MB
25//!     max_file_age: Duration::from_secs(24 * 60 * 60), // 24 hours
26//!     retention_days: 90,
27//! };
28//!
29//! let storage = FileAuditStorage::new(config)?;
30//!
31//! // Create and write an audit event
32//! let event = AuditEvent::new(
33//!     AuditEventType::CryptoOperation,
34//!     "encrypt_data",
35//!     AuditOutcome::Success,
36//! );
37//! storage.write(&event)?;
38//! # Ok(())
39//! # }
40//! ```

130    ///
131    /// Bounds memory + hashing cost when caller-supplied input is routed
132    /// through audit. Beyond the cap, `with_metadata` silently drops the
133    /// entry rather than rejecting the build (audit emission must never
134    /// fail-open by aborting the operation that produced the event).
135    pub const MAX_METADATA_ENTRIES: usize = 32;
136    /// Maximum byte length per metadata key.
137    pub const MAX_METADATA_KEY_LEN: usize = 256;
138    /// Maximum byte length per metadata value.
139    pub const MAX_METADATA_VALUE_LEN: usize = 4096;

141    /// Add metadata to this event.
142    ///
143    /// Caps:
144    /// - Keys longer than [`MAX_METADATA_KEY_LEN`](Self::MAX_METADATA_KEY_LEN)
145    ///   are truncated.
146    /// - Values longer than [`MAX_METADATA_VALUE_LEN`](Self::MAX_METADATA_VALUE_LEN)
147    ///   are truncated.
148    /// - Beyond [`MAX_METADATA_ENTRIES`](Self::MAX_METADATA_ENTRIES) entries
149    ///   the call is a no-op (audit emission must not abort the operation
150    ///   that triggered it).
151    ///
152    /// These caps prevent a DoS amplification path where attacker-controlled
153    /// strings (e.g. SignedData.scheme, a request header) are routed through
154    /// audit-event metadata: without them, every audit event could carry
155    /// unbounded heap + hashing cost.
156    #[must_use]
157    pub fn with_metadata(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
158        if self.metadata.len() >= Self::MAX_METADATA_ENTRIES {
159            return self;
160        }
161        let mut k = key.into();
162        let mut v = value.into();
163        truncate_utf8_safe(&mut k, Self::MAX_METADATA_KEY_LEN);
164        truncate_utf8_safe(&mut v, Self::MAX_METADATA_VALUE_LEN);
165        self.metadata.insert(k, v); // LINT-OK: canonical-with-metadata-impl
166        self
167    }

224/// Truncate a UTF-8 String to at most `max_bytes` bytes without splitting
225/// a multi-byte code point. `String::truncate` panics on a non-boundary
226/// index — for caller-controlled text we round DOWN to the nearest valid
227/// boundary so attacker-supplied UTF-8 can't crash the audit pipeline.
228fn truncate_utf8_safe(s: &mut String, max_bytes: usize) {
229    if s.len() <= max_bytes {
230        return;
231    }
232    let mut cut = max_bytes;
233    while cut > 0 && !s.is_char_boundary(cut) {
234        cut = cut.saturating_sub(1);
235    }
236    s.truncate(cut);
237}

442/// Parse the timestamp embedded in an audit-log filename.
443///
444/// Audit logs are named `audit-YYYY-MM-DDTHH-MM-SS.jsonl` (created by
445/// `FileAuditStorage::create_new_file`). This helper inverts that
446/// formatting and returns the embedded creation time. Used by the
447/// retention sweep so cleanup decisions don't depend on filesystem
448/// `mtime`, which a privileged attacker can rewrite with `touch`.
449fn parse_audit_filename_timestamp(file_name: &str) -> Option<DateTime<Utc>> {
450    let stem = file_name.strip_prefix("audit-")?.strip_suffix(".jsonl")?;
451    // The format string mirrors `create_new_file`'s
452    // `now.format("%Y-%m-%dT%H-%M-%S")`. Treat the parsed `NaiveDateTime`
453    // as UTC since the producer writes UTC clock readings.
454    let naive = chrono::NaiveDateTime::parse_from_str(stem, "%Y-%m-%dT%H-%M-%S").ok()?;
455    Some(DateTime::<Utc>::from_naive_utc_and_offset(naive, Utc))
456}

533    config: AuditConfig,

540/// Filename for the per-storage genesis anchor.
541///
542/// The genesis anchor is the `previous_hash` value used for the very first
543/// audit event in a storage. It is derived from a domain label, a fresh
544/// random nonce, and the creation timestamp, and is persisted alongside
545/// the audit log so that:
546///
547///   * a truncate-only attack (delete log entries, leave genesis intact)
548///     is detectable: the next event chains from `genesis`, not from the
549///     deleted entry's hash, and any verifier replaying the chain spots
550///     the gap.
551///   * an attacker who deletes both the log and the genesis to fabricate
552///     a "fresh storage" ends up with a different genesis on the next
553///     create — an external pinning store (HSM, write-once log) can spot
554///     the genesis change. Total erasure with no external state is
555///     fundamentally undetectable; this fix closes the
556///     truncate-and-restart-only path the empty-genesis behaviour left
557///     wide open.
558const AUDIT_GENESIS_FILENAME: &str = "genesis";
559
560/// Domain-separation label baked into the genesis hash. Distinct from
561/// every other transcript prefix in the crate so a genesis bytestring
562/// cannot collide with any other artifact.
563const AUDIT_GENESIS_DOMAIN_LABEL: &[u8] = b"latticearc-audit-genesis-v1";
564
565impl FileAuditStorage {
566    /// Create a new file-based audit storage.
567    ///
568    /// Creates the storage directory if it doesn't exist, and ensures a
569    /// persisted genesis anchor file exists (creating one with a fresh
570    /// random nonce on first run, reading the existing one on subsequent
571    /// runs).
572    ///
573    /// # Errors
574    ///
575    /// Returns an error if the storage directory cannot be created, the
576    /// genesis file cannot be read or written, or the random nonce
577    /// generation for a new genesis fails.
578    pub fn new(config: AuditConfig) -> Result<Arc<Self>> {
579        // Create storage directory if it doesn't exist
580        fs::create_dir_all(&config.storage_path).map_err(|e| {
581            CoreError::AuditError(format!(
582                "Failed to create audit directory '{}': {}",
583                config.storage_path.display(),
584                e
585            ))
586        })?;
587
588        let genesis = Self::load_or_create_genesis(&config.storage_path)?;
589
590        let storage = Arc::new(Self {
591            config,
592            file_state: Mutex::new(None),
593            previous_hash: RwLock::new(genesis),
594        });
595
596        // Clean up old files based on retention policy
597        storage.cleanup_old_files()?;
598
599        Ok(storage)
600    }
601
602    /// Read the on-disk genesis if it exists, otherwise create one.
603    ///
604    /// The created genesis is `SHA-256(domain_label || nonce || timestamp)`
605    /// hex-encoded, where `nonce` is 32 bytes from the system CSPRNG and
606    /// `timestamp` is the RFC 3339 creation time.
607    fn load_or_create_genesis(storage_path: &std::path::Path) -> Result<String> {
608        let genesis_path = storage_path.join(AUDIT_GENESIS_FILENAME);
609
610        match fs::read_to_string(&genesis_path) {
611            Ok(existing) => {
612                let trimmed = existing.trim().to_string();
613                if trimmed.is_empty() {
614                    return Err(CoreError::AuditError(format!(
615                        "Audit genesis file '{}' exists but is empty; refusing to start \
616                         with an empty chain anchor",
617                        genesis_path.display()
618                    )));
619                }
620                Ok(trimmed)
621            }
622            Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
623                let nonce = crate::primitives::rand::csprng::random_bytes(32);
624                let timestamp = Utc::now().to_rfc3339();
625                let mut buf = Vec::with_capacity(
626                    AUDIT_GENESIS_DOMAIN_LABEL
627                        .len()
628                        .saturating_add(4)
629                        .saturating_add(nonce.len())
630                        .saturating_add(4)
631                        .saturating_add(timestamp.len()),
632                );
633                buf.extend_from_slice(AUDIT_GENESIS_DOMAIN_LABEL);
634                let nonce_len = u32::try_from(nonce.len()).map_err(|_e| {
635                    CoreError::AuditError("Genesis nonce length exceeds u32::MAX".to_string())
636                })?;
637                buf.extend_from_slice(&nonce_len.to_be_bytes());
638                buf.extend_from_slice(&nonce);
639                let ts_len = u32::try_from(timestamp.len()).map_err(|_e| {
640                    CoreError::AuditError("Genesis timestamp length exceeds u32::MAX".to_string())
641                })?;
642                buf.extend_from_slice(&ts_len.to_be_bytes());
643                buf.extend_from_slice(timestamp.as_bytes());
644
645                let digest = crate::primitives::hash::sha2::sha256(&buf).map_err(|e| {
646                    CoreError::AuditError(format!("Failed to hash genesis material: {e}"))
647                })?;
648                let hex = hex::encode(digest);
649
650                // Persist atomically with restrictive permissions on Unix.
651                #[cfg(unix)]
652                {
653                    use std::io::Write as _;
654                    use std::os::unix::fs::OpenOptionsExt;
655                    let mut f = OpenOptions::new()
656                        .create_new(true)
657                        .write(true)
658                        .mode(0o600)
659                        // LINT-OK: mode-on-prev-line — chained above.
660                        .open(&genesis_path)
661                        .map_err(|err| {
662                            CoreError::AuditError(format!(
663                                "Failed to create audit genesis file '{}': {}",
664                                genesis_path.display(),
665                                err
666                            ))
667                        })?;
668                    f.write_all(hex.as_bytes()).map_err(|err| {

702    /// Compute the integrity hash for an event.
703    ///
704    /// The hash includes the previous event's hash to create a chain,
705    /// making tampering detectable. Routes through the
706    /// [`crate::primitives::hash::sha2::sha256`] wrapper so audit integrity
707    /// uses the same hash call path as the rest of the crate.
708    ///
709    /// made `pub(crate)` so the public
710    /// `verify_chain` helper below can re-use it without duplicating
711    /// the field-encoding rules.
712    ///
713    /// # Errors
714    /// Returns an error if the SHA-256 primitive fails (input exceeds 1 GiB guard).
715    pub(crate) fn compute_integrity_hash(
716        event: &AuditEvent,
717        previous_hash: &str,
718    ) -> Result<String> {
719        // Length-prefix every field so prefix-collision attacks are
720        // impossible (`"ab" + "c"` and `"a" + "bc"` no longer hash to the
721        // same digest). The metadata caps already bound each field, but a
722        // 4-byte BE length per element is the cheap and definitive fix.
723        // (Encoding is BE per the L3 transcript-convention migration in
724        // 85e2bd79e — see `append_lenp_field` doc comment for details.)
725        let mut buf = Vec::new();
726        Self::append_lenp_field(&mut buf, previous_hash.as_bytes())?;
727        Self::append_lenp_field(&mut buf, event.id.as_bytes())?;
728        Self::append_lenp_field(&mut buf, event.timestamp.to_rfc3339().as_bytes())?;
729        Self::append_lenp_field(&mut buf, event.event_type.to_string().as_bytes())?;
730
731        // Optional fields are encoded as a discriminator byte followed by
732        // the length-prefixed bytes; absence vs empty is now distinguishable.
733        match event.actor.as_ref() {
734            Some(a) => {
735                buf.push(1);
736                Self::append_lenp_field(&mut buf, a.as_bytes())?;
737            }
738            None => buf.push(0),
739        }
740        match event.resource.as_ref() {
741            Some(r) => {
742                buf.push(1);
743                Self::append_lenp_field(&mut buf, r.as_bytes())?;
744            }
745            None => buf.push(0),
746        }
747
748        Self::append_lenp_field(&mut buf, event.action.as_bytes())?;
749        Self::append_lenp_field(&mut buf, event.outcome.to_string().as_bytes())?;
750
751        // Metadata: include count, then sorted (key, value) pairs each
752        // with their own length prefix. Overflow at `u32::MAX` (4 G
753        // entries) propagates as `AuditError`, symmetric with the
754        // length-prefix overflow path above. Length prefixes are
755        // big-endian to match the transcript convention used by
756        // `zkp::sigma::compute_challenge` (round-12 audit fix L3 —
757        // unifies endianness across all transcript-style hashing in
758        // the crate).
759        let mut metadata_keys: Vec<&String> = event.metadata.keys().collect();
760        metadata_keys.sort();
761        let count = u32::try_from(metadata_keys.len()).map_err(|_e| {
762            CoreError::AuditError("integrity hash metadata count exceeds 2^32".to_string())
763        })?;
764        buf.extend_from_slice(&count.to_be_bytes());
765        for key in metadata_keys {
766            Self::append_lenp_field(&mut buf, key.as_bytes())?;
767            if let Some(value) = event.metadata.get(key) {
768                Self::append_lenp_field(&mut buf, value.as_bytes())?;
769            } else {
770                Self::append_lenp_field(&mut buf, &[])?;
771            }
772        }
773
774        // `sha256` only fails on inputs larger than 1 GiB (resource-limit guard),
775        // which no reasonable audit event approaches.
776        let digest = crate::primitives::hash::sha2::sha256(&buf)
777            .map_err(|e| CoreError::AuditError(format!("integrity hash failed: {}", e)))?;
778        Ok(hex::encode(digest))
779    }
780
781    /// Append `field.len() as u32 BE` followed by `field` to `buf`.
782    ///
783    /// Returns `Err(CoreError::AuditError)` if `field.len()` exceeds
784    /// `u32::MAX` bytes (4 GiB) — symmetric with the overflow handling
785    /// in `zkp::sigma::compute_challenge` (round-21 audit fix #7). The
786    /// length is encoded big-endian to match the transcript convention
787    /// used by `zkp::sigma::compute_challenge` (round-12 audit fix L3 —
788    /// the previous LE encoding was an isolated outlier within the
789    /// crate's transcript-style hashing).
790    /// The previous saturating-to-`u32::MAX` form was a silent collapse
791    /// that would let two distinct field values share the same length
792    /// prefix; while the SHA-256 backend's 1 GiB cap makes this
793    /// unreachable today, an explicit error preserves the asymmetric
794    /// defensive posture the rest of the crate now uses.
795    fn append_lenp_field(buf: &mut Vec<u8>, field: &[u8]) -> Result<()> {
796        let len = u32::try_from(field.len()).map_err(|_e| {
797            CoreError::AuditError("integrity hash field exceeds 2^32 bytes".to_string())
798        })?;
799        buf.extend_from_slice(&len.to_be_bytes());
800        buf.extend_from_slice(field);
801        Ok(())
802    }

804    /// Check if the current file needs rotation.
805    fn needs_rotation(&self, state: &FileState) -> bool {
806        // Check size limit
807        if state.current_size >= self.config.max_file_size_bytes {
808            return true;
809        }
810
811        // Check age limit
812        let age =
813            Utc::now().signed_duration_since(state.created_at).to_std().unwrap_or(Duration::ZERO);
814
815        age >= self.config.max_file_age
816    }

818    /// Rotate the current file if needed.
819    fn rotate_if_needed(&self, state: &mut Option<FileState>) -> Result<()> {
820        let should_rotate = state.as_ref().is_some_and(|s| self.needs_rotation(s));
821
822        if should_rotate {
823            // Close current file and create new one
824            if let Some(mut old_state) = state.take() {
825                tracing::info!(
826                    "Rotating audit file: {} (size: {} bytes)",
827                    old_state.current_path.display(),
828                    old_state.current_size
829                );
830                old_state.writer.flush().map_err(|e| {
831                    CoreError::AuditError(format!("Failed to flush audit file: {}", e))
832                })?;
833            }
834        }
835
836        // Create new file if needed
837        if state.is_none() {
838            *state = Some(self.create_new_file()?);
839        }
840
841        Ok(())
842    }

844    /// Create a new audit file.
845    fn create_new_file(&self) -> Result<FileState> {
846        let now = Utc::now();
847        let filename = format!("audit-{}.jsonl", now.format("%Y-%m-%dT%H-%M-%S"));
848        let path = self.config.storage_path.join(&filename);
849
850        let file = {
851            // On Unix, set 0o600 atomically via OpenOptions::mode() so the
852            // file is never world-readable, even briefly. Audit logs may
853            // contain operation context (key IDs, paths, actors) and must
854            // not inherit the default umask the way the previous bare
855            // `OpenOptions::new().create(...).open(...)` did.
856            #[cfg(unix)]
857            {
858                use std::os::unix::fs::OpenOptionsExt;
859                OpenOptions::new().create(true).append(true).mode(0o600).open(&path).map_err(
860                    |e| {
861                        CoreError::AuditError(format!(
862                            "Failed to create audit file '{}': {}",
863                            path.display(),
864                            e
865                        ))
866                    },
867                )?
868            }
869            #[cfg(not(unix))]
870            {
871                // LINT-OK: cfg-not-unix — Windows / non-Unix has no
872                // mode bits; ACL hardening is the caller's job.
873                OpenOptions::new().create(true).append(true).open(&path).map_err(|e| {
874                    CoreError::AuditError(format!(
875                        "Failed to create audit file '{}': {}",
876                        path.display(),
877                        e
878                    ))
879                })?
880            }
881        };
882
883        tracing::debug!("Created new audit file: {}", path.display());
884
885        Ok(FileState {
886            writer: BufWriter::new(file),
887            current_path: path,
888            current_size: 0,
889            created_at: now,
890        })
891    }

893    /// Clean up old audit files based on retention policy.
894    fn cleanup_old_files(&self) -> Result<()> {
895        // Belt-and-braces: even with `with_retention_days` rejecting
896        // `0`, a `Default` or struct-literal construction could still
897        // land here with `retention_days = 0`. Fail closed instead of
898        // wiping the directory.
899        if self.config.retention_days == 0 {
900            return Err(CoreError::AuditError(
901                "AuditConfig.retention_days = 0 would delete every audit file on \
902                 cleanup; refusing to proceed (set with_retention_days(>= 1))"
903                    .to_string(),
904            ));
905        }
906
907        let retention_duration = chrono::Duration::days(i64::from(self.config.retention_days));
908        let Some(cutoff) = Utc::now().checked_sub_signed(retention_duration) else {
909            return Err(CoreError::AuditError(format!(
910                "Retention period of {} days overflows date arithmetic",
911                self.config.retention_days
912            )));
913        };
914
915        let entries = fs::read_dir(&self.config.storage_path).map_err(|e| {
916            CoreError::AuditError(format!(
917                "Failed to read audit directory '{}': {}",
918                self.config.storage_path.display(),
919                e
920            ))
921        })?;
922
923        for entry in entries {
924            let Ok(entry) = entry else { continue };
925
926            let path = entry.path();
927
928            // Only process .jsonl files
929            if path.extension().and_then(|e| e.to_str()) != Some("jsonl") {
930                continue;
931            }
932
933            // Use the timestamp embedded in the filename
934            // (`audit-YYYY-MM-DDTHH-MM-SS.jsonl`, set by
935            // `create_new_file`) instead of the filesystem mtime,
936            // which a privileged attacker can rewrite with `touch`.
937            // The filename is set at file creation by this process and
938            // is not modifiable without also changing the file's name,
939            // which is observable in directory listings and recorded
940            // in the audit-genesis chain. Files whose names don't
941            // parse — including operator-imported logs from another
942            // tool — are skipped rather than deleted.
943            let Some(file_name) = path.file_name().and_then(|n| n.to_str()) else {
944                continue;
945            };
946            let Some(created_at) = parse_audit_filename_timestamp(file_name) else {
947                tracing::debug!("Audit cleanup: skipping non-conforming filename '{}'", file_name);
948                continue;
949            };
950
951            if created_at < cutoff {
952                // Delete old file
953                if let Err(e) = fs::remove_file(&path) {
954                    tracing::warn!("Failed to remove old audit file '{}': {}", path.display(), e);
955                } else {
956                    tracing::info!("Removed old audit file: {}", path.display());
957                }
958            }
959        }
960
961        Ok(())
962    }

964    /// public hash-chain verification.
965    ///
966    /// Walks every `audit-*.jsonl` file in the storage directory in
967    /// filename-timestamp order, recomputes each entry's
968    /// `integrity_hash` from the persisted genesis, and reports the
969    /// first divergence (or `Ok(report)` with no mismatches when the
970    /// chain is intact). Tamper detection previously required
971    /// hand-writing a verifier in the right encoding — the asymmetry
972    /// tamper-evident logging exists to remove.
973    ///
974    /// The returned [`ChainVerificationReport`] carries the line count
975    /// inspected and an `Option<Mismatch>` describing the first event
976    /// whose recomputed hash differed from its persisted
977    /// `integrity_hash` field.
978    ///
979    /// # Errors
980    /// Returns an error if the storage directory cannot be read, a
981    /// log file cannot be opened, a line cannot be parsed as JSON, or
982    /// the hash recomputation fails.
983    pub fn verify_chain(&self) -> Result<ChainVerificationReport> {
984        let mut log_files: Vec<PathBuf> = Vec::new();
985        let entries = fs::read_dir(&self.config.storage_path)
986            .map_err(|e| CoreError::AuditError(format!("Failed to read audit dir: {}", e)))?;
987        for entry in entries.flatten() {
988            let path = entry.path();
989            let Some(name) = path.file_name().and_then(|n| n.to_str()) else { continue };
990            // Match `audit-*.jsonl` (case-insensitive on extension; round-26
991            // audit fix L for case-sensitive .jsonl bug).
992            let lower = name.to_ascii_lowercase();
993            if lower.starts_with("audit-") && lower.ends_with(".jsonl") {
994                log_files.push(path);
995            }
996        }
997        // Filename-timestamp ordering: `audit-YYYYMMDD-HHMMSS-NNN.jsonl`
998        // is lexicographically chronological.
999        log_files.sort();
1000
1001        // Recompute from genesis. Reads the persisted genesis directly
1002        // rather than the in-memory `previous_hash` so verification
1003        // works on a freshly-mounted storage with no prior writes in
1004        // this process.
1005        let genesis = Self::load_or_create_genesis(&self.config.storage_path)?;
1006        let mut prev_hash = genesis;
1007        let mut events_checked: u64 = 0;
1008        let mut mismatch: Option<ChainMismatch> = None;
1009
1010        for file in &log_files {
1011            use std::io::BufRead;
1012            let f = File::open(file).map_err(|e| {
1013                CoreError::AuditError(format!("Failed to open {}: {}", file.display(), e))
1014            })?;
1015            let reader = std::io::BufReader::new(f);
1016            for (line_idx, line_res) in reader.lines().enumerate() {
1017                let line = line_res
1018                    .map_err(|e| CoreError::AuditError(format!("Failed to read line: {}", e)))?;
1019                if line.trim().is_empty() {
1020                    continue;
1021                }
1022                let event: AuditEvent = serde_json::from_str(&line).map_err(|e| {
1023                    CoreError::AuditError(format!(
1024                        "Failed to parse line {} of {}: {}",
1025                        line_idx,
1026                        file.display(),
1027                        e
1028                    ))
1029                })?;
1030                let stored = event.integrity_hash.clone();
1031                let recomputed = Self::compute_integrity_hash(&event, &prev_hash)?;
1032                events_checked = events_checked.saturating_add(1);
1033                if stored != recomputed {
1034                    mismatch = Some(ChainMismatch {
1035                        file: file.clone(),
1036                        line: line_idx,
1037                        event_id: event.id.clone(),
1038                        stored_hash: stored,
1039                        expected_hash: recomputed,
1040                    });
1041                    break;
1042                }
1043                prev_hash = recomputed;
1044            }
1045            if mismatch.is_some() {
1046                break;
1047            }
1048        }
1049
1050        Ok(ChainVerificationReport { files_checked: log_files.len(), events_checked, mismatch })
1051    }

1053    /// Write an audit event to the current file.
1054    fn write_event_to_file(&self, event: &mut AuditEvent) -> Result<()> {
1055        let mut file_state = self.file_state.lock();
1056
1057        // Rotate if needed
1058        self.rotate_if_needed(&mut file_state)?;
1059
1060        let state = file_state
1061            .as_mut()
1062            .ok_or_else(|| CoreError::AuditError("No active audit file".to_string()))?;
1063
1064        // Compute integrity hash with chain
1065        let previous_hash = self.previous_hash.read().clone();
1066        event.integrity_hash = Self::compute_integrity_hash(event, &previous_hash)?;
1067
1068        // Update previous hash for next event
1069        {
1070            let mut prev = self.previous_hash.write();
1071            prev.clone_from(&event.integrity_hash);
1072        }
1073
1074        // Serialize event to JSON
1075        let json = serde_json::to_string(event).map_err(|e| {
1076            CoreError::AuditError(format!("Failed to serialize audit event: {}", e))
1077        })?;
1078
1079        // Write JSON line
1080        let line = format!("{}\n", json);
1081        let line_bytes = line.as_bytes();
1082
1083        state
1084            .writer
1085            .write_all(line_bytes)
1086            .map_err(|e| CoreError::AuditError(format!("Failed to write audit event: {}", e)))?;
1087
1088        // Update size tracking. `usize → u64` is widening on 64-bit and
1089        // equal on 32-bit, so the conversion is always lossless — but
1090        // route via `try_from` for consistency with the rest of this file
1091        // and to silence `clippy::cast_possible_truncation` (round-20
1092        // audit fix #24).
1093        let line_len_u64 = u64::try_from(line_bytes.len()).unwrap_or(u64::MAX);
1094        state.current_size = state.current_size.saturating_add(line_len_u64);
1095
1096        Ok(())
1097    }
1098}

1793        if let Ok(dir) = temp_dir {
1794            let dir_tiny = dir.path().join("tiny");
1795            fs::create_dir_all(&dir_tiny).unwrap();
1796            let storage_tiny =
1797                FileAuditStorage::new(AuditConfig::new(dir_tiny.clone()).with_max_file_size(1));
1798            if let Ok(s) = storage_tiny {
1799                // Write one event — its JSON is >1 byte, so current_size > threshold after write.
1800                // The next write will trigger needs_rotation() == true.
1801                let event1 = AuditEvent::new(
1802                    AuditEventType::CryptoOperation,
1803                    "op_first",
1804                    AuditOutcome::Success,
1805                );
1806                s.write(&event1).unwrap();
1807
1808                // Verify that writing a second event also succeeds (rotation runs, old file
1809                // is flushed and a new FileState is opened).
1810                let event2 = AuditEvent::new(
1811                    AuditEventType::CryptoOperation,
1812                    "op_second",
1813                    AuditOutcome::Success,
1814                );
1815                // This write must succeed — rotation must handle the 1-byte overflow gracefully.
1816                assert!(
1817                    s.write(&event2).is_ok(),
1818                    "Write after size-triggered rotation must succeed"
1819                );
1820                s.flush().unwrap();
1821            }
1822
1823            // With a large limit, writing many events never triggers overflow errors.
1824            let dir_large = dir.path().join("large");
1825            fs::create_dir_all(&dir_large).unwrap();
1826            let storage_large = FileAuditStorage::new(
1827                AuditConfig::new(dir_large.clone()).with_max_file_size(100 * 1024 * 1024),
1828            );
1829            if let Ok(s) = storage_large {
1830                for i in 0..5 {
1831                    let event = AuditEvent::new(
1832                        AuditEventType::CryptoOperation,
1833                        &format!("op_{}", i),
1834                        AuditOutcome::Success,
1835                    );
1836                    s.write(&event).unwrap();
1837                }
1838                s.flush().unwrap();
1839
1840                // All events land in a single file (no rotation needed).
1841                let file_count = fs::read_dir(&dir_large)
1842                    .unwrap()
1843                    .filter_map(|e| e.ok())
1844                    .filter(|e| e.path().extension().and_then(|x| x.to_str()) == Some("jsonl"))
1845                    .count();
1846                assert_eq!(
1847                    file_count, 1,
1848                    "max_file_size_bytes=100MB must not rotate for 5 small events (got {})",
1849                    file_count
1850                );
1851            }
1852        }
1853    }
