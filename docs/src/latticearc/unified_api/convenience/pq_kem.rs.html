1//! Post-quantum KEM operations (ML-KEM)
2//!
3//! This module provides post-quantum key encapsulation mechanism operations
4//! using ML-KEM (FIPS 203) with **FIPS 140-3 validated** aws-lc-rs.
5//!
6//! Encryption uses ML-KEM encapsulation to derive a shared secret, then encrypts
7//! the payload with AES-GCM using that shared secret as the key. The wire format
8//! is: `ML-KEM ciphertext || AES-GCM encrypted data`.
9//!
10//! # Unified API with SecurityMode
11//!
12//! All cryptographic operations use `SecurityMode` to specify verification behavior:
13//! - `SecurityMode::Verified(&session)`: Validates session before operation
14//! - `SecurityMode::Unverified`: Skips session validation
15//!
16//! The `_unverified` variants are opt-out functions for scenarios where Zero Trust
17//! verification is not required or not possible.
18

32/// Internal implementation of ML-KEM encryption.
33fn encrypt_pq_ml_kem_internal(
34    data: &[u8],
35    ml_kem_pk: &[u8],
36    security_level: MlKemSecurityLevel,
37) -> Result<Vec<u8>> {
38    crate::log_crypto_operation_start!(
39        "encrypt_pq_ml_kem",
40        security_level = ?security_level,
41        data_len = data.len()
42    );
43
44    validate_encryption_size(data.len()).map_err(|e| {
45        crate::log_crypto_operation_error!("encrypt_pq_ml_kem", "resource limit exceeded");
46        CoreError::ResourceExceeded(e.to_string())
47    })?;
48
49    let pk =
50        crate::primitives::kem::ml_kem::MlKemPublicKey::new(security_level, ml_kem_pk.to_vec())
51            .map_err(|e| {
52                crate::log_crypto_operation_error!("encrypt_pq_ml_kem", e);
53                CoreError::InvalidInput("Invalid ML-KEM public key format".to_string())
54            })?;
55
56    let mut rng = rand::rngs::OsRng;
57    let (shared_secret, ciphertext) = MlKem::encapsulate(&mut rng, &pk).map_err(|e| {
58        crate::log_crypto_operation_error!("encrypt_pq_ml_kem", "encapsulation failed");
59        CoreError::EncryptionFailed(format!("ML-KEM encapsulation failed: {}", e))
60    })?;
61
62    // Use shared secret to encrypt data with AES-GCM
63    let symmetric_key = shared_secret.as_bytes();
64    let encrypted_data = encrypt_aes_gcm_internal(data, symmetric_key)?;
65
66    // Combine ciphertext and encrypted data
67    let mut result = ciphertext.into_bytes();
68    result.extend_from_slice(&encrypted_data);
69
70    crate::log_crypto_operation_complete!(
71        "encrypt_pq_ml_kem",
72        security_level = ?security_level,
73        result_len = result.len()
74    );
75
76    Ok(result)
77}

79/// Internal implementation of ML-KEM decryption.
80///
81/// Decrypts data that was encrypted with [`encrypt_pq_ml_kem_internal`]. The wire
82/// format is: `ML-KEM ciphertext || AES-GCM encrypted payload`.
83///
84/// # Errors
85///
86/// Returns an error if:
87/// - The encrypted data is shorter than the expected ciphertext size
88/// - The ML-KEM decapsulation fails (invalid key or ciphertext)
89/// - The AES-GCM decryption of the payload fails
90fn decrypt_pq_ml_kem_internal(
91    encrypted_data: &[u8],
92    ml_kem_sk: &[u8],
93    security_level: MlKemSecurityLevel,
94) -> Result<Vec<u8>> {
95    use super::aes_gcm::decrypt_aes_gcm_internal;
96    use crate::primitives::kem::ml_kem::{MlKemCiphertext, MlKemSecretKey};
97
98    crate::log_crypto_operation_start!(
99        "decrypt_pq_ml_kem",
100        security_level = ?security_level,
101        data_len = encrypted_data.len()
102    );
103
104    let ct_size = security_level.ciphertext_size();
105    if encrypted_data.len() < ct_size {
106        crate::log_crypto_operation_error!("decrypt_pq_ml_kem", "encrypted data too short");
107        return Err(CoreError::DecryptionFailed(format!(
108            "Encrypted data ({} bytes) shorter than ML-KEM {:?} ciphertext size ({} bytes)",
109            encrypted_data.len(),
110            security_level,
111            ct_size,
112        )));
113    }
114
115    let (ct_bytes, aes_encrypted) = encrypted_data.split_at(ct_size);
116
117    let sk = MlKemSecretKey::new(security_level, ml_kem_sk.to_vec()).map_err(|e| {
118        crate::log_crypto_operation_error!("decrypt_pq_ml_kem", "invalid secret key");
119        CoreError::DecryptionFailed(format!("Invalid ML-KEM secret key: {}", e))
120    })?;
121
122    let ct = MlKemCiphertext::new(security_level, ct_bytes.to_vec()).map_err(|e| {
123        crate::log_crypto_operation_error!("decrypt_pq_ml_kem", "invalid ciphertext");
124        CoreError::DecryptionFailed(format!("Invalid ML-KEM ciphertext: {}", e))
125    })?;
126
127    let shared_secret = MlKem::decapsulate(&sk, &ct).map_err(|e| {
128        crate::log_crypto_operation_error!("decrypt_pq_ml_kem", "decapsulation failed");
129        CoreError::DecryptionFailed(format!("ML-KEM decapsulation failed: {}", e))
130    })?;
131
132    let plaintext = decrypt_aes_gcm_internal(aes_encrypted, shared_secret.as_bytes())?;
133
134    crate::log_crypto_operation_complete!(
135        "decrypt_pq_ml_kem",
136        security_level = ?security_level,
137        result_len = plaintext.len()
138    );
139
140    Ok(plaintext)
141}

147/// Encrypt data using ML-KEM.
148///
149/// Uses `SecurityMode` to specify verification behavior:
150/// - `SecurityMode::Verified(&session)`: Validates session before encryption
151/// - `SecurityMode::Unverified`: Skips session validation
152///
153/// # Example
154///
155/// ```no_run
156/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
157/// use latticearc::unified_api::{encrypt_pq_ml_kem, SecurityMode, VerifiedSession};
158/// use latticearc::primitives::kem::ml_kem::MlKemSecurityLevel;
159/// # let data = b"example data";
160/// # let ml_kem_pk = vec![0u8; 1184]; // ML-KEM-768 public key size
161/// # let pk = [0u8; 32];
162/// # let sk = [0u8; 32];
163///
164/// // With Zero Trust (recommended)
165/// let session = VerifiedSession::establish(&pk, &sk)?;
166/// let encrypted = encrypt_pq_ml_kem(data, &ml_kem_pk, MlKemSecurityLevel::MlKem768, SecurityMode::Verified(&session))?;
167///
168/// // Without verification (opt-out)
169/// let encrypted = encrypt_pq_ml_kem(data, &ml_kem_pk, MlKemSecurityLevel::MlKem768, SecurityMode::Unverified)?;
170/// # Ok(())
171/// # }
172/// ```
173///
174/// # Errors
175///
176/// Returns an error if:
177/// - The mode is `Verified` and the session has expired (`CoreError::SessionExpired`)
178/// - The data size exceeds resource limits
179/// - The public key is invalid for the specified security level
180/// - The ML-KEM encapsulation operation fails
181/// - The AES-GCM encryption of the payload fails
182pub fn encrypt_pq_ml_kem(
183    data: &[u8],
184    ml_kem_pk: &[u8],
185    security_level: MlKemSecurityLevel,
186    mode: SecurityMode,
187) -> Result<Vec<u8>> {
188    mode.validate()?;
189    encrypt_pq_ml_kem_internal(data, ml_kem_pk, security_level)
190}
