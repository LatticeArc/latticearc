6//! LatticeArc Portable Key Format (LPK v1)
7//!
8//! Schema-first dual-format key serialization supporting JSON (human-readable)
9//! and CBOR (compact binary, RFC 8949) for all LatticeArc key types.
10//!
11//! # Design Principles
12//!
13//! - **Schema-first**: One Rust struct, two wire formats (JSON + CBOR)
14//! - **CBOR primary**: Wire protocol, database storage, enterprise containers
15//! - **JSON secondary**: CLI display, REST APIs, debugging, key export
16//! - **Standards-aligned**: Algorithm IDs from FIPS 203-206 / RFC 7748 / RFC 8032;
17//!   composite key layout follows `draft-ietf-lamps-pq-composite-kem` (concatenation);
18//!   JWK-compatible structure (per `draft-ietf-jose-pqc-kem`)
19//!
20//! # Key Identity
21//!
22//! Keys are identified by **use case** or **security level** — mirroring
23//! how the library's API works. The algorithm is auto-derived and stored
24//! internally for version-stability.
25//!
26//! # JSON Format
27//!
28//! ```json
29//! {
30//!   "version": 1,
31//!   "use_case": "file-storage",
32//!   "algorithm": "hybrid-ml-kem-1024-x25519",
33//!   "key_type": "public",
34//!   "key_data": { "raw": "Base64..." },
35//!   "created": "2026-03-19T..."
36//! }
37//! ```
38//!
39//! # CBOR Format
40//!
41//! Same logical schema. Key material stored as CBOR byte strings (`bstr`) —
42//! no base64 encoding, no string overhead. See [`crate::unified_api::key_format::PortableKey::to_cbor`].
43//!
44//! # Enterprise Extension Model
45//!
46//! The `metadata` field is an open `BTreeMap<String, serde_json::Value>`.
47//! Enterprise crates add dimensions, key expiry,
48//! hardware binding, etc. via extension traits — the base library preserves
49//! unknown metadata keys during roundtrips without modification:
50//!
51//! ```rust,ignore
52//! // Enterprise crate — typed accessors over the metadata map
53//! impl EnterpriseKeyExt for PortableKey {
54//!     fn dimensions(&self) -> Option<Vec<String>> { ... }
55//!     fn key_expiry(&self) -> Option<DateTime<Utc>> { ... }
56//!     fn hsm_binding(&self) -> Option<HsmSlot> { ... }
57//! }
58//! ```

320/// Key material container — single, composite (hybrid), or passphrase-encrypted.
321///
322/// In JSON: uses base64-encoded strings.
323/// In CBOR: uses raw byte strings (`bstr`) — no base64 encoding.
324///
325/// Uses untagged serde. Variants are disambiguated by their field names:
326/// `"enc"` (with KDF metadata) → [`KeyData::Encrypted`], `"raw"` → [`KeyData::Single`],
327/// `"pq"` + `"classical"` → [`KeyData::Composite`]. `Encrypted` is listed first
328/// so serde matches it before falling back to Single/Composite.
329#[non_exhaustive]
330#[derive(Clone, Serialize, Deserialize)]
331#[serde(untagged)]
332pub enum KeyData {
333    /// Passphrase-encrypted key material.
334    ///
335    /// The inner `KeyData` (`Single` or `Composite`) is serialized to JSON and
336    /// then encrypted with AES-256-GCM using a key derived from a user
337    /// passphrase via PBKDF2-HMAC-SHA256 (SP 800-132). Layout:
338    ///
339    /// 1. Derive 32-byte AES key: `PBKDF2-HMAC-SHA256(passphrase, salt, iters)`
340    /// 2. Serialize plaintext `KeyData` to JSON bytes
341    /// 3. Encrypt JSON bytes with AES-256-GCM using a fresh random nonce.
342    ///    AAD binds every envelope field (version, algorithm, key_type,
343    ///    KDF name, iteration count, salt, AEAD name) plus the enclosing
344    ///    `PortableKey`'s `algorithm` and `key_type`, so tampering with
345    ///    any metadata on disk causes AEAD authentication to fail.
346    /// 4. Store KDF params, nonce, and `ciphertext || tag` base64-encoded
347    ///
348    /// Format version `1` fixes the algorithm choices to
349    /// `PBKDF2-HMAC-SHA256` + `AES-256-GCM`; future versions may add alternatives.
350    Encrypted {
351        /// Envelope format version (currently `1`).
352        enc: u32,
353        /// KDF identifier (currently `"PBKDF2-HMAC-SHA256"`).
354        kdf: String,
355        /// PBKDF2 iteration count (recommended ≥ 600_000 per OWASP 2023).
356        kdf_iterations: u32,
357        /// Base64-encoded PBKDF2 salt (16 bytes recommended).
358        kdf_salt: String,
359        /// AEAD identifier (currently `"AES-256-GCM"`).
360        aead: String,
361        /// Base64-encoded AES-GCM nonce (12 bytes).
362        nonce: String,
363        /// Base64-encoded `ciphertext || tag` (tag is the trailing 16 bytes).
364        ciphertext: String,
365    },
366    /// Single-component key (e.g., ML-KEM public key, AES symmetric key).
367    Single {
368        /// Base64-encoded key bytes (JSON) or raw bytes (CBOR).
369        raw: String,
370    },
371    /// Composite hybrid key with separate PQ and classical components.
372    Composite {
373        /// Base64-encoded post-quantum key component.
374        pq: String,
375        /// Base64-encoded classical key component.
376        classical: String,
377    },
378}

403impl Drop for KeyData {
404    fn drop(&mut self) {
405        match self {
406            Self::Single { raw } => {
407                raw.zeroize();
408            }
409            Self::Composite { pq, classical } => {
410                pq.zeroize();
411                classical.zeroize();
412            }
413            Self::Encrypted { ciphertext, nonce, kdf_salt, .. } => {
414                // The ciphertext is already encrypted, but zeroize the base64
415                // string residue anyway so nothing lingers in memory.
416                ciphertext.zeroize();
417                nonce.zeroize();
418                kdf_salt.zeroize();
419            }
420        }
421    }
422}

424impl KeyData {
425    /// Decode the single raw key bytes (returns error if composite or encrypted).
426    ///
427    /// # Security
428    ///
429    /// The returned `Vec<u8>` is **not** automatically zeroized on drop. When
430    /// this method is called for secret or symmetric key material, callers are
431    /// responsible for zeroizing the returned bytes after use. Prefer
432    /// [`decode_raw_zeroized`](Self::decode_raw_zeroized) for secret key data.
433    ///
434    /// # Errors
435    /// Returns an error if this is a composite key, the key is passphrase-encrypted
436    /// (call [`PortableKey::decrypt_with_passphrase`] first), or Base64 decoding fails.
437    pub fn decode_raw(&self) -> Result<Vec<u8>> {
438        match self {
439            Self::Single { raw } => BASE64_ENGINE
440                .decode(raw)
441                .map_err(|e| CoreError::SerializationError(format!("Invalid key base64: {e}"))),
442            Self::Composite { .. } => Err(CoreError::InvalidKey(
443                "Expected single key data but found composite".to_string(),
444            )),
445            Self::Encrypted { .. } => Err(CoreError::InvalidKey(
446                "Key is passphrase-encrypted; call PortableKey::decrypt_with_passphrase first"
447                    .to_string(),
448            )),
449        }
450    }
451
452    /// Decode the single raw key bytes into a zeroizing buffer (returns error if composite).
453    ///
454    /// Equivalent to [`decode_raw`](Self::decode_raw) but wraps the result in
455    /// [`zeroize::Zeroizing`] so the bytes are wiped from memory when dropped.
456    /// Use this variant whenever the data may be secret key material.
457    ///
458    /// # Errors
459    /// Returns an error if this is a composite key or Base64 decoding fails.
460    pub fn decode_raw_zeroized(&self) -> Result<zeroize::Zeroizing<Vec<u8>>> {
461        self.decode_raw().map(zeroize::Zeroizing::new)
462    }
463
464    /// Decode composite key components (returns error if single or encrypted).
465    ///
466    /// # Security
467    ///
468    /// The returned `Vec<u8>` values are **not** automatically zeroized on drop.
469    /// When this method is called for secret or symmetric key material, callers
470    /// are responsible for zeroizing the returned bytes after use. Prefer
471    /// [`decode_composite_zeroized`](Self::decode_composite_zeroized) for
472    /// secret key data.
473    ///
474    /// # Errors
475    /// Returns an error if this is a single key, the key is passphrase-encrypted
476    /// (call [`PortableKey::decrypt_with_passphrase`] first), or Base64 decoding fails.
477    pub fn decode_composite(&self) -> Result<(Vec<u8>, Vec<u8>)> {
478        match self {
479            Self::Composite { pq, classical } => {
480                let pq_bytes = BASE64_ENGINE.decode(pq).map_err(|e| {
481                    CoreError::SerializationError(format!("Invalid PQ key base64: {e}"))
482                })?;
483                let classical_bytes = BASE64_ENGINE.decode(classical).map_err(|e| {
484                    CoreError::SerializationError(format!("Invalid classical key base64: {e}"))
485                })?;
486                Ok((pq_bytes, classical_bytes))
487            }
488            Self::Single { .. } => Err(CoreError::InvalidKey(
489                "Expected composite key data but found single".to_string(),
490            )),
491            Self::Encrypted { .. } => Err(CoreError::InvalidKey(
492                "Key is passphrase-encrypted; call PortableKey::decrypt_with_passphrase first"
493                    .to_string(),
494            )),
495        }
496    }
497
498    /// Decode composite key components into zeroizing buffers (returns error if single).
499    ///
500    /// Equivalent to [`decode_composite`](Self::decode_composite) but wraps
501    /// each component in [`zeroize::Zeroizing`] so the bytes are wiped from
502    /// memory when dropped. Use this variant whenever either component may be
503    /// secret key material.
504    ///
505    /// # Errors
506    /// Returns an error if this is a single key or Base64 decoding fails.
507    pub fn decode_composite_zeroized(&self) -> Result<ZeroizingKeyPair> {
508        let (pq, classical) = self.decode_composite()?;
509        Ok((zeroize::Zeroizing::new(pq), zeroize::Zeroizing::new(classical)))
510    }
511
512    /// Create single key data from raw bytes.
513    #[must_use]
514    pub fn from_raw(bytes: &[u8]) -> Self {
515        Self::Single { raw: BASE64_ENGINE.encode(bytes) }
516    }
517
518    /// Create composite key data from PQ and classical components.
519    #[must_use]
520    pub fn from_composite(pq_bytes: &[u8], classical_bytes: &[u8]) -> Self {
521        Self::Composite {
522            pq: BASE64_ENGINE.encode(pq_bytes),
523            classical: BASE64_ENGINE.encode(classical_bytes),
524        }
525    }
526}

532/// Portable, versioned key container for all LatticeArc key types.
533///
534/// Keys are identified by **use case** or **security level** — mirroring how
535/// the library's API works. The specific algorithm is auto-derived from these
536/// and stored internally for version-stability (the mapping may change between
537/// library releases). Users never specify algorithms directly.
538///
539/// At least one of `use_case` or `security_level` must be present. If both
540/// are set, `security_level` takes precedence for algorithm resolution
541/// (matching `CryptoConfig` behavior).
542///
543/// # Dual-Format Serialization
544///
545/// - **JSON** — human-readable: CLI, REST APIs, debugging
546/// - **CBOR** (RFC 8949) — compact binary: wire protocol, database, containers
547///
548/// # Enterprise Extensions
549///
550/// The `metadata` map is the extension point for enterprise features.
551/// Enterprise crates store additional fields (expiry, hardware binding,
552/// dimensions, etc.) as metadata entries and provide typed accessor traits.
553/// The base library preserves all metadata during roundtrips.
554///
555/// # Security Note: Clone
556///
557/// `PortableKey` derives `Clone` because it is a **serialization type** — it
558/// is designed to be written to disk, sent over the wire, or stored in a
559/// database. Cloning is required for serialization roundtrip testing and for
560/// passing keys through API boundaries.
561///
562/// **Do not use `PortableKey` as a long-lived runtime key holder.** Extract
563/// the concrete key type (e.g., `HybridPublicKey`, `HybridSecretKey`) via the
564/// `to_hybrid_*` bridge methods, and work with those instead. Secret key
565/// material inside `KeyData` is zeroized on drop via the explicit `Drop` impl.
566///
567/// # Constant-Time Comparison
568///
569/// AUDIT-ACCEPTED: ConstantTimeEq not implemented because key material is stored
570/// as Base64-encoded strings inside the `KeyData` enum, not as raw bytes amenable
571/// to byte-level constant-time comparison. This type is a serialization container
572/// (not a runtime key), and is not compared in any production code path. Use the
573/// concrete key types extracted via `to_hybrid_*` for cryptographic operations.
574///
575/// # Example
576///
577/// Real-world flow: generate keypair → wrap in PortableKey → serialize →
578/// deserialize → use for crypto operations.
579///
580/// ```rust,no_run
581/// use latticearc::{PortableKey, UseCase};
582///
583/// // 1. Generate a hybrid keypair
584/// let (pk, sk) = latticearc::generate_hybrid_keypair().expect("keygen");
585///
586/// // 2. Wrap in PortableKey (UseCase determines algorithm automatically)
587/// let (portable_pk, portable_sk) =
588///     PortableKey::from_hybrid_kem_keypair(UseCase::FileStorage, &pk, &sk)
589///         .expect("wrap");
590///
591/// // 3. Serialize (JSON for files/REST, CBOR for wire/storage)
592/// let json = portable_pk.to_json().expect("serialize");
593/// let cbor = portable_pk.to_cbor().expect("serialize");
594///
595/// // 4. Deserialize and extract for crypto operations
596/// let restored_pk = PortableKey::from_json(&json)
597///     .expect("deserialize")
598///     .to_hybrid_public_key()
599///     .expect("extract");
600/// ```
601#[derive(Clone, Serialize, Deserialize)]
602pub struct PortableKey {
603    /// Format version (currently `1`).
604    version: u32,
605
606    // --- Primary identifiers (at least one required) ---
607    /// Use case that determined algorithm selection.
608    /// This is how the library's API works: users pick a use case,
609    /// the policy engine selects the optimal algorithm.
610    #[serde(default, skip_serializing_if = "Option::is_none")]
611    use_case: Option<crate::types::types::UseCase>,
612
613    /// Security level (NIST Level 1/3/5). If both `use_case` and
614    /// `security_level` are set, `security_level` takes precedence
615    /// for algorithm resolution (matching `CryptoConfig` behavior).
616    #[serde(default, skip_serializing_if = "Option::is_none")]
617    security_level: Option<crate::types::types::SecurityLevel>,
618
619    /// Resolved algorithm identifier. Auto-derived from `use_case` or
620    /// `security_level` at creation time. Stored for version-stability —
621    /// if the policy engine mapping changes in a future release, existing
622    /// keys still parse correctly using this field.
623    algorithm: KeyAlgorithm,
624
625    /// Key type (public, secret, symmetric).
626    key_type: KeyType,
627    /// Key material (single or composite).
628    key_data: KeyData,
629    /// Creation timestamp (UTC, ISO 8601).
630    created: DateTime<Utc>,
631
632    /// Open metadata map for enterprise extensions.
633    /// Enterprise crates store additional fields (expiry, hardware binding,
634    /// dimensions, etc.) here via extension traits. The base library preserves
635    /// all entries during roundtrips without interpretation.
636    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
637    metadata: BTreeMap<String, serde_json::Value>,
638}

660impl subtle::ConstantTimeEq for PortableKey {
661    fn ct_eq(&self, other: &Self) -> subtle::Choice {
662        // Constant-time comparison of key data to prevent timing side-channels
663        // when comparing secret or symmetric keys. Encrypted variants are not
664        // compared by content — their ciphertexts are not directly meaningful
665        // (different nonces → different ciphertexts for the same plaintext).
666        match (&self.key_data, &other.key_data) {
667            (KeyData::Single { raw: a }, KeyData::Single { raw: b }) => {
668                a.as_bytes().ct_eq(b.as_bytes())
669            }
670            (
671                KeyData::Composite { pq: a_pq, classical: a_cl },
672                KeyData::Composite { pq: b_pq, classical: b_cl },
673            ) => a_pq.as_bytes().ct_eq(b_pq.as_bytes()) & a_cl.as_bytes().ct_eq(b_cl.as_bytes()),
674            _ => subtle::Choice::from(0),
675        }
676    }

901    /// Wrap a hybrid KEM keypair (from `generate_hybrid_keypair()`) into
902    /// a pair of `PortableKey`s (public + secret).
903    ///
904    /// Algorithm is auto-derived from the keypair's `security_level`.
905    ///
906    /// # Errors
907    /// Returns an error if secret key export fails.
908    pub fn from_hybrid_kem_keypair(
909        use_case: crate::types::types::UseCase,
910        pk: &crate::hybrid::kem_hybrid::HybridKemPublicKey,
911        sk: &crate::hybrid::kem_hybrid::HybridKemSecretKey,
912    ) -> Result<(Self, Self)> {
913        let algorithm = match pk.security_level() {
914            crate::primitives::kem::MlKemSecurityLevel::MlKem512 => {
915                KeyAlgorithm::HybridMlKem512X25519
916            }
917            crate::primitives::kem::MlKemSecurityLevel::MlKem768 => {
918                KeyAlgorithm::HybridMlKem768X25519
919            }
920            crate::primitives::kem::MlKemSecurityLevel::MlKem1024 => {
921                KeyAlgorithm::HybridMlKem1024X25519
922            }
923        };
924
925        let pub_key = Self {
926            version: Self::CURRENT_VERSION,
927            use_case: Some(use_case),
928            security_level: None,
929            algorithm,
930            key_type: KeyType::Public,
931            key_data: KeyData::from_composite(pk.ml_kem_pk(), pk.ecdh_pk()),
932            created: Utc::now(),
933            metadata: BTreeMap::new(),
934        };
935
936        let ml_kem_sk = sk
937            .ml_kem_sk_bytes()
938            .map_err(|e| CoreError::InvalidKey(format!("ML-KEM SK export: {e}")))?;
939        let ecdh_seed = sk
940            .ecdh_seed_bytes()
941            .map_err(|e| CoreError::InvalidKey(format!("ECDH seed export: {e}")))?;
942
943        // Store ML-KEM public key in metadata so the secret key file is
944        // self-contained for decryption (no separate public key file needed).
945        // This follows the PKCS#12 pattern of bundling public + private material.
946        // Note: ECDH public key is not stored — it's derived from the seed at
947        // reconstruction time via X25519StaticKeyPair::from_seed_bytes().
948        let mut sk_metadata = BTreeMap::new();
949        sk_metadata.insert(
950            ML_KEM_PK_METADATA_KEY.to_string(),
951            serde_json::Value::String(BASE64_ENGINE.encode(pk.ml_kem_pk())),
952        );
953
954        let sec_key = Self {
955            version: Self::CURRENT_VERSION,
956            use_case: Some(use_case),
957            security_level: None,
958            algorithm,
959            key_type: KeyType::Secret,
960            key_data: KeyData::from_composite(&ml_kem_sk, &*ecdh_seed),
961            created: Utc::now(),
962            metadata: sk_metadata,
963        };
964
965        Ok((pub_key, sec_key))
966    }

995    /// Reconstruct a `HybridSecretKey` from a portable key pair.
996    ///
997    /// Requires the corresponding public key PortableKey because ML-KEM
998    /// key reconstruction needs both the secret key bytes and public key bytes.
999    ///
1000    /// # Arguments
1001    /// * `public_key` - The corresponding `PortableKey` with `KeyType::Public`
1002    ///
1003    /// # Errors
1004    /// Returns an error if the algorithm is not hybrid KEM, key data is invalid,
1005    /// or key reconstruction fails.
1006    ///
1007    /// The ML-KEM public key is extracted from the secret key file's metadata
1008    /// (stored at keygen time), making the secret key file fully self-contained.
1009    /// No separate public key file is needed for decryption.
1010    pub fn to_hybrid_secret_key(&self) -> Result<crate::hybrid::kem_hybrid::HybridKemSecretKey> {
1011        let level = match self.algorithm {
1012            KeyAlgorithm::HybridMlKem512X25519 => {
1013                crate::primitives::kem::MlKemSecurityLevel::MlKem512
1014            }
1015            KeyAlgorithm::HybridMlKem768X25519 => {
1016                crate::primitives::kem::MlKemSecurityLevel::MlKem768
1017            }
1018            KeyAlgorithm::HybridMlKem1024X25519 => {
1019                crate::primitives::kem::MlKemSecurityLevel::MlKem1024
1020            }
1021            other => {
1022                return Err(CoreError::InvalidKey(format!(
1023                    "Not a hybrid KEM algorithm: {other:?}"
1024                )));
1025            }
1026        };
1027
1028        if self.key_type != KeyType::Secret {
1029            return Err(CoreError::InvalidKey(
1030                "Cannot reconstruct secret key from a public key".to_string(),
1031            ));
1032        }
1033
1034        let (ml_kem_sk, ecdh_seed_vec) = self.key_data.decode_composite()?;
1035
1036        // Extract ML-KEM public key from metadata (stored at keygen time)
1037        let ml_kem_pk = self
1038            .metadata
1039            .get(ML_KEM_PK_METADATA_KEY)
1040            .and_then(|v| v.as_str())
1041            .ok_or_else(|| {
1042                CoreError::InvalidKey(
1043                    "Secret key file missing 'ml_kem_pk' metadata. \
1044                     Re-generate the keypair with the latest CLI."
1045                        .to_string(),
1046                )
1047            })
1048            .and_then(|b64| {
1049                BASE64_ENGINE
1050                    .decode(b64)
1051                    .map_err(|e| CoreError::InvalidKey(format!("Invalid ml_kem_pk base64: {e}")))
1052            })?;
1053
1054        if ecdh_seed_vec.len() != 32 {
1055            return Err(CoreError::InvalidKey(format!(
1056                "X25519 seed must be 32 bytes, got {}",
1057                ecdh_seed_vec.len()
1058            )));
1059        }
1060
1061        let mut ecdh_seed = zeroize::Zeroizing::new([0u8; 32]);
1062        ecdh_seed.copy_from_slice(&ecdh_seed_vec);
1063
1064        crate::hybrid::kem_hybrid::HybridKemSecretKey::from_serialized(
1065            level, &ml_kem_sk, &ml_kem_pk, &ecdh_seed,
1066        )
1067        .map_err(|e| CoreError::InvalidKey(format!("Secret key reconstruction: {e}")))
1068    }

1464    // --- Validation ---
1465
1466    /// Validate internal consistency.
1467    ///
1468    /// Checks:
1469    /// - Format version matches [`CURRENT_VERSION`](Self::CURRENT_VERSION)
1470    /// - Symmetric algorithms (`aes-256`, `chacha20`) require `KeyType::Symmetric`
1471    /// - Hybrid algorithms require composite `KeyData`
1472    /// - Non-hybrid algorithms require single `KeyData`
1473    /// - Base64 key data decodes successfully
1474    ///
1475    /// # Errors
1476    /// Returns `CoreError::InvalidKey` on validation failure.
1477    pub fn validate(&self) -> Result<()> {
1478        // Version check — reject keys serialized by future or incompatible versions
1479        if self.version != Self::CURRENT_VERSION {
1480            return Err(CoreError::InvalidKey(format!(
1481                "Unsupported key format version {}, expected {}",
1482                self.version,
1483                Self::CURRENT_VERSION
1484            )));
1485        }
1486
1487        // Symmetric algorithm ↔ key type
1488        if self.algorithm.is_symmetric() && self.key_type != KeyType::Symmetric {
1489            return Err(CoreError::InvalidKey(format!(
1490                "Algorithm {:?} requires KeyType::Symmetric, got {:?}",
1491                self.algorithm, self.key_type
1492            )));
1493        }
1494        if !self.algorithm.is_symmetric() && self.key_type == KeyType::Symmetric {
1495            return Err(CoreError::InvalidKey(format!(
1496                "KeyType::Symmetric is not valid for algorithm {:?}",
1497                self.algorithm
1498            )));
1499        }
1500
1501        // Hybrid ↔ composite key data (encrypted variant is opaque — skip this check).
1502        match (&self.key_data, self.algorithm.is_hybrid()) {
1503            (KeyData::Composite { .. }, false) => {
1504                return Err(CoreError::InvalidKey(format!(
1505                    "Non-hybrid algorithm {:?} must use single key data",
1506                    self.algorithm
1507                )));
1508            }
1509            (KeyData::Single { .. }, true) => {
1510                return Err(CoreError::InvalidKey(format!(
1511                    "Hybrid algorithm {:?} must use composite key data",
1512                    self.algorithm
1513                )));
1514            }
1515            _ => {}
1516        }
1517
1518        // Verify base64 decodes / envelope shape
1519        match &self.key_data {
1520            KeyData::Single { raw } => {
1521                let _ = BASE64_ENGINE
1522                    .decode(raw)
1523                    .map_err(|e| CoreError::SerializationError(format!("Invalid base64: {e}")))?;
1524            }
1525            KeyData::Composite { pq, classical } => {
1526                let _ = BASE64_ENGINE.decode(pq).map_err(|e| {
1527                    CoreError::SerializationError(format!("Invalid PQ base64: {e}"))
1528                })?;
1529                let _ = BASE64_ENGINE.decode(classical).map_err(|e| {
1530                    CoreError::SerializationError(format!("Invalid classical base64: {e}"))
1531                })?;
1532            }
1533            KeyData::Encrypted { enc, kdf, kdf_iterations, kdf_salt, aead, nonce, ciphertext } => {
1534                Self::validate_encrypted_envelope_fields(
1535                    *enc,
1536                    kdf,
1537                    *kdf_iterations,
1538                    kdf_salt,
1539                    aead,
1540                    nonce,
1541                    ciphertext,
1542                )?;
1543            }
1544        }
1545
1546        Ok(())
1547    }

1549    /// Validate an encrypted-envelope's metadata and base64-decodable sizes.
1550    ///
1551    /// Shared between [`Self::validate`] (which rejects malformed keys at
1552    /// load time) and [`Self::decrypt_with_passphrase`] (which uses this to
1553    /// defend against direct construction of an unvalidated encrypted
1554    /// variant). Does not verify the AEAD tag — that happens in
1555    /// `decrypt_with_passphrase` after key derivation.
1556    fn validate_encrypted_envelope_fields(
1557        enc: u32,
1558        kdf: &str,
1559        kdf_iterations: u32,
1560        kdf_salt: &str,
1561        aead: &str,
1562        nonce: &str,
1563        ciphertext: &str,
1564    ) -> Result<()> {
1565        if enc != ENCRYPTED_ENVELOPE_VERSION {
1566            return Err(CoreError::InvalidKey(format!(
1567                "Unsupported encrypted key envelope version {enc}, expected {ENCRYPTED_ENVELOPE_VERSION}",
1568            )));
1569        }
1570        if kdf != PBKDF2_KDF_ID {
1571            return Err(CoreError::InvalidKey(format!(
1572                "Unsupported KDF {kdf:?}, expected {PBKDF2_KDF_ID:?}",
1573            )));
1574        }
1575        if aead != AES_GCM_AEAD_ID {
1576            return Err(CoreError::InvalidKey(format!(
1577                "Unsupported AEAD {aead:?}, expected {AES_GCM_AEAD_ID:?}",
1578            )));
1579        }
1580        if kdf_iterations < PBKDF2_MIN_ITERATIONS {
1581            return Err(CoreError::InvalidKey(format!(
1582                "PBKDF2 iteration count {kdf_iterations} below minimum {PBKDF2_MIN_ITERATIONS}",
1583            )));
1584        }
1585        let salt = BASE64_ENGINE
1586            .decode(kdf_salt)
1587            .map_err(|e| CoreError::SerializationError(format!("Invalid KDF salt base64: {e}")))?;
1588        if salt.len() < PBKDF2_MIN_SALT_LEN {
1589            return Err(CoreError::InvalidKey(format!(
1590                "PBKDF2 salt length {} below minimum {PBKDF2_MIN_SALT_LEN}",
1591                salt.len(),
1592            )));
1593        }
1594        let nonce_bytes = BASE64_ENGINE
1595            .decode(nonce)
1596            .map_err(|e| CoreError::SerializationError(format!("Invalid nonce base64: {e}")))?;
1597        if nonce_bytes.len() != AES_GCM_NONCE_LEN {
1598            return Err(CoreError::InvalidKey(format!(
1599                "AES-GCM nonce length {} != {AES_GCM_NONCE_LEN}",
1600                nonce_bytes.len(),
1601            )));
1602        }
1603        let ct_bytes = BASE64_ENGINE.decode(ciphertext).map_err(|e| {
1604            CoreError::SerializationError(format!("Invalid ciphertext base64: {e}"))
1605        })?;
1606        if ct_bytes.len() < AES_GCM_TAG_LEN {
1607            return Err(CoreError::InvalidKey(
1608                "Encrypted key ciphertext shorter than AES-GCM tag".to_string(),
1609            ));
1610        }
1611        Ok(())
1612    }

1626    /// Encrypt the key material in place using a passphrase.
1627    ///
1628    /// Derives a 32-byte AES key via PBKDF2-HMAC-SHA256 (600,000 iterations,
1629    /// 16-byte random salt) and encrypts the JSON-serialized `KeyData` under
1630    /// AES-256-GCM with a fresh 12-byte random nonce. The full envelope
1631    /// (version, algorithm, key_type, KDF name, iteration count, salt, AEAD
1632    /// name) is mixed into the AEAD AAD, so tampering with any metadata
1633    /// field on disk causes decryption to fail at the tag check. See
1634    /// [`Self::encryption_aad`] for the exact byte layout.
1635    ///
1636    /// # Errors
1637    ///
1638    /// Returns an error if the key is already encrypted, if the passphrase is
1639    /// empty, or if the KDF / AEAD operation fails.
1640    pub fn encrypt_with_passphrase(&mut self, passphrase: &[u8]) -> Result<()> {
1641        if self.is_encrypted() {
1642            return Err(CoreError::InvalidKey("Key is already passphrase-encrypted".to_string()));
1643        }
1644        if passphrase.is_empty() {
1645            return Err(CoreError::InvalidKey("Passphrase must not be empty".to_string()));
1646        }
1647
1648        // 1. Serialize the current plaintext KeyData to its own JSON. We
1649        //    serialize only the KeyData (not the whole PortableKey) so the
1650        //    ciphertext is self-contained and round-trips cleanly.
1651        let plaintext_json = serde_json::to_vec(&self.key_data).map_err(|e| {
1652            CoreError::SerializationError(format!(
1653                "Failed to serialize key data for encryption: {e}"
1654            ))
1655        })?;
1656        let plaintext = zeroize::Zeroizing::new(plaintext_json);
1657
1658        // 2. Generate a fresh random salt.
1659        let salt = crate::primitives::rand::csprng::random_bytes(PBKDF2_SALT_LEN);
1660
1661        // 3. Derive the AES key via PBKDF2-HMAC-SHA256.
1662        let kdf_params = crate::primitives::kdf::pbkdf2::Pbkdf2Params::with_salt(&salt)
1663            .iterations(PBKDF2_DEFAULT_ITERATIONS)
1664            .key_length(AES_256_KEY_LEN);
1665        let derived = crate::primitives::kdf::pbkdf2::pbkdf2(passphrase, &kdf_params)
1666            .map_err(|e| CoreError::InvalidKey(format!("PBKDF2 derivation failed: {e}")))?;
1667
1668        // 4. Encrypt via AES-256-GCM with a fresh random nonce. Bind the
1669        //    full envelope (version, algorithm, key_type, KDF name,
1670        //    iterations, salt, AEAD name) to the ciphertext via AAD so an
1671        //    attacker who modifies any of these fields on disk breaks the
1672        //    AEAD tag.
1673        use crate::primitives::aead::AeadCipher;
1674        let cipher = crate::primitives::aead::aes_gcm::AesGcm256::new(derived.key())
1675            .map_err(|e| CoreError::InvalidKey(format!("Failed to initialize AES-256-GCM: {e}")))?;
1676        let aad = Self::encryption_aad(
1677            ENCRYPTED_ENVELOPE_VERSION,
1678            self.algorithm,
1679            self.key_type,
1680            PBKDF2_KDF_ID,
1681            PBKDF2_DEFAULT_ITERATIONS,
1682            &salt,
1683            AES_GCM_AEAD_ID,
1684        );
1685        let (nonce, mut ct, tag) = cipher
1686            .seal(&plaintext, Some(&aad))
1687            .map_err(|e| CoreError::InvalidKey(format!("AES-256-GCM sealing failed: {e}")))?;
1688
1689        // 5. Pack ciphertext || tag for on-wire storage.
1690        ct.extend_from_slice(&tag);
1691
1692        // 6. Replace the plaintext KeyData with the encrypted envelope.
1693        self.key_data = KeyData::Encrypted {
1694            enc: ENCRYPTED_ENVELOPE_VERSION,
1695            kdf: PBKDF2_KDF_ID.to_string(),
1696            kdf_iterations: PBKDF2_DEFAULT_ITERATIONS,
1697            kdf_salt: BASE64_ENGINE.encode(&salt),
1698            aead: AES_GCM_AEAD_ID.to_string(),
1699            nonce: BASE64_ENGINE.encode(nonce),
1700            ciphertext: BASE64_ENGINE.encode(&ct),
1701        };
1702
1703        Ok(())
1704    }

1706    /// Decrypt the key material in place using a passphrase.
1707    ///
1708    /// Reverses [`Self::encrypt_with_passphrase`]. On success the `key_data`
1709    /// field is replaced with the underlying `Single` or `Composite` variant.
1710    ///
1711    /// # Errors
1712    ///
1713    /// Returns an error if the key is not encrypted, the envelope is
1714    /// malformed, the passphrase is wrong (AEAD authentication fails), or
1715    /// the decrypted plaintext is not a valid `KeyData` serialization.
1716    pub fn decrypt_with_passphrase(&mut self, passphrase: &[u8]) -> Result<()> {
1717        if passphrase.is_empty() {
1718            return Err(CoreError::InvalidKey("Passphrase must not be empty".to_string()));
1719        }
1720
1721        // Borrow the envelope to validate shape and decode its fields into
1722        // owned bytes. After validation the `kdf` and `aead` string fields
1723        // are known to equal the envelope's fixed constants, so we don't
1724        // bother cloning them into `Decoded` — we pass the constants to
1725        // `encryption_aad` at the call site instead.
1726        struct Decoded {
1727            enc: u32,
1728            kdf_iterations: u32,
1729            salt: Vec<u8>,
1730            nonce: [u8; AES_GCM_NONCE_LEN],
1731            ct_and_tag: Vec<u8>,
1732        }
1733        let decoded = match &self.key_data {
1734            KeyData::Encrypted { enc, kdf, kdf_iterations, kdf_salt, aead, nonce, ciphertext } => {
1735                Self::validate_encrypted_envelope_fields(
1736                    *enc,
1737                    kdf,
1738                    *kdf_iterations,
1739                    kdf_salt,
1740                    aead,
1741                    nonce,
1742                    ciphertext,
1743                )?;
1744                let salt = BASE64_ENGINE.decode(kdf_salt).map_err(|e| {
1745                    CoreError::SerializationError(format!("Invalid KDF salt base64: {e}"))
1746                })?;
1747                let nonce_bytes = BASE64_ENGINE.decode(nonce).map_err(|e| {
1748                    CoreError::SerializationError(format!("Invalid nonce base64: {e}"))
1749                })?;
1750                let mut nonce_array = [0u8; AES_GCM_NONCE_LEN];
1751                nonce_array.copy_from_slice(&nonce_bytes);
1752                let ct_and_tag = BASE64_ENGINE.decode(ciphertext).map_err(|e| {
1753                    CoreError::SerializationError(format!("Invalid ciphertext base64: {e}"))
1754                })?;
1755                Decoded {
1756                    enc: *enc,
1757                    kdf_iterations: *kdf_iterations,
1758                    salt,
1759                    nonce: nonce_array,
1760                    ct_and_tag,
1761                }
1762            }
1763            _ => {
1764                return Err(CoreError::InvalidKey("Key is not passphrase-encrypted".to_string()));
1765            }
1766        };
1767        let Decoded { enc, kdf_iterations, salt, nonce: nonce_array, ct_and_tag } = decoded;
1768
1769        let tag_offset = ct_and_tag
1770            .len()
1771            .checked_sub(AES_GCM_TAG_LEN)
1772            .ok_or_else(|| CoreError::InvalidKey("Ciphertext shorter than tag".to_string()))?;
1773        let (ct_bytes, tag_bytes) = ct_and_tag.split_at(tag_offset);
1774        let mut tag_array = [0u8; AES_GCM_TAG_LEN];
1775        tag_array.copy_from_slice(tag_bytes);
1776
1777        // Derive the AES key from the passphrase + salt.
1778        let kdf_params = crate::primitives::kdf::pbkdf2::Pbkdf2Params::with_salt(&salt)
1779            .iterations(kdf_iterations)
1780            .key_length(AES_256_KEY_LEN);
1781        let derived = crate::primitives::kdf::pbkdf2::pbkdf2(passphrase, &kdf_params)
1782            .map_err(|e| CoreError::InvalidKey(format!("PBKDF2 derivation failed: {e}")))?;
1783
1784        // Decrypt. A wrong passphrase produces a wrong AES key, which causes
1785        // AEAD authentication to fail with an opaque error — we do NOT leak
1786        // whether the passphrase was wrong vs. the envelope was corrupted.
1787        // The AAD binds the full envelope so any tampered metadata field
1788        // also breaks the tag.
1789        use crate::primitives::aead::AeadCipher;
1790        let cipher = crate::primitives::aead::aes_gcm::AesGcm256::new(derived.key())
1791            .map_err(|e| CoreError::InvalidKey(format!("Failed to initialize AES-256-GCM: {e}")))?;
1792        // `kdf` and `aead` are the constants: `validate_encrypted_envelope_fields`
1793        // rejected any other value, so we can pass the literals directly
1794        // instead of cloning them out of the borrowed envelope.
1795        let aad = Self::encryption_aad(
1796            enc,
1797            self.algorithm,
1798            self.key_type,
1799            PBKDF2_KDF_ID,
1800            kdf_iterations,
1801            &salt,
1802            AES_GCM_AEAD_ID,
1803        );
1804        let plaintext = cipher
1805            .decrypt(&nonce_array, ct_bytes, &tag_array, Some(&aad))
1806            .map_err(|_e| {
1807                CoreError::InvalidKey(
1808                    "Passphrase-protected key unwrap failed (wrong passphrase or corrupted envelope)"
1809                        .to_string(),
1810                )
1811            })?;
1812
1813        // Deserialize the plaintext bytes back into a KeyData variant.
1814        let new_key_data: KeyData = serde_json::from_slice(&plaintext).map_err(|e| {
1815            CoreError::SerializationError(format!("Failed to deserialize decrypted key data: {e}"))
1816        })?;
1817        // Reject nested encryption (prevents re-wrap confusion).
1818        if matches!(new_key_data, KeyData::Encrypted { .. }) {
1819            return Err(CoreError::InvalidKey(
1820                "Decrypted payload was itself an encrypted envelope".to_string(),
1821            ));
1822        }
1823
1824        self.key_data = new_key_data;
1825        Ok(())
1826    }

1828    /// Build the AAD bound to an encrypted key envelope.
1829    ///
1830    /// AEAD AAD is authenticated (not encrypted), so fields folded in here
1831    /// are protected against tampering but not against disclosure. Binding
1832    /// all envelope parameters — version, algorithm, key type, KDF name,
1833    /// iteration count, salt, and AEAD name — ensures that any attacker
1834    /// modification of a stored key file's metadata or KDF parameters
1835    /// causes `cipher.decrypt` to fail with an opaque error.
1836    ///
1837    /// Uses stable kebab-case / lowercase names (`ml-kem-768`, `secret`)
1838    /// via `canonical_name` accessors on the enums. These are load-bearing:
1839    /// changing the returned strings breaks every existing encrypted key
1840    /// file. A pinned byte-level test in the `tests` module guards this.
1841    ///
1842    /// # Byte layout
1843    ///
1844    /// ```text
1845    /// "latticearc-lpk-v1-enc" || 0x00
1846    /// || enc (u32 BE)
1847    /// || algorithm_name || 0x00
1848    /// || key_type_name || 0x00
1849    /// || kdf_name || 0x00
1850    /// || kdf_iterations (u32 BE)
1851    /// || kdf_salt_len (u32 BE) || kdf_salt_raw_bytes
1852    /// || aead_name
1853    /// ```
1854    ///
1855    /// Length prefixes and null separators prevent ambiguity between
1856    /// adjacent variable-length fields. The salt is included as its raw
1857    /// (base64-decoded) bytes, not the base64 string, so an attacker
1858    /// cannot use base64 non-canonical encodings to get past the check.
1859    fn encryption_aad(
1860        enc: u32,
1861        algorithm: KeyAlgorithm,
1862        key_type: KeyType,
1863        kdf: &str,
1864        kdf_iterations: u32,
1865        kdf_salt: &[u8],
1866        aead: &str,
1867    ) -> Vec<u8> {
1868        let algorithm_name = algorithm.canonical_name();
1869        let key_type_name = key_type.canonical_name();
1870
1871        let mut aad = Vec::with_capacity(
1872            b"latticearc-lpk-v1-enc"
1873                .len()
1874                .saturating_add(1) // null
1875                .saturating_add(4) // enc
1876                .saturating_add(algorithm_name.len())
1877                .saturating_add(1)
1878                .saturating_add(key_type_name.len())
1879                .saturating_add(1)
1880                .saturating_add(kdf.len())
1881                .saturating_add(1)
1882                .saturating_add(4) // kdf_iterations
1883                .saturating_add(4) // salt len
1884                .saturating_add(kdf_salt.len())
1885                .saturating_add(aead.len()),
1886        );
1887        aad.extend_from_slice(b"latticearc-lpk-v1-enc");
1888        aad.push(0);
1889        aad.extend_from_slice(&enc.to_be_bytes());
1890        aad.extend_from_slice(algorithm_name.as_bytes());
1891        aad.push(0);
1892        aad.extend_from_slice(key_type_name.as_bytes());
1893        aad.push(0);
1894        aad.extend_from_slice(kdf.as_bytes());
1895        aad.push(0);
1896        aad.extend_from_slice(&kdf_iterations.to_be_bytes());
1897        let salt_len_u32 = u32::try_from(kdf_salt.len()).unwrap_or(u32::MAX);
1898        aad.extend_from_slice(&salt_len_u32.to_be_bytes());
1899        aad.extend_from_slice(kdf_salt);
1900        aad.extend_from_slice(aead.as_bytes());
1901        aad
1902    }

2334    /// Snapshot of a freshly-encrypted envelope's fields, used to build
2335    /// tampered copies that exercise each rejection branch of
2336    /// `validate_encrypted_envelope_fields`. Cloning via this helper
2337    /// sidesteps `KeyData`'s `Drop` impl, which forbids partial moves.
2338    #[derive(Clone)]
2339    struct EnvelopeSnapshot {
2340        enc: u32,
2341        kdf: String,
2342        kdf_iterations: u32,
2343        kdf_salt: String,
2344        aead: String,
2345        nonce: String,
2346        ciphertext: String,
2347    }
2348
2349    impl EnvelopeSnapshot {
2350        fn capture(key: &PortableKey) -> Option<Self> {
2351            match &key.key_data {
2352                KeyData::Encrypted {
2353                    enc,
2354                    kdf,
2355                    kdf_iterations,
2356                    kdf_salt,
2357                    aead,
2358                    nonce,
2359                    ciphertext,
2360                } => Some(Self {
2361                    enc: *enc,
2362                    kdf: kdf.clone(),
2363                    kdf_iterations: *kdf_iterations,
2364                    kdf_salt: kdf_salt.clone(),
2365                    aead: aead.clone(),
2366                    nonce: nonce.clone(),
2367                    ciphertext: ciphertext.clone(),
2368                }),
2369                _ => None,
2370            }
2371        }
2372
2373        fn into_key_data(self) -> KeyData {
2374            KeyData::Encrypted {
2375                enc: self.enc,
2376                kdf: self.kdf,
2377                kdf_iterations: self.kdf_iterations,
2378                kdf_salt: self.kdf_salt,
2379                aead: self.aead,
2380                nonce: self.nonce,
2381                ciphertext: self.ciphertext,
2382            }
2383        }
2384    }
