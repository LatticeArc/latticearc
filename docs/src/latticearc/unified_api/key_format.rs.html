6//! LatticeArc Portable Key Format (LPK v1)
7//!
8//! Schema-first dual-format key serialization supporting JSON (human-readable)
9//! and CBOR (compact binary, RFC 8949) for all LatticeArc key types.
10//!
11//! # Design Principles
12//!
13//! - **Schema-first**: One Rust struct, two wire formats (JSON + CBOR)
14//! - **CBOR primary**: Wire protocol, database storage, enterprise containers
15//! - **JSON secondary**: CLI display, REST APIs, debugging, key export
16//! - **Standards-aligned**: Algorithm IDs from FIPS 203-206 / RFC 7748 / RFC 8032;
17//!   composite key layout follows `draft-ietf-lamps-pq-composite-kem` (concatenation);
18//!   JWK-compatible structure (per `draft-ietf-jose-pqc-kem`)
19//!
20//! # Key Identity
21//!
22//! Keys are identified by **use case** or **security level** — mirroring
23//! how the library's API works. The algorithm is auto-derived and stored
24//! internally for version-stability.
25//!
26//! # JSON Format
27//!
28//! ```json
29//! {
30//!   "version": 1,
31//!   "use_case": "file-storage",
32//!   "algorithm": "hybrid-ml-kem-1024-x25519",
33//!   "key_type": "public",
34//!   "key_data": { "raw": "Base64..." },
35//!   "created": "2026-03-19T..."
36//! }
37//! ```
38//!
39//! # CBOR Format
40//!
41//! Same logical schema. Key material stored as CBOR byte strings (`bstr`) —
42//! no base64 encoding, no string overhead. See [`crate::unified_api::key_format::PortableKey::to_cbor`].
43//!
44//! # Enterprise Extension Model
45//!
46//! The `metadata` field is an open `BTreeMap<String, serde_json::Value>`.
47//! Enterprise crates add dimensions, key expiry,
48//! hardware binding, etc. via extension traits — the base library preserves
49//! unknown metadata keys during roundtrips without modification:
50//!
51//! ```rust,ignore
52//! // Enterprise crate — typed accessors over the metadata map
53//! impl EnterpriseKeyExt for PortableKey {
54//!     fn dimensions(&self) -> Option<Vec<String>> { ... }
55//!     fn key_expiry(&self) -> Option<DateTime<Utc>> { ... }
56//!     fn hsm_binding(&self) -> Option<HsmSlot> { ... }
57//! }
58//! ```

484/// Key material container — single, composite (hybrid), or passphrase-encrypted.
485///
486/// In JSON: uses base64-encoded strings.
487/// In CBOR: uses raw byte strings (`bstr`) — no base64 encoding.
488///
489/// Uses untagged serde. Variants are disambiguated by their field names:
490/// `"enc"` (with KDF metadata) → [`KeyData::Encrypted`], `"raw"` → [`KeyData::Single`],
491/// `"pq"` + `"classical"` → [`KeyData::Composite`]. `Encrypted` is listed first
492/// so serde matches it before falling back to Single/Composite.
493#[non_exhaustive]
494#[derive(Clone, Serialize, Deserialize)]
495#[serde(untagged)]
496pub enum KeyData {
497    /// Passphrase-encrypted key material.
498    ///
499    /// The inner `KeyData` (`Single` or `Composite`) is serialized to JSON and
500    /// then encrypted with AES-256-GCM using a key derived from a user
501    /// passphrase via PBKDF2-HMAC-SHA256 (SP 800-132). Layout:
502    ///
503    /// 1. Derive 32-byte AES key: `PBKDF2-HMAC-SHA256(passphrase, salt, iters)`
504    /// 2. Serialize plaintext `KeyData` to JSON bytes
505    /// 3. Encrypt JSON bytes with AES-256-GCM using a fresh random nonce.
506    ///    AAD binds every envelope field (version, algorithm, key_type,
507    ///    KDF name, iteration count, salt, AEAD name) plus the enclosing
508    ///    `PortableKey`'s `algorithm` and `key_type`, so tampering with
509    ///    any metadata on disk causes AEAD authentication to fail.
510    /// 4. Store KDF params, nonce, and `ciphertext || tag` base64-encoded
511    ///
512    /// Format version `1` fixes the algorithm choices to
513    /// `PBKDF2-HMAC-SHA256` + `AES-256-GCM`; future versions may add alternatives.
514    Encrypted {
515        /// Envelope format version (currently `1`).
516        enc: u32,
517        /// KDF identifier (currently `"PBKDF2-HMAC-SHA256"`).
518        kdf: String,

589    /// Decode the single raw key bytes (returns error if composite or encrypted).
590    ///
591    /// # Security
592    ///
593    /// The returned `Vec<u8>` is **not** automatically zeroized on drop. When
594    /// this method is called for secret or symmetric key material, callers are
595    /// responsible for zeroizing the returned bytes after use. Prefer
596    /// [`decode_raw_zeroized`](Self::decode_raw_zeroized) for secret key data.
597    ///
598    /// # Errors
599    /// Returns an error if this is a composite key, the key is passphrase-encrypted
600    /// (call [`PortableKey::decrypt_with_passphrase`] first), or Base64 decoding fails.
601    pub fn decode_raw(&self) -> Result<Vec<u8>> {
602        match self {
603            Self::Single { raw } => BASE64_ENGINE
604                .decode(raw)
605                .map_err(|e| CoreError::SerializationError(format!("Invalid key base64: {e}"))),
606            Self::Composite { .. } => Err(CoreError::InvalidKey(
607                "Expected single key data but found composite".to_string(),
608            )),
609            Self::Encrypted { .. } => Err(CoreError::InvalidKey(
610                "Key is passphrase-encrypted; call PortableKey::decrypt_with_passphrase first"
611                    .to_string(),
612            )),
613        }
614    }
615
616    /// Decode the single raw key bytes into a zeroizing buffer (returns error if composite).
617    ///
618    /// Equivalent to [`decode_raw`](Self::decode_raw) but wraps the result in
619    /// [`zeroize::Zeroizing`] so the bytes are wiped from memory when dropped.
620    /// Use this variant whenever the data may be secret key material.
621    ///
622    /// # Errors
623    /// Returns an error if this is a composite key or Base64 decoding fails.
624    pub fn decode_raw_zeroized(&self) -> Result<zeroize::Zeroizing<Vec<u8>>> {
625        self.decode_raw().map(zeroize::Zeroizing::new)
626    }
627
628    /// Decode composite key components (returns error if single or encrypted).
629    ///
630    /// # Security
631    ///
632    /// The returned `Vec<u8>` values are **not** automatically zeroized on drop.
633    /// When this method is called for secret or symmetric key material, callers
634    /// are responsible for zeroizing the returned bytes after use. Prefer
635    /// [`decode_composite_zeroized`](Self::decode_composite_zeroized) for
636    /// secret key data.
637    ///
638    /// # Errors
639    /// Returns an error if this is a single key, the key is passphrase-encrypted
640    /// (call [`PortableKey::decrypt_with_passphrase`] first), or Base64 decoding fails.
641    pub fn decode_composite(&self) -> Result<(Vec<u8>, Vec<u8>)> {
642        match self {
643            Self::Composite { pq, classical } => {
644                let pq_bytes = BASE64_ENGINE.decode(pq).map_err(|e| {
645                    CoreError::SerializationError(format!("Invalid PQ key base64: {e}"))
646                })?;
647                let classical_bytes = BASE64_ENGINE.decode(classical).map_err(|e| {
648                    CoreError::SerializationError(format!("Invalid classical key base64: {e}"))
649                })?;
650                Ok((pq_bytes, classical_bytes))
651            }
652            Self::Single { .. } => Err(CoreError::InvalidKey(
653                "Expected composite key data but found single".to_string(),
654            )),
655            Self::Encrypted { .. } => Err(CoreError::InvalidKey(
656                "Key is passphrase-encrypted; call PortableKey::decrypt_with_passphrase first"
657                    .to_string(),
658            )),
659        }
660    }
661
662    /// Decode composite key components into zeroizing buffers (returns error if single).
663    ///
664    /// Equivalent to [`decode_composite`](Self::decode_composite) but wraps
665    /// each component in [`zeroize::Zeroizing`] so the bytes are wiped from
666    /// memory when dropped. Use this variant whenever either component may be
667    /// secret key material.
668    ///
669    /// # Errors
670    /// Returns an error if this is a single key or Base64 decoding fails.
671    pub fn decode_composite_zeroized(&self) -> Result<ZeroizingKeyPair> {
672        let (pq, classical) = self.decode_composite()?;
673        Ok((zeroize::Zeroizing::new(pq), zeroize::Zeroizing::new(classical)))
674    }
675
676    /// Create single key data from raw bytes.
677    #[must_use]
678    pub fn from_raw(bytes: &[u8]) -> Self {
679        Self::Single { raw: BASE64_ENGINE.encode(bytes) }
680    }
681
682    /// Create composite key data from PQ and classical components.
683    #[must_use]
684    pub fn from_composite(pq_bytes: &[u8], classical_bytes: &[u8]) -> Self {
685        Self::Composite {
686            pq: BASE64_ENGINE.encode(pq_bytes),
687            classical: BASE64_ENGINE.encode(classical_bytes),
688        }
689    }

696/// Portable, versioned key container for all LatticeArc key types.
697///
698/// Keys are identified by **use case** or **security level** — mirroring how
699/// the library's API works. The specific algorithm is auto-derived from these
700/// and stored internally for version-stability (the mapping may change between
701/// library releases). Users never specify algorithms directly.
702///
703/// At least one of `use_case` or `security_level` must be present. If both
704/// are set, `security_level` takes precedence for algorithm resolution
705/// (matching `CryptoConfig` behavior).
706///
707/// # Dual-Format Serialization
708///
709/// - **JSON** — human-readable: CLI, REST APIs, debugging
710/// - **CBOR** (RFC 8949) — compact binary: wire protocol, database, containers
711///
712/// # Enterprise Extensions
713///
714/// The `metadata` map is the extension point for enterprise features.
715/// Enterprise crates store additional fields (expiry, hardware binding,
716/// dimensions, etc.) as metadata entries and provide typed accessor traits.
717/// The base library preserves all metadata during roundtrips.
718///
719/// # Security Note: Clone
720///
721/// `PortableKey` derives `Clone` because it is a **serialization type** — it
722/// is designed to be written to disk, sent over the wire, or stored in a
723/// database. Cloning is required for serialization roundtrip testing and for
724/// passing keys through API boundaries.
725///
726/// **Do not use `PortableKey` as a long-lived runtime key holder.** Extract
727/// the concrete key type (e.g., `HybridPublicKey`, `HybridSecretKey`) via the
728/// `to_hybrid_*` bridge methods, and work with those instead. Secret key
729/// material inside `KeyData` is zeroized on drop via the explicit `Drop` impl.
730///
731/// # Constant-Time Comparison
732///
733/// `PortableKey` implements [`subtle::ConstantTimeEq`]. Metadata fields
734/// (`version`, `algorithm`, `key_type`, `use_case`, `security_level`, `created`,
735/// `metadata`) are compared with non-CT equality because they are serialized
736/// in plaintext on the wire — their equality is not a secret. The `key_data`
737/// field (containing actual key material, including encrypted envelopes) is
738/// compared in constant time via the canonical `subtle::ConstantTimeEq` pattern.
739///
740/// Cross-variant comparisons (`Single` vs `Composite`, etc.) always return
741/// `Choice(0)`. `PortableKey` deliberately does not derive [`PartialEq`] — use
742/// `ct_eq` explicitly when comparing key material, and prefer the concrete key
743/// types extracted via `to_hybrid_*` for cryptographic operations.
744///
745/// # Example
746///
747/// Real-world flow: generate keypair → wrap in PortableKey → serialize →
748/// deserialize → use for crypto operations.
749///
750/// ```rust,no_run
751/// use latticearc::{PortableKey, UseCase};
752///
753/// // 1. Generate a hybrid keypair
754/// let (pk, sk) = latticearc::generate_hybrid_keypair().expect("keygen");
755///
756/// // 2. Wrap in PortableKey (UseCase determines algorithm automatically)
757/// let (portable_pk, portable_sk) =
758///     PortableKey::from_hybrid_kem_keypair(UseCase::FileStorage, &pk, &sk)
759///         .expect("wrap");
760///
761/// // 3. Serialize (JSON for files/REST, CBOR for wire/storage)
762/// let json = portable_pk.to_json().expect("serialize");
763/// let cbor = portable_pk.to_cbor().expect("serialize");
764///
765/// // 4. Deserialize and extract for crypto operations
766/// let restored_pk = PortableKey::from_json(&json)
767///     .expect("deserialize")
768///     .to_hybrid_public_key()
769///     .expect("extract");
770/// ```
771#[derive(Clone, Serialize, Deserialize)]
772pub struct PortableKey {
773    /// Format version (currently `1`).
774    version: u32,
775
776    // --- Primary identifiers (at least one required) ---
777    /// Use case that determined algorithm selection.
778    /// This is how the library's API works: users pick a use case,
779    /// the policy engine selects the optimal algorithm.
780    #[serde(default, skip_serializing_if = "Option::is_none")]
781    use_case: Option<crate::types::types::UseCase>,
782
783    /// Security level (NIST Level 1/3/5). If both `use_case` and
784    /// `security_level` are set, `security_level` takes precedence
785    /// for algorithm resolution (matching `CryptoConfig` behavior).
786    #[serde(default, skip_serializing_if = "Option::is_none")]
787    security_level: Option<crate::types::types::SecurityLevel>,
788
789    /// Resolved algorithm identifier. Auto-derived from `use_case` or
790    /// `security_level` at creation time. Stored for version-stability —
791    /// if the policy engine mapping changes in a future release, existing
792    /// keys still parse correctly using this field.
793    algorithm: KeyAlgorithm,
794
795    /// Key type (public, secret, symmetric).
796    key_type: KeyType,
797    /// Key material (single or composite).
798    key_data: KeyData,
799    /// Creation timestamp (UTC, ISO 8601).

830impl subtle::ConstantTimeEq for PortableKey {
831    fn ct_eq(&self, other: &Self) -> subtle::Choice {
832        // Metadata fields are serialized in plaintext on the wire (version,
833        // algorithm identifiers, timestamps, extension map). Their equality
834        // is not a secret, so a non-CT short-circuit here is intentional —
835        // it avoids needlessly comparing key material when metadata already
836        // differs.
837        if self.version != other.version
838            || self.algorithm != other.algorithm
839            || self.key_type != other.key_type
840            || self.use_case != other.use_case
841            || self.security_level != other.security_level
842            || self.created != other.created
843            || self.metadata != other.metadata
844        {
845            return subtle::Choice::from(0);
846        }
847
848        // Key material: the CT-sensitive field.
849        self.key_data.ct_eq(&other.key_data)
850    }
851}

853impl subtle::ConstantTimeEq for KeyData {
854    fn ct_eq(&self, other: &Self) -> subtle::Choice {
855        use subtle::Choice;
856
857        match (self, other) {
858            (Self::Single { raw: a }, Self::Single { raw: b }) => a.as_bytes().ct_eq(b.as_bytes()),
859            (
860                Self::Composite { pq: a_pq, classical: a_cl },
861                Self::Composite { pq: b_pq, classical: b_cl },
862            ) => a_pq.as_bytes().ct_eq(b_pq.as_bytes()) & a_cl.as_bytes().ct_eq(b_cl.as_bytes()),
863            (
864                Self::Encrypted {
865                    enc: a_enc,
866                    kdf: a_kdf,
867                    kdf_iterations: a_iter,
868                    kdf_salt: a_salt,
869                    aead: a_aead,
870                    nonce: a_nonce,
871                    ciphertext: a_ct,
872                },
873                Self::Encrypted {
874                    enc: b_enc,
875                    kdf: b_kdf,
876                    kdf_iterations: b_iter,
877                    kdf_salt: b_salt,
878                    aead: b_aead,
879                    nonce: b_nonce,
880                    ciphertext: b_ct,
881                },
882            ) => {
883                // All envelope fields are stored in plaintext, so none are
884                // strictly secret. We short-circuit on the algorithm-identifier
885                // fields (fast path) and fall through to a CT compare on the
886                // variable-length byte fields for a uniform pattern with the
887                // other arms.
888                if a_enc != b_enc || a_kdf != b_kdf || a_iter != b_iter || a_aead != b_aead {
889                    return Choice::from(0);
890                }
891                a_salt.as_bytes().ct_eq(b_salt.as_bytes())
892                    & a_nonce.as_bytes().ct_eq(b_nonce.as_bytes())
893                    & a_ct.as_bytes().ct_eq(b_ct.as_bytes())
894            }
895            // Variant mismatches, enumerated explicitly (no `_` wildcard) so
896            // that adding a new `KeyData` variant fails to compile here —
897            // forcing the author to decide how it compares against every
898            // existing variant rather than silently defaulting to not-equal.
899            (Self::Single { .. }, Self::Composite { .. } | Self::Encrypted { .. })
900            | (Self::Composite { .. }, Self::Single { .. } | Self::Encrypted { .. })
901            | (Self::Encrypted { .. }, Self::Single { .. } | Self::Composite { .. }) => {
902                Choice::from(0)
903            }
904        }
905    }
906}

1213    /// Reconstruct a `HybridSecretKey` from a portable key pair.
1214    ///
1215    /// Requires the corresponding public key PortableKey because ML-KEM
1216    /// key reconstruction needs both the secret key bytes and public key bytes.
1217    ///
1218    /// # Arguments
1219    /// * `public_key` - The corresponding `PortableKey` with `KeyType::Public`
1220    ///
1221    /// # Errors
1222    /// Returns an error if the algorithm is not hybrid KEM, key data is invalid,
1223    /// or key reconstruction fails.
1224    ///
1225    /// The ML-KEM public key is extracted from the secret key file's metadata
1226    /// (stored at keygen time), making the secret key file fully self-contained.
1227    /// No separate public key file is needed for decryption.
1228    pub fn to_hybrid_secret_key(&self) -> Result<crate::hybrid::kem_hybrid::HybridKemSecretKey> {
1229        let level =
1230            crate::primitives::kem::MlKemSecurityLevel::try_from(self.algorithm).map_err(|()| {
1231                CoreError::InvalidKey(format!(
1232                    "Not a hybrid KEM algorithm: {alg:?}",
1233                    alg = self.algorithm
1234                ))
1235            })?;
1236
1237        if self.key_type != KeyType::Secret {
1238            return Err(CoreError::InvalidKey(
1239                "Cannot reconstruct secret key from a public key".to_string(),
1240            ));
1241        }
1242
1243        // Use the zeroizing decoder so the ML-KEM secret-key bytes and
1244        // the X25519 seed are wiped from heap memory when this scope
1245        // ends, regardless of which downstream branch we take.
1246        let (ml_kem_sk, ecdh_seed_vec) = self.key_data.decode_composite_zeroized()?;
1247
1248        // Extract ML-KEM public key from metadata (stored at keygen time)
1249        let ml_kem_pk = self
1250            .metadata
1251            .get(ML_KEM_PK_METADATA_KEY)
1252            .and_then(|v| v.as_str())
1253            .ok_or_else(|| {
1254                CoreError::InvalidKey(
1255                    "Secret key file missing 'ml_kem_pk' metadata. \
1256                     Re-generate the keypair with the latest CLI."
1257                        .to_string(),
1258                )
1259            })
1260            .and_then(|b64| {
1261                BASE64_ENGINE
1262                    .decode(b64)
1263                    .map_err(|e| CoreError::InvalidKey(format!("Invalid ml_kem_pk base64: {e}")))
1264            })?;
1265
1266        if ecdh_seed_vec.len() != 32 {
1267            return Err(CoreError::InvalidKey(format!(
1268                "X25519 seed must be 32 bytes, got {}",
1269                ecdh_seed_vec.len()
1270            )));
1271        }
1272
1273        let mut ecdh_seed = zeroize::Zeroizing::new([0u8; 32]);
1274        ecdh_seed.copy_from_slice(&ecdh_seed_vec);
1275
1276        crate::hybrid::kem_hybrid::HybridKemSecretKey::from_serialized(
1277            level, &ml_kem_sk, &ml_kem_pk, &ecdh_seed,
1278        )
1279        .map_err(|e| CoreError::InvalidKey(format!("Secret key reconstruction: {e}")))
1280    }

1695    /// - Non-hybrid algorithms require single `KeyData`
1696    /// - Base64 key data decodes successfully
1697    ///
1698    /// # Errors
1699    /// Returns `CoreError::InvalidKey` on validation failure.
1700    pub fn validate(&self) -> Result<()> {
1701        // Version check — reject keys serialized by future or incompatible versions
1702        if self.version != Self::CURRENT_VERSION {
1703            return Err(CoreError::InvalidKey(format!(
1704                "Unsupported key format version {}, expected {}",
1705                self.version,
1706                Self::CURRENT_VERSION
1707            )));
1708        }
1709
1710        // Symmetric algorithm ↔ key type
1711        if self.algorithm.is_symmetric() && self.key_type != KeyType::Symmetric {
1712            return Err(CoreError::InvalidKey(format!(
1713                "Algorithm {:?} requires KeyType::Symmetric, got {:?}",
1714                self.algorithm, self.key_type
1715            )));
1716        }
1717        if !self.algorithm.is_symmetric() && self.key_type == KeyType::Symmetric {
1718            return Err(CoreError::InvalidKey(format!(
1719                "KeyType::Symmetric is not valid for algorithm {:?}",
1720                self.algorithm
1721            )));
1722        }
1723
1724        // Hybrid ↔ composite key data (encrypted variant is opaque — skip this check).
1725        match (&self.key_data, self.algorithm.is_hybrid()) {
1726            (KeyData::Composite { .. }, false) => {
1727                return Err(CoreError::InvalidKey(format!(
1728                    "Non-hybrid algorithm {:?} must use single key data",
1729                    self.algorithm
1730                )));
1731            }
1732            (KeyData::Single { .. }, true) => {
1733                return Err(CoreError::InvalidKey(format!(
1734                    "Hybrid algorithm {:?} must use composite key data",
1735                    self.algorithm
1736                )));
1737            }
1738            _ => {}
1739        }
1740
1741        // Verify base64 decodes / envelope shape
1742        match &self.key_data {
1743            KeyData::Single { raw } => {
1744                let _ = BASE64_ENGINE
1745                    .decode(raw)
1746                    .map_err(|e| CoreError::SerializationError(format!("Invalid base64: {e}")))?;
1747            }
1748            KeyData::Composite { pq, classical } => {
1749                let _ = BASE64_ENGINE.decode(pq).map_err(|e| {
1750                    CoreError::SerializationError(format!("Invalid PQ base64: {e}"))
1751                })?;
1752                let _ = BASE64_ENGINE.decode(classical).map_err(|e| {
1753                    CoreError::SerializationError(format!("Invalid classical base64: {e}"))
1754                })?;
1755            }
1756            KeyData::Encrypted { enc, kdf, kdf_iterations, kdf_salt, aead, nonce, ciphertext } => {
1757                Self::validate_encrypted_envelope_fields(
1758                    *enc,
1759                    kdf,
1760                    *kdf_iterations,
1761                    kdf_salt,
1762                    aead,
1763                    nonce,
1764                    ciphertext,
1765                )?;
1766            }
1767        }
1768
1769        Ok(())
1770    }

1772    /// Validate an encrypted-envelope's metadata and base64-decodable sizes.
1773    ///
1774    /// Shared between [`Self::validate`] (which rejects malformed keys at
1775    /// load time) and [`Self::decrypt_with_passphrase`] (which uses this to
1776    /// defend against direct construction of an unvalidated encrypted
1777    /// variant). Does not verify the AEAD tag — that happens in
1778    /// `decrypt_with_passphrase` after key derivation.
1779    fn validate_encrypted_envelope_fields(
1780        enc: u32,
1781        kdf: &str,
1782        kdf_iterations: u32,
1783        kdf_salt: &str,
1784        aead: &str,
1785        nonce: &str,
1786        ciphertext: &str,
1787    ) -> Result<()> {
1788        if enc != ENCRYPTED_ENVELOPE_VERSION {
1789            return Err(CoreError::InvalidKey(format!(
1790                "Unsupported encrypted key envelope version {enc}, expected {ENCRYPTED_ENVELOPE_VERSION}",
1791            )));
1792        }
1793        if kdf != PBKDF2_KDF_ID {
1794            return Err(CoreError::InvalidKey(format!(
1795                "Unsupported KDF {kdf:?}, expected {PBKDF2_KDF_ID:?}",
1796            )));
1797        }
1798        if aead != AES_GCM_AEAD_ID {
1799            return Err(CoreError::InvalidKey(format!(
1800                "Unsupported AEAD {aead:?}, expected {AES_GCM_AEAD_ID:?}",
1801            )));
1802        }
1803        if kdf_iterations < PBKDF2_MIN_ITERATIONS {
1804            return Err(CoreError::InvalidKey(format!(
1805                "PBKDF2 iteration count {kdf_iterations} below minimum {PBKDF2_MIN_ITERATIONS}",
1806            )));
1807        }
1808        if kdf_iterations > PBKDF2_MAX_ITERATIONS {
1809            // Round-11 audit fix: reject adversary-supplied envelopes with
1810            // pathological `kdf_iterations` before any HMAC-SHA256 rounds run.
1811            return Err(CoreError::InvalidKey(format!(
1812                "PBKDF2 iteration count {kdf_iterations} exceeds maximum {PBKDF2_MAX_ITERATIONS}",
1813            )));
1814        }
1815        // Warn when a loaded key uses fewer iterations than the current
1816        // OWASP-recommended default. The minimum (100k) is retained as a
1817        // hard floor for backwards compatibility with keys generated under
1818        // OWASP 2018 guidance; the default (600k) is OWASP 2023 for
1819        // HMAC-SHA256. Callers should re-protect keys below the default.
1820        //
1821        // Deduped per *distinct iteration count* per process: a mixed-fleet
1822        // operator loading key-A at 10k and key-B at 1k must see two
1823        // warnings (one per cohort), not one — otherwise the audit trail
1824        // hides the lower-iteration cohort behind the first cohort that
1825        // happened to trigger. Using `Mutex<HashSet<u32>>` here is fine
1826        // because key load is cold-path; the mutex is never contended on
1827        // a hot loop.
1828        if kdf_iterations < PBKDF2_DEFAULT_ITERATIONS {
1829            // Capacity cap prevents unbounded memory growth from an adversary
1830            // feeding distinct iteration-count keys. 256 distinct values is
1831            // far more than any realistic operator fleet (cohorts cluster
1832            // around OWASP guidance dates); beyond the cap we silence
1833            // further warnings — a single previously-emitted warning is
1834            // already enough audit signal that the cohort exists.
1835            const MAX_DISTINCT_ITERATION_COUNTS: usize = 256;
1836            static LOW_ITER_WARNED: std::sync::OnceLock<
1837                std::sync::Mutex<std::collections::HashSet<u32>>,
1838            > = std::sync::OnceLock::new();
1839            let table = LOW_ITER_WARNED.get_or_init(|| std::sync::Mutex::new(Default::default()));
1840            // Lock-poisoning here is non-fatal — emit the warning anyway,
1841            // since not warning is strictly worse than a duplicate warning.
1842            let mut seen = table.lock().unwrap_or_else(std::sync::PoisonError::into_inner);
1843            let should_warn = if seen.contains(&kdf_iterations) {
1844                false
1845            } else if seen.len() < MAX_DISTINCT_ITERATION_COUNTS {
1846                seen.insert(kdf_iterations);
1847                true
1848            } else {
1849                // Capacity reached — drop the new value silently. The
1850                // operator already has 256 prior warnings to act on; a
1851                // 257th cohort would not change the response.
1852                false
1853            };
1854            if should_warn {
1855                tracing::warn!(
1856                    kdf_iterations,
1857                    recommended = PBKDF2_DEFAULT_ITERATIONS,
1858                    "PBKDF2 iteration count is below the current OWASP recommendation; \
1859                     re-protect this key (decrypt + re-encrypt with passphrase) at the \
1860                     next opportunity."
1861                );
1862            }
1863        }
1864        let salt = BASE64_ENGINE
1865            .decode(kdf_salt)
1866            .map_err(|e| CoreError::SerializationError(format!("Invalid KDF salt base64: {e}")))?;
1867        if salt.len() < PBKDF2_MIN_SALT_LEN {
1868            return Err(CoreError::InvalidKey(format!(
1869                "PBKDF2 salt length {} below minimum {PBKDF2_MIN_SALT_LEN}",
1870                salt.len(),
1871            )));
1872        }
1873        let nonce_bytes = BASE64_ENGINE
1874            .decode(nonce)
1875            .map_err(|e| CoreError::SerializationError(format!("Invalid nonce base64: {e}")))?;
1876        if nonce_bytes.len() != AES_GCM_NONCE_LEN {
1877            return Err(CoreError::InvalidKey(format!(
1878                "AES-GCM nonce length {} != {AES_GCM_NONCE_LEN}",
1879                nonce_bytes.len(),
1880            )));
1881        }
1882        let ct_bytes = BASE64_ENGINE.decode(ciphertext).map_err(|e| {
1883            CoreError::SerializationError(format!("Invalid ciphertext base64: {e}"))
1884        })?;
1885        if ct_bytes.len() < AES_GCM_TAG_LEN {
1886            return Err(CoreError::InvalidKey(
1887                "Encrypted key ciphertext shorter than AES-GCM tag".to_string(),
1888            ));
1889        }
1890        Ok(())
1891    }

1905    /// Encrypt the key material in place using a passphrase.
1906    ///
1907    /// Derives a 32-byte AES key via PBKDF2-HMAC-SHA256 (600,000 iterations,
1908    /// 16-byte random salt) and encrypts the JSON-serialized `KeyData` under
1909    /// AES-256-GCM with a fresh 12-byte random nonce. The full envelope
1910    /// (version, algorithm, key_type, KDF name, iteration count, salt, AEAD
1911    /// name) is mixed into the AEAD AAD, so tampering with any metadata
1912    /// field on disk causes decryption to fail at the tag check. See
1913    /// [`Self::encryption_aad`] for the exact byte layout.
1914    ///
1915    /// # Errors
1916    ///
1917    /// Returns an error if the key is already encrypted, if the passphrase is
1918    /// empty, or if the KDF / AEAD operation fails.
1919    pub fn encrypt_with_passphrase(&mut self, passphrase: &[u8]) -> Result<()> {
1920        if self.is_encrypted() {
1921            return Err(CoreError::InvalidKey("Key is already passphrase-encrypted".to_string()));
1922        }
1923        if passphrase.is_empty() {
1924            return Err(CoreError::InvalidKey("Passphrase must not be empty".to_string()));
1925        }
1926
1927        // 1. Serialize the current plaintext KeyData to its own JSON. We
1928        //    serialize only the KeyData (not the whole PortableKey) so the
1929        //    ciphertext is self-contained and round-trips cleanly.
1930        let plaintext_json = serde_json::to_vec(&self.key_data).map_err(|e| {
1931            CoreError::SerializationError(format!(
1932                "Failed to serialize key data for encryption: {e}"
1933            ))
1934        })?;
1935        let plaintext = zeroize::Zeroizing::new(plaintext_json);
1936
1937        // 2. Generate a fresh random salt.
1938        let salt = crate::primitives::rand::csprng::random_bytes(PBKDF2_SALT_LEN);
1939
1940        // 3. Derive the AES key via PBKDF2-HMAC-SHA256. Salt-length
1941        // validation lives at the `pbkdf2(...)` call below; `with_salt`
1942        // is intentionally infallible so wire-format parsers can also
1943        // round-trip pre-0.8.0 short-salt envelopes for inspection.
1944        let kdf_params = crate::primitives::kdf::pbkdf2::Pbkdf2Params::with_salt(&salt)
1945            .iterations(PBKDF2_DEFAULT_ITERATIONS)
1946            .key_length(AES_256_KEY_LEN);
1947        let derived = crate::primitives::kdf::pbkdf2::pbkdf2(passphrase, &kdf_params)
1948            .map_err(|e| CoreError::InvalidKey(format!("PBKDF2 derivation failed: {e}")))?;
1949
1950        // 4. Encrypt via AES-256-GCM with a fresh random nonce. Bind the
1951        //    full envelope (version, algorithm, key_type, KDF name,
1952        //    iterations, salt, AEAD name) to the ciphertext via AAD so an
1953        //    attacker who modifies any of these fields on disk breaks the
1954        //    AEAD tag.
1955        use crate::primitives::aead::AeadCipher;
1956        let cipher = crate::primitives::aead::aes_gcm::AesGcm256::new(derived.expose_secret())
1957            .map_err(|e| CoreError::InvalidKey(format!("Failed to initialize AES-256-GCM: {e}")))?;
1958        let aad = Self::encryption_aad(
1959            ENCRYPTED_ENVELOPE_VERSION,
1960            self.algorithm,
1961            self.key_type,
1962            PBKDF2_KDF_ID,
1963            PBKDF2_DEFAULT_ITERATIONS,
1964            &salt,
1965            AES_GCM_AEAD_ID,
1966            &self.metadata,
1967        )?;
1968        let (nonce, mut ct, tag) = cipher
1969            .seal(&plaintext, Some(&aad))
1970            .map_err(|e| CoreError::InvalidKey(format!("AES-256-GCM sealing failed: {e}")))?;
1971
1972        // 5. Pack ciphertext || tag for on-wire storage.
1973        ct.extend_from_slice(&tag);
1974
1975        // 6. Replace the plaintext KeyData with the encrypted envelope.
1976        self.key_data = KeyData::Encrypted {
1977            enc: ENCRYPTED_ENVELOPE_VERSION,
1978            kdf: PBKDF2_KDF_ID.to_string(),
1979            kdf_iterations: PBKDF2_DEFAULT_ITERATIONS,
1980            kdf_salt: BASE64_ENGINE.encode(&salt),
1981            aead: AES_GCM_AEAD_ID.to_string(),
1982            nonce: BASE64_ENGINE.encode(nonce),
1983            ciphertext: BASE64_ENGINE.encode(&ct),
1984        };
1985
1986        Ok(())
1987    }

1989    /// Decrypt the key material in place using a passphrase.
1990    ///
1991    /// Reverses [`Self::encrypt_with_passphrase`]. On success the `key_data`
1992    /// field is replaced with the underlying `Single` or `Composite` variant.
1993    ///
1994    /// # Errors
1995    ///
1996    /// Returns an error if the key is not encrypted, the envelope is
1997    /// malformed, the passphrase is wrong (AEAD authentication fails), or
1998    /// the decrypted plaintext is not a valid `KeyData` serialization.
1999    pub fn decrypt_with_passphrase(&mut self, passphrase: &[u8]) -> Result<()> {
2000        if passphrase.is_empty() {
2001            return Err(CoreError::InvalidKey("Passphrase must not be empty".to_string()));
2002        }
2003
2004        // Borrow the envelope to validate shape and decode its fields into
2005        // owned bytes. After validation the `kdf` and `aead` string fields
2006        // are known to equal the envelope's fixed constants, so we don't
2007        // bother cloning them into `Decoded` — we pass the constants to
2008        // `encryption_aad` at the call site instead.
2009        struct Decoded {
2010            enc: u32,
2011            kdf_iterations: u32,
2012            salt: Vec<u8>,
2013            nonce: [u8; AES_GCM_NONCE_LEN],
2014            ct_and_tag: Vec<u8>,
2015        }
2016        let decoded = match &self.key_data {
2017            KeyData::Encrypted { enc, kdf, kdf_iterations, kdf_salt, aead, nonce, ciphertext } => {
2018                Self::validate_encrypted_envelope_fields(
2019                    *enc,
2020                    kdf,
2021                    *kdf_iterations,
2022                    kdf_salt,
2023                    aead,
2024                    nonce,
2025                    ciphertext,
2026                )?;
2027                let salt = BASE64_ENGINE.decode(kdf_salt).map_err(|e| {
2028                    CoreError::SerializationError(format!("Invalid KDF salt base64: {e}"))
2029                })?;
2030                let nonce_bytes = BASE64_ENGINE.decode(nonce).map_err(|e| {
2031                    CoreError::SerializationError(format!("Invalid nonce base64: {e}"))
2032                })?;
2033                let mut nonce_array = [0u8; AES_GCM_NONCE_LEN];
2034                nonce_array.copy_from_slice(&nonce_bytes);
2035                let ct_and_tag = BASE64_ENGINE.decode(ciphertext).map_err(|e| {
2036                    CoreError::SerializationError(format!("Invalid ciphertext base64: {e}"))
2037                })?;
2038                Decoded {
2039                    enc: *enc,
2040                    kdf_iterations: *kdf_iterations,
2041                    salt,
2042                    nonce: nonce_array,
2043                    ct_and_tag,
2044                }
2045            }
2046            _ => {
2047                return Err(CoreError::InvalidKey("Key is not passphrase-encrypted".to_string()));
2048            }
2049        };
2050        let Decoded { enc, kdf_iterations, salt, nonce: nonce_array, ct_and_tag } = decoded;
2051
2052        let tag_offset = ct_and_tag
2053            .len()
2054            .checked_sub(AES_GCM_TAG_LEN)
2055            .ok_or_else(|| CoreError::InvalidKey("Ciphertext shorter than tag".to_string()))?;
2056        let (ct_bytes, tag_bytes) = ct_and_tag.split_at(tag_offset);
2057        let mut tag_array = [0u8; AES_GCM_TAG_LEN];
2058        tag_array.copy_from_slice(tag_bytes);
2059
2060        // Derive the AES key from the passphrase + salt.
2061        // `with_salt` is the right tool here: `salt` was
2062        // deserialized from an externally-supplied envelope and may be
2063        // a pre-0.8.0 short salt that `with_salt`'s NIST SP 800-132 §5.1
2064        // floor would reject. The validating min-length check happens at
2065        // the `pbkdf2(...)` call below instead, so the unchecked
2066        // construction cannot smuggle a short salt past actual
2067        // derivation. This is the documented wire-format-parser use
2068        // case for `with_salt`.
2069        let kdf_params = crate::primitives::kdf::pbkdf2::Pbkdf2Params::with_salt(&salt)
2070            .iterations(kdf_iterations)
2071            .key_length(AES_256_KEY_LEN);
2072        // Use the legacy load-side floor (PBKDF2_MIN_ITERATIONS = 100k)
2073        // rather than the public OWASP 2023 floor: the iteration count
2074        // here is integrity-protected by the envelope AAD, so a count
2075        // below 600k means the legitimate keyholder wrote it under
2076        // OWASP 2018 guidance. Refusing to load it would strand
2077        // historical keys; the caller already saw a re-protect warning
2078        // in `validate_encrypted_envelope_fields` above.
2079        let derived = crate::primitives::kdf::pbkdf2::pbkdf2_with_floor(
2080            passphrase,
2081            &kdf_params,
2082            PBKDF2_MIN_ITERATIONS,
2083        )
2084        .map_err(|e| CoreError::InvalidKey(format!("PBKDF2 derivation failed: {e}")))?;
2085
2086        // Decrypt. A wrong passphrase produces a wrong AES key, which causes
2087        // AEAD authentication to fail with an opaque error — we do NOT leak
2088        // whether the passphrase was wrong vs. the envelope was corrupted.
2089        // The AAD binds the full envelope so any tampered metadata field
2090        // also breaks the tag.
2091        use crate::primitives::aead::AeadCipher;
2092        let cipher = crate::primitives::aead::aes_gcm::AesGcm256::new(derived.expose_secret())
2093            .map_err(|e| CoreError::InvalidKey(format!("Failed to initialize AES-256-GCM: {e}")))?;
2094        // `kdf` and `aead` are the constants: `validate_encrypted_envelope_fields`
2095        // rejected any other value, so we can pass the literals directly
2096        // instead of cloning them out of the borrowed envelope.
2097        let aad = Self::encryption_aad(
2098            enc,
2099            self.algorithm,
2100            self.key_type,
2101            PBKDF2_KDF_ID,
2102            kdf_iterations,
2103            &salt,
2104            AES_GCM_AEAD_ID,
2105            &self.metadata,
2106        )?;
2107        let plaintext = cipher
2108            .decrypt(&nonce_array, ct_bytes, &tag_array, Some(&aad))
2109            .map_err(|_e| {
2110                CoreError::InvalidKey(
2111                    "Passphrase-protected key unwrap failed (wrong passphrase or corrupted envelope)"
2112                        .to_string(),
2113                )
2114            })?;
2115
2116        // Deserialize the plaintext bytes back into a KeyData variant.
2117        let new_key_data: KeyData = serde_json::from_slice(&plaintext).map_err(|e| {
2118            CoreError::SerializationError(format!("Failed to deserialize decrypted key data: {e}"))
2119        })?;
2120        // Reject nested encryption (prevents re-wrap confusion).
2121        if matches!(new_key_data, KeyData::Encrypted { .. }) {
2122            return Err(CoreError::InvalidKey(
2123                "Decrypted payload was itself an encrypted envelope".to_string(),
2124            ));
2125        }
2126
2127        // Stage the new KeyData and run the same coherence check that
2128        // `from_json`/`load_from_file` apply on entry. An attacker who
2129        // crafted an envelope where the AEAD-bound metadata declares one
2130        // algorithm but the decrypted KeyData payload describes another
2131        // would otherwise install a structurally-incoherent PortableKey
2132        // that the rest of the API would only catch much later (or, in
2133        // some paths, not at all). Run validation before mutating self
2134        // so a failed decrypt leaves the receiver untouched.
2135        let original = std::mem::replace(&mut self.key_data, new_key_data);
2136        if let Err(e) = self.validate() {
2137            // Restore the original encrypted envelope on validation
2138            // failure so the caller can retry, inspect, or re-wrap.
2139            self.key_data = original;
2140            return Err(e);
2141        }
2142        Ok(())
2143    }

2145    /// Build the AAD bound to an encrypted key envelope.
2146    ///
2147    /// AEAD AAD is authenticated (not encrypted), so fields folded in here
2148    /// are protected against tampering but not against disclosure. Binding
2149    /// all envelope parameters — version, algorithm, key type, KDF name,
2150    /// iteration count, salt, and AEAD name — ensures that any attacker
2151    /// modification of a stored key file's metadata or KDF parameters
2152    /// causes `cipher.decrypt` to fail with an opaque error.
2153    ///
2154    /// Uses stable kebab-case / lowercase names (`ml-kem-768`, `secret`)
2155    /// via `canonical_name` accessors on the enums. These are load-bearing:
2156    /// changing the returned strings breaks every existing encrypted key
2157    /// file. A pinned byte-level test in the `tests` module guards this.
2158    ///
2159    /// # Byte layout
2160    ///
2161    /// ```text
2162    /// "latticearc-lpk-v1-enc" || 0x00
2163    /// || enc (u32 BE)
2164    /// || algorithm_name || 0x00
2165    /// || key_type_name || 0x00
2166    /// || kdf_name || 0x00
2167    /// || kdf_iterations (u32 BE)
2168    /// || kdf_salt_len (u32 BE) || kdf_salt_raw_bytes
2169    /// || aead_name
2170    /// ```
2171    ///
2172    /// Length prefixes and null separators prevent ambiguity between
2173    /// adjacent variable-length fields. The salt is included as its raw
2174    /// (base64-decoded) bytes, not the base64 string, so an attacker
2175    /// cannot use base64 non-canonical encodings to get past the check.
2176    #[allow(clippy::too_many_arguments)] // 8 fields are inherent to the AAD layout; bundling into a struct adds indirection without simplifying the call sites.
2177    fn encryption_aad(
2178        enc: u32,
2179        algorithm: KeyAlgorithm,
2180        key_type: KeyType,
2181        kdf: &str,
2182        kdf_iterations: u32,
2183        kdf_salt: &[u8],
2184        aead: &str,
2185        metadata: &BTreeMap<String, serde_json::Value>,
2186    ) -> Result<Vec<u8>> {
2187        let algorithm_name = algorithm.canonical_name();
2188        let key_type_name = key_type.canonical_name();
2189
2190        // Canonical metadata serialization: BTreeMap iterates in
2191        // lexicographic key order, so `serde_json::to_vec` produces
2192        // deterministic bytes for flat string→string maps (the actual
2193        // shape used by this format — `ml_kem_pk` and friends are
2194        // base64 strings). Length-prefix the result so a metadata
2195        // append/truncate cannot collide with an adjacent AAD field.
2196        let metadata_bytes = serde_json::to_vec(metadata).map_err(|e| {
2197            CoreError::SerializationError(format!(
2198                "Failed to canonicalize key-format metadata for AEAD AAD: {e}"
2199            ))
2200        })?;
2201        let metadata_len_u32 = u32::try_from(metadata_bytes.len()).map_err(|_e| {
2202            CoreError::SerializationError(
2203                "Encrypted-envelope metadata exceeds u32::MAX bytes".to_string(),
2204            )
2205        })?;
2206
2207        let mut aad = Vec::with_capacity(
2208            b"latticearc-lpk-v2-enc"
2209                .len()
2210                .saturating_add(1) // null
2211                .saturating_add(4) // enc
2212                .saturating_add(algorithm_name.len())

2392    /// Build an `AtomicWrite` with the project's mode policy for this
2393    /// key type pre-applied.
2394    ///
2395    /// Mode policy:
2396    ///   Secret / Symmetric → 0o600 (owner read+write only)
2397    ///   Public             → 0o644 (owner rw, others r) — public
2398    ///                        keys are MEANT to be readable; without
2399    ///                        this an explicit 0o644, tempfile's
2400    ///                        default 0o600 would lock pub keys to
2401    ///                        the creator and break key-distribution
2402    ///                        flows.
2403    ///
2404    /// Round-9 audit fix #4: lifted out of `write_to_file_with_overwrite`
2405    /// so the JSON and CBOR write paths share one source of truth for
2406    /// mode selection. Drift between the two would silently regress the
2407    /// secret-key threat model.
2408    fn make_atomic_writer<'a>(
2409        &self,
2410        bytes: &'a [u8],
2411        overwrite: bool,
2412    ) -> crate::unified_api::atomic_write::AtomicWrite<'a> {
2413        let writer =
2414            crate::unified_api::atomic_write::AtomicWrite::new(bytes).overwrite_existing(overwrite);
2415        if self.key_type == KeyType::Secret || self.key_type == KeyType::Symmetric {
2416            writer.secret_mode()
2417        } else {
2418            writer.unix_mode(0o644)
2419        }
2420    }

2473    ///
2474    /// The CLI v1 format uses `"key"` for raw key bytes and `"algorithm"` as a string.
2475    ///
2476    /// ```json
2477    /// {
2478    ///   "algorithm": "ML-DSA-65",
2479    ///   "key_type": "public",

2481    /// }
2482    /// ```
2483    ///
2484    /// # Errors
2485    /// Returns an error if the JSON is invalid or the algorithm is unrecognized.
2486    pub fn from_legacy_json(json: &str) -> Result<Self> {
2487        // Round-11 audit fix: parallel `from_json` enforces a 1 MiB size
2488        // guard before parsing to prevent memory exhaustion from
2489        // maliciously crafted payloads. The legacy path was missing this
2490        // check, leaving a DoS vector when migrating older keyfiles.
2491        if json.len() > Self::MAX_KEY_JSON_SIZE {
2492            return Err(CoreError::ResourceExceeded(format!(
2493                "Legacy key JSON size {} exceeds limit {}",
2494                json.len(),
2495                Self::MAX_KEY_JSON_SIZE
2496            )));
2497        }
2498
2499        #[derive(Deserialize)]
2500        struct LegacyKeyFile {
2501            algorithm: String,
2502            key_type: String,
2503            key: String,
2504            #[serde(default)]
2505            label: Option<String>,
2506        }
2507
2508        let legacy: LegacyKeyFile = serde_json::from_str(json)
2509            .map_err(|e| CoreError::SerializationError(format!("Legacy JSON parse failed: {e}")))?;
2510
2511        let algorithm = parse_legacy_algorithm(&legacy.algorithm)?;
2512        let key_type = match legacy.key_type.to_lowercase().as_str() {
2513            "public" | "pub" => KeyType::Public,
2514            "secret" | "private" | "sk" => KeyType::Secret,
2515            "symmetric" | "sym" => KeyType::Symmetric,
2516            other => {
2517                return Err(CoreError::InvalidKey(format!(

2519                )));

2521        };

2524

2526        if let Some(label) = legacy.label {
2527            key.set_label(label);
2528        }
2529        key.validate()?;
2530        Ok(key)
2531    }
