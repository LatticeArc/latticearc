19/// Threshold (in bytes) below which classical-only encryption may be used for
20/// performance optimization when explicitly configured with low security + speed.
21///
22/// **Rationale**: ML-KEM ciphertext overhead is ~1000-1500 bytes depending on
23/// security level. For messages under this threshold, the hybrid overhead is
24/// significant relative to message size.
25///
26/// **Security Note**: Classical fallback ONLY occurs when ALL of:
27/// 1. Security level is `Medium` or `Low` (user accepts reduced security)
28/// 2. Performance preference is `Speed` (user prioritizes performance)
29/// 3. Data size is below this threshold
30///
31/// If quantum safety is required for all messages regardless of size, use
32/// `SecurityLevel::High` or `SecurityLevel::Maximum`.
33pub const CLASSICAL_FALLBACK_SIZE_THRESHOLD: usize = 4096;
34
35/// Main cryptographic policy engine.
36///
37/// Analyzes data and configuration to recommend optimal cryptographic schemes
38/// based on security policies, use cases, and runtime context.
39///
40/// # Modes
41///
42/// The engine supports three cryptographic modes:
43/// - **Hybrid** (default): ML-KEM + X25519 + AES-256-GCM for defense-in-depth
44/// - **PQ-Only**: ML-KEM + AES-256-GCM for pure post-quantum security
45/// - **Classical**: X25519 + AES-256-GCM for legacy compatibility
46pub struct CryptoPolicyEngine;

162    /// Selects encryption scheme based on data, config, and optional use case.
163    ///
164    /// When `hardware_acceleration` is `false` and no use case is specified,
165    /// prefers `chacha20-poly1305` (fast in software without AES-NI).
166    ///
167    /// When data is provided, analyzes data characteristics to optimize
168    /// scheme selection for the data's entropy and size patterns.
169    ///
170    /// # Errors
171    ///
172    /// This function currently does not return errors, but returns `Result`
173    /// for future compatibility with validation logic.
174    pub fn select_encryption_scheme(
175        data: &[u8],
176        config: &CoreConfig,
177        use_case: Option<&UseCase>,
178    ) -> Result<String> {
179        if let Some(use_case) = use_case {
180            return Self::recommend_scheme(use_case, config);
181        }
182
183        // When hardware acceleration is disabled, prefer ChaCha20-Poly1305
184        // which is fast in pure software (no AES-NI needed)
185        if !config.hardware_acceleration {
186            return Ok(CHACHA20_POLY1305.to_string());
187        }
188
189        // Data-aware adjustments when data is non-empty
190        if !data.is_empty() {
191            let characteristics = Self::analyze_data_characteristics(data);
192
193            match (&config.performance_preference, &characteristics.pattern_type) {
194                // High-entropy data with speed preference: smaller KEM is sufficient
195                (PerformancePreference::Speed, PatternType::Random) => {
196                    if matches!(config.security_level, SecurityLevel::High) {
197                        return Ok(HYBRID_ENCRYPTION_512.to_string());
198                    }
199                }
200                // Memory-constrained with small data: downgrade if not at maximum
201                (PerformancePreference::Memory, _)
202                    if data.len() < CLASSICAL_FALLBACK_SIZE_THRESHOLD =>
203                {
204                    if matches!(config.security_level, SecurityLevel::High) {
205                        return Ok(HYBRID_ENCRYPTION_512.to_string());
206                    }
207                }
208                _ => {}
209            }
210        }
211
212        Ok(Self::select_for_security_level(config))
213    }

240    /// Select encryption scheme based on runtime performance metrics.
241    ///
242    /// Implementation: Thresholds on `memory_usage_mb` and `encryption_speed_ms`
243    /// select smaller KEM parameters when resources are constrained.
244    ///
245    /// # Errors
246    ///
247    /// This function currently does not return errors, but returns `Result`
248    /// for future compatibility with validation logic.
249    pub fn adaptive_selection(
250        data: &[u8],
251        performance_metrics: &PerformanceMetrics,
252        config: &CoreConfig,
253    ) -> Result<String> {
254        let characteristics = Self::analyze_data_characteristics(data);
255        let base_scheme = Self::select_encryption_scheme(data, config, None)?;
256
257        match (&config.performance_preference, performance_metrics) {
258            (PerformancePreference::Memory, metrics) if metrics.memory_usage_mb > 500.0 => {
259                Ok("hybrid-ml-kem-768-aes-256-gcm".to_string())
260            }
261            (PerformancePreference::Speed, metrics)
262                if metrics.encryption_speed_ms > 1000.0
263                    && matches!(characteristics.pattern_type, PatternType::Repetitive) =>
264            {
265                Ok("hybrid-ml-kem-512-aes-256-gcm".to_string())
266            }
267            _ => Ok(base_scheme),
268        }
269    }

350            SecurityLevel::Standard => EncryptionScheme::HybridMlKem512Aes256Gcm,

354    /// Validate that the key variant matches the scheme requirements.
355    ///
356    /// Returns `Ok(())` if the key type is compatible with the scheme,
357    /// or `Err(TypeError::ConfigurationError)` describing the mismatch.
358    ///
359    /// # Errors
360    ///
361    /// Returns an error if:
362    /// - A symmetric key is provided for a hybrid scheme
363    /// - A hybrid key is provided for a symmetric scheme
364    pub fn validate_key_matches_scheme(
365        key: &EncryptKey<'_>,
366        scheme: &EncryptionScheme,
367    ) -> Result<()> {
368        match (key, scheme.requires_hybrid_key()) {
369            (EncryptKey::Symmetric(_), true) => Err(TypeError::ConfigurationError(format!(
370                "Scheme '{}' requires a hybrid key (EncryptKey::Hybrid), \
371                 but a symmetric key was provided. Use generate_hybrid_keypair() \
372                 to create a hybrid keypair.",
373                scheme
374            ))),
375            (EncryptKey::Hybrid(_), false) => Err(TypeError::ConfigurationError(format!(
376                "Scheme '{}' requires a symmetric key (EncryptKey::Symmetric), \
377                 but a hybrid key was provided.",
378                scheme
379            ))),
380            _ => {
381                // For hybrid keys, verify ML-KEM security level matches scheme
382                if let (EncryptKey::Hybrid(pk), Some(expected_level)) = (key, scheme.ml_kem_level())
383                    && pk.security_level() != expected_level
384                {
385                    return Err(TypeError::ConfigurationError(format!(
386                        "Scheme '{}' requires ML-KEM-{} key, but the provided key \
387                         was generated at ML-KEM-{} level. Use \
388                         generate_hybrid_keypair_with_level({:?}).",
389                        scheme,
390                        expected_level.name(),
391                        pk.security_level().name(),
392                        expected_level
393                    )));
394                }
395                Ok(())
396            }
397        }
398    }

416                "Scheme '{}' requires a symmetric key (DecryptKey::Symmetric), \
417                 but a hybrid key was provided.",
418                scheme
419            ))),
420            _ => {
421                // For hybrid keys, verify ML-KEM security level matches scheme
422                if let (DecryptKey::Hybrid(sk), Some(expected_level)) = (key, scheme.ml_kem_level())
423                    && sk.security_level() != expected_level

425                    return Err(TypeError::ConfigurationError(format!(

429                        expected_level.name(),
430                        sk.security_level().name(),
431                    )));
432                }
433                Ok(())
434            }
435        }
436    }

475/// Calculates the Shannon entropy of the data in bits per byte.
476fn calculate_entropy(data: &[u8]) -> f64 {
477    if data.is_empty() {
478        return 0.0;
479    }
480
481    let mut frequency = [0u64; 256];
482    for &byte in data {
483        let index = usize::from(byte);
484        if let Some(count) = frequency.get_mut(index) {
485            *count = count.saturating_add(1);
486        }
487    }
488
489    #[allow(clippy::cast_precision_loss)]
490    let len = data.len() as f64;
491    let mut entropy = 0.0_f64;
492
493    for &count in &frequency {
494        if count > 0 {
495            #[allow(clippy::cast_precision_loss)]
496            let probability = count as f64 / len;
497            entropy -= probability * probability.log2();
498        }
499    }
500
501    entropy
502}

504/// Detects the pattern type of the input data.
505fn detect_pattern_type(data: &[u8]) -> PatternType {
506    if data.is_empty() {
507        return PatternType::Random;
508    }
509
510    let entropy = calculate_entropy(data);
511
512    if entropy > 7.5 {
513        return PatternType::Random;
514    }
515
516    let mut is_text = true;
517    for &byte in data {
518        if !(byte.is_ascii_graphic() || byte.is_ascii_whitespace()) {
519            is_text = false;
520            break;
521        }
522    }
523
524    if is_text && entropy > 4.0 {
525        return PatternType::Text;
526    }
527
528    let mut repetitive = true;
529    if data.len() > 8 {
530        let chunk_size = std::cmp::min(8, data.len() / 4);
531        let first_chunk = data.get(..chunk_size);
532
533        if let Some(first) = first_chunk {
534            for chunk in data.chunks(chunk_size).skip(1) {
535                if chunk != first {
536                    repetitive = false;
537                    break;
538                }
539            }
540        } else {
541            repetitive = false;
542        }
543    } else {
544        repetitive = false;
545    }
546
547    if repetitive {
548        return PatternType::Repetitive;
549    }
550
551    let has_structure = data.windows(4).any(|window| {
552        let w0 = window.first().copied().unwrap_or(0);
553        let w1 = window.get(1).copied().unwrap_or(0);
554        let w2 = window.get(2).copied().unwrap_or(0);
555        let w3 = window.get(3).copied().unwrap_or(0);
556        w0.wrapping_add(1) == w1 && w1.wrapping_add(1) == w2 && w2.wrapping_add(1) == w3
557    });
558
559    if has_structure || entropy < 6.0 { PatternType::Structured } else { PatternType::Binary }
560}

612/// Runtime performance metrics for adaptive scheme selection.
613///
614/// Implementation: `CryptoPolicyEngine::adaptive_selection()` — thresholds on `memory_usage_mb` and `encryption_speed_ms`
615#[derive(Debug, Clone)]
616pub struct PerformanceMetrics {
617    /// Average encryption time in milliseconds.
618    /// Consumer: `CryptoPolicyEngine::adaptive_selection()`
619    pub encryption_speed_ms: f64,
620    /// Current memory usage in megabytes.
621    /// Consumer: `CryptoPolicyEngine::adaptive_selection()`
622    pub memory_usage_mb: f64,
623}
624
625impl Default for PerformanceMetrics {
626    fn default() -> Self {
627        Self { encryption_speed_ms: 100.0, memory_usage_mb: 100.0 }
628    }
