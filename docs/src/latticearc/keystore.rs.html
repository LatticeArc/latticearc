1//! Persistent key storage with password-protected key wrapping.
2//!
3//! Keys are encrypted at rest using AES-256-GCM with a key encryption key (KEK)
4//! derived from a master password via PBKDF2-HMAC-SHA256 (600K iterations).
5//! Each key's associated data (AAD) is the SHA-256 hash of its metadata,
6//! preventing key/metadata substitution attacks.
7//!
8//! ## File layout
9//!
10//! ```text
11//! ~/.latticearc/keys/
12//!   keystore.json          # Index + wrapped keys
13//! ```
14//!
15//! ## Security properties
16//!
17//! - KEK derived from password with PBKDF2 (600K iterations, 16-byte random salt)
18//! - Each key wrapped individually with AES-256-GCM + per-key random nonce
19//! - AAD = SHA-256(key_id || algorithm || key_type || created) — binds ciphertext to metadata
20//! - Password verification via encrypted sentinel (no password hash stored)
21//! - All secret material zeroized on drop
22//! - File permissions restricted to owner (0600 on Unix)
23

36/// PBKDF2 iteration count (OWASP 2023 recommendation for HMAC-SHA256).
37const PBKDF2_ITERATIONS: u32 = 600_000;
38
39/// PBKDF2 salt length in bytes.
40const PBKDF2_SALT_LEN: usize = 16;
41
42/// KEK length in bytes (AES-256).
43const KEK_LEN: usize = 32;
44
45/// Sentinel plaintext for password verification.
46/// Encrypted with the KEK on creation; decrypted on open to verify password.
47const SENTINEL_PLAINTEXT: &[u8] = b"latticearc-keystore-v1-sentinel";
48
49/// Sentinel AAD — domain-separated from key wrapping AAD.
50const SENTINEL_AAD: &[u8] = b"latticearc-keystore-sentinel-aad";
51
52/// Default keystore directory name under home.
53const KEYSTORE_DIR: &str = ".latticearc/keys";
54
55/// Keystore index filename.
56const KEYSTORE_FILE: &str = "keystore.json";

62/// Key type stored in the keystore.
63#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
64#[serde(rename_all = "snake_case")]
65pub(crate) enum StoredKeyType {
66    /// Symmetric key (e.g., AES-256).
67    Symmetric,
68    /// Asymmetric keypair (public + secret stored together).
69    Keypair,
70    /// Public key only (exported from a keypair).
71    Public,
72}
73
74/// Persisted keystore file format (JSON on disk).
75#[derive(Serialize, Deserialize)]
76pub(crate) struct KeyStoreFile {
77    /// Format version.
78    pub version: u8,
79    /// ISO 8601 creation timestamp.
80    pub created: String,
81    /// Base64-encoded PBKDF2 salt.
82    pub salt: String,
83    /// PBKDF2 iteration count.
84    pub iterations: u32,
85    /// Base64-encoded encrypted sentinel (password verification).
86    pub sentinel: String,
87    /// Wrapped key entries keyed by user-chosen label.
88    pub entries: BTreeMap<String, WrappedEntry>,
89}

91/// A single wrapped key entry on disk.
92#[derive(Serialize, Deserialize, Clone)]
93pub(crate) struct WrappedEntry {
94    /// Unique key identifier (hex-encoded random bytes).
95    pub id: String,
96    /// Algorithm identifier (e.g., "ml-kem-768", "aes-256").
97    pub algorithm: String,
98    /// Key type.
99    pub key_type: StoredKeyType,
100    /// ISO 8601 creation timestamp.
101    pub created: String,
102    /// Base64-encoded AES-256-GCM wrapped key material (nonce || ciphertext || tag).
103    pub wrapped_key: String,
104    /// Base64-encoded public key (only for keypair entries).
105    #[serde(skip_serializing_if = "Option::is_none")]
106    pub public_key: Option<String>,
107    /// Number of times this key has been rotated.
108    pub rotation_count: u32,
109    /// ISO 8601 timestamp of last rotation (None if never rotated).
110    #[serde(skip_serializing_if = "Option::is_none")]
111    pub last_rotated: Option<String>,
112}

114impl std::fmt::Debug for WrappedEntry {
115    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
116        f.debug_struct("WrappedEntry")
117            .field("id", &self.id)
118            .field("algorithm", &self.algorithm)
119            .field("key_type", &self.key_type)
120            .field("created", &self.created)
121            .field("wrapped_key", &"[ENCRYPTED]")
122            .field("public_key", &self.public_key.as_ref().map(|_| "[PRESENT]"))
123            .field("rotation_count", &self.rotation_count)
124            .field("last_rotated", &self.last_rotated)
125            .finish()
126    }
127}

129/// In-memory keystore with unlocked KEK.
130///
131/// The KEK is held in `Zeroizing<Vec<u8>>` and is wiped when the `KeyStore`
132/// is dropped. All key material loaded from entries is also zeroized.
133pub(crate) struct KeyStore {
134    /// Path to the keystore.json file.
135    path: PathBuf,
136    /// Deserialized file contents.
137    file: KeyStoreFile,
138    /// Derived key encryption key (zeroized on drop).
139    kek: Zeroizing<Vec<u8>>,
140}
141
142impl std::fmt::Debug for KeyStore {
143    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
144        f.debug_struct("KeyStore")
145            .field("path", &self.path)
146            .field("entries", &self.file.entries.len())
147            .field("kek", &"[REDACTED]")
148            .finish()
149    }
150}

157    /// Create a new keystore protected by the given master password.
158    ///
159    /// Creates the directory structure and writes the initial keystore.json.
160    pub fn create(password: &[u8], store_dir: Option<&Path>) -> Result<Self> {
161        let dir = match store_dir {
162            Some(d) => d.to_path_buf(),
163            None => default_keystore_dir()?,
164        };
165
166        std::fs::create_dir_all(&dir)
167            .with_context(|| format!("Failed to create keystore directory: {}", dir.display()))?;
168
169        let path = dir.join(KEYSTORE_FILE);
170        if path.exists() {
171            bail!("Keystore already exists at {}", path.display());
172        }
173
174        // Derive KEK from password
175        let params = latticearc::primitives::kdf::pbkdf2::Pbkdf2Params::new(PBKDF2_SALT_LEN)?
176            .iterations(PBKDF2_ITERATIONS)
177            .key_length(KEK_LEN);
178
179        let kek_result = latticearc::primitives::kdf::pbkdf2::pbkdf2(password, &params)
180            .map_err(|e| anyhow::anyhow!("PBKDF2 key derivation failed: {e}"))?;
181
182        let kek = Zeroizing::new(kek_result.key().to_vec());
183
184        // Encrypt sentinel for password verification
185        let sentinel_encrypted =
186            latticearc::encrypt_aes_gcm_with_aad_unverified(SENTINEL_PLAINTEXT, &kek, SENTINEL_AAD)
187                .map_err(|e| anyhow::anyhow!("Sentinel encryption failed: {e}"))?;
188
189        let file = KeyStoreFile {
190            version: 1,
191            created: chrono::Utc::now().to_rfc3339(),
192            salt: B64.encode(&params.salt),
193            iterations: params.iterations,
194            sentinel: B64.encode(&sentinel_encrypted),
195            entries: BTreeMap::new(),
196        };
197
198        let store = Self { path, file, kek };
199        store.save()?;
200
201        Ok(store)
202    }

204    /// Open an existing keystore with the given master password.
205    pub fn open(password: &[u8], store_dir: Option<&Path>) -> Result<Self> {
206        let dir = match store_dir {
207            Some(d) => d.to_path_buf(),
208            None => default_keystore_dir()?,
209        };
210
211        let path = dir.join(KEYSTORE_FILE);
212        if !path.exists() {
213            bail!("No keystore found at {}", path.display());
214        }
215
216        let data = std::fs::read_to_string(&path)
217            .with_context(|| format!("Failed to read keystore: {}", path.display()))?;
218        let file: KeyStoreFile = serde_json::from_str(&data).context("Invalid keystore format")?;
219
220        if file.version != 1 {
221            bail!("Unsupported keystore version: {}", file.version);
222        }
223
224        // Derive KEK from password using stored salt + iterations
225        let salt = B64.decode(&file.salt).context("Invalid salt encoding")?;
226        let params = latticearc::primitives::kdf::pbkdf2::Pbkdf2Params::with_salt(&salt)
227            .iterations(file.iterations)
228            .key_length(KEK_LEN);
229
230        let kek_result = latticearc::primitives::kdf::pbkdf2::pbkdf2(password, &params)
231            .map_err(|e| anyhow::anyhow!("PBKDF2 key derivation failed: {e}"))?;
232
233        let kek = Zeroizing::new(kek_result.key().to_vec());
234
235        // Verify password by decrypting sentinel
236        let sentinel_bytes = B64.decode(&file.sentinel).context("Invalid sentinel encoding")?;
237        let decrypted =
238            latticearc::decrypt_aes_gcm_with_aad_unverified(&sentinel_bytes, &kek, SENTINEL_AAD)
239                .map_err(|_err| anyhow::anyhow!("Wrong password or corrupted keystore"))?;
240
241        if decrypted != SENTINEL_PLAINTEXT {
242            bail!("Sentinel mismatch — keystore may be corrupted");
243        }
244
245        Ok(Self { path, file, kek })
246    }

248    /// Store a key in the keystore.
249    ///
250    /// For asymmetric keys, pass the secret key as `key_bytes` and optionally
251    /// provide the public key via `public_key_bytes`.
252    pub fn store(
253        &mut self,
254        label: &str,
255        algorithm: &str,
256        key_type: StoredKeyType,
257        key_bytes: &[u8],
258        public_key_bytes: Option<&[u8]>,
259    ) -> Result<String> {
260        if self.file.entries.contains_key(label) {
261            bail!("Key with label '{label}' already exists. Use rotate or delete first.");
262        }
263
264        let id = generate_key_id();
265        let created = chrono::Utc::now().to_rfc3339();
266
267        // Compute AAD = SHA-256(id || algorithm || key_type || created)
268        let aad = compute_entry_aad(&id, algorithm, &key_type, &created);
269
270        // Wrap key with KEK + AAD
271        let wrapped = latticearc::encrypt_aes_gcm_with_aad_unverified(key_bytes, &self.kek, &aad)
272            .map_err(|e| anyhow::anyhow!("Key wrapping failed: {e}"))?;
273
274        let entry = WrappedEntry {
275            id: id.clone(),
276            algorithm: algorithm.to_string(),
277            key_type,
278            created,
279            wrapped_key: B64.encode(&wrapped),
280            public_key: public_key_bytes.map(|pk| B64.encode(pk)),
281            rotation_count: 0,
282            last_rotated: None,
283        };
284
285        self.file.entries.insert(label.to_string(), entry);
286        self.save()?;
287
288        Ok(id)
289    }

291    /// Load (unwrap) a key from the keystore by label.
292    ///
293    /// Returns the decrypted key bytes wrapped in `Zeroizing` for automatic cleanup.
294    pub fn load(&self, label: &str) -> Result<Zeroizing<Vec<u8>>> {
295        let entry = self
296            .file
297            .entries
298            .get(label)
299            .with_context(|| format!("No key with label '{label}' in keystore"))?;
300
301        let wrapped = B64.decode(&entry.wrapped_key).context("Invalid wrapped key encoding")?;
302        let aad = compute_entry_aad(&entry.id, &entry.algorithm, &entry.key_type, &entry.created);
303
304        let plaintext = latticearc::decrypt_aes_gcm_with_aad_unverified(&wrapped, &self.kek, &aad)
305            .map_err(|_err| {
306                anyhow::anyhow!("Key unwrap failed for '{label}' — keystore may be corrupted")
307            })?;
308
309        Ok(Zeroizing::new(plaintext))
310    }

317    /// Rotate a key: re-wrap new key material under the same label.
318    ///
319    /// Increments the rotation counter and updates the last_rotated timestamp.
320    pub fn rotate(
321        &mut self,
322        label: &str,
323        new_key_bytes: &[u8],
324        new_public_key_bytes: Option<&[u8]>,
325    ) -> Result<()> {
326        let entry = self
327            .file
328            .entries
329            .get(label)
330            .with_context(|| format!("No key with label '{label}' to rotate"))?;
331
332        // Preserve metadata but generate new wrapping
333        let id = generate_key_id();
334        let created = chrono::Utc::now().to_rfc3339();
335        let algorithm = entry.algorithm.clone();
336        let key_type = entry.key_type.clone();
337        let rotation_count = entry.rotation_count.saturating_add(1);
338
339        let aad = compute_entry_aad(&id, &algorithm, &key_type, &created);
340        let wrapped =
341            latticearc::encrypt_aes_gcm_with_aad_unverified(new_key_bytes, &self.kek, &aad)
342                .map_err(|e| anyhow::anyhow!("Key wrapping failed during rotation: {e}"))?;
343
344        let new_entry = WrappedEntry {
345            id,
346            algorithm,
347            key_type,
348            created,
349            wrapped_key: B64.encode(&wrapped),
350            public_key: new_public_key_bytes.map(|pk| B64.encode(pk)),
351            rotation_count,
352            last_rotated: Some(chrono::Utc::now().to_rfc3339()),
353        };
354
355        self.file.entries.insert(label.to_string(), new_entry);
356        self.save()?;
357
358        Ok(())
359    }

361    /// Delete a key from the keystore.
362    pub fn delete(&mut self, label: &str) -> Result<()> {
363        if self.file.entries.remove(label).is_none() {
364            bail!("No key with label '{label}' in keystore");
365        }
366        self.save()
367    }

370    ///
371    /// Returns the raw public key bytes, or an error if the entry has no public key.
372    pub fn export_public(&self, label: &str) -> Result<Vec<u8>> {

374            .file
375            .entries
376            .get(label)
377            .with_context(|| format!("No key with label '{label}' in keystore"))?;
378
379        let pk_b64 = entry
380            .public_key
381            .as_ref()
382            .with_context(|| format!("Key '{label}' has no public key (symmetric key?)"))?;
383
384        B64.decode(pk_b64).context("Invalid public key encoding")
385    }

408    /// Write the keystore to disk with restricted permissions.
409    fn save(&self) -> Result<()> {
410        let json =
411            serde_json::to_string_pretty(&self.file).context("Failed to serialize keystore")?;
412        std::fs::write(&self.path, &json)
413            .with_context(|| format!("Failed to write keystore: {}", self.path.display()))?;
414
415        restrict_file_permissions(&self.path)?;
416        Ok(())
417    }

424/// Compute the AAD for a key entry: SHA-256(id || algorithm || key_type || created).
425fn compute_entry_aad(
426    id: &str,
427    algorithm: &str,
428    key_type: &StoredKeyType,
429    created: &str,
430) -> Vec<u8> {
431    let key_type_str = match key_type {
432        StoredKeyType::Symmetric => "symmetric",
433        StoredKeyType::Keypair => "keypair",
434        StoredKeyType::Public => "public",
435    };
436
437    let mut hasher = Sha256::new();
438    hasher.update(id.as_bytes());
439    hasher.update(b"|");
440    hasher.update(algorithm.as_bytes());
441    hasher.update(b"|");
442    hasher.update(key_type_str.as_bytes());
443    hasher.update(b"|");
444    hasher.update(created.as_bytes());
445    hasher.finalize().to_vec()
446}

448/// Generate a random key ID (32 hex chars = 16 random bytes).
449fn generate_key_id() -> String {
450    let mut bytes = [0u8; 16];
451    use rand::RngCore;
452    rand::rngs::OsRng.fill_bytes(&mut bytes);
453    hex::encode(bytes)
454}

456/// Resolve the default keystore directory (~/.latticearc/keys/).
457fn default_keystore_dir() -> Result<PathBuf> {
458    let home = dirs_or_home()?;
459    Ok(home.join(KEYSTORE_DIR))
460}
461
462/// Get the user's home directory.
463fn dirs_or_home() -> Result<PathBuf> {
464    std::env::var("HOME")
465        .map(PathBuf::from)
466        .or_else(|_| std::env::var("USERPROFILE").map(PathBuf::from))
467        .context("Cannot determine home directory (HOME or USERPROFILE not set)")
468}

470/// Set owner-only read/write permissions on the keystore file (Unix).
471#[cfg(unix)]
472fn restrict_file_permissions(path: &Path) -> Result<()> {
473    use std::os::unix::fs::PermissionsExt;
474    std::fs::set_permissions(path, std::fs::Permissions::from_mode(0o600))
475        .with_context(|| format!("Failed to restrict permissions on {}", path.display()))
476}
477
478/// No-op on non-Unix platforms.
479#[cfg(not(unix))]
480fn restrict_file_permissions(_path: &Path) -> Result<()> {
481    Ok(())
482}

719    fn test_multiple_keys_independent() {
720        let dir = temp_keystore_dir();
721        let password = b"multi-test";
722        let mut store = KeyStore::create(password, Some(dir.path())).unwrap();
723
724        let key_a = [0xAAu8; 32];
725        let key_b = [0xBBu8; 32];
726        let key_c = [0xCCu8; 32];
727
728        store.store("key-a", "aes-256", StoredKeyType::Symmetric, &key_a, None).unwrap();
729        store.store("key-b", "aes-256", StoredKeyType::Symmetric, &key_b, None).unwrap();
730        store.store("key-c", "aes-256", StoredKeyType::Symmetric, &key_c, None).unwrap();
731
732        assert_eq!(store.load("key-a").unwrap().as_slice(), &key_a);
733        assert_eq!(store.load("key-b").unwrap().as_slice(), &key_b);
734        assert_eq!(store.load("key-c").unwrap().as_slice(), &key_c);
735
736        // Delete middle key, others still accessible
737        store.delete("key-b").unwrap();
738        assert_eq!(store.len(), 2);
739        assert_eq!(store.load("key-a").unwrap().as_slice(), &key_a);
740        assert_eq!(store.load("key-c").unwrap().as_slice(), &key_c);
