6//! Session management for TLS session resumption
7//!
8//! This module provides session storage configuration for faster TLS reconnections.
9//!
10//! ## Performance Benefits
11//!
12//! TLS session resumption significantly reduces handshake latency:
13//! - Full handshake: ~2 round trips + cryptographic operations
14//! - Resumed session: ~1 round trip, minimal crypto
15//!
16//! For hybrid PQ-TLS, the savings are even more significant due to
17//! the larger key sizes involved in ML-KEM key exchange.
18//!
19//! ## Session Storage Options
20//!
21//! - **In-memory cache**: Fast, volatile (lost on restart)
22//! - **Persistent storage**: Reserved for future use when rustls supports serialization
23//!
24//! ## Security Considerations
25//!
26//! - Session tickets are encrypted by the server
27//! - Sessions automatically expire based on server policy
28//! - In-memory storage is cleared on process exit
29

33/// Enhanced session store with configurable capacity
34///
35/// Wraps rustls's in-memory session cache with additional configuration options.
36#[derive(Debug)]
37pub struct ConfigurableSessionStore {
38    /// Maximum number of sessions to cache
39    max_sessions: usize,
40    /// The underlying session store
41    inner: Arc<dyn ClientSessionStore>,
42}
43
44impl ConfigurableSessionStore {
45    /// Create a new configurable session store
46    ///
47    /// # Arguments
48    /// * `max_sessions` - Maximum number of sessions to cache
49    ///
50    /// # Returns
51    /// A session store with the specified capacity
52    ///
53    /// # Example
54    /// ```
55    /// use latticearc::tls::session_store::ConfigurableSessionStore;
56    ///
57    /// let store = ConfigurableSessionStore::new(100);
58    /// ```
59    #[must_use]
60    pub fn new(max_sessions: usize) -> Self {
61        Self {
62            max_sessions,
63            inner: Arc::new(rustls::client::ClientSessionMemoryCache::new(max_sessions)),
64        }
65    }
66
67    /// Get the maximum number of sessions this store can hold
68    #[must_use]
69    pub fn capacity(&self) -> usize {
70        self.max_sessions
71    }
72
73    /// Get the underlying session store for use with rustls
74    #[must_use]
75    pub fn as_store(&self) -> Arc<dyn ClientSessionStore> {
76        self.inner.clone()
77    }
78}

150/// Create a session store based on configuration
151///
152/// Returns either a persistent store (when supported) or an in-memory store.
153///
154/// # Arguments
155/// * `persistence` - Optional persistence configuration
156///
157/// # Returns
158/// An Arc-wrapped session store suitable for use with rustls
159///
160/// # Example
161/// ```
162/// use latticearc::tls::session_store::create_session_store;
163/// use latticearc::tls::SessionPersistenceConfig;
164///
165/// // In-memory store (default)
166/// let store = create_session_store(None);
167///
168/// // Configured store (in-memory for now, path reserved for future use)
169/// let config = SessionPersistenceConfig::new(std::env::temp_dir().join("sessions.bin"), 500);
170/// let store = create_session_store(Some(&config));
171/// ```
172#[must_use]
173pub fn create_session_store(
174    persistence: Option<&crate::tls::SessionPersistenceConfig>,
175) -> Arc<dyn ClientSessionStore> {
176    if let Some(config) = persistence {
177        Arc::new(rustls::client::ClientSessionMemoryCache::new(config.max_sessions))
178    } else {
179        // Default: 32 sessions in memory
180        Arc::new(rustls::client::ClientSessionMemoryCache::new(32))
181    }
182}

184/// Create a session resumption configuration for rustls client
185///
186/// # Arguments
187/// * `persistence` - Optional persistence configuration
188///
189/// # Returns
190/// A rustls Resumption configuration
191#[must_use]
192pub fn create_resumption_config(
193    persistence: Option<&crate::tls::SessionPersistenceConfig>,
194) -> rustls::client::Resumption {
195    let store = create_session_store(persistence);
196    rustls::client::Resumption::store(store)
197}
