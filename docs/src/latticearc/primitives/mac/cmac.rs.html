10//! CMAC (Cipher-based Message Authentication Code)
11//!
12//! This module provides CMAC (AES-CMAC) implementation as specified in
13//! NIST SP 800-38B: Recommendation for Block Cipher Modes of Operation:
14//! The CMAC Mode for Authentication.
15//!
16//! CMAC provides data integrity and authenticity verification for AES-encrypted data.
17//! This implementation follows the full NIST SP 800-38B specification including:
18//! - Subkey generation (K1, K2) from encryption key K
19//! - Proper padding and block processing
20//! - Constant-time operations to prevent timing attacks

89/// Constant-time XOR of two byte arrays
90#[inline(always)]
91fn xor_block(a: &mut [u8; 16], b: &[u8; 16]) {
92    // Perform constant-time XOR operation using iterator pattern
93    for (a_byte, b_byte) in a.iter_mut().zip(b.iter()) {
94        *a_byte ^= b_byte;
95    }
96}

98/// Left shift a 128-bit block by 1 bit (constant-time)
99///
100/// Returns the MSB that was shifted out (0 or 1)
101///
102/// Bitwise shifts in Rust are safe and cannot overflow - they are defined
103/// to shift in zeros and the result fits in the same integer type.
104#[inline(always)]
105#[allow(clippy::arithmetic_side_effects)]
106fn left_shift_block(block: &[u8; 16]) -> ([u8; 16], u8) {
107    let mut result = [0u8; 16];
108    let mut overflow = 0u8;
109
110    // Process bytes with carry propagation for left shift (reverse iteration)
111    for (i, &current_byte) in block.iter().enumerate().rev() {
112        let new_byte = (current_byte << 1) | overflow;
113        if let Some(r) = result.get_mut(i) {
114            *r = new_byte;
115        }
116        overflow = (current_byte >> 7) & 1;
117    }
118
119    (result, overflow)
120}

122/// Generate CMAC subkeys K1 and K2 from the encryption key
123///
124/// # Algorithm (NIST SP 800-38B Section 5.3)
125/// 1. L = AES_K(0^128) - encrypt a zero block
126/// 2. K1 = (L << 1) XOR R_b (where R_b = 0x87 if MSB(L) = 1)
127/// 3. K2 = (K1 << 1) XOR R_b (where R_b = 0x87 if MSB(K1) = 1)
128///
129/// # Arguments
130/// * `key` - The encryption key (16, 24, or 32 bytes for AES-128/192/256)
131///
132/// # Returns
133/// CmacSubkeys containing K1 and K2
134///
135/// # Errors
136/// Returns InvalidKeyLength if key length is not 16, 24, or 32 bytes
137fn generate_subkeys(key: &[u8]) -> Result<CmacSubkeys, CmacError> {
138    // Validate key length
139    match key.len() {
140        16 | 24 | 32 => {}
141        _ => return Err(CmacError::InvalidKeyLength { actual: key.len() }),
142    }
143
144    // Step 1: Compute L = AES_K(0^128)
145    let l_block = match key.len() {
146        16 => {
147            let cipher = Aes128::new_from_slice(key)
148                .map_err(|e| CmacError::ComputationError(e.to_string()))?;
149            let mut block = [0u8; 16];
150            cipher.encrypt_block((&mut block).into());
151            block
152        }
153        24 => {
154            // AES-192 requires 24-byte keys
155            let cipher = Aes192::new_from_slice(key)
156                .map_err(|e| CmacError::ComputationError(e.to_string()))?;
157            let mut block = [0u8; 16];
158            cipher.encrypt_block((&mut block).into());
159            block
160        }
161        32 => {
162            let cipher = Aes256::new_from_slice(key)
163                .map_err(|e| CmacError::ComputationError(e.to_string()))?;
164            let mut block = [0u8; 16];
165            cipher.encrypt_block((&mut block).into());
166            block
167        }
168        _ => return Err(CmacError::InvalidKeyLength { actual: key.len() }),
169    };
170
171    // RB constant for 128-bit blocks = 0x87
172    const RB: [u8; 16] = [
173        0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
174        0x87,
175    ];
176
177    // Step 2: Derive K1
178    let (k1_shifted, k1_msb) = left_shift_block(&l_block);
179    let mut k1 = k1_shifted;
180
181    // If MSB(L) was 1, XOR with RB
182    if k1_msb == 1 {
183        xor_block(&mut k1, &RB);
184    }
185
186    // Step 3: Derive K2
187    let (k2_shifted, k2_msb) = left_shift_block(&k1);
188    let mut k2 = k2_shifted;
189
190    // If MSB(K1) was 1, XOR with RB
191    if k2_msb == 1 {
192        xor_block(&mut k2, &RB);
193    }
194
195    Ok(CmacSubkeys { k1, k2 })
196}

198/// Apply padding and compute CMAC tag
199///
200/// # Algorithm (NIST SP 800-38B Section 6)
201/// 1. If message length is a multiple of block size:
202///    - Pad last block: M_n XOR K1, then CBC-MAC
203/// 2. If message length is NOT a multiple of block size:
204///    - Pad incomplete block with 10...0, then XOR with K2, then CBC-MAC
205///
206/// # Arguments
207/// * `key` - The encryption key (16, 24, or 32 bytes)
208/// * `data` - The message to authenticate
209///
210/// # Returns
211/// The 16-byte CMAC tag
212fn compute_cmac_internal(key: &[u8], data: &[u8]) -> Result<[u8; 16], CmacError> {
213    // Validate key length
214    match key.len() {
215        16 | 24 | 32 => {}
216        _ => return Err(CmacError::InvalidKeyLength { actual: key.len() }),
217    }
218
219    // Generate subkeys K1 and K2
220    let subkeys = generate_subkeys(key)?;
221
222    // Initialize cipher for CBC-MAC
223    let cipher = match key.len() {
224        16 => CipherType::Aes128(
225            Aes128::new_from_slice(key).map_err(|e| CmacError::ComputationError(e.to_string()))?,
226        ),
227        24 => CipherType::Aes192(
228            Aes192::new_from_slice(key).map_err(|e| CmacError::ComputationError(e.to_string()))?,
229        ),
230        32 => CipherType::Aes256(
231            Aes256::new_from_slice(key).map_err(|e| CmacError::ComputationError(e.to_string()))?,
232        ),
233        _ => return Err(CmacError::InvalidKeyLength { actual: key.len() }),
234    };
235
236    let data_len = data.len();
237    let num_complete_blocks = data_len / 16;
238    let incomplete_block_size = data_len % 16;
239
240    // CBC-MAC state: C_0 = 0^128
241    let mut c_i = [0u8; 16];
242
243    // Determine how many blocks to process as regular CBC blocks
244    // All blocks except the last one are processed normally
245    let total_blocks = if data_len == 0 { 1 } else { data_len.div_ceil(16) };
246
247    if total_blocks > 1 {
248        for i in 0..(total_blocks - 1) {
249            let mut block = [0u8; 16];
250            let start = i * 16;
251            let end = (i + 1) * 16;
252            let data_slice = data
253                .get(start..end)
254                .ok_or_else(|| CmacError::ComputationError(format!("Block {} out of bounds", i)))?;
255            block.copy_from_slice(data_slice);
256            xor_block(&mut block, &c_i);
257
258            match &cipher {
259                CipherType::Aes128(c) => c.encrypt_block((&mut block).into()),
260                CipherType::Aes192(c) => c.encrypt_block((&mut block).into()),
261                CipherType::Aes256(c) => c.encrypt_block((&mut block).into()),
262            }
263
264            c_i = block;
265        }
266    }
267
268    // Process final block with padding
269    let mut final_block = [0u8; 16];
270
271    if data_len == 0 {
272        // Empty message: pad with 10...0 and XOR with K2
273        final_block[0] = 0x80;
274        xor_block(&mut final_block, &subkeys.k2);
275    } else if incomplete_block_size == 0 {
276        // Message is multiple of block size
277        // Last block is the nth block, XOR with K1
278        let block_idx = num_complete_blocks - 1;
279        let start = block_idx * 16;
280        let end = (block_idx + 1) * 16;
281        let data_slice = data
282            .get(start..end)
283            .ok_or_else(|| CmacError::ComputationError("Final block out of bounds".to_string()))?;
284        final_block.copy_from_slice(data_slice);
285        xor_block(&mut final_block, &c_i);
286        xor_block(&mut final_block, &subkeys.k1);
287    } else {
288        // Message has incomplete final block
289        // Pad with 10...0 and XOR with K2
290        let start = num_complete_blocks * 16;
291        let incomplete_data = data.get(start..).ok_or_else(|| {
292            CmacError::ComputationError("Incomplete block out of bounds".to_string())
293        })?;
294        let dest_slice = final_block.get_mut(..incomplete_block_size).ok_or_else(|| {
295            CmacError::ComputationError("Final block slice out of bounds".to_string())
296        })?;
297        dest_slice.copy_from_slice(incomplete_data);
298        // Safe: incomplete_block_size = data_len % 16, always < 16
299        if let Some(pad_byte) = final_block.get_mut(incomplete_block_size) {
300            *pad_byte = 0x80; // Padding with 1 bit followed by zeros
301        }
302        xor_block(&mut final_block, &c_i);
303        xor_block(&mut final_block, &subkeys.k2);
304    }
305
306    // Encrypt final block
307    match &cipher {
308        CipherType::Aes128(c) => c.encrypt_block((&mut final_block).into()),
309        CipherType::Aes192(c) => c.encrypt_block((&mut final_block).into()),
310        CipherType::Aes256(c) => c.encrypt_block((&mut final_block).into()),
311    }
312
313    Ok(final_block)
314}

323/// Compute AES-128-CMAC for given data
324///
325/// # NIST SP 800-38B Specification
326/// - Key length: 128 bits (16 bytes)
327/// - Tag length: 128 bits (16 bytes)
328/// - Block size: 128 bits (16 bytes)
329///
330/// # Security Requirements
331/// - The key must be cryptographically secure
332/// - Use fresh keys for each context (never reuse keys across applications)
333/// - This implementation uses constant-time operations to prevent timing attacks
334///
335/// # Example
336/// ```no_run
337/// use latticearc::primitives::mac::cmac::{cmac_128, CmacError};
338///
339/// let key = [0u8; 16]; // 128-bit key
340/// let data = b"message to authenticate";
341///
342/// let result = cmac_128(&key, data);
343/// assert!(result.is_ok());
344/// ```
345///
346/// # Errors
347/// Returns an error if the key length is not exactly 16 bytes.
348pub fn cmac_128(key: &[u8], data: &[u8]) -> Result<Cmac128, CmacError> {
349    // Input validation - key must be exactly 16 bytes
350    if key.len() != 16 {
351        return Err(CmacError::InvalidKeyLength { actual: key.len() });
352    }
353
354    // Compute CMAC tag using AES-128
355    let tag = compute_cmac_internal(key, data)?;
356
357    Ok(Cmac128 { tag })
358}

434/// Verify CMAC-128 tag using constant-time comparison
435///
436/// This function computes the CMAC tag for the given data and compares it
437/// with the provided tag in constant-time to prevent timing attacks.
438///
439/// # Security Notice
440/// Always use constant-time comparison for tag verification to prevent timing attacks.
441///
442/// # Arguments
443/// * `key` - The AES-128 key (16 bytes)
444/// * `data` - The message to verify
445/// * `tag` - The CMAC tag to verify against (16 bytes)
446///
447/// # Returns
448/// `true` if the tag is valid, `false` otherwise
449///
450/// # Example
451/// ```no_run
452/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
453/// use latticearc::primitives::mac::cmac::{cmac_128, verify_cmac_128};
454///
455/// let key = [0u8; 16];
456/// let data = b"message to authenticate";
457///
458/// let cmac = cmac_128(&key, data)?;
459/// let is_valid = verify_cmac_128(&key, data, cmac.tag());
460/// assert!(is_valid);
461/// # Ok(())
462/// # }
463/// ```
464#[must_use]
465pub fn verify_cmac_128(key: &[u8], data: &[u8], tag: &[u8]) -> bool {
466    use subtle::ConstantTimeEq;
467
468    // Always compute CMAC to prevent timing side-channels
469    let tag_valid: bool = tag.len().ct_eq(&16).into();
470
471    // Compute CMAC regardless of tag length validation (timing-safe)
472    let expected_tag_result = cmac_128(key, data);
473
474    // Constant-time comparison: valid only if tag length valid AND CMAC computation succeeded AND tags match
475    match expected_tag_result {
476        Ok(cmac) => {
477            let tags_match = cmac.tag.ct_eq(tag).into();
478            tag_valid && tags_match
479        }
480        Err(_) => false,
481    }
482}
