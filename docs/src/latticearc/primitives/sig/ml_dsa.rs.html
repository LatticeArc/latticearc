6//! # ML-DSA (FIPS 204) Digital Signatures
7//!
8//! ## FIPS 140-3 Certification Notice
9//!
10//! **Current Implementation**: Uses the `fips204` crate (pure Rust, NOT independently audited)
11//!
12//! **For FIPS 140-3 certification**, this module needs to migrate to `aws-lc-rs`
13//! once its ML-DSA Rust API is stabilized. The current `fips204` crate is not
14//! independently FIPS-validated.
15//!
16//! ## Usage for Non-FIPS Applications
17//!
18//! The current implementation is functionally correct and suitable for:
19//! - Development and testing
20//! - Non-regulated applications
21//! - Applications not requiring FIPS 140-3 certification
22//!
23//! ## FIPS 204 Standard
24//!
25//! FIPS 204 specifies the Module-Lattice-Based Digital Signature Algorithm (ML-DSA),
26//! which provides post-quantum security for digital signatures.
27//!
28//! ## Security Level
29//!
30//! ML-DSA provides EUF-CMA (Existential Unforgeability under Chosen Message Attacks)
31//! security and is believed to be secure against quantum adversaries.
32//!
33//! ## Parameter Sets
34//!
35//! | Parameter Set | Public Key | Signature | NIST Level |
36//! |---------------|------------|-----------|------------|
37//! | ML-DSA-44     | ~1.3 KB    | ~2.4 KB   | 2          |
38//! | ML-DSA-65     | ~2.0 KB    | ~3.3 KB   | 3          |
39//! | ML-DSA-87     | ~2.6 KB    | ~4.6 KB   | 5          |
40//!
41//! ## Backend Selection (Future)
42//!
43//! When aws-lc-rs ML-DSA support is available, the backend will be selectable via
44//! feature flags:
45//!
46//! ```toml
47//! # Future configuration (not yet available)
48//! [dependencies]
49//! arc-primitives = { version = "0.1", features = ["ml-dsa-aws-lc"] }
50//! ```

61/// ML-DSA parameter sets for different security levels
62#[derive(Debug, Clone, Copy, PartialEq, Eq)]
63#[non_exhaustive]
64pub enum MlDsaParameterSet {
65    /// ML-DSA-44: NIST Level 2 security (~128-bit classical security)
66    MLDSA44,
67    /// ML-DSA-65: NIST Level 3 security (~192-bit classical security)
68    MLDSA65,
69    /// ML-DSA-87: NIST Level 5 security (~256-bit classical security)
70    MLDSA87,
71}
72
73impl MlDsaParameterSet {
74    /// Returns the name of the parameter set
75    #[must_use]
76    pub const fn name(&self) -> &'static str {
77        match self {
78            Self::MLDSA44 => "ML-DSA-44",
79            Self::MLDSA65 => "ML-DSA-65",
80            Self::MLDSA87 => "ML-DSA-87",
81        }
82    }
83
84    /// Returns the public key size in bytes
85    #[must_use]
86    pub const fn public_key_size(&self) -> usize {
87        match self {
88            Self::MLDSA44 => 1312,
89            Self::MLDSA65 => 1952,
90            Self::MLDSA87 => 2592,
91        }
92    }
93
94    /// Returns the secret key size in bytes
95    #[must_use]
96    pub const fn secret_key_size(&self) -> usize {
97        match self {
98            Self::MLDSA44 => 2560,
99            Self::MLDSA65 => 4032,
100            Self::MLDSA87 => 4896,
101        }
102    }
103
104    /// Returns the signature size in bytes
105    #[must_use]
106    pub const fn signature_size(&self) -> usize {
107        match self {
108            Self::MLDSA44 => 2420,
109            Self::MLDSA65 => 3309,
110            Self::MLDSA87 => 4627,
111        }
112    }
113
114    /// Returns the NIST security level
115    #[must_use]
116    pub const fn nist_security_level(&self) -> u8 {
117        match self {
118            Self::MLDSA44 => 2,
119            Self::MLDSA65 => 3,
120            Self::MLDSA87 => 5,
121        }
122    }
123}

180impl MlDsaPublicKey {
181    /// Creates a new ML-DSA public key from raw bytes
182    ///
183    /// # Errors
184    /// Returns an error if the key length does not match the expected size for the parameter set.
185    pub fn new(parameter_set: MlDsaParameterSet, data: Vec<u8>) -> Result<Self, MlDsaError> {
186        let expected_size = parameter_set.public_key_size();
187        if data.len() != expected_size {
188            return Err(MlDsaError::InvalidKeyLength {
189                expected: expected_size,
190                actual: data.len(),
191            });
192        }
193        Ok(Self { parameter_set, data })
194    }

199        self.data.len()
200    }

203    #[must_use]
204    pub fn is_empty(&self) -> bool {
205        self.data.is_empty()
206    }
207
208    /// Serializes the public key to bytes
209    #[must_use]
210    pub fn as_bytes(&self) -> &[u8] {
211        &self.data
212    }
213}

348/// Generate an ML-DSA keypair for the specified parameter set
349///
350/// This function generates a new ML-DSA keypair and performs a FIPS 140-3
351/// Pairwise Consistency Test (PCT) to verify the keypair is valid before
352/// returning it.
353///
354/// # Errors
355/// Returns an error if key generation fails, the ml_dsa feature is not enabled,
356/// or the PCT fails (indicating a corrupted keypair).
357#[instrument(level = "debug", fields(parameter_set = ?parameter_set))]
358pub fn generate_keypair(
359    parameter_set: MlDsaParameterSet,
360) -> Result<(MlDsaPublicKey, MlDsaSecretKey), MlDsaError> {
361    let (pk, sk) = match parameter_set {
362        MlDsaParameterSet::MLDSA44 => {
363            let (pk, sk) = ml_dsa_44::try_keygen().map_err(|e| {
364                MlDsaError::KeyGenerationError(format!("ML-DSA-44 key generation failed: {}", e))
365            })?;
366            (
367                MlDsaPublicKey { parameter_set, data: pk.into_bytes().to_vec() },
368                MlDsaSecretKey { parameter_set, data: sk.into_bytes().to_vec() },
369            )
370        }
371        MlDsaParameterSet::MLDSA65 => {
372            let (pk, sk) = ml_dsa_65::try_keygen().map_err(|e| {
373                MlDsaError::KeyGenerationError(format!("ML-DSA-65 key generation failed: {}", e))
374            })?;
375            (
376                MlDsaPublicKey { parameter_set, data: pk.into_bytes().to_vec() },
377                MlDsaSecretKey { parameter_set, data: sk.into_bytes().to_vec() },
378            )
379        }
380        MlDsaParameterSet::MLDSA87 => {
381            let (pk, sk) = ml_dsa_87::try_keygen().map_err(|e| {
382                MlDsaError::KeyGenerationError(format!("ML-DSA-87 key generation failed: {}", e))
383            })?;
384            (
385                MlDsaPublicKey { parameter_set, data: pk.into_bytes().to_vec() },
386                MlDsaSecretKey { parameter_set, data: sk.into_bytes().to_vec() },
387            )
388        }
389    };
390
391    // FIPS 140-3 Pairwise Consistency Test (PCT)
392    // Sign and verify a test message to ensure the keypair is consistent
393    crate::primitives::pct::pct_ml_dsa(&pk, &sk)
394        .map_err(|e| MlDsaError::KeyGenerationError(format!("PCT failed: {}", e)))?;
395
396    Ok((pk, sk))
397}

399/// Sign a message using ML-DSA
400///
401/// # Errors
402/// Returns an error if signing fails, the key is invalid, or the ml_dsa feature is not enabled.
403#[instrument(level = "debug", skip(secret_key, message, context), fields(parameter_set = ?secret_key.parameter_set, message_len = message.len(), context_len = context.len()))]
404pub fn sign(
405    secret_key: &MlDsaSecretKey,
406    message: &[u8],
407    context: &[u8],
408) -> Result<MlDsaSignature, MlDsaError> {
409    let parameter_set = secret_key.parameter_set();
410
411    let signature = match parameter_set {
412        MlDsaParameterSet::MLDSA44 => {
413            let sk_bytes: [u8; 2560] = secret_key.as_bytes().try_into().map_err(|_e| {
414                MlDsaError::InvalidKeyLength { expected: 2560, actual: secret_key.len() }
415            })?;
416            let sk = ml_dsa_44::PrivateKey::try_from_bytes(sk_bytes).map_err(|e| {
417                MlDsaError::SigningError(format!(
418                    "Failed to deserialize ML-DSA-44 secret key: {}",
419                    e
420                ))
421            })?;
422            let sig = sk.try_sign(message, context).map_err(|e| {
423                MlDsaError::SigningError(format!("ML-DSA-44 signing failed: {}", e))
424            })?;
425            MlDsaSignature { parameter_set, data: sig.to_vec() }
426        }
427        MlDsaParameterSet::MLDSA65 => {
428            let sk_bytes: [u8; 4032] = secret_key.as_bytes().try_into().map_err(|_e| {
429                MlDsaError::InvalidKeyLength { expected: 4032, actual: secret_key.len() }
430            })?;
431            let sk = ml_dsa_65::PrivateKey::try_from_bytes(sk_bytes).map_err(|e| {
432                MlDsaError::SigningError(format!(
433                    "Failed to deserialize ML-DSA-65 secret key: {}",
434                    e
435                ))
436            })?;
437            let sig = sk.try_sign(message, context).map_err(|e| {
438                MlDsaError::SigningError(format!("ML-DSA-65 signing failed: {}", e))
439            })?;
440            MlDsaSignature { parameter_set, data: sig.to_vec() }
441        }
442        MlDsaParameterSet::MLDSA87 => {
443            let sk_bytes: [u8; 4896] = secret_key.as_bytes().try_into().map_err(|_e| {
444                MlDsaError::InvalidKeyLength { expected: 4896, actual: secret_key.len() }
445            })?;
446            let sk = ml_dsa_87::PrivateKey::try_from_bytes(sk_bytes).map_err(|e| {
447                MlDsaError::SigningError(format!(
448                    "Failed to deserialize ML-DSA-87 secret key: {}",
449                    e
450                ))
451            })?;
452            let sig = sk.try_sign(message, context).map_err(|e| {
453                MlDsaError::SigningError(format!("ML-DSA-87 signing failed: {}", e))
454            })?;
455            MlDsaSignature { parameter_set, data: sig.to_vec() }
456        }
457    };
458
459    Ok(signature)
460}

462/// Verify a signature using ML-DSA
463///
464/// # Errors
465/// Returns an error if verification fails due to invalid key or signature format.
466#[instrument(level = "debug", skip(public_key, message, signature, context), fields(parameter_set = ?public_key.parameter_set, message_len = message.len(), signature_len = signature.data.len()))]
467pub fn verify(
468    public_key: &MlDsaPublicKey,
469    message: &[u8],
470    signature: &MlDsaSignature,
471    context: &[u8],
472) -> Result<bool, MlDsaError> {
473    if public_key.parameter_set != signature.parameter_set {
474        return Ok(false);
475    }
476
477    let is_valid = match public_key.parameter_set {
478        MlDsaParameterSet::MLDSA44 => {
479            let pk_bytes: [u8; 1312] = public_key.data.as_slice().try_into().map_err(|_e| {
480                MlDsaError::InvalidKeyLength { expected: 1312, actual: public_key.data.len() }
481            })?;
482            let pk = ml_dsa_44::PublicKey::try_from_bytes(pk_bytes).map_err(|e| {
483                MlDsaError::VerificationError(format!(
484                    "Failed to deserialize ML-DSA-44 public key: {}",
485                    e
486                ))
487            })?;
488            let sig_bytes: [u8; 2420] = signature.data.as_slice().try_into().map_err(|_e| {
489                MlDsaError::InvalidSignatureLength { expected: 2420, actual: signature.data.len() }
490            })?;
491            pk.verify(message, &sig_bytes, context)
492        }
493        MlDsaParameterSet::MLDSA65 => {
494            let pk_bytes: [u8; 1952] = public_key.data.as_slice().try_into().map_err(|_e| {
495                MlDsaError::InvalidKeyLength { expected: 1952, actual: public_key.data.len() }
496            })?;
497            let pk = ml_dsa_65::PublicKey::try_from_bytes(pk_bytes).map_err(|e| {
498                MlDsaError::VerificationError(format!(
499                    "Failed to deserialize ML-DSA-65 public key: {}",
500                    e
501                ))
502            })?;
503            let sig_bytes: [u8; 3309] = signature.data.as_slice().try_into().map_err(|_e| {
504                MlDsaError::InvalidSignatureLength { expected: 3309, actual: signature.data.len() }
505            })?;
506            pk.verify(message, &sig_bytes, context)
507        }
508        MlDsaParameterSet::MLDSA87 => {
509            let pk_bytes: [u8; 2592] = public_key.data.as_slice().try_into().map_err(|_e| {
510                MlDsaError::InvalidKeyLength { expected: 2592, actual: public_key.data.len() }
511            })?;
512            let pk = ml_dsa_87::PublicKey::try_from_bytes(pk_bytes).map_err(|e| {
513                MlDsaError::VerificationError(format!(
514                    "Failed to deserialize ML-DSA-87 public key: {}",
515                    e
516                ))
517            })?;
518            let sig_bytes: [u8; 4627] = signature.data.as_slice().try_into().map_err(|_e| {
519                MlDsaError::InvalidSignatureLength { expected: 4627, actual: signature.data.len() }
520            })?;
521            pk.verify(message, &sig_bytes, context)
522        }
523    };
524
525    Ok(is_valid)
526}
