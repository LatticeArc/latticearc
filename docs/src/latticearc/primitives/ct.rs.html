1//! Constant-time helpers usable across primitive modules.
2//!
3//! Centralises CT predicates that several primitives share (AEAD weak-key
4//! check, KDF salt validation, ...) so each call site cannot accidentally
5//! reach for the early-exiting `iter().all()` form. All functions here use
6//! [`subtle::ConstantTimeEq`] under the hood.
7
8use subtle::ConstantTimeEq;
9
10/// Returns `true` iff every byte of `bytes` is zero, in constant time.
11///
12/// Empty slices return `false` so that a caller passing a zero-length
13/// secret cannot trigger a false-positive weak-input rejection.
14///
15/// # Constant-time behaviour
16///
17/// Runtime is independent of the position of the first non-zero byte.
18/// Internally compares `bytes` against an all-zero stack buffer in
19/// 32-byte chunks via [`subtle::ConstantTimeEq::ct_eq`]; longer inputs
20/// (e.g. PBKDF2 salts greater than 32 bytes) are handled chunk-by-chunk
21/// without falling back to a heap allocation, and without an early exit
22/// on chunk boundaries (each chunk's `Choice` is `&`-folded).
23#[inline]
24#[must_use]
25pub fn is_all_zero_bytes(bytes: &[u8]) -> bool {
26    if bytes.is_empty() {
27        return false;
28    }
29    const CHUNK: usize = 32;
30    let zero = [0u8; CHUNK];
31    let mut acc = subtle::Choice::from(1u8);
32    let mut i = 0usize;
33    while i < bytes.len() {
34        let end = i.saturating_add(CHUNK).min(bytes.len());
35        let len = end.saturating_sub(i);
36        let chunk = bytes.get(i..end).unwrap_or(&[]);
37        let zero_slice = zero.get(..len).unwrap_or(&zero);
38        acc &= chunk.ct_eq(zero_slice);
39        i = end;
40    }
41    bool::from(acc)
42}

44#[cfg(test)]
45mod tests {
46    use super::is_all_zero_bytes;
47
48    #[test]
49    fn test_all_zero_returns_true() {
50        assert!(is_all_zero_bytes(&[0u8; 16]));
51        assert!(is_all_zero_bytes(&[0u8; 32]));
52        assert!(is_all_zero_bytes(&[0u8; 33]));
53        assert!(is_all_zero_bytes(&[0u8; 64]));
54        assert!(is_all_zero_bytes(&[0u8; 100]));
55    }
56
57    #[test]
58    fn test_any_nonzero_returns_false() {
59        let mut bytes = [0u8; 32];
60        bytes[0] = 1;
61        assert!(!is_all_zero_bytes(&bytes));
62        let mut bytes = [0u8; 32];
63        bytes[31] = 1;
64        assert!(!is_all_zero_bytes(&bytes));
65        let mut bytes = [0u8; 64];
66        bytes[63] = 1;
67        assert!(!is_all_zero_bytes(&bytes));
68    }
69
70    #[test]
71    fn test_empty_returns_false() {
72        // Empty inputs are not "weak" — they're just unsupported. Returning
73        // `false` prevents downstream callers from rejecting an empty key
74        // as weak-zero when the actual issue is length validation.
75        assert!(!is_all_zero_bytes(&[]));
76    }
77
78    #[test]
79    fn test_oversize_chunked_path() {
80        // Larger than CHUNK (32) and not a multiple, exercises the partial-
81        // last-chunk branch. PBKDF2 salts in the wild are 16-64 bytes.
82        assert!(is_all_zero_bytes(&[0u8; 65]));
83        let mut bytes = [0u8; 65];
84        bytes[64] = 1;
85        assert!(!is_all_zero_bytes(&bytes));
86    }
87}
