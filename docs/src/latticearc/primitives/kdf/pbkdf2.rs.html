6//! SP 800-132: Password-Based Key Derivation Function (PBKDF2)
7//!
8//! PBKDF2 is a password-based key derivation function that applies a pseudorandom
9//! function (typically HMAC) to derive keys from passwords. It includes salting and
10//! iteration to make brute-force attacks more difficult.
11//!
12//! This implementation provides NIST SP 800-132 compliant PBKDF2 with:
13//! - Configurable iteration counts for adjustable computational cost
14//! - Salt support for key uniqueness
15//! - Multiple PRF options (HMAC-SHA256, HMAC-SHA512)
16//! - Secure memory handling with zeroization
17

20use subtle::ConstantTimeEq;
21use zeroize::{Zeroize, ZeroizeOnDrop, Zeroizing};

23/// PBKDF2 pseudorandom function types
24#[non_exhaustive]
25#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
26pub enum PrfType {
27    /// HMAC-SHA256 (recommended for most applications)
28    HmacSha256,
29    /// HMAC-SHA512 (for higher security requirements)
30    HmacSha512,
31}

33/// PBKDF2 parameters structure.
34///
35/// Salt is public protocol data, but scrubbing it on drop matches the rest of
36/// the library's zeroization hygiene and prevents stale-copy buildup when the
37/// struct is cloned around a request pipeline.
38#[derive(Debug, Clone, Zeroize, ZeroizeOnDrop)]
39pub struct Pbkdf2Params {
40    /// Salt value (minimum 16 bytes recommended)
41    /// Consumer: pbkdf2()
42    pub salt: Vec<u8>,
43    /// Iteration count (minimum 1000, default 600,000 per OWASP 2023 for HMAC-SHA256)
44    /// Consumer: pbkdf2()
45    pub iterations: u32,
46    /// Desired key length in bytes
47    /// Consumer: pbkdf2()
48    pub key_length: usize,
49    /// PRF to use
50    /// Consumer: pbkdf2()
51    #[zeroize(skip)] // PrfType is a Copy enum with no secret content
52    pub prf: PrfType,
53}
54
55impl Pbkdf2Params {
56    /// NIST SP 800-132 §5.1 minimum salt length (128 bits = 16 bytes).
57    ///
58    /// > "A randomly-generated salt SHALL be used. The salt SHALL be a
59    /// > randomly-generated value of at least 128 bits."
60    ///
61    /// Enforced as a hard floor in [`Self::new`] and [`pbkdf2`] — fail-closed
62    /// rather than warn, matching the rest of the v0.8.0 strict-by-default
63    /// posture (AEAD `WeakKey`, X25519 low-order rejection, etc.).
64    pub const MIN_SALT_LEN: usize = 16;
65
66    /// Create PBKDF2 parameters with a securely generated random salt.
67    ///
68    /// This constructor generates a cryptographically secure random salt to ensure
69    /// uniqueness and prevent precomputation attacks.
70    ///
71    /// # Arguments
72    /// * `salt_length` - Length of the salt to generate (must be at least
73    ///   [`MIN_SALT_LEN`] = 16 bytes per NIST SP 800-132 §5.1).
74    ///
75    /// # Security Note
76    /// Using a fresh random salt for each key derivation is essential for security.
77    /// Never reuse salts across different passwords or applications.
78    ///
79    /// # Errors
80    /// Returns [`LatticeArcError::InvalidParameter`] if `salt_length < 16`.
81    ///
82    /// [`MIN_SALT_LEN`]: Self::MIN_SALT_LEN
83    pub fn new(salt_length: usize) -> Result<Self> {
84        if salt_length < Self::MIN_SALT_LEN {
85            return Err(LatticeArcError::InvalidParameter(format!(
86                "Salt length {salt_length} below NIST SP 800-132 §5.1 minimum of \
87                 {} bytes (128 bits)",
88                Self::MIN_SALT_LEN
89            )));
90        }
91
92        let mut salt = vec![0u8; salt_length];
93        get_random_bytes(&mut salt);
94
95        Ok(Self { salt, iterations: 600_000, key_length: 32, prf: PrfType::HmacSha256 })
96    }
97
98    /// Create PBKDF2 parameters with the given salt and OWASP-2023
99    /// defaults (`iterations = 600_000`, `key_length = 32`,
100    /// `prf = HmacSha256`).
101    ///
102    /// # Salt validation
103    ///
104    /// Salt-length validation against NIST SP 800-132 §5.1 (≥ 16 bytes)
105    /// is enforced at the [`pbkdf2`] call site, **not** here. This is
106    /// deliberate so that wire-format parsers (e.g.
107    /// [`crate::unified_api::key_format`]) can deserialize a
108    /// `Pbkdf2Params` from a possibly-short pre-0.8.0 envelope for
109    /// inspection or re-protection. A `Pbkdf2Params` carrying a short
110    /// salt cannot reach key derivation: `pbkdf2(password, params)`
111    /// returns [`LatticeArcError::InvalidParameter`] before any HMAC
112    /// rounds run, so the only observable behaviour of a short-salt
113    /// `Pbkdf2Params` is a derivation-time error.
114    ///
115    /// For the alternative "validate at construction time" path that
116    /// generates a fresh random salt of a guaranteed-safe length, use
117    /// [`Self::new`].
118    ///
119    /// # Arguments
120    /// * `salt` - The salt value. Must be ≥ [`Self::MIN_SALT_LEN`]
121    ///   (16) bytes when this `Pbkdf2Params` is later passed to
122    ///   [`pbkdf2`]. Wire-format parsers may pass a shorter salt
123    ///   provided the resulting struct is not used for derivation.
124    ///
125    /// # Security Note
126    /// The salt must be cryptographically random and unique for each
127    /// password. For derivation-bound use, prefer [`Self::new`] which
128    /// generates the salt itself.
129    #[must_use]
130    pub fn with_salt(salt: &[u8]) -> Self {
131        Self { salt: salt.to_vec(), iterations: 600_000, key_length: 32, prf: PrfType::HmacSha256 }
132    }
133
134    /// Set iteration count
135    #[must_use]
136    pub fn iterations(mut self, iterations: u32) -> Self {
137        self.iterations = iterations;
138        self
139    }
140
141    /// Set key length
142    #[must_use]
143    pub fn key_length(mut self, key_length: usize) -> Self {
144        self.key_length = key_length;
145        self
146    }
147
148    /// Set PRF type
149    #[must_use]
150    pub fn prf(mut self, prf: PrfType) -> Self {
151        self.prf = prf;
152        self
153    }
154}

181impl Pbkdf2Result {
182    /// Borrow the derived key bytes.
183    ///
184    /// Named `expose_secret()` to align with Secret Type Invariant I-6
185    /// (universal accessor; see `docs/SECRET_TYPE_INVARIANTS.md`).
186    #[must_use]
187    pub fn expose_secret(&self) -> &[u8] {
188        &self.key
189    }

192    #[must_use]
193    pub fn key_length(&self) -> usize {
194        self.key.len()
195    }
196
197    /// Get the parameters used for derivation
198    #[must_use]
199    pub fn params(&self) -> &Pbkdf2Params {
200        &self.params
201    }
202
203    /// Verify a password against this result
204    ///
205    /// # Errors
206    /// Returns an error if the password derivation fails.
207    pub fn verify_password(&self, password: &[u8]) -> Result<bool> {
208        let derived = pbkdf2(password, &self.params)?;
209        let len_eq = self.key.len().ct_eq(&derived.key.len());
210        let bytes_eq = self
211            .key
212            .iter()
213            .zip(derived.key.iter())
214            .fold(subtle::Choice::from(1u8), |acc, (x, y)| acc & x.ct_eq(y));
215        Ok((len_eq & bytes_eq).into())
216    }

219/// PBKDF2 key derivation function
220///
221/// Derives a cryptographic key from a password using PBKDF2 as specified in
222/// NIST SP 800-132. The function applies a pseudorandom function (HMAC) multiple
223/// times to increase computational cost and make brute-force attacks more difficult.
224///
225/// # Arguments
226/// * `password` - The password to derive key from
227/// * `params` - PBKDF2 parameters (salt, iterations, key length, PRF)
228///
229/// # Returns
230/// Derived key wrapped in Pbkdf2Result
231///
232/// # Security Considerations
233/// - Use at least 16 bytes of random salt
234/// - Use at least 10,000 iterations (higher for better security)
235/// - Store salt alongside derived key for password verification
236///
237/// # Errors
238/// Returns an error if parameters are invalid (empty salt, too few iterations, or zero key length).
239pub fn pbkdf2(password: &[u8], params: &Pbkdf2Params) -> Result<Pbkdf2Result> {
240    // Validate parameters per SP 800-132 §5.1: salt MUST be at least 128 bits
241    // (16 bytes). Enforced as a hard floor — fail-closed defence in depth,
242    // matching the rest of the v0.8.0 strict-by-default posture.
243    if params.salt.len() < Pbkdf2Params::MIN_SALT_LEN {
244        return Err(LatticeArcError::InvalidParameter(format!(
245            "Salt length {} below NIST SP 800-132 §5.1 minimum of {} bytes (128 bits)",
246            params.salt.len(),
247            Pbkdf2Params::MIN_SALT_LEN,
248        )));
249    }
250
251    // Reject all-zero salt (defends against uninitialised-memory bugs, same
252    // spirit as `AeadError::WeakKey`). Use the shared CT helper rather than
253    // `iter().all`, which short-circuits on the first non-zero byte and so
254    // leaks salt-prefix structure via timing. PBKDF2 salts can exceed
255    // 32 bytes — go through `primitives::ct` (no MAX cap) rather than the
256    // AEAD-shaped `is_all_zero_key` re-export.
257    if crate::primitives::ct::is_all_zero_bytes(&params.salt) {
258        return Err(LatticeArcError::InvalidParameter(
259            "Salt must not be all zeros - use a cryptographically random salt".to_string(),
260        ));
261    }
262
263    if params.iterations < 1000 {
264        return Err(LatticeArcError::InvalidParameter(
265            "Iteration count must be at least 1000".to_string(),
266        ));
267    }
268
269    // Cap iterations to prevent denial-of-service via excessive computation
270    if params.iterations > 10_000_000 {
271        return Err(LatticeArcError::InvalidParameter(
272            "Iteration count must not exceed 10,000,000".to_string(),
273        ));
274    }
275
276    if params.key_length == 0 {
277        return Err(LatticeArcError::InvalidParameter(
278            "Key length must be greater than 0".to_string(),
279        ));
280    }
281
282    // Calculate number of blocks needed
283    let prf_output_len = match params.prf {
284        PrfType::HmacSha256 => 32,
285        PrfType::HmacSha512 => 64,
286    };
287
288    let block_count = params.key_length.div_ceil(prf_output_len);
289    // Wrap in Zeroizing so partial key material is erased on early return via `?`.
290    let mut derived_key: Zeroizing<Vec<u8>> = Zeroizing::new(vec![0u8; params.key_length]);
291    let mut offset = 0;
292
293    // Generate each block of the derived key
294    for block_index in 1..=block_count {
295        let block_index_u32 = u32::try_from(block_index).map_err(|_e| {
296            LatticeArcError::InvalidParameter(format!(
297                "Block index {} exceeds u32::MAX",
298                block_index
299            ))
300        })?;
301        let block =
302            generate_block(password, &params.salt, params.iterations, block_index_u32, params.prf);
303        let copy_len = std::cmp::min(block.len(), params.key_length.saturating_sub(offset));
304        let end_offset = offset.checked_add(copy_len).ok_or_else(|| {
305            LatticeArcError::InvalidParameter("Derived key offset overflow".to_string())
306        })?;
307        let dest_slice = derived_key.get_mut(offset..end_offset).ok_or_else(|| {
308            LatticeArcError::InvalidParameter("Derived key buffer overflow".to_string())
309        })?;
310        let src_slice = block.get(..copy_len).ok_or_else(|| {
311            LatticeArcError::InvalidParameter("Block slice out of bounds".to_string())
312        })?;
313        dest_slice.copy_from_slice(src_slice);
314        offset = end_offset;
315    }
316
317    Ok(Pbkdf2Result { key: derived_key, params: params.clone() })
318}

320/// Generate a single block of the PBKDF2 output.
321///
322/// The returned buffer contains derived key material and is wrapped in
323/// `Zeroizing` to erase it automatically on drop, including on early error
324/// returns from the caller. Infallible: all underlying HMAC operations are
325/// infallible for valid key lengths (checked in `pbkdf2()`).
326fn generate_block(
327    password: &[u8],
328    salt: &[u8],
329    iterations: u32,
330    block_index: u32,
331    prf: PrfType,
332) -> Zeroizing<Vec<u8>> {
333    // Convert block index to bytes (big-endian)
334    let mut block_input = salt.to_vec();
335    block_input.extend_from_slice(&block_index.to_be_bytes());
336
337    // U_1 = PRF(password, salt || INT(block_index))
338    let mut u = compute_prf(password, &block_input, prf);
339    let mut result: Zeroizing<Vec<u8>> = Zeroizing::new(u.to_vec());
340
341    // U_2 = PRF(password, U_1) ⊕ U_1
342    // U_3 = PRF(password, U_2) ⊕ U_2
343    // ...
344    // U_c = PRF(password, U_{c-1}) ⊕ U_{c-1}
345    for _ in 1..iterations {
346        u = compute_prf(password, &u, prf);
347        for (res_byte, u_byte) in result.iter_mut().zip(u.iter()) {
348            *res_byte ^= u_byte;
349        }
350    }
351
352    result
353}

355/// Compute PRF (HMAC) for PBKDF2.
356///
357/// Returns the HMAC output wrapped in `Zeroizing` because it is used directly
358/// as derived key material (both `U_1` and the XOR-chained iterates `U_i`).
359/// Infallible: aws-lc-rs `hmac::Key::new` accepts any key length, and
360/// `hmac::sign` cannot fail for HMAC-SHA256/SHA-512.
361fn compute_prf(password: &[u8], data: &[u8], prf: PrfType) -> Zeroizing<Vec<u8>> {
362    // Delegates to aws-lc-rs HMAC for FIPS-validated constant-time operation.
363    // The same backend is used by HKDF, SP800-108, and the mac::hmac wrapper.
364    let algorithm = match prf {
365        PrfType::HmacSha256 => HMAC_SHA256,
366        PrfType::HmacSha512 => HMAC_SHA512,
367    };
368    let key = hmac::Key::new(algorithm, password);
369    let tag = hmac::sign(&key, data);
370    Zeroizing::new(tag.as_ref().to_vec())
371}

373/// Password-based key derivation with default parameters
374///
375/// Convenience function that uses recommended default parameters:
376/// - 16-byte random salt
377/// - 600,000 iterations (OWASP 2023 recommendation for HMAC-SHA256)
378/// - 32-byte key length
379/// - HMAC-SHA256 PRF
380///
381/// # Errors
382/// Returns an error if key derivation fails.
383pub fn pbkdf2_simple(password: &[u8]) -> Result<Pbkdf2Result> {
384    let params = Pbkdf2Params::new(16)?.iterations(600_000).key_length(32).prf(PrfType::HmacSha256);
385
386    pbkdf2(password, &params)
387}

389/// Verify a password against a previously derived key
390///
391/// # Errors
392/// Returns an error if the password derivation fails.
393pub fn verify_password(
394    password: &[u8],
395    derived_key: &[u8],
396    salt: &[u8],
397    iterations: u32,
398) -> Result<bool> {
399    let params = Pbkdf2Params::with_salt(salt)
400        .iterations(iterations)
401        .key_length(derived_key.len())
402        .prf(PrfType::HmacSha256);
403
404    let result = pbkdf2(password, &params)?;
405    let len_eq = derived_key.len().ct_eq(&result.key.len());
406    let bytes_eq = derived_key
407        .iter()
408        .zip(result.key.iter())
409        .fold(subtle::Choice::from(1u8), |acc, (x, y)| acc & x.ct_eq(y));
410    Ok((len_eq & bytes_eq).into())
411}

618    #[test]
619    fn test_pbkdf2_short_salt_rejected_at_derivation() {
620        // Pin the documented invariant: `with_salt` is infallible, but
621        // a `Pbkdf2Params` carrying a salt below MIN_SALT_LEN cannot
622        // reach key derivation — `pbkdf2()` rejects with
623        // `InvalidParameter` before any HMAC rounds run. This is the
624        // single enforcement point for NIST SP 800-132 §5.1 on the
625        // wire-format-parser path.
626        let short = vec![0xAAu8; Pbkdf2Params::MIN_SALT_LEN.saturating_sub(1)];
627        let params = Pbkdf2Params::with_salt(&short).iterations(1000).key_length(32);
628        let result = pbkdf2(b"password", &params);
629        assert!(
630            result.is_err(),
631            "pbkdf2 must reject {short_len}-byte salt",
632            short_len = short.len()
633        );
634    }
