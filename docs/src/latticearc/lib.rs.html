6//! LatticeArc - Post-Quantum Cryptography Library
7//!
8//! Comprehensive post-quantum cryptography library providing advanced encryption,
9//! digital signatures, and security features for modern applications.
10//!
11//! ## Unified API with CryptoConfig
12//!
13//! All cryptographic operations use [`CryptoConfig`] for configuration. This builder
14//! pattern provides automatic algorithm selection based on use case or security level,
15//! with optional Zero Trust session verification.
16//!
17//! ### Basic Usage (Hybrid — Recommended)
18//!
19//! ```rust,no_run
20//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
21//! use latticearc::{encrypt, decrypt, CryptoConfig, EncryptKey, DecryptKey};
22//!
23//! // Hybrid encryption: ML-KEM-768 + X25519 + HKDF + AES-256-GCM
24//! let (pk, sk) = latticearc::generate_hybrid_keypair()?;
25//! let encrypted = encrypt(b"secret", EncryptKey::Hybrid(&pk), CryptoConfig::new())?;
26//! let decrypted = decrypt(&encrypted, DecryptKey::Hybrid(&sk), CryptoConfig::new())?;
27//! # Ok(())
28//! # }
29//! ```
30//!
31//! ### Symmetric Encryption
32//!
33//! ```rust,no_run
34//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
35//! use latticearc::{encrypt, decrypt, CryptoConfig, CryptoScheme, EncryptKey, DecryptKey};
36//!
37//! let key = [0u8; 32];  // 256-bit key for AES-256-GCM
38//! let encrypted = encrypt(b"secret", EncryptKey::Symmetric(&key),
39//!     CryptoConfig::new().force_scheme(CryptoScheme::Symmetric))?;
40//! let decrypted = decrypt(&encrypted, DecryptKey::Symmetric(&key), CryptoConfig::new())?;
41//! # Ok(())
42//! # }
43//! ```
44//!
45//! ### With Use Case Selection
46//!
47//! ```rust,no_run
48//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
49//! use latticearc::{encrypt, CryptoConfig, UseCase, EncryptKey};
50//!
51//! // Library selects optimal hybrid algorithm for the use case
52//! let (pk, _sk) = latticearc::generate_hybrid_keypair()?;
53//! let encrypted = encrypt(b"data", EncryptKey::Hybrid(&pk),
54//!     CryptoConfig::new().use_case(UseCase::FileStorage))?;
55//! # Ok(())
56//! # }
57//! ```
58//!
59//! ## Zero Trust Session Verification
60//!
61//! For production deployments, use [`VerifiedSession`] to enable Zero Trust
62//! verification before each operation:
63//!
64//! ```rust,no_run
65//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
66//! use latticearc::{
67//!     encrypt, decrypt, CryptoConfig, VerifiedSession, generate_keypair,
68//!     EncryptKey, DecryptKey,
69//! };
70//!
71//! // Step 1: Generate a keypair (done once, typically at provisioning)
72//! let (pk, sk) = generate_keypair()?;
73//!
74//! // Step 2: Establish a verified session (performs challenge-response)
75//! let session = VerifiedSession::establish(&pk, sk.as_ref())?;
76//!
77//! // Step 3: Hybrid encryption with session verification
78//! let (enc_pk, enc_sk) = latticearc::generate_hybrid_keypair()?;
79//! let encrypted = encrypt(b"secret", EncryptKey::Hybrid(&enc_pk),
80//!     CryptoConfig::new().session(&session))?;
81//! let decrypted = decrypt(&encrypted, DecryptKey::Hybrid(&enc_sk),
82//!     CryptoConfig::new().session(&session))?;
83//! # Ok(())
84//! # }
85//! ```
86//!
87//! **Benefits of session verification:**
88//! - Session expiration is checked before each operation
89//! - Provides audit context (session ID, trust level, timestamp)
90//! - Supports continuous verification workflows
91//!
92//! ## Digital Signatures
93//!
94//! ```rust,no_run
95//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
96//! use latticearc::{generate_signing_keypair, sign_with_key, verify, CryptoConfig};
97//!
98//! let message = b"Document to sign";
99//!
100//! // Generate a persistent signing keypair (ML-DSA-65 + Ed25519 hybrid)
101//! let (pk, sk, scheme) = generate_signing_keypair(CryptoConfig::new())?;
102//!
103//! // Sign with the persistent keypair
104//! let signed = sign_with_key(message, &sk, &pk, CryptoConfig::new())?;
105//!
106//! // Verify (uses public key embedded in SignedData)
107//! let is_valid = verify(&signed, CryptoConfig::new())?;
108//! # Ok(())
109//! # }
110//! ```
111//!
112//! ## Hybrid Encryption (ML-KEM-768 + X25519)
113//!
114//! Use the unified API with `EncryptKey::Hybrid` / `DecryptKey::Hybrid`:
115//!
116//! ```rust,no_run
117//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
118//! use latticearc::{encrypt, decrypt, CryptoConfig, EncryptKey, DecryptKey};
119//!
120//! let (pk, sk) = latticearc::generate_hybrid_keypair()?;
121//! let encrypted = encrypt(b"secret data", EncryptKey::Hybrid(&pk), CryptoConfig::new())?;
122//! let decrypted = decrypt(&encrypted, DecryptKey::Hybrid(&sk), CryptoConfig::new())?;
123//! # Ok(())
124//! # }
125//! ```
126//!
127//! ## Hybrid Signatures (ML-DSA-65 + Ed25519)
128//!
129//! ```rust,no_run
130//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
131//! use latticearc::{generate_hybrid_signing_keypair, sign_hybrid, verify_hybrid_signature, SecurityMode};
132//!
133//! // Generate a hybrid signing keypair (ML-DSA-65 + Ed25519)
134//! let (pk, sk) = generate_hybrid_signing_keypair(SecurityMode::Unverified)?;
135//!
136//! // Sign (both ML-DSA and Ed25519 signatures generated)
137//! let signature = sign_hybrid(b"document", &sk, SecurityMode::Unverified)?;
138//!
139//! // Verify (both must pass for signature to be valid)
140//! let valid = verify_hybrid_signature(b"document", &signature, &pk, SecurityMode::Unverified)?;
141//! # Ok(())
142//! # }
143//! ```
144//!
145//! ## Session Lifecycle
146//!
147//! Sessions have a 30-minute default lifetime:
148//!
149//! ```rust,no_run
150//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
151//! use latticearc::{encrypt, CryptoConfig, VerifiedSession, generate_keypair, CoreError, EncryptKey};
152//!
153//! let (pk, sk) = generate_keypair()?;
154//! let session = VerifiedSession::establish(&pk, sk.as_ref())?;
155//!
156//! // Check session properties
157//! assert!(session.is_valid());  // Not expired
158//! let _ = session.session_id(); // Unique ID for audit
159//! let _ = session.expires_at(); // Expiration time
160//!
161//! // Validate before critical operations
162//! session.verify_valid()?;  // Returns Err(SessionExpired) if expired
163//!
164//! // Refresh if expired
165//! if !session.is_valid() {
166//!     let new_session = VerifiedSession::establish(&pk, sk.as_ref())?;
167//! }
168//! # Ok(())
169//! # }
170//! ```
171//!
172//! ## Complete Example
173//!
174//! ```rust,no_run
175//! use latticearc::{
176//!     encrypt, decrypt, generate_signing_keypair, sign_with_key, verify,
177//!     generate_hybrid_keypair, CryptoConfig, CoreError,
178//!     EncryptKey, DecryptKey,
179//! };
180//!
181//! fn secure_workflow() -> Result<(), CoreError> {
182//!     // --- Hybrid Encryption (unified API) ---
183//!     let (enc_pk, enc_sk) = generate_hybrid_keypair()?;
184//!     let encrypted = encrypt(b"confidential", EncryptKey::Hybrid(&enc_pk),
185//!         CryptoConfig::new())?;
186//!     let decrypted = decrypt(&encrypted, DecryptKey::Hybrid(&enc_sk),
187//!         CryptoConfig::new())?;
188//!
189//!     // --- Digital Signatures ---
190//!     let (sign_pk, sign_sk, _scheme) = generate_signing_keypair(CryptoConfig::new())?;
191//!     let signed = sign_with_key(b"important document", &sign_sk, &sign_pk, CryptoConfig::new())?;
192//!     let is_valid = verify(&signed, CryptoConfig::new())?;
193//!     assert!(is_valid);
194//!
195//!     Ok(())
196//! }
197//! ```
198//!
199//! ## Feature Flags
200//!
201//! | Feature | Description |
202//! |---------|-------------|
203//! | `fips` | FIPS 140-3 validated backend via aws-lc-rs. Requires CMake + Go build tools. Without this feature, aws-lc-rs uses its default non-FIPS backend (C compiler only). |
204//! | `fips-self-test` | Power-up KAT self-tests for all FIPS-boundary algorithms (ML-KEM, AES-GCM, SHA-2, ML-DSA, SLH-DSA). |
205//! | `zkp-serde` | Serialization support for ZKP types (enables `serde_with` for Schnorr/Sigma protocol structs). |
206//! | `formal-verification` | Compilation marker: enables formal verification harness code (Kani proofs). Does not run proofs — use `cargo kani` separately. |
207//! | `kani` | Compilation marker: enables Kani bounded model checking proof harnesses. Requires `cargo kani` to execute proofs. |
208//! | `saw` | Compilation marker: enables SAW formal verification markers (inherited from aws-lc-rs). Does not run SAW proofs at build time. |
209//!
