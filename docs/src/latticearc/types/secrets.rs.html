1//! # Secret byte containers
2//!
3//! Provides the two primitive secret types used throughout LatticeArc:
4//!
5//! - [`SecretBytes<N>`]: fixed-size secret bytes, stack-allocated `[u8; N]` backing.
6//!   Preferred whenever the length is compile-time known.
7//! - [`SecretVec`]: variable-length secret bytes, heap-allocated `Vec<u8>` backing.
8//!   Used only when the length genuinely varies at runtime.
9//!
10//! Both types uphold the full Secret Type Invariant set — see
11//! [`docs/SECRET_TYPE_INVARIANTS.md`](../../../docs/SECRET_TYPE_INVARIANTS.md)
12//! for the canonical, normative specification.
13//!
14//! ## Quick reference
15//!
16//! | Invariant | Mechanism |
17//! |---|---|
18//! | Wipe on drop | `#[derive(Zeroize, ZeroizeOnDrop)]` — volatile writes |
19//! | Redacted `Debug` | Manual `fmt::Debug` impl emits `"[REDACTED]"` |
20//! | Timing-safe equality | `impl ConstantTimeEq` |
21//! | No `PartialEq` / `Eq` | Enforced at `tests/no_partial_eq_on_secret_types.rs` |
22//! | No `Clone` | Explicit `clone_for_transmission(&self) -> Self` |
23//! | Sealed accessor | `expose_secret()` only — no `AsRef`, no `Deref` |
24//!
25//! ## When to use which
26//!
27//! ```ignore
28//! // 32-byte HKDF-Extract output: fixed size → SecretBytes<32>
29//! fn hkdf_extract(...) -> SecretBytes<32>;
30//!
31//! // 64-byte hybrid shared secret: fixed size → SecretBytes<64>
32//! struct EncapsulatedKey { shared_secret: SecretBytes<64>, ... }
33//!
34//! // AEAD plaintext: length varies with input → SecretVec
35//! fn decrypt_aes_gcm(...) -> SecretVec;
36//!
37//! // PBKDF2 output with user-specified key_length → SecretVec
38//! fn pbkdf2(..., key_length: usize) -> SecretVec;
39//! ```

46// ============================================================================
47// SecretBytes<N> — fixed-size, stack-allocated
48// ============================================================================
49
50/// Fixed-size secret byte container, stack-allocated.
51///
52/// The `[u8; N]` backing store never reallocates, never migrates across the heap,
53/// and is wiped by volatile writes on drop. Prefer this type over [`SecretVec`]
54/// whenever `N` is compile-time known.
55///
56/// # Security invariants
57///
58/// - **Wipe on drop**: `#[derive(Zeroize, ZeroizeOnDrop)]` emits volatile writes
59///   that the compiler cannot optimize away.
60/// - **Redacted `Debug`**: the bytes are never printed; `"[REDACTED]"` is emitted.
61/// - **Timing-safe equality**: [`ConstantTimeEq`] compares all `N` bytes in time
62///   independent of their contents; `PartialEq`/`Eq` are not implemented.
63/// - **No `Clone`**: duplication must go through [`clone_for_transmission`] so
64///   every copy is a grep-able audit checkpoint.
65/// - **Sealed access**: the bytes are exposed only through [`expose_secret`]. No
66///   `AsRef<[u8]>`, no `Deref`, no public field.
67///
68/// [`clone_for_transmission`]: Self::clone_for_transmission
69/// [`expose_secret`]: Self::expose_secret
70///
71/// # Why stack-allocated matters
72///
73/// - Heap allocators retain size metadata after `free`, which is a side-channel
74///   leak about secret sizes in adversary-controlled environments.
75/// - A `Vec<u8>` with a single `.push()` past capacity silently reallocates and
76///   frees the old buffer **without zeroization**. `[u8; N]` has no such escape.
77/// - Stack memory is reclaimed in deterministic scope order, not at allocator
78///   whim.
79///
80/// # Example
81///
82/// ```
83/// use latticearc::types::SecretBytes;
84///
85/// // Construct from raw bytes (e.g., after filling a buffer with KDF output)
86/// let key: SecretBytes<32> = SecretBytes::new([0x42u8; 32]);
87/// assert_eq!(key.len(), 32);
88/// assert_eq!(key.expose_secret()[0], 0x42);
89///
90/// // Debug is redacted
91/// let debug = format!("{:?}", key);
92/// assert!(debug.contains("[REDACTED]"));
93/// assert!(!debug.contains("42"));
94/// ```
95#[derive(Zeroize, ZeroizeOnDrop)]
96pub struct SecretBytes<const N: usize> {
97    bytes: [u8; N],
98}
99
100impl<const N: usize> SecretBytes<N> {
101    /// Construct from raw bytes.
102    ///
103    /// The passed array is moved in; no copy is made beyond the move itself.
104    #[must_use]
105    #[inline]
106    pub const fn new(bytes: [u8; N]) -> Self {
107        Self { bytes }
108    }
109
110    /// Construct an all-zero buffer.
111    ///
112    /// Build the final byte pattern in a plain `[u8; N]` on the stack, then
113    /// hand the array to [`Self::new`]; use [`Self::zero`] when the caller
114    /// wants an all-zero placeholder (e.g., a struct field that is overwritten
115    /// by a downstream operation).
116    #[must_use]
117    #[inline]
118    pub const fn zero() -> Self {
119        Self { bytes: [0u8; N] }
120    }
121
122    /// Length in bytes. Always equals the const generic `N`.
123    #[must_use]
124    #[inline]
125    pub const fn len(&self) -> usize {
126        N
127    }
128
129    /// Returns `true` if `N == 0`.
130    #[must_use]
131    #[inline]
132    pub const fn is_empty(&self) -> bool {
133        N == 0
134    }
135
136    /// Expose the inner bytes.
137    ///
138    /// This is the **only** public read accessor. Every call site is a grep-able
139    /// audit checkpoint: `rg '\.expose_secret\(\)'` enumerates every point at
140    /// which secret bytes are read.
141    ///
142    /// Consumers should hold the returned slice only as long as needed and
143    /// should never copy it into a non-secret container (a subsequent `.to_vec()`
144    /// or struct field of type `Vec<u8>` would evade zeroization).
145    #[must_use]
146    #[inline]
147    pub const fn expose_secret(&self) -> &[u8; N] {
148        &self.bytes
149    }
150
151    /// Create an independent copy.
152    ///
153    /// `Clone` is intentionally not derived. Use this method when duplication is
154    /// semantically required — for example, sending a shared secret across a
155    /// channel boundary — so every duplication is grep-able.
156    #[must_use]
157    #[inline]
158    pub fn clone_for_transmission(&self) -> Self {
159        Self { bytes: self.bytes }
160    }
161}
162
163impl<const N: usize> fmt::Debug for SecretBytes<N> {
164    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
165        f.debug_struct("SecretBytes").field("len", &N).field("bytes", &"[REDACTED]").finish()
166    }
167}
168
169impl<const N: usize> ConstantTimeEq for SecretBytes<N> {
170    fn ct_eq(&self, other: &Self) -> Choice {
171        self.bytes.ct_eq(&other.bytes)
172    }
173}

175// ============================================================================
176// SecretVec — variable-length, heap-allocated
177// ============================================================================
178
179/// Variable-length secret byte container, heap-allocated.
180///
181/// Use only when the length genuinely varies at runtime (e.g., user-specified
182/// PBKDF2 output length, AEAD plaintext of arbitrary input). For any
183/// compile-time-known length, prefer [`SecretBytes<N>`].
184///
185/// # Security invariants
186///
187/// Same as [`SecretBytes`]. See type-level docs for the full list.
188///
189/// # Reallocation safety
190///
191/// The internal `Vec<u8>` is never reallocated by this type's public API: there
192/// is no `push`, no `extend`, no `reserve`. All construction paths allocate the
193/// final size up front. If a caller needs to build a secret incrementally, they
194/// must build the full `Vec<u8>` first (outside this wrapper), then pass it to
195/// [`Self::new`] — at which point reallocation is no longer a concern because
196/// subsequent mutation is not exposed.
197///
198/// # OS-level memory locking (`secret-mlock` feature)
199///
200/// With the `secret-mlock` feature enabled, each `SecretVec` locks its backing
201/// buffer into RAM via `mlock(2)` (Linux/macOS) or `VirtualLock` (Windows) so
202/// the bytes cannot be swapped to disk or captured in a core dump. This
203/// implements invariant I-10. If locking fails at construction time (e.g.,
204/// `RLIMIT_MEMLOCK` exceeded), the `SecretVec` is still returned with an
205/// unlocked backing — the bytes remain zeroized on drop, but OS-level leakage
206/// protection is not active. Without the feature, `mlock` is not called and
207/// the type has no dependency on the `region` crate.
208///
209/// `SecretBytes<N>` is deliberately NOT covered by this feature: its `[u8; N]`
210/// backing is stack-allocated, so mlocking it would require per-instance
211/// page-aligned locking of arbitrary stack addresses — impractical and
212/// incompatible with Rust's move semantics. Callers who need OS-level
213/// protection for small fixed-size secrets should wrap them in `SecretVec` at
214/// a boundary where the sensitive bytes leave the stack.
215///
216/// # Example
217///
218/// ```
219/// use latticearc::types::SecretVec;
220///
221/// let key = SecretVec::new(vec![0x42u8; 48]);
222/// assert_eq!(key.len(), 48);
223/// assert_eq!(key.expose_secret()[0], 0x42);
224///
225/// let debug = format!("{:?}", key);
226/// assert!(debug.contains("[REDACTED]"));
227/// ```
228#[derive(Zeroize, ZeroizeOnDrop)]
229pub struct SecretVec {
230    // Drop order when `secret-mlock` is enabled:
231    //   1. `ZeroizeOnDrop`'s generated `Drop` runs first, calling `.zeroize()`
232    //      on `bytes` while the region is still mlocked.
233    //   2. Fields drop in declaration order: `_lock` first (unlocks the
234    //      region), then `bytes` (returns the zeroed buffer to the allocator).
235    // `#[zeroize(skip)]` is required because `LockGuard` does not implement
236    // `Zeroize` and there's nothing secret inside it — it only holds a handle
237    // to an OS lock.
238    #[cfg(feature = "secret-mlock")]
239    #[zeroize(skip)]
240    _lock: Option<region::LockGuard>,
241    bytes: Vec<u8>,
242}
243
244impl SecretVec {
245    /// Construct from a `Vec<u8>`.
246    ///
247    /// The vector is moved in; its capacity is taken as-is. For best security
248    /// hygiene, construct the passed `Vec` at its final length (e.g.,
249    /// `vec![0u8; len]`) rather than growing it, so no unzeroized intermediate
250    /// buffer is ever freed.
251    #[must_use]
252    #[inline]
253    pub fn new(bytes: Vec<u8>) -> Self {
254        Self::from_bytes(bytes)
255    }
256
257    /// Construct an all-zero buffer of the given length.
258    #[must_use]
259    #[inline]
260    pub fn zero(len: usize) -> Self {
261        Self::from_bytes(vec![0u8; len])
262    }
263
264    /// Single construction path. Shrinks the backing `Vec` so its capacity
265    /// matches its length (so `Vec::zeroize` covers every byte of the
266    /// allocation — `zeroize` 1.8 only wipes `..len`, not `..capacity`), then
267    /// attaches the mlock guard when the `secret-mlock` feature is enabled.
268    #[inline]
269    fn from_bytes(mut bytes: Vec<u8>) -> Self {
270        bytes.shrink_to_fit();
271        #[cfg(feature = "secret-mlock")]
272        let _lock = Self::try_lock(&bytes);
273        Self {
274            #[cfg(feature = "secret-mlock")]
275            _lock,
276            bytes,
277        }
278    }
279
280    /// Attempt to lock the backing buffer into RAM. Returns `None` on empty
281    /// buffers (locking a dangling pointer is UB) or on `mlock` failure
282    /// (e.g. `RLIMIT_MEMLOCK` exceeded). Failures are swallowed: the bytes
283    /// are still zeroized on drop, we simply lose OS-level leakage protection.
284    #[cfg(feature = "secret-mlock")]
285    #[inline]
286    fn try_lock(bytes: &[u8]) -> Option<region::LockGuard> {
287        if bytes.is_empty() {
288            return None;
289        }
290        region::lock(bytes.as_ptr(), bytes.len()).ok()
291    }
292
293    /// Length in bytes.
294    #[must_use]
295    #[inline]
296    pub fn len(&self) -> usize {
297        self.bytes.len()
298    }
299
300    /// Returns `true` if the backing buffer is empty.
301    #[must_use]
302    #[inline]
303    pub fn is_empty(&self) -> bool {
304        self.bytes.is_empty()
305    }
306
307    /// Expose the inner bytes. The only public read accessor.
308    #[must_use]
309    #[inline]
310    pub fn expose_secret(&self) -> &[u8] {
311        &self.bytes
312    }
313
314    /// Create an independent copy. See [`SecretBytes::clone_for_transmission`].
315    ///
316    /// With the `secret-mlock` feature, the cloned buffer is independently
317    /// locked into RAM (the original's `LockGuard` does not transfer).
318    #[must_use]
319    pub fn clone_for_transmission(&self) -> Self {
320        Self::from_bytes(self.bytes.clone())
321    }
322}
323
324impl fmt::Debug for SecretVec {
325    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
326        f.debug_struct("SecretVec")
327            .field("len", &self.bytes.len())
328            .field("bytes", &"[REDACTED]")
329            .finish()
330    }
331}
332
333impl ConstantTimeEq for SecretVec {
334    fn ct_eq(&self, other: &Self) -> Choice {
335        self.bytes.ct_eq(&other.bytes)
336    }
337}
