1//! SP 800-57 Key Lifecycle Management
2//!
3//! This module implements formal key lifecycle management per NIST SP 800-57
4//! requirements, including state transitions, custodianship, and audit trails.
5//!
6//! # Key States (SP 800-57 Section 3)
7//!
8//! - **Generation**: Key material is being generated
9//! - **Active**: Key is ready for use
10//! - **Rotating**: Key rotation in progress (overlap period)
11//! - **Retired**: Key scheduled for retirement
12//! - **Destroyed**: Key material zeroized
13//!
14//! # Example
15//!
16//! ```
17//! use latticearc::types::key_lifecycle::{KeyLifecycleRecord, KeyLifecycleState};
18//!
19//! let mut record = KeyLifecycleRecord::new(
20//!     "key-123".to_string(),
21//!     "ML-KEM-768".to_string(),
22//!     3,   // security level
23//!     365, // rotation interval (days)
24//!     30,  // overlap period (days)
25//! );
26//!
27//! // Activate the key
28//! record.transition(
29//!     KeyLifecycleState::Active,
30//!     "alice".to_string(),
31//!     "Key generation complete".to_string(),
32//!     Some("approval-123".to_string()),
33//! ).expect("Valid transition");
34//!
35//! assert!(record.is_valid_for_use());
36//! ```
37
38use crate::types::error::{Result, TypeError};
39use serde::{Deserialize, Serialize};

41/// Maximum length (in bytes) for a single audit-trail string field.
42/// Keeps state_history bounded and JSONL audit consumers happy.
43const MAX_AUDIT_FIELD_LEN: usize = 512;
44
45/// Cap on the number of `StateTransition` entries in
46/// `KeyLifecycleRecord::state_history`. The state machine has 5
47/// terminal states; legitimate records have under a dozen entries.
48/// 1024 is generous slack for Active⇄Rotating cycling, while
49/// preventing a tampered or pathologically-cycled record from driving
50/// unbounded growth (and making the linear scans in
51/// `KeyLifecycleRecordRaw::try_from` O(n²) — see L8).
52const MAX_STATE_HISTORY: usize = 1024;
53
54/// Cap on the number of approver IDs in
55/// `KeyLifecycleRecord::approvers`. Real-world quorum schemes top out
56/// at a few dozen approvers; 256 is generous slack with the same
57/// motivation as `MAX_STATE_HISTORY`.
58const MAX_APPROVERS: usize = 256;
59
60/// Validate an audit-trail string. Rejects:
61/// - empty
62/// - control characters (including newlines, which break JSONL)
63/// - more than `MAX_AUDIT_FIELD_LEN` bytes
64///
65/// `Option<String>` callers should validate inside the `Some(_)` arm:
66/// presence is decided at the caller; content is validated here.
67fn validate_audit_field(name: &str, value: &str) -> Result<()> {
68    if value.is_empty() {
69        return Err(TypeError::InvalidAuditInput(format!("{} must not be empty", name)));
70    }
71    if value.len() > MAX_AUDIT_FIELD_LEN {
72        return Err(TypeError::InvalidAuditInput(format!(
73            "{} exceeds {} bytes",
74            name, MAX_AUDIT_FIELD_LEN
75        )));
76    }
77    if value.chars().any(char::is_control) {
78        return Err(TypeError::InvalidAuditInput(format!("{} contains control characters", name)));
79    }
80    Ok(())
81}

83/// SP 800-57 Section 3: Key Lifecycle States
84#[non_exhaustive]
85#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
86#[cfg_attr(kani, derive(kani::Arbitrary))]
87pub enum KeyLifecycleState {
88    /// Key material generation
89    Generation,
90    /// Key activation - ready for use
91    Active,
92    /// Key rotation in progress (overlap period)
93    Rotating,
94    /// Key scheduled for retirement
95    Retired,
96    /// Key destruction - material zeroized
97    Destroyed,
98}
99
100/// SP 800-57 Key State Transitions (formally verifiable)
101pub struct KeyStateMachine;
102
103impl KeyStateMachine {
104    /// Verify state transition is valid per SP 800-57
105    ///
106    /// Valid transitions:
107    /// - None -> Generation (initial state)
108    /// - Generation -> Active (initialization complete)
109    /// - Active -> Rotating (rotation initiated)
110    /// - Active -> Retired (direct retirement)
111    /// - Rotating -> Retired (rotation complete)
112    /// - Retired -> Destroyed (cleanup)
113    #[must_use]
114    pub fn is_valid_transition(from: Option<KeyLifecycleState>, to: KeyLifecycleState) -> bool {
115        match (from, to) {
116            // Generation is always valid initial state
117            (None, KeyLifecycleState::Generation) => true,
118
119            // Generation -> Active (initialization complete)
120            (Some(KeyLifecycleState::Generation), KeyLifecycleState::Active) => true,
121
122            // Active -> Rotating (rotation initiated)
123            (Some(KeyLifecycleState::Active), KeyLifecycleState::Rotating) => true,
124
125            // Rotating -> Retired (rotation complete)
126            (Some(KeyLifecycleState::Rotating), KeyLifecycleState::Retired) => true,
127
128            // Active -> Retired (direct retirement)
129            (Some(KeyLifecycleState::Active), KeyLifecycleState::Retired) => true,
130
131            // Retired -> Destroyed (cleanup)
132            (Some(KeyLifecycleState::Retired), KeyLifecycleState::Destroyed) => true,
133
134            // All other transitions are invalid
135            _ => false,
136        }
137    }
138
139    /// Get allowed next states from current state
140    #[must_use]
141    pub fn allowed_next_states(current: KeyLifecycleState) -> Vec<KeyLifecycleState> {
142        match current {
143            KeyLifecycleState::Generation => vec![KeyLifecycleState::Active],
144            KeyLifecycleState::Active => {
145                vec![KeyLifecycleState::Rotating, KeyLifecycleState::Retired]
146            }
147            KeyLifecycleState::Rotating => vec![KeyLifecycleState::Retired],
148            KeyLifecycleState::Retired => vec![KeyLifecycleState::Destroyed],
149            KeyLifecycleState::Destroyed => vec![],
150        }
151    }
152}

183/// Key lifecycle record with audit trail.
184///
185/// State-machine fields are private so the only way to mutate
186/// `current_state`, `state_history`, the per-state timestamps, or
187/// custodianship is through [`Self::transition`] /
188/// [`Self::add_approver`]. Construction-time fields (`key_id`,
189/// `key_type`, `security_level`, `generated_at`,
190/// `rotation_interval_days`, `overlap_period_days`) stay public — they
191/// are set in [`Self::new`] and never reassigned.
192#[derive(Debug, Clone, Serialize, Deserialize)]
193#[serde(try_from = "KeyLifecycleRecordRaw")]
194pub struct KeyLifecycleRecord {
195    /// Unique key identifier
196    pub key_id: String,
197    /// Algorithm/key type (e.g., "ML-KEM-768", "ML-DSA-65")
198    pub key_type: String,
199    /// Security level (1-5)
200    pub security_level: u32,
201
202    // State management — private so transitions only happen via `transition()`
203    current_state: KeyLifecycleState,
204    state_history: Vec<StateTransition>,
205
206    // Custodianship — private so updates only happen via `transition()` /
207    // `add_approver()`
208    generator: Option<String>,
209    approvers: Vec<String>,
210    destroyer: Option<String>,
211
212    /// When the key was generated
213    pub generated_at: chrono::DateTime<chrono::Utc>,
214
215    // Per-state timestamps — private so they can only be set by
216    // `transition()`
217    activated_at: Option<chrono::DateTime<chrono::Utc>>,
218    rotated_at: Option<chrono::DateTime<chrono::Utc>>,
219    retired_at: Option<chrono::DateTime<chrono::Utc>>,
220    destroyed_at: Option<chrono::DateTime<chrono::Utc>>,
221
222    // SP 800-57 requirements
223    /// How often the key should be rotated (days)
224    pub rotation_interval_days: u32,
225    /// Overlap period during rotation (days)
226    pub overlap_period_days: u32,
227}

229/// Wire-shape mirror of `KeyLifecycleRecord` used by the serde
230/// `try_from` hook. Loading a persisted record goes
231/// JSON → `KeyLifecycleRecordRaw` → `validate()` → `KeyLifecycleRecord`,
232/// so a tampered or corrupted file with state-machine inconsistencies
233/// (e.g. `Destroyed` with empty `state_history`, or `Generation` with
234/// a populated `activated_at`) is rejected at deserialize time rather
235/// than producing a silently-broken record.
236#[derive(Debug, Clone, Deserialize)]
237#[doc(hidden)]
238struct KeyLifecycleRecordRaw {
239    key_id: String,
240    key_type: String,
241    security_level: u32,
242    current_state: KeyLifecycleState,
243    state_history: Vec<StateTransition>,
244    generator: Option<String>,
245    approvers: Vec<String>,
246    destroyer: Option<String>,
247    generated_at: chrono::DateTime<chrono::Utc>,
248    activated_at: Option<chrono::DateTime<chrono::Utc>>,
249    rotated_at: Option<chrono::DateTime<chrono::Utc>>,
250    retired_at: Option<chrono::DateTime<chrono::Utc>>,
251    destroyed_at: Option<chrono::DateTime<chrono::Utc>>,
252    rotation_interval_days: u32,
253    overlap_period_days: u32,
254}

256impl TryFrom<KeyLifecycleRecordRaw> for KeyLifecycleRecord {
257    type Error = TypeError;
258
259    fn try_from(raw: KeyLifecycleRecordRaw) -> Result<Self> {
260        // Re-validate state-machine invariants on load so a corrupted
261        // or tampered persisted record cannot bypass the rules
262        // `transition()` enforces in memory.
263
264        // (0) Vector caps: same bounds the in-memory mutators
265        //     enforce. Without these, the linear scans below
266        //     (`entered`, last-history check) become O(n²) on a
267        //     pathologically-large persisted record, and the
268        //     in-memory record can grow past `transition()`'s cap
269        //     after a roundtrip.
270        if raw.state_history.len() > MAX_STATE_HISTORY {
271            return Err(TypeError::InvalidAuditInput(format!(
272                "state_history length {} exceeds cap of {}",
273                raw.state_history.len(),
274                MAX_STATE_HISTORY
275            )));
276        }
277        if raw.approvers.len() > MAX_APPROVERS {
278            return Err(TypeError::InvalidAuditInput(format!(
279                "approvers length {} exceeds cap of {}",
280                raw.approvers.len(),
281                MAX_APPROVERS
282            )));
283        }
284
285        // (1) `current_state` must be reachable from the persisted
286        //     `activated_at` value. Rotating and Retired both require
287        //     having passed through Active, so `activated_at` MUST be
288        //     `Some` for those — and Generation MUST NOT have one.
289        match (raw.current_state, raw.activated_at) {
290            (KeyLifecycleState::Generation, Some(_)) => {
291                return Err(TypeError::InvalidAuditInput(
292                    "current_state=Generation with activated_at set".to_string(),
293                ));
294            }
295            (
296                KeyLifecycleState::Active
297                | KeyLifecycleState::Rotating
298                | KeyLifecycleState::Retired,
299                None,
300            ) => {
301                return Err(TypeError::InvalidAuditInput(format!(
302                    "current_state={:?} requires activated_at",
303                    raw.current_state
304                )));
305            }
306            _ => {}
307        }
308
309        // (2) For non-Generation states, the last state_history entry
310        //     must agree with `current_state`. Otherwise a tampered
311        //     file with `current_state = Active` and a `state_history`
312        //     ending in `Rotating` (or empty) would deserialize cleanly.
313        match (raw.current_state, raw.state_history.last()) {
314            (KeyLifecycleState::Generation, _) => {}
315            (_, None) => {
316                return Err(TypeError::InvalidAuditInput(format!(
317                    "current_state={:?} with empty state_history",
318                    raw.current_state
319                )));
320            }
321            (current, Some(t)) if t.to_state != current => {
322                return Err(TypeError::InvalidAuditInput(format!(
323                    "current_state={:?} but last state_history entry is {:?}",
324                    current, t.to_state
325                )));
326            }
327            _ => {}
328        }
329
330        // (3) Per-state timestamp consistency: the timestamp for a
331        //     state must be present iff that state has been entered
332        //     (i.e. appears in state_history).
333        let entered = |st: KeyLifecycleState| raw.state_history.iter().any(|t| t.to_state == st);
334        if entered(KeyLifecycleState::Rotating) != raw.rotated_at.is_some() {
335            return Err(TypeError::InvalidAuditInput(
336                "rotated_at presence disagrees with state_history".to_string(),
337            ));
338        }
339        if entered(KeyLifecycleState::Retired) != raw.retired_at.is_some() {
340            return Err(TypeError::InvalidAuditInput(
341                "retired_at presence disagrees with state_history".to_string(),
342            ));
343        }
344        if entered(KeyLifecycleState::Destroyed) != raw.destroyed_at.is_some() {
345            return Err(TypeError::InvalidAuditInput(
346                "destroyed_at presence disagrees with state_history".to_string(),
347            ));
348        }
349        Ok(KeyLifecycleRecord {
350            key_id: raw.key_id,
351            key_type: raw.key_type,
352            security_level: raw.security_level,
353            current_state: raw.current_state,
354            state_history: raw.state_history,
355            generator: raw.generator,
356            approvers: raw.approvers,
357            destroyer: raw.destroyer,
358            generated_at: raw.generated_at,
359            activated_at: raw.activated_at,
360            rotated_at: raw.rotated_at,
361            retired_at: raw.retired_at,
362            destroyed_at: raw.destroyed_at,
363            rotation_interval_days: raw.rotation_interval_days,
364            overlap_period_days: raw.overlap_period_days,
365        })
366    }
367}

369/// Record of a state transition
370#[derive(Debug, Clone, Serialize, Deserialize)]
371pub struct StateTransition {
372    /// Previous state (None if initial)
373    pub from_state: Option<KeyLifecycleState>,
374    /// New state
375    pub to_state: KeyLifecycleState,
376    /// When the transition occurred
377    pub timestamp: chrono::DateTime<chrono::Utc>,
378    /// ID of the custodian who performed the transition
379    pub custodian_id: String,
380    /// Reason for the transition
381    pub justification: String,
382    /// Approval reference (if applicable)
383    pub approval_id: Option<String>,
384}
385
386impl KeyLifecycleRecord {
387    /// Create new key lifecycle record
388    ///
389    /// # Arguments
390    ///
391    /// * `key_id` - Unique identifier for the key
392    /// * `key_type` - Algorithm/key type (e.g., "ML-KEM-768")
393    /// * `security_level` - Security level (1-5)
394    /// * `rotation_interval_days` - How often to rotate the key
395    /// * `overlap_period_days` - Overlap period during rotation
396    #[must_use]
397    pub fn new(
398        key_id: String,
399        key_type: String,
400        security_level: u32,
401        rotation_interval_days: u32,
402        overlap_period_days: u32,
403    ) -> Self {
404        Self {
405            key_id,
406            key_type,
407            security_level,
408            current_state: KeyLifecycleState::Generation,
409            state_history: Vec::new(),
410            generator: None,
411            approvers: Vec::new(),
412            destroyer: None,
413            generated_at: chrono::Utc::now(),
414            activated_at: None,
415            rotated_at: None,
416            retired_at: None,
417            destroyed_at: None,
418            rotation_interval_days,
419            overlap_period_days,
420        }
421    }
422
423    /// Transition key to new state with custodianship tracking
424    ///
425    /// # Arguments
426    ///
427    /// * `to_state` - Target state
428    /// * `custodian_id` - ID of the custodian performing the transition
429    /// * `justification` - Reason for the transition
430    /// * `approval_id` - Optional approval reference
431    ///
432    /// # Errors
433    ///
434    /// Returns `TypeError::InvalidStateTransition` if the transition is invalid
435    pub fn transition(
436        &mut self,
437        to_state: KeyLifecycleState,
438        custodian_id: String,
439        justification: String,
440        approval_id: Option<String>,
441    ) -> Result<()> {
442        // Audit-trail input gates: empty / control-chars / oversized
443        // strings break JSONL audit consumers and produce attribution-
444        // free entries. Reject up front rather than persisting them.
445        validate_audit_field("custodian_id", &custodian_id)?;
446        validate_audit_field("justification", &justification)?;
447        // Inside `Some(_)` the caller has explicitly chosen to send a
448        // value; empty-string is a sentinel that confuses
449        // presence-vs-content matching downstream. Reject it here.
450        if let Some(ref approval_id) = approval_id {
451            validate_audit_field("approval_id", approval_id)?;
452        }
453
454        if !KeyStateMachine::is_valid_transition(Some(self.current_state), to_state) {
455            return Err(TypeError::InvalidStateTransition {
456                from: self.current_state,
457                to: to_state,
458            });
459        }
460        // Capture `now` once so the state_history timestamp and the
461        // per-state timestamp below match exactly. An auditor comparing
462        // `state_history[i].timestamp` against `activated_at` (etc.)
463        // would otherwise see sub-tick skew on every transition.
464        let now = chrono::Utc::now();
465        let transition = StateTransition {
466            from_state: Some(self.current_state),
467            to_state,
468            timestamp: now,
469            custodian_id: custodian_id.clone(),
470            justification,
471            approval_id,
472        };
473
474        if self.state_history.len() >= MAX_STATE_HISTORY {
475            return Err(TypeError::InvalidAuditInput(format!(
476                "state_history at cap of {} entries; cannot record further transitions",
477                MAX_STATE_HISTORY
478            )));
479        }
480        self.state_history.push(transition);
481        self.current_state = to_state;
482
483        // Update timestamps (use the captured `now`).
484        match to_state {
485            KeyLifecycleState::Active => self.activated_at = Some(now),
486            KeyLifecycleState::Rotating => self.rotated_at = Some(now),
487            KeyLifecycleState::Retired => self.retired_at = Some(now),
488            KeyLifecycleState::Destroyed => self.destroyed_at = Some(now),
489            _ => {}
490        }
491
492        // Update custodianship
493        // Generator is the custodian who completes key generation (moves to Active)
494        // Check the last transition we just added to see if it was from Generation
495        if let Some(last_transition) = self.state_history.last()
496            && last_transition.from_state == Some(KeyLifecycleState::Generation)
497            && to_state == KeyLifecycleState::Active
498        {
499            self.generator = Some(custodian_id.clone());
500        }
501        if to_state == KeyLifecycleState::Destroyed {
502            self.destroyer = Some(custodian_id);
503        }
504
505        Ok(())
506    }
507
508    /// Days elapsed since `activated`, signed (negative for future
509    /// activation due to clock skew). Internal helper backing
510    /// [`Self::requires_rotation`] and [`Self::age_days`] so the
511    /// `Utc::now().signed_duration_since(...).num_days()` chain lives
512    /// in one place.
513    fn days_since_activation(activated: chrono::DateTime<chrono::Utc>) -> i64 {
514        chrono::Utc::now().signed_duration_since(activated).num_days()
515    }
516
517    /// Check if key is due for rotation per SP 800-57
518    #[must_use]
519    pub fn requires_rotation(&self) -> bool {
520        let Some(activated_at) = self.activated_at else { return false };
521        let age_days_i64 = Self::days_since_activation(activated_at);
522        if age_days_i64 < 0 {
523            // Future activation date — clock skew or misconfiguration
524            tracing::warn!(
525                "Key has negative age ({age_days_i64} days); activation_at in the future"
526            );
527            return false;
528        }
529        // Ages larger than u32::MAX → always require rotation
530        let age_days = u32::try_from(age_days_i64).unwrap_or(u32::MAX);
531        age_days >= self.rotation_interval_days
532    }
533
534    /// Get key age in days since activation
535    #[must_use]
536    pub fn age_days(&self) -> Option<u32> {
537        self.activated_at.map(|activated| {
538            // Negative ages (future activation) treated as 0.
539            u32::try_from(Self::days_since_activation(activated)).unwrap_or(0)
540        })
541    }
542
543    /// Check if key is in valid state for use
544    #[must_use]
545    pub fn is_valid_for_use(&self) -> bool {
546        matches!(self.current_state, KeyLifecycleState::Active | KeyLifecycleState::Rotating)
547    }
548
549    /// Add an approver to the key. Silently ignored if the approver
550    /// is already in the list, or if the cap (`MAX_APPROVERS`) has
551    /// been reached. Returns `false` on cap rejection so callers can
552    /// surface a warning if needed — the `#[must_use]` here forces
553    /// every caller to acknowledge the bool, since round-34 M8
554    /// changed the return type from `()` precisely to expose this
555    /// case and a caller that just `.;` it would silently lose the
556    /// signal.
557    #[must_use = "add_approver returns false when the cap is reached; surface that to the caller"]
558    pub fn add_approver(&mut self, approver_id: impl Into<String>) -> bool {
559        let approver_id = approver_id.into();
560        if self.approvers.contains(&approver_id) {
561            return true;
562        }
563        if self.approvers.len() >= MAX_APPROVERS {
564            return false;
565        }
566        self.approvers.push(approver_id);
567        true
568    }

574    // ----------------------------------------------------------------

579        self.current_state
580    }

586    }

589    #[must_use]
590    pub fn generator(&self) -> Option<&str> {
591        self.generator.as_deref()
592    }
593

595    #[must_use]
596    pub fn approvers(&self) -> &[String] {
597        &self.approvers
598    }
599

601    #[must_use]
602    pub fn destroyer(&self) -> Option<&str> {
603        self.destroyer.as_deref()
604    }
605
606    /// When the key was activated.
607    #[must_use]
608    pub fn activated_at(&self) -> Option<chrono::DateTime<chrono::Utc>> {
609        self.activated_at
610    }
611
612    /// When key rotation was initiated.
613    #[must_use]
614    pub fn rotated_at(&self) -> Option<chrono::DateTime<chrono::Utc>> {
615        self.rotated_at
616    }
617
618    /// When the key was retired.

621        self.retired_at
622    }

636    // --- Existing proofs ---
637
638    #[kani::proof]
639    fn key_state_machine_destroyed_cannot_transition() {
640        let to: KeyLifecycleState = kani::any();
641
642        // Property: Destroyed keys cannot transition to any state
643        let is_valid = KeyStateMachine::is_valid_transition(Some(KeyLifecycleState::Destroyed), to);
644        kani::assert(!is_valid, "Destroyed keys should not transition");
645    }
646
647    #[kani::proof]
648    fn key_state_machine_no_backward_to_generation() {
649        let from: KeyLifecycleState = kani::any();
650
651        // Property: Cannot go back to Generation from any state

661    /// Security property: keys must begin in the Generation state.

671    /// Proves that `is_valid_transition` matches the SP 800-57 transition
672    /// specification for all state pairs. This is an independent encoding of
673    /// the allowed transitions — if `is_valid_transition` has a bug, this
674    /// proof will catch the discrepancy.
675    ///
676    /// Note: `allowed_next_states()` returns `Vec`, which Kani cannot
677    /// efficiently verify (unbounded loop unwinding on heap iteration).
678    /// Unit tests verify `allowed_next_states()` separately.
679    #[kani::proof]
680    fn key_state_machine_transitions_match_spec() {
681        let from: KeyLifecycleState = kani::any();
682        let to: KeyLifecycleState = kani::any();
683
684        let transition_valid = KeyStateMachine::is_valid_transition(Some(from), to);
685
686        // Independent specification of SP 800-57 allowed transitions
687        let spec_allows = match from {
688            KeyLifecycleState::Generation => to == KeyLifecycleState::Active,
689            KeyLifecycleState::Active => {
690                to == KeyLifecycleState::Rotating || to == KeyLifecycleState::Retired
691            }
692            KeyLifecycleState::Rotating => to == KeyLifecycleState::Retired,
693            KeyLifecycleState::Retired => to == KeyLifecycleState::Destroyed,
694            KeyLifecycleState::Destroyed => false,
695        };
696
697        kani::assert(
698            transition_valid == spec_allows,
699            "is_valid_transition must match SP 800-57 specification",
700        );
701    }
702
703    /// Proves that retired keys can only transition to Destroyed.
704    /// Security property: once retired, the only path is secure destruction.
705    #[kani::proof]
706    fn key_state_machine_retired_only_to_destroyed() {
707        let to: KeyLifecycleState = kani::any();
708        kani::assume(to != KeyLifecycleState::Destroyed);
709
710        let is_valid = KeyStateMachine::is_valid_transition(Some(KeyLifecycleState::Retired), to);
711        kani::assert(!is_valid, "Retired keys can only transition to Destroyed");
712    }
