27/// Initialize FIPS mode with power-on self-tests.
28///
29/// # Aborts
30///
31/// This function calls `std::process::abort()` if power-on self-tests fail or
32/// no security level is achieved, as required by FIPS 140-3. Callers cannot
33/// recover from self-test failure.
34///
35/// # Errors
36///
37/// Returns an error only if the validation result lock cannot be acquired.
38/// All crypto failures result in process abort, not `Err`.
39pub fn init() -> Result<(), LatticeArcError> {
40    if FIPS_INITIALIZED.load(Ordering::Acquire) {
41        return Ok(());
42    }
43
44    tracing::info!("Starting FIPS power-on self-tests");
45
46    let validator = FIPSValidator::new(ValidationScope::FullModule);
47    let result = validator.validate_module()?;
48
49    if !result.is_valid {
50        tracing::error!("FIPS power-on self-tests failed - aborting library initialization");
51        std::process::abort();
52    }
53
54    if let Some(level) = result.level {
55        tracing::info!("FIPS power-on self-tests passed - Level {:?}", level);
56    } else {
57        tracing::error!(
58            "FIPS power-on self-tests passed but no security level achieved - aborting"
59        );
60        std::process::abort();
61    }
62
63    FIPS_VALIDATION_RESULT
64        .lock()
65        .map_err(|e| LatticeArcError::ValidationError {
66            message: format!("Failed to acquire FIPS validation result lock: {}", e),
67        })?
68        .replace(result);
69
70    FIPS_INITIALIZED.store(true, Ordering::Release);
71    tracing::info!("FIPS validation completed successfully");
72
73    Ok(())
74}

76/// Run conditional self-test for a specific algorithm.
77///
78/// # Errors
79/// Returns an error if initialization fails or the specified algorithm test fails.
80pub fn run_conditional_self_test(algorithm: &str) -> Result<(), LatticeArcError> {
81    if !FIPS_INITIALIZED.load(Ordering::Acquire) {
82        init()?;
83    }
84
85    let validator = FIPSValidator::new(ValidationScope::AlgorithmsOnly);
86
87    match algorithm {
88        "aes" | "AES" => {
89            let result = validator.test_aes_algorithm()?;
90            if !result.passed {
91                return Err(LatticeArcError::ValidationError {
92                    message: format!(
93                        "AES conditional self-test failed: {}",
94                        result.error_message.unwrap_or_default()
95                    ),
96                });
97            }
98        }
99        "sha3" | "SHA3" => {
100            let result = validator.test_sha3_algorithm()?;
101            if !result.passed {
102                return Err(LatticeArcError::ValidationError {
103                    message: format!(
104                        "SHA-3 conditional self-test failed: {}",
105                        result.error_message.unwrap_or_default()
106                    ),
107                });
108            }
109        }
110        "mlkem" | "MLKEM" => {
111            let result = validator.test_mlkem_algorithm()?;
112            if !result.passed {
113                return Err(LatticeArcError::ValidationError {
114                    message: format!(
115                        "ML-KEM conditional self-test failed: {}",
116                        result.error_message.unwrap_or_default()
117                    ),
118                });
119            }
120        }
121        _ => {
122            let result = validator.test_self_tests()?;
123            if !result.passed {
124                return Err(LatticeArcError::ValidationError {
125                    message: format!(
126                        "Self-test conditional check failed: {}",
127                        result.error_message.unwrap_or_default()
128                    ),
129                });
130            }
131        }
132    }
133
134    Ok(())
135}

137/// Perform continuous RNG health test per FIPS 140-3
138///
139/// # Errors
140/// Returns an error if initialization fails or if the RNG produces identical consecutive samples.
141pub fn continuous_rng_test() -> Result<(), LatticeArcError> {
142    if !FIPS_INITIALIZED.load(Ordering::Acquire) {
143        init()?;
144    }
145
146    let mut sample1 = [0u8; 32];
147    let mut sample2 = [0u8; 32];
148
149    rand::rngs::OsRng.fill_bytes(&mut sample1);
150    rand::rngs::OsRng.fill_bytes(&mut sample2);
151
152    if sample1 == sample2 {
153        return Err(LatticeArcError::ValidationError {
154            message: "RNG continuous test failed: identical samples".to_string(),
155        });
156    }
157
158    let mut bits_set = 0;
159    for byte in sample1.iter().chain(sample2.iter()) {
160        bits_set += byte.count_ones();
161    }
162
163    let total_bits = 64 * 8;
164    let ones_ratio = f64::from(bits_set) / f64::from(total_bits);
165
166    if !(0.4..=0.6).contains(&ones_ratio) {
167        return Err(LatticeArcError::ValidationError {
168            message: format!(
169                "RNG continuous test failed: bit distribution out of range: {:.3}",
170                ones_ratio
171            ),
172        });
173    }
174
175    Ok(())
176}

188/// Auto-initialize FIPS on library load
189/// Can be disabled by setting FIPS_SKIP_AUTO_INIT=1 environment variable
190///
191/// # Note
192/// Auto-init is DISABLED by default in library builds to avoid interfering with
193/// test harnesses and applications that need control over initialization timing.
194/// Applications should call `init()` explicitly when FIPS mode is required.
195#[ctor::ctor]
196fn fips_auto_init() {
197    // Skip auto-init when explicitly disabled (default behavior)
198    // To enable auto-init, set FIPS_ENABLE_AUTO_INIT=1
199    if std::env::var("FIPS_ENABLE_AUTO_INIT").is_err() {
200        return;
201    }
202
203    // Allow explicit skip as well
204    if std::env::var("FIPS_SKIP_AUTO_INIT").is_ok() {
205        return;
206    }
207
208    if let Err(e) = init() {
209        // Use tracing instead of eprintln! for library code
210        tracing::error!("FIPS initialization failed: {}", e);
211        std::process::abort();
212    }
213}
