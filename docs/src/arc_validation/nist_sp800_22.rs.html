55    /// Test a bit sequence using NIST SP 800-22 tests.
56    ///
57    /// # Errors
58    /// This function is infallible but returns Result for API consistency.
59    #[allow(clippy::arithmetic_side_effects)] // Statistical calculations require arithmetic
60    pub fn test_bit_sequence(&self, data: &[u8]) -> Result<RngTestResults> {
61        let min_bytes = self.min_sequence_length.saturating_div(8);
62        if data.len() < min_bytes {
63            return Ok(RngTestResults {
64                algorithm: "unknown".to_string(),
65                bits_tested: data.len().saturating_mul(8),
66                test_results: vec![],
67                passed: false,
68                entropy_estimate: 0.0,
69            });
70        }
71
72        let bits = self.bytes_to_bits(data);
73        let mut test_results = Vec::new();
74
75        test_results.push(self.frequency_test(&bits)?);
76        test_results.push(self.frequency_within_block_test(&bits)?);
77        test_results.push(self.runs_test(&bits)?);
78        test_results.push(self.longest_run_of_ones_test(&bits)?);
79        test_results.push(self.serial_test(&bits)?);
80        test_results.push(self.approximate_entropy_test(&bits)?);
81
82        let entropy_estimate = self.estimate_entropy(&bits);
83        let passed = test_results.iter().all(|r| r.passed);
84
85        Ok(RngTestResults {
86            algorithm: "NIST SP 800-22".to_string(),
87            bits_tested: bits.len(),
88            test_results,
89            passed,
90            entropy_estimate,
91        })
92    }

95    #[allow(clippy::arithmetic_side_effects)] // Bit manipulation is safe
96    #[must_use]
97    pub fn bytes_to_bits(&self, bytes: &[u8]) -> Vec<bool> {
98        bytes
99            .iter()
100            .flat_map(|&b| (0..8).map(move |i| (b >> (7_u8.saturating_sub(i))) & 1 == 1))
101            .collect()
102    }

104    /// Frequency (Monobit) Test
105    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
106    fn frequency_test(&self, bits: &[bool]) -> Result<NistStatisticalTestResult> {
107        let n = bits.len() as f64;
108        let ones = bits.iter().filter(|&&b| b).count() as f64;
109        let proportion = ones / n;
110
111        let s_obs = fabs(ones - n / 2.0) / sqrt(n / 4.0);
112        let p_value = erf(s_obs / sqrt(2.0));
113
114        Ok(NistStatisticalTestResult {
115            test_name: "Frequency (Monobit) Test".to_string(),
116            p_value,
117            passed: p_value > self.significance_level,
118            parameters: serde_json::json!({
119                "significance_level": self.significance_level,
120                "proportion_of_ones": proportion,
121                "s_obs": s_obs
122            }),
123        })
124    }

126    /// Frequency Within Block Test
127    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
128    fn frequency_within_block_test(&self, bits: &[bool]) -> Result<NistStatisticalTestResult> {
129        let n = bits.len();
130        let block_size = if n >= 1000 { 10000 } else { n.saturating_div(10) };
131        let num_blocks = n.saturating_div(block_size);
132
133        if num_blocks < 1 || block_size == 0 {
134            return Ok(NistStatisticalTestResult {
135                test_name: "Frequency Within Block Test".to_string(),
136                p_value: 0.0,
137                passed: false,
138                parameters: serde_json::json!({"error": "insufficient data"}),
139            });
140        }
141
142        let mut chi_squared = 0.0;
143        for i in 0..num_blocks {
144            let start = i.saturating_mul(block_size);
145            let end = start.saturating_add(block_size).min(bits.len());
146
147            let block = match bits.get(start..end) {
148                Some(b) => b,
149                None => continue,
150            };
151
152            if block.is_empty() {
153                continue;
154            }
155
156            let ones = block.iter().filter(|&&b| b).count() as f64;
157            let block_len = block.len() as f64;
158            let proportion = ones / block_len;
159            chi_squared += 4.0 * block_len * pow(proportion - 0.5, 2.0);
160        }
161
162        let degrees_of_freedom = num_blocks as f64;
163        let p_value = 1.0 - self.igamc(degrees_of_freedom / 2.0, chi_squared / 2.0);
164
165        Ok(NistStatisticalTestResult {
166            test_name: "Frequency Within Block Test".to_string(),
167            p_value,
168            passed: p_value > self.significance_level,
169            parameters: serde_json::json!({
170                "block_size": block_size,
171                "num_blocks": num_blocks,
172                "chi_squared": chi_squared,
173                "degrees_of_freedom": degrees_of_freedom
174            }),
175        })
176    }

178    /// Runs Test
179    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
180    fn runs_test(&self, bits: &[bool]) -> Result<NistStatisticalTestResult> {
181        let n = bits.len() as f64;
182        let ones = bits.iter().filter(|&&b| b).count() as f64;
183        let proportion = ones / n;
184
185        if fabs(proportion - 0.5) < 1e-10 {
186            return Ok(NistStatisticalTestResult {
187                test_name: "Runs Test".to_string(),
188                p_value: 0.0,
189                passed: false,
190                parameters: serde_json::json!({"error": "proportion too close to 0.5"}),
191            });
192        }
193
194        let runs = self.count_runs(bits) as f64;
195
196        let expected_runs = 2.0 * n * proportion * (1.0 - proportion);
197        let variance = 2.0 * n * proportion * (1.0 - proportion);
198
199        let test_statistic = fabs(runs - expected_runs) / sqrt(variance);
200        let p_value = erf(test_statistic / sqrt(2.0));
201
202        Ok(NistStatisticalTestResult {
203            test_name: "Runs Test".to_string(),
204            p_value,
205            passed: p_value > self.significance_level,
206            parameters: serde_json::json!({
207                "runs": runs,
208                "expected_runs": expected_runs,
209                "proportion": proportion,
210                "test_statistic": test_statistic
211            }),
212        })
213    }

215    /// Longest Run of Ones Test
216    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
217    fn longest_run_of_ones_test(&self, bits: &[bool]) -> Result<NistStatisticalTestResult> {
218        let n = bits.len();
219        let (block_size, k, expected_probabilities): (usize, usize, Vec<f64>) = match n {
220            128..=6272 => (8, 3, vec![0.2148, 0.3672, 0.2305, 0.1875]),
221            6273..=75000 => (128, 5, vec![0.1174, 0.2430, 0.2493, 0.1752, 0.1027, 0.1124]),
222            _ => (10000, 6, vec![0.0882, 0.2092, 0.2483, 0.1933, 0.1208, 0.0675, 0.0727]),
223        };
224
225        if n < block_size {
226            return Ok(NistStatisticalTestResult {
227                test_name: "Longest Run of Ones Test".to_string(),
228                p_value: 0.0,
229                passed: false,
230                parameters: serde_json::json!({"error": "insufficient data for test"}),
231            });
232        }
233
234        let num_blocks = n.saturating_div(block_size);
235        let mut category_counts = vec![0.0; k.saturating_add(1)];
236
237        for i in 0..num_blocks {
238            let start = i.saturating_mul(block_size);
239            let end = start.saturating_add(block_size).min(bits.len());
240
241            let block = match bits.get(start..end) {
242                Some(b) => b,
243                None => continue,
244            };
245
246            let longest_run = self.longest_run_of_ones_in_block(block);
247            let category = if longest_run <= k { longest_run } else { k };
248            if let Some(count) = category_counts.get_mut(category) {
249                *count += 1.0;
250            }
251        }
252
253        let chi_squared =
254            self.compute_chi_squared(&category_counts, &expected_probabilities, num_blocks as f64);
255        let p_value = 1.0 - self.igamc(k as f64 / 2.0, chi_squared / 2.0);
256
257        Ok(NistStatisticalTestResult {
258            test_name: "Longest Run of Ones in a Block Test".to_string(),
259            p_value,
260            passed: p_value > self.significance_level,
261            parameters: serde_json::json!({
262                "block_size": block_size,
263                "num_blocks": num_blocks,
264                "chi_squared": chi_squared,
265                "k": k
266            }),
267        })
268    }

270    /// Serial Test
271    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
272    fn serial_test(&self, bits: &[bool]) -> Result<NistStatisticalTestResult> {
273        let log2_len = bits.len().checked_ilog2().unwrap_or(0) as usize;
274        let m = std::cmp::min(16, log2_len.saturating_sub(3));
275
276        if m < 2 {
277            return Ok(NistStatisticalTestResult {
278                test_name: "Serial Test".to_string(),
279                p_value: 0.0,
280                passed: false,
281                parameters: serde_json::json!({"error": "insufficient data for serial test"}),
282            });
283        }
284
285        let psi_m = self.compute_psi_m(bits, m);
286        let psi_m_minus_1 = self.compute_psi_m(bits, m.saturating_sub(1));
287        let psi_m_minus_2 = self.compute_psi_m(bits, m.saturating_sub(2));
288
289        let del1 = psi_m - psi_m_minus_1;
290        let del2 = psi_m - 2.0 * psi_m_minus_1 + psi_m_minus_2;
291
292        let degrees1 = pow(2.0, (m as f64) - 1.0);
293        let degrees2 = pow(2.0, (m as f64) - 2.0);
294
295        let p_value1 = self.igamc(degrees1 / 2.0, del1 / 2.0);
296        let p_value2 = self.igamc(degrees2 / 2.0, del2 / 2.0);
297
298        let passed = p_value1 > self.significance_level && p_value2 > self.significance_level;
299
300        Ok(NistStatisticalTestResult {
301            test_name: "Serial Test".to_string(),
302            p_value: p_value1.min(p_value2),
303            passed,
304            parameters: serde_json::json!({
305                "m": m,
306                "p_value1": p_value1,
307                "p_value2": p_value2,
308                "del1": del1,
309                "del2": del2,
310                "degrees1": degrees1,
311                "degrees2": degrees2
312            }),
313        })
314    }

316    /// Approximate Entropy Test
317    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
318    fn approximate_entropy_test(&self, bits: &[bool]) -> Result<NistStatisticalTestResult> {
319        let log2_len = bits.len().checked_ilog2().unwrap_or(0) as usize;
320        let m = std::cmp::min(16, log2_len.saturating_sub(3));
321
322        if m < 2 {
323            return Ok(NistStatisticalTestResult {
324                test_name: "Approximate Entropy Test".to_string(),
325                p_value: 0.0,
326                passed: false,
327                parameters: serde_json::json!({"error": "insufficient data for entropy test"}),
328            });
329        }
330
331        let phi_m = self.compute_approximate_entropy(bits, m);
332        let phi_m_plus_1 = self.compute_approximate_entropy(bits, m.saturating_add(1));
333        let chi_squared = 2.0 * (bits.len() as f64) * (log(2.0) * (phi_m - phi_m_plus_1));
334
335        let degrees_of_freedom = pow(2.0, m as f64);
336        let p_value = self.igamc(degrees_of_freedom / 2.0, chi_squared / 2.0);
337
338        Ok(NistStatisticalTestResult {
339            test_name: "Approximate Entropy Test".to_string(),
340            p_value,
341            passed: p_value > self.significance_level,
342            parameters: serde_json::json!({
343                "m": m,
344                "chi_squared": chi_squared,
345                "phi_m": phi_m,
346                "phi_m_plus_1": phi_m_plus_1,
347                "degrees_of_freedom": degrees_of_freedom
348            }),
349        })
350    }

352    /// Count runs in bit sequence
353    #[allow(clippy::unused_self)] // Method of struct for consistency
354    fn count_runs(&self, bits: &[bool]) -> usize {
355        if bits.is_empty() {
356            return 0;
357        }
358
359        let mut runs = 1usize;
360        for i in 1..bits.len() {
361            if let (Some(&current_bit), Some(&prev_bit)) =
362                (bits.get(i), bits.get(i.saturating_sub(1)))
363                && current_bit != prev_bit
364            {
365                runs = runs.saturating_add(1);
366            }
367        }
368        runs
369    }

371    /// Find longest run of ones in a block
372    #[allow(clippy::unused_self)] // Method of struct for consistency
373    fn longest_run_of_ones_in_block(&self, block: &[bool]) -> usize {
374        let mut max_run = 0usize;
375        let mut current_run = 0usize;
376
377        for &bit in block {
378            if bit {
379                current_run = current_run.saturating_add(1);
380                max_run = max_run.max(current_run);
381            } else {
382                current_run = 0;
383            }
384        }
385        max_run
386    }

388    /// Compute chi-squared statistic
389    #[allow(clippy::arithmetic_side_effects, clippy::unused_self)] // Statistical calculation
390    fn compute_chi_squared(&self, observed: &[f64], expected: &[f64], total: f64) -> f64 {
391        observed
392            .iter()
393            .zip(expected.iter())
394            .map(|(&o, &e)| {
395                let exp = e * total;
396                if exp == 0.0 { 0.0 } else { pow(o - exp, 2.0) / exp }
397            })
398            .sum()
399    }

401    /// Compute psi_m statistic for serial test
402    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss, clippy::unused_self)] // Statistical math
403    fn compute_psi_m(&self, bits: &[bool], m: usize) -> f64 {
404        let mut counts = HashMap::new();
405        let n = bits.len();
406
407        if n < m || m == 0 {
408            return 0.0;
409        }
410
411        let iterations = n.saturating_sub(m).saturating_add(1);
412        for i in 0..iterations {
413            let mut pattern = 0u64;
414            for j in 0..m {
415                if let Some(&bit) = bits.get(i.saturating_add(j)) {
416                    pattern = (pattern << 1) | if bit { 1 } else { 0 };
417                }
418            }
419            *counts.entry(pattern).or_insert(0.0) += 1.0;
420        }
421
422        let total_sequences = iterations as f64;
423        if total_sequences == 0.0 {
424            return 0.0;
425        }
426        counts.values().map(|&count| pow(count, 2.0)).sum::<f64>() / total_sequences
427    }

429    /// Compute approximate entropy
430    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss, clippy::unused_self)] // Statistical math
431    fn compute_approximate_entropy(&self, bits: &[bool], m: usize) -> f64 {
432        let mut counts = HashMap::new();
433        let n = bits.len();
434
435        if n < m || m == 0 {
436            return 0.0;
437        }
438
439        let iterations = n.saturating_sub(m).saturating_add(1);
440        for i in 0..iterations {
441            let mut pattern = 0u64;
442            for j in 0..m {
443                if let Some(&bit) = bits.get(i.saturating_add(j)) {
444                    pattern = (pattern << 1) | if bit { 1 } else { 0 };
445                }
446            }
447            *counts.entry(pattern).or_insert(0.0) += 1.0;
448        }
449
450        let total_sequences = iterations as f64;
451        if total_sequences == 0.0 {
452            return 0.0;
453        }
454
455        let mut entropy = 0.0;
456
457        for &count in counts.values() {
458            let p = count / total_sequences;
459            if p > 0.0 {
460                entropy += p * log(p) / log(2.0);
461            }
462        }
463
464        -entropy
465    }

467    /// Estimate entropy of bit sequence
468    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss)] // Statistical math
469    #[must_use]
470    pub fn estimate_entropy(&self, bits: &[bool]) -> f64 {
471        if bits.is_empty() {
472            return 0.0;
473        }
474
475        let ones = bits.iter().filter(|&&b| b).count() as f64;
476        let proportion = ones / (bits.len() as f64);
477
478        if proportion <= 0.0 || proportion >= 1.0 {
479            return 0.0;
480        }
481
482        -proportion * log(proportion) / log(2.0)
483            - (1.0 - proportion) * log(1.0 - proportion) / log(2.0)
484    }

486    /// Incomplete gamma function (upper)
487    #[allow(clippy::arithmetic_side_effects, clippy::unused_self)] // Mathematical function
488    fn igamc(&self, a: f64, x: f64) -> f64 {
489        if x <= 0.0 {
490            return 1.0;
491        }
492        if a <= 0.0 {
493            return 0.0;
494        }
495        if x < a + 1.0 { 1.0 - self.igam(a, x) } else { self.igamc_series(a, x) }
496    }

498    /// Incomplete gamma function (lower)
499    #[allow(clippy::arithmetic_side_effects, clippy::unused_self)] // Mathematical function
500    fn igam(&self, a: f64, x: f64) -> f64 {
501        if x <= 0.0 {
502            return 0.0;
503        }
504        if a <= 0.0 {
505            return 1.0;
506        }
507        if x < a + 1.0 { self.igam_series(a, x) } else { 1.0 - self.igamc_series(a, x) }
508    }

510    /// Incomplete gamma function series expansion
511    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss, clippy::unused_self)] // Mathematical function
512    fn igam_series(&self, a: f64, x: f64) -> f64 {
513        let ax = a * log(x) - x - self.log_gamma(a);
514        if ax < -709.782_712_893_384 {
515            return 0.0;
516        }
517
518        let term = exp(ax);
519        let mut sum = term;
520        let mut c = 1.0;
521
522        for n in 1..1000 {
523            let denom = a + f64::from(n) - 1.0;
524            if denom == 0.0 {
525                break;
526            }
527            c *= x / denom;
528            let next_term = c * term;
529            sum += next_term;
530            if fabs(next_term / sum) < 1e-15 {
531                break;
532            }
533        }
534
535        if a == 0.0 { 0.0 } else { sum / a }
536    }

538    /// Incomplete gamma function continued fraction
539    #[allow(clippy::arithmetic_side_effects, clippy::cast_precision_loss, clippy::unused_self)] // Mathematical function
540    fn igamc_series(&self, a: f64, x: f64) -> f64 {
541        let ax = a * log(x) - x - self.log_gamma(a);
542        if ax < -709.782_712_893_384 {
543            return 0.0;
544        }
545
546        if x == 0.0 {
547            return 0.0;
548        }
549
550        let term = exp(ax) / x;
551        let mut sum = term;
552        let mut c = 1.0;
553
554        for n in 1..1000 {
555            c *= (a - f64::from(n)) / x;
556            let next_term = c * term;
557            sum += next_term;
558            if fabs(next_term / sum) < 1e-15 {
559                break;
560            }
561        }
562
563        sum
564    }

566    /// Log gamma function (Lanczos approximation)
567    #[allow(clippy::arithmetic_side_effects, clippy::unused_self)] // Mathematical function
568    fn log_gamma(&self, x: f64) -> f64 {
569        let cof = [
570            76.18009172947146,
571            -86.50532032941677,
572            24.01409824083091,
573            -1.231739572450155,
574            0.1208650973866179e-2,
575            -0.5395239384953e-5,
576        ];
577
578        let mut y = x;
579        let mut tmp = x + 5.5;
580        tmp -= (x + 0.5) * log(tmp);
581        let mut ser = 1.000000000190015;
582
583        for j in 0..6 {
584            y += 1.0;
585            if let Some(&cof_val) = cof.get(j)
586                && y != 0.0
587            {
588                ser += cof_val / y;
589            }
590        }
591
592        if x == 0.0 {
593            return 0.0;
594        }
595
596        -tmp + log(2.5066282746310005 * ser / x)
597    }
