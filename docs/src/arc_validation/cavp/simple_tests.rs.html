10/// Positive test: Verify CAVP framework basic functionality

13    let storage = crate::cavp::MemoryCavpStorage::new();
14
15    let algorithm = crate::cavp::CavpAlgorithm::MlKem { variant: "768".to_string() };
16
17    let test_result = crate::cavp::CavpTestResult::new(
18        "test-001".to_string(),
19        algorithm.clone(),
20        "vector-001".to_string(),
21        vec![1, 2, 3, 4],
22        vec![1, 2, 3, 4],
23        std::time::Duration::from_millis(100),
24        crate::cavp::CavpTestMetadata::default(),
25    );
26
27    storage.store_result(&test_result).unwrap();
28
29    let retrieved = storage.retrieve_result("test-001").unwrap();

32    assert_eq!(result.test_id, "test-001");

124/// Negative test: Verify CAVP framework handles errors correctly
125#[test]
126fn test_cavp_framework_error_handling() {
127    let storage = crate::cavp::MemoryCavpStorage::new();
128
129    let algorithm = crate::cavp::CavpAlgorithm::SlhDsa { variant: "128".to_string() };
130
131    let failed_result = crate::cavp::CavpTestResult::failed(
132        "test-fail".to_string(),
133        algorithm.clone(),
134        "vector-fail".to_string(),
135        vec![1, 2, 3],
136        vec![4, 5, 6],
137        std::time::Duration::from_millis(50),
138        "Expected mismatch".to_string(),
139        crate::cavp::CavpTestMetadata::default(),
140    );
141
142    storage.store_result(&failed_result).unwrap();
143
144    let retrieved = storage.retrieve_result("test-fail").unwrap();
145    assert!(retrieved.is_some());
146
147    let result = retrieved.unwrap();
148    assert!(!result.passed);
149    assert_eq!(result.error_message, Some("Expected mismatch".to_string()));
150}

152/// Positive test: Verify CAVP framework algorithm enumeration
153#[test]
154fn test_cavp_algorithm_enumeration() {
155    let mlkem_768 = crate::cavp::CavpAlgorithm::MlKem { variant: "768".to_string() };
156    let mldsa_65 = crate::cavp::CavpAlgorithm::MlDsa { variant: "65".to_string() };
157    let slhdsa_128 = crate::cavp::CavpAlgorithm::SlhDsa { variant: "128".to_string() };
158    let fndsa_512 = crate::cavp::CavpAlgorithm::FnDsa { variant: "512".to_string() };
159    let hybrid_kem = crate::cavp::CavpAlgorithm::HybridKem;
160
161    assert_eq!(mlkem_768.name(), "ML-KEM-768");
162    assert_eq!(mldsa_65.name(), "ML-DSA-65");
163    assert_eq!(slhdsa_128.name(), "SLH-DSA-128");
164    assert_eq!(fndsa_512.name(), "FN-DSA-512");
165    assert_eq!(hybrid_kem.name(), "Hybrid-KEM");
166
167    assert_eq!(mlkem_768.fips_standard(), "FIPS 203");
168    assert_eq!(mldsa_65.fips_standard(), "FIPS 204");
169    assert_eq!(slhdsa_128.fips_standard(), "FIPS 205");
170    assert_eq!(fndsa_512.fips_standard(), "FIPS 206");
171    assert_eq!(hybrid_kem.fips_standard(), "FIPS 203 + FIPS 197");
